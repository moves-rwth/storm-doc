126 static RelevancyInformation determineRelevantStatesAndLabels(storm::models::sparse::Model<T> const& model,
127 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
128 storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates,
129 storm::storage::FlatSet<uint_fast64_t> const& dontCareLabels) {
130 // Create result.
131 RelevancyInformation relevancyInformation;
132
133 // Compute all relevant states, i.e. states for which there exists a scheduler that has a non-zero
134 // probabilitiy of satisfying phi until psi.
135 storm::storage::SparseMatrix<T> backwardTransitions = model.getBackwardTransitions();
136 relevancyInformation.relevantStates = storm::utility::graph::performProbGreater0E(backwardTransitions, phiStates, psiStates);
137 relevancyInformation.relevantStates &= ~psiStates;
138
139 STORM_LOG_DEBUG("Found " << relevancyInformation.relevantStates.getNumberOfSetBits() << " relevant states.");
140
141 // Retrieve some references for convenient access.
142 storm::storage::SparseMatrix<T> const& transitionMatrix = model.getTransitionMatrix();
143 std::vector<uint_fast64_t> const& nondeterministicChoiceIndices = transitionMatrix.getRowGroupIndices();
144
145 // Now traverse all choices of all relevant states and check whether there is a successor target state.
146 // If so, the associated labels become relevant. Also, if a choice of a relevant state has at least one
147 // relevant successor, the choice becomes relevant.
148 for (auto state : relevancyInformation.relevantStates) {
149 relevancyInformation.relevantChoicesForRelevantStates.emplace(state, std::list<uint_fast64_t>());
150
151 for (uint_fast64_t row = nondeterministicChoiceIndices[state]; row < nondeterministicChoiceIndices[state + 1]; ++row) {
152 for (auto const& entry : transitionMatrix.getRow(row)) {
153 // If there is a relevant successor, we need to add the labels of the current choice.
154 if (relevancyInformation.relevantStates.get(entry.getColumn()) || psiStates.get(entry.getColumn())) {
155 relevancyInformation.relevantChoicesForRelevantStates[state].push_back(row);
156
157 for (auto const& label : labelSets[row]) {
158 relevancyInformation.relevantLabels.insert(label);
159 }
160
161 relevancyInformation.relevantLabelSets.insert(labelSets[row]);
162 break;
163 }
164 }
165 }
166 }
167
168 // Compute the set of labels that are known to be taken in any case.
169 relevancyInformation.knownLabels = storm::counterexamples::getGuaranteedLabelSet(model, labelSets, psiStates, relevancyInformation.relevantLabels);
170 if (!relevancyInformation.knownLabels.empty()) {
171 storm::storage::FlatSet<uint_fast64_t> remainingLabels;
172 std::set_difference(relevancyInformation.relevantLabels.begin(), relevancyInformation.relevantLabels.end(),
173 relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
174 std::inserter(remainingLabels, remainingLabels.end()));
175 relevancyInformation.relevantLabels = remainingLabels;
176 }
177
178 relevancyInformation.dontCareLabels = dontCareLabels;
179 std::set_difference(relevancyInformation.relevantLabels.begin(), relevancyInformation.relevantLabels.end(), dontCareLabels.begin(),
180 dontCareLabels.end(), std::inserter(relevancyInformation.minimalityLabels, relevancyInformation.minimalityLabels.begin()));
181
182 STORM_LOG_DEBUG("Found " << relevancyInformation.relevantLabels.size() << " relevant and " << relevancyInformation.knownLabels.size()
183 << " known labels.");
184 STORM_LOG_DEBUG("Found " << relevancyInformation.minimalityLabels.size() << " labels to minize over.");
185 return relevancyInformation;
186 }

195 static VariableInformation createVariables(std::shared_ptr<storm::expressions::ExpressionManager> const& manager,
196 storm::models::sparse::Model<T> const& model, storm::storage::BitVector const& psiStates,
197 RelevancyInformation const& relevancyInformation, bool createReachabilityVariables) {
198 VariableInformation variableInformation;
199 variableInformation.manager = manager;
200
201 // Create stringstream to build expression names.
202 std::stringstream variableName;
203
204 // Create the variables for the relevant labels.
205 for (auto label : relevancyInformation.relevantLabels) {
206 variableInformation.labelToIndexMap[label] = variableInformation.labelVariables.size();
207
208 // Clear contents of the stream to construct new expression name.
209 variableName.clear();
210 variableName.str("");
211 variableName << "c" << label;
212
213 variableInformation.labelVariables.push_back(manager->declareBooleanVariable(variableName.str()));
214
215 // Record if the label is among the ones that matter for minimality.
216 if (relevancyInformation.minimalityLabels.find(label) != relevancyInformation.minimalityLabels.end()) {
217 variableInformation.minimalityLabelVariables.push_back(variableInformation.labelVariables.back());
218 }
219
220 // Clear contents of the stream to construct new expression name.
221 variableName.clear();
222 variableName.str("");
223 variableName << "h" << label;
224
225 variableInformation.originalAuxiliaryVariables.push_back(manager->declareBooleanVariable(variableName.str()));
226 }
227
228 // A mapping from each pair of adjacent relevant states to their index in the corresponding variable vector.
229 std::map<std::pair<uint_fast64_t, uint_fast64_t>, uint_fast64_t> statePairToIndexMap;
230
231 // A vector of variables associated with each pair of relevant states (s, s') such that s' is
232 // a successor of s.
233 std::vector<storm::expressions::Variable> statePairVariables;
234
235 // Create variables needed for encoding reachability of a target state if requested.
236 if (createReachabilityVariables) {
237 variableInformation.hasReachabilityVariables = true;
238
239 storm::storage::SparseMatrix<T> const& transitionMatrix = model.getTransitionMatrix();
240
241 for (auto state : relevancyInformation.relevantStates) {
242 variableInformation.relevantStatesToOrderVariableIndexMap[state] = variableInformation.stateOrderVariables.size();
243
244 // Clear contents of the stream to construct new expression name.
245 variableName.clear();
246 variableName.str("");
247 variableName << "o" << state;
248
249 variableInformation.stateOrderVariables.push_back(manager->declareRationalVariable(variableName.str()));
250
251 for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(state)) {
252 for (auto const& entry : transitionMatrix.getRow(relevantChoice)) {
253 // If the successor state is neither the state itself nor an irrelevant state, we need to add a variable for the transition.
254 if (state != entry.getColumn() && (relevancyInformation.relevantStates.get(entry.getColumn()) || psiStates.get(entry.getColumn()))) {
255 // Make sure that there is not already one variable for the state pair. This may happen because of several nondeterministic choices
256 // targeting the same state.
257 if (variableInformation.statePairToIndexMap.find(std::make_pair(state, entry.getColumn())) !=
258 variableInformation.statePairToIndexMap.end()) {
259 continue;
260 }
261
262 // At this point we know that the state-pair does not have an associated variable.
263 variableInformation.statePairToIndexMap[std::make_pair(state, entry.getColumn())] = variableInformation.statePairVariables.size();
264
265 // Clear contents of the stream to construct new expression name.
266 variableName.clear();
267 variableName.str("");
268 variableName << "t" << state << "_" << entry.getColumn();
269
270 variableInformation.statePairVariables.push_back(manager->declareBooleanVariable(variableName.str()));
271 }
272 }
273 }
274 }
275
276 for (auto psiState : psiStates) {
277 variableInformation.relevantStatesToOrderVariableIndexMap[psiState] = variableInformation.stateOrderVariables.size();
278
279 // Clear contents of the stream to construct new expression name.
280 variableName.clear();
281 variableName.str("");
282 variableName << "o" << psiState;
283
284 variableInformation.stateOrderVariables.push_back(manager->declareRationalVariable(variableName.str()));
285 }
286 }
287
288 return variableInformation;
289 }

291 static storm::expressions::Expression getOtherSynchronizationEnabledFormula(
292 storm::storage::FlatSet<uint_fast64_t> const& labelSet,
293 std::map<uint_fast64_t, std::set<storm::storage::FlatSet<uint_fast64_t>>> const& synchronizingLabels,
294 boost::container::flat_map<storm::storage::FlatSet<uint_fast64_t>, storm::expressions::Expression> const& labelSetToFormula,
295 VariableInformation const& variableInformation, RelevancyInformation const& relevancyInformation) {
296 // Taking all commands of a combination does not necessarily mean that a following label set needs to be taken.
297 // This is because it could be that the commands are taken to enable other synchronizations. Therefore, we need
298 // to add an additional clause that says that the right-hand side of the implication is also true if all commands
299 // of the current choice have enabled synchronization options.
300 if (labelSet.size() > 1) {
301 storm::expressions::Expression result = variableInformation.manager->boolean(true);
302 for (auto label : labelSet) {
303 storm::expressions::Expression alternativeExpressionForLabel = variableInformation.manager->boolean(false);
304 std::set<storm::storage::FlatSet<uint_fast64_t>> const& synchsForCommand = synchronizingLabels.at(label);
305
306 for (auto const& synchSet : synchsForCommand) {
307 storm::expressions::Expression alternativeExpression = variableInformation.manager->boolean(true);
308
309 // If the current synchSet is the same as left-hand side of the implication, we can to skip it.
310 if (synchSet == labelSet)
311 continue;
312
313 // Build labels that are missing for this synchronization option.
314 std::set<uint_fast64_t> unknownSynchSetLabels;
315 std::set_difference(synchSet.begin(), synchSet.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
316 std::inserter(unknownSynchSetLabels, unknownSynchSetLabels.end()));
317
318 for (auto label : unknownSynchSetLabels) {
319 alternativeExpression = alternativeExpression && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
320 }
321
322 alternativeExpressionForLabel = alternativeExpressionForLabel || (alternativeExpression && labelSetToFormula.at(synchSet));
323 }
324
325 result = result && alternativeExpressionForLabel;
326 }
327
328 return result;
329 } else {
330 return variableInformation.manager->boolean(false);
331 }
332 }

343 static std::chrono::milliseconds assertCuts(storm::storage::SymbolicModelDescription const& symbolicModel, storm::models::sparse::Model<T> const& model,
344 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
345 storm::storage::BitVector const& psiStates, VariableInformation const& variableInformation,
346 RelevancyInformation const& relevancyInformation, storm::solver::SmtSolver& solver,
347 bool addBackwardImplications) {
348 // Walk through the model and
349 // * identify labels enabled in initial states
350 // * identify labels that can directly precede a given action
351 // * identify labels that directly reach a target state
352 // * identify labels that can directly follow a given action
353 auto assertCutsClock = std::chrono::high_resolution_clock::now();
354
355 //
356 if (addBackwardImplications) {
357 STORM_LOG_THROW(!symbolicModel.isJaniModel() || !symbolicModel.asJaniModel().usesAssignmentLevels(), storm::exceptions::NotSupportedException,
358 "Counterexample generation with backward implications is not supported for indexed assignments");
359 }
360
361 storm::storage::FlatSet<uint_fast64_t> initialLabels;
362 std::set<storm::storage::FlatSet<uint_fast64_t>> initialCombinations;
363 storm::storage::FlatSet<uint_fast64_t> targetLabels;
364 storm::storage::FlatSet<storm::storage::FlatSet<uint_fast64_t>> targetCombinations;
365 std::map<storm::storage::FlatSet<uint_fast64_t>, std::set<storm::storage::FlatSet<uint_fast64_t>>> precedingLabels;
366 std::map<storm::storage::FlatSet<uint_fast64_t>, std::set<storm::storage::FlatSet<uint_fast64_t>>> followingLabels;
367 std::map<uint_fast64_t, std::set<storm::storage::FlatSet<uint_fast64_t>>> synchronizingLabels;
368
369 // Get some data from the model for convenient access.
370 storm::storage::SparseMatrix<T> const& transitionMatrix = model.getTransitionMatrix();
371 storm::storage::SparseMatrix<T> backwardTransitions = model.getBackwardTransitions();
372 storm::storage::BitVector const& initialStates = model.getInitialStates();
373
374 for (auto currentState : relevancyInformation.relevantStates) {
375 bool isInitial = initialStates.get(currentState);
376 for (auto currentChoice : relevancyInformation.relevantChoicesForRelevantStates.at(currentState)) {
377 // If the choice is a synchronization choice, we need to record it.
378 if (labelSets[currentChoice].size() > 1) {
379 for (auto label : labelSets[currentChoice]) {
380 synchronizingLabels[label].emplace(labelSets[currentChoice]);
381 }
382 }
383
384 // If the state is initial, we need to add all the choice labels to the initial label set.
385 if (isInitial) {
386 initialLabels.insert(labelSets[currentChoice].begin(), labelSets[currentChoice].end());
387 initialCombinations.insert(labelSets[currentChoice]);
388 }
389
390 // Iterate over successors and add relevant choices of relevant successors to the following label set.
391 bool canReachTargetState = false;
392 for (auto const& entry : transitionMatrix.getRow(currentChoice)) {
393 if (relevancyInformation.relevantStates.get(entry.getColumn())) {
394 for (auto relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(entry.getColumn())) {
395 if (labelSets[currentChoice] == labelSets[relevantChoice])
396 continue;
397
398 followingLabels[labelSets[currentChoice]].insert(labelSets[relevantChoice]);
399 }
400 } else if (psiStates.get(entry.getColumn())) {
401 canReachTargetState = true;
402 }
403 }
404
405 // If the choice can reach a target state directly, we add all the labels to the target label set.
406 if (canReachTargetState) {
407 targetLabels.insert(labelSets[currentChoice].begin(), labelSets[currentChoice].end());
408 targetCombinations.insert(labelSets[currentChoice]);
409 }
410
411 // Iterate over predecessors and add all choices that target the current state to the preceding
412 // label set of all labels of all relevant choices of the current state.
413 for (auto const& predecessorEntry : backwardTransitions.getRow(currentState)) {
414 if (relevancyInformation.relevantStates.get(predecessorEntry.getColumn())) {
415 for (auto predecessorChoice : relevancyInformation.relevantChoicesForRelevantStates.at(predecessorEntry.getColumn())) {
416 bool choiceTargetsCurrentState = false;
417 for (auto const& successorEntry : transitionMatrix.getRow(predecessorChoice)) {
418 if (successorEntry.getColumn() == currentState) {
419 choiceTargetsCurrentState = true;
420 break;
421 }
422 }
423
424 if (choiceTargetsCurrentState) {
425 precedingLabels[labelSets.at(currentChoice)].insert(labelSets.at(predecessorChoice));
426 }
427 }
428 }
429 }
430 }
431 }
432
433 // Store the found implications in a container similar to the preceding label sets.
434 std::map<storm::storage::FlatSet<uint_fast64_t>, std::set<storm::storage::FlatSet<uint_fast64_t>>> backwardImplications;
435 if (addBackwardImplications) {
436 // Create a new solver over the same variables as the given symbolic model description to use it for
437 // determining the symbolic cuts.
438 std::unique_ptr<storm::solver::SmtSolver> localSolver;
439 if (symbolicModel.isPrismProgram()) {
440 storm::prism::Program const& program = symbolicModel.asPrismProgram();
441 localSolver = std::make_unique<storm::solver::Z3SmtSolver>(program.getManager());
442
443 // Then add the constraints for bounds of the integer variables.
444 for (auto const& integerVariable : program.getGlobalIntegerVariables()) {
445 localSolver->add(integerVariable.getRangeExpression());
446 }
447 for (auto const& module : program.getModules()) {
448 for (auto const& integerVariable : module.getIntegerVariables()) {
449 localSolver->add(integerVariable.getRangeExpression());
450 }
451 }
452 } else {
453 storm::jani::Model const& janiModel = symbolicModel.asJaniModel();
454 localSolver = std::make_unique<storm::solver::Z3SmtSolver>(janiModel.getManager());
455
456 for (auto const& integerVariable : janiModel.getGlobalVariables().getBoundedIntegerVariables()) {
457 if (integerVariable.isTransient()) {
458 continue;
459 }
460 localSolver->add(integerVariable.getRangeExpression());
461 }
462 for (auto const& automaton : janiModel.getAutomata()) {
463 for (auto const& integerVariable : automaton.getVariables().getBoundedIntegerVariables()) {
464 if (integerVariable.isTransient()) {
465 continue;
466 }
467 localSolver->add(integerVariable.getRangeExpression());
468 }
469 }
470 }
471
472 // Construct an expression that exactly characterizes the initial state.
473 storm::expressions::Expression initialStateExpression;
474 if (symbolicModel.isPrismProgram()) {
475 initialStateExpression = symbolicModel.asPrismProgram().getInitialStatesExpression();
476 } else {
477 initialStateExpression = symbolicModel.asJaniModel().getInitialStatesExpression();
478 }
479
480 // Now check for possible backward cuts.
481 for (auto const& labelSetAndPrecedingLabelSetsPair : precedingLabels) {
482 bool backwardImplicationAdded = false;
483 // std::cout << "labelSetAndPrecedingLabelSetsPair.first ";
484 // for (auto const& e : labelSetAndPrecedingLabelSetsPair.first) {
485 // std::cout << e << ", ";
486 // }
487 // std::cout << '\n';
488 // Find out the commands for the currently considered label set.
489 storm::expressions::Expression guardConjunction;
490
491 if (symbolicModel.isPrismProgram()) {
492 storm::prism::Program const& program = symbolicModel.asPrismProgram();
493
494 for (uint_fast64_t moduleIndex = 0; moduleIndex < program.getNumberOfModules(); ++moduleIndex) {
495 storm::prism::Module const& module = program.getModule(moduleIndex);
496
497 for (uint_fast64_t commandIndex = 0; commandIndex < module.getNumberOfCommands(); ++commandIndex) {
498 storm::prism::Command const& command = module.getCommand(commandIndex);
499
500 // If the current command is one of the commands we need to consider, add its guard.
501 if (labelSetAndPrecedingLabelSetsPair.first.find(command.getGlobalIndex()) != labelSetAndPrecedingLabelSetsPair.first.end()) {
502 guardConjunction = guardConjunction && command.getGuardExpression();
503 }
504 }
505 }
506 } else {
507 storm::jani::Model const& janiModel = symbolicModel.asJaniModel();
508
509 for (uint_fast64_t automatonIndex = 0; automatonIndex < janiModel.getNumberOfAutomata(); ++automatonIndex) {
510 storm::jani::Automaton const& automaton = janiModel.getAutomaton(automatonIndex);
511
512 for (uint_fast64_t edgeIndex = 0; edgeIndex < automaton.getNumberOfEdges(); ++edgeIndex) {
513 // If the current edge is one of the edges we need to consider, add its guard.
514 if (labelSetAndPrecedingLabelSetsPair.first.find(janiModel.encodeAutomatonAndEdgeIndices(automatonIndex, edgeIndex)) !=
515 labelSetAndPrecedingLabelSetsPair.first.end()) {
516 storm::jani::Edge const& edge = automaton.getEdge(edgeIndex);
517
518 guardConjunction = guardConjunction && edge.getGuard();
519 }
520 }
521 }
522 }
523
524 // Save the state of the solver so we can easily backtrack.
525 localSolver->push();
526
527 // Push initial state expression.
528 STORM_LOG_DEBUG("About to assert that combination is not enabled in the current state.");
529 localSolver->add(initialStateExpression);
530
531 // Check if the label set is enabled in the initial state.
532 localSolver->add(guardConjunction);
533
534 storm::solver::SmtSolver::CheckResult checkResult = localSolver->check();
535 localSolver->pop();
536 localSolver->push();
537
538 // If the solver reports unsat, then we know that the current selection is not enabled in the initial state.
539 if (checkResult == storm::solver::SmtSolver::CheckResult::Unsat) {
540 STORM_LOG_DEBUG("Selection not enabled in initial state.");
541
542 // std::cout << "not gc: " << !guardConjunction << '\n';
543 localSolver->add(!guardConjunction);
544 STORM_LOG_DEBUG("Asserted disjunction of negated guards.");
545
546 // Now check the possible preceding label sets for the essential ones.
547 for (auto const& precedingLabelSet : labelSetAndPrecedingLabelSetsPair.second) {
548 if (labelSetAndPrecedingLabelSetsPair.first == precedingLabelSet)
549 continue;
550
551 // std::cout << "push\n";
552 // Create a restore point so we can easily pop-off all weakest precondition expressions.
553 localSolver->push();
554
555 // Find out the commands for the currently considered preceding label set.
556 std::vector<std::vector<boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression>>>
557 currentPreceedingVariableUpdates;
558 storm::expressions::Expression preceedingGuardConjunction;
559 if (symbolicModel.isPrismProgram()) {
560 storm::prism::Program const& program = symbolicModel.asPrismProgram();
561 for (uint_fast64_t moduleIndex = 0; moduleIndex < program.getNumberOfModules(); ++moduleIndex) {
562 storm::prism::Module const& module = program.getModule(moduleIndex);
563
564 for (uint_fast64_t commandIndex = 0; commandIndex < module.getNumberOfCommands(); ++commandIndex) {
565 storm::prism::Command const& command = module.getCommand(commandIndex);
566
567 // If the current command is one of the commands we need to consider, store a reference to it in the container.
568 if (precedingLabelSet.find(command.getGlobalIndex()) != precedingLabelSet.end()) {
569 preceedingGuardConjunction = preceedingGuardConjunction && command.getGuardExpression();
570
571 currentPreceedingVariableUpdates.emplace_back();
572
573 for (uint64_t updateIndex = 0; updateIndex < command.getNumberOfUpdates(); ++updateIndex) {
574 storm::prism::Update const& update = command.getUpdate(updateIndex);
575 boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression> variableUpdates;
576 for (auto const& assignment : update.getAssignments()) {
577 variableUpdates.emplace(assignment.getVariable(), assignment.getExpression());
578 }
579 currentPreceedingVariableUpdates.back().emplace_back(std::move(variableUpdates));
580 }
581 }
582 }
583 }
584 } else {
585 storm::jani::Model const& janiModel = symbolicModel.asJaniModel();
586 for (uint_fast64_t automatonIndex = 0; automatonIndex < janiModel.getNumberOfAutomata(); ++automatonIndex) {
587 storm::jani::Automaton const& automaton = janiModel.getAutomaton(automatonIndex);
588
589 for (uint_fast64_t edgeIndex = 0; edgeIndex < automaton.getNumberOfEdges(); ++edgeIndex) {
590 // If the current command is one of the commands we need to consider, store a reference to it in the container.
591 if (precedingLabelSet.find(janiModel.encodeAutomatonAndEdgeIndices(automatonIndex, edgeIndex)) != precedingLabelSet.end()) {
592 storm::jani::Edge const& edge = automaton.getEdge(edgeIndex);
593
594 preceedingGuardConjunction = preceedingGuardConjunction && edge.getGuard();
595
596 currentPreceedingVariableUpdates.emplace_back();
597
598 for (uint64_t destinationIndex = 0; destinationIndex < edge.getNumberOfDestinations(); ++destinationIndex) {
599 storm::jani::EdgeDestination const& destination = edge.getDestination(destinationIndex);
600 boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression> variableUpdates;
601 for (auto const& assignment : destination.getOrderedAssignments().getNonTransientAssignments()) {
602 variableUpdates.emplace(assignment.getVariable().getExpressionVariable(), assignment.getAssignedExpression());
603 }
604 currentPreceedingVariableUpdates.back().emplace_back(std::move(variableUpdates));
605 }
606 }
607 }
608 }
609 }
610
611 // std::cout << "pgc: " << preceedingGuardConjunction << '\n';
612
613 // Assert all the guards of the preceding command set.
614 localSolver->add(preceedingGuardConjunction);
615
616 std::vector<std::vector<boost::container::flat_map<storm::expressions::Variable, storm::expressions::Expression>>::const_iterator>
617 iteratorVector;
618 for (auto const& variableUpdates : currentPreceedingVariableUpdates) {
619 iteratorVector.push_back(variableUpdates.begin());
620 }
621
622 // Iterate over all possible combinations of updates of the preceding command set.
623 std::vector<storm::expressions::Expression> formulae;
624 while (true) {
625 std::map<storm::expressions::Variable, storm::expressions::Expression> currentVariableUpdateCombinationMap;
626 for (auto const& updateIterator : iteratorVector) {
627 for (auto const& variableUpdatePair : *updateIterator) {
628 currentVariableUpdateCombinationMap.emplace(variableUpdatePair.first, variableUpdatePair.second);
629 }
630 }
631
632 STORM_LOG_DEBUG("About to assert a weakest precondition.");
633 storm::expressions::Expression wp = guardConjunction.substitute(currentVariableUpdateCombinationMap);
634 formulae.push_back(wp);
635 STORM_LOG_DEBUG("Asserted weakest precondition.");
636
637 // Now try to move iterators to the next position if possible. If we could properly
638 // move it, we can directly move on to the next combination of updates. If we have
639 // to reset it to the start, we do so unless there is no more iterator to reset.
640 uint_fast64_t k = iteratorVector.size();
641 for (; k > 0; --k) {
642 ++iteratorVector[k - 1];
643 if (iteratorVector[k - 1] == currentPreceedingVariableUpdates[k - 1].end()) {
644 iteratorVector[k - 1] = currentPreceedingVariableUpdates[k - 1].begin();
645 } else {
646 break;
647 }
648 }
649
650 // If we had to reset all iterator to the start, we are done.
651 if (k == 0) {
652 break;
653 }
654 }
655
656 // Now assert the disjunction of all weakest preconditions of all considered update combinations.
657 assertDisjunction(
658 *localSolver, formulae,
659 symbolicModel.isPrismProgram() ? symbolicModel.asPrismProgram().getManager() : symbolicModel.asJaniModel().getManager());
660
661 STORM_LOG_DEBUG("Asserted disjunction of all weakest preconditions.");
662 storm::solver::SmtSolver::CheckResult result = localSolver->check();
663
664 if (result == storm::solver::SmtSolver::CheckResult::Sat) {
665 backwardImplications[labelSetAndPrecedingLabelSetsPair.first].insert(precedingLabelSet);
666 backwardImplicationAdded = true;
667 } else if (result == storm::solver::SmtSolver::CheckResult::Unknown) {
668 STORM_LOG_ERROR("The SMT solver does not come to a conclusive answer. Does your model contain integer division?");
669 }
670
671 localSolver->pop();
672 }
673
674 // Popping the disjunction of negated guards from the solver stack.
675 localSolver->pop();
676 STORM_LOG_ERROR_COND(backwardImplicationAdded,
677 "Error in adding cuts for counterexample generation (backward implication misses a label set).");
678 } else {
679 STORM_LOG_DEBUG("Selection is enabled in initial state.");
680 }
681 }
682 } else if (symbolicModel.isJaniModel()) {
683 STORM_LOG_WARN("Model uses assignment levels, did not assert backward implications.");
684 }
685
686 STORM_LOG_DEBUG("Successfully gathered data for cuts.");
687
688 // Compute the sets of labels such that the transitions labeled with this set possess at least one known label.
689 storm::storage::FlatSet<storm::storage::FlatSet<uint_fast64_t>> hasKnownSuccessor;
690 for (auto const& labelSetFollowingSetsPair : followingLabels) {
691 for (auto const& set : labelSetFollowingSetsPair.second) {
692 if (std::includes(relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), set.begin(), set.end())) {
693 hasKnownSuccessor.insert(set);
694 break;
695 }
696 }
697 }
698
699 // Compute the sets of labels such that the transitions labeled with this set possess at least one known predecessor.
700 storm::storage::FlatSet<storm::storage::FlatSet<uint_fast64_t>> hasKnownPredecessor;
701 for (auto const& labelSetImplicationsPair : backwardImplications) {
702 for (auto const& set : labelSetImplicationsPair.second) {
703 if (std::includes(relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(), set.begin(), set.end())) {
704 hasKnownPredecessor.insert(labelSetImplicationsPair.first);
705 break;
706 }
707 }
708 }
709
710 STORM_LOG_DEBUG("Asserting initial combination is taken.");
711 {
712 std::vector<storm::expressions::Expression> formulae;
713
714 // Start by asserting that we take at least one initial label. We may do so only if there is no initial
715 // combination that is already known. Otherwise this condition would be too strong.
716 bool initialCombinationKnown = false;
717 for (auto const& combination : initialCombinations) {
718 storm::storage::FlatSet<uint_fast64_t> tmpSet;
719 std::set_difference(combination.begin(), combination.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
720 std::inserter(tmpSet, tmpSet.end()));
721 if (tmpSet.size() == 0) {
722 initialCombinationKnown = true;
723 break;
724 } else {
725 storm::expressions::Expression conj = variableInformation.manager->boolean(true);
726 for (auto label : tmpSet) {
727 conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
728 }
729 formulae.push_back(conj);
730 }
731 }
732 if (!initialCombinationKnown) {
733 assertDisjunction(solver, formulae, *variableInformation.manager);
734 }
735 }
736
737 STORM_LOG_DEBUG("Asserting target combination is taken.");
738 {
739 std::vector<storm::expressions::Expression> formulae;
740
741 // Likewise, if no target combination is known, we may assert that there is at least one.
742 bool targetCombinationKnown = false;
743 for (auto const& combination : targetCombinations) {
744 storm::storage::FlatSet<uint_fast64_t> tmpSet;
745 std::set_difference(combination.begin(), combination.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
746 std::inserter(tmpSet, tmpSet.end()));
747 if (tmpSet.size() == 0) {
748 targetCombinationKnown = true;
749 break;
750 } else {
751 storm::expressions::Expression conj = variableInformation.manager->boolean(true);
752 for (auto label : tmpSet) {
753 conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
754 }
755 formulae.push_back(conj);
756 }
757 }
758 if (!targetCombinationKnown) {
759 assertDisjunction(solver, formulae, *variableInformation.manager);
760 }
761 }
762
763 if (addBackwardImplications) {
764 STORM_LOG_DEBUG(
765 "Asserting taken labels are followed and preceeded by another label if they are not a target label or an initial label, respectively.");
766 boost::container::flat_map<storm::storage::FlatSet<uint_fast64_t>, storm::expressions::Expression> labelSetToFormula;
767 for (auto const& labelSet : relevancyInformation.relevantLabelSets) {
768 storm::expressions::Expression labelSetFormula = variableInformation.manager->boolean(false);
769
770 // Compute the set of unknown labels on the left-hand side of the implication.
771 storm::storage::FlatSet<uint_fast64_t> unknownLhsLabels;
772 std::set_difference(labelSet.begin(), labelSet.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
773 std::inserter(unknownLhsLabels, unknownLhsLabels.end()));
774 for (auto label : unknownLhsLabels) {
775 labelSetFormula = labelSetFormula || !variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
776 }
777
778 // Only build a constraint if the combination does not lead to a target state and
779 // no successor set is already known.
780 storm::expressions::Expression successorExpression;
781 if (targetCombinations.find(labelSet) == targetCombinations.end() && hasKnownSuccessor.find(labelSet) == hasKnownSuccessor.end()) {
782 successorExpression = variableInformation.manager->boolean(false);
783
784 auto const& followingLabelSets = followingLabels.at(labelSet);
785
786 for (auto const& followingSet : followingLabelSets) {
787 storm::storage::FlatSet<uint_fast64_t> tmpSet;
788
789 // Check which labels of the current following set are not known. This set must be non-empty, because
790 // otherwise a successor combination would already be known and control cannot reach this point.
791 std::set_difference(followingSet.begin(), followingSet.end(), relevancyInformation.knownLabels.begin(),
792 relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
793
794 // Construct an expression that enables all unknown labels of the current following set.
795 storm::expressions::Expression conj = variableInformation.manager->boolean(true);
796 for (auto label : tmpSet) {
797 conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
798 }
799 successorExpression = successorExpression || conj;
800 }
801 } else {
802 successorExpression = variableInformation.manager->boolean(true);
803 }
804 // Constructed following cuts at this point.
805
806 // Only build a constraint if the combination is no initial combination and no
807 // predecessor set is already known.
808 storm::expressions::Expression predecessorExpression;
809 if (initialCombinations.find(labelSet) == initialCombinations.end() && hasKnownPredecessor.find(labelSet) == hasKnownPredecessor.end()) {
810 predecessorExpression = variableInformation.manager->boolean(false);
811
812 // std::cout << "labelSet\n";
813 // for (auto const& e : labelSet) {
814 // std::cout << e << ", ";
815 // }
816 // std::cout << '\n';
817
818 auto const& preceedingLabelSets = backwardImplications.at(labelSet);
819
820 for (auto const& preceedingSet : preceedingLabelSets) {
821 storm::storage::FlatSet<uint_fast64_t> tmpSet;
822
823 // Check which labels of the current following set are not known. This set must be non-empty, because
824 // otherwise a predecessor combination would already be known and control cannot reach this point.
825 std::set_difference(preceedingSet.begin(), preceedingSet.end(), relevancyInformation.knownLabels.begin(),
826 relevancyInformation.knownLabels.end(), std::inserter(tmpSet, tmpSet.end()));
827
828 // Construct an expression that enables all unknown labels of the current following set.
829 storm::expressions::Expression conj = variableInformation.manager->boolean(true);
830 for (auto label : tmpSet) {
831 conj = conj && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
832 }
833 predecessorExpression = predecessorExpression || conj;
834 }
835 } else {
836 predecessorExpression = variableInformation.manager->boolean(true);
837 }
838
839 labelSetFormula = labelSetFormula || (successorExpression && predecessorExpression);
840
841 labelSetToFormula[labelSet] = labelSetFormula;
842 }
843
844 for (auto const& labelSetFormula : labelSetToFormula) {
845 solver.add(labelSetFormula.second || getOtherSynchronizationEnabledFormula(labelSetFormula.first, synchronizingLabels, labelSetToFormula,
846 variableInformation, relevancyInformation));
847 }
848 }
849
850 STORM_LOG_DEBUG("Asserting synchronization cuts.");
851 // Finally, assert that if we take one of the synchronizing labels, we also take one of the combinations
852 // the label appears in.
853 for (auto const& labelSynchronizingSetsPair : synchronizingLabels) {
854 std::vector<storm::expressions::Expression> formulae;
855
856 if (relevancyInformation.knownLabels.find(labelSynchronizingSetsPair.first) == relevancyInformation.knownLabels.end()) {
857 formulae.push_back(!variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(labelSynchronizingSetsPair.first)));
858 }
859
860 // We need to be careful, because there may be one synchronisation set out of which all labels are
861 // known, which means we must not assert anything.
862 bool allImplicantsKnownForOneSet = false;
863 for (auto const& synchronizingSet : labelSynchronizingSetsPair.second) {
864 storm::expressions::Expression currentCombination = variableInformation.manager->boolean(true);
865 bool allImplicantsKnownForCurrentSet = true;
866 for (auto label : synchronizingSet) {
867 if (relevancyInformation.knownLabels.find(label) == relevancyInformation.knownLabels.end() && label != labelSynchronizingSetsPair.first) {
868 currentCombination = currentCombination && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label));
869 }
870 }
871 formulae.push_back(currentCombination);
872
873 // If all implicants of the current set are known, we do not need to further build the constraint.
874 if (allImplicantsKnownForCurrentSet) {
875 allImplicantsKnownForOneSet = true;
876 break;
877 }
878 }
879
880 if (!allImplicantsKnownForOneSet) {
881 assertDisjunction(solver, formulae, *variableInformation.manager);
882 }
883 }
884
885 auto endTime = std::chrono::high_resolution_clock::now();
886 return std::chrono::duration_cast<std::chrono::milliseconds>(endTime - assertCutsClock);
887 }

892 static void assertReachabilityCuts(storm::models::sparse::Model<T> const& model, std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
893 storm::storage::BitVector const& psiStates, VariableInformation const& variableInformation,
894 RelevancyInformation const& relevancyInformation, storm::solver::SmtSolver& solver) {
895 if (!variableInformation.hasReachabilityVariables) {
896 throw storm::exceptions::InvalidStateException() << "Impossible to assert reachability cuts without the necessary variables.";
897 }
898
899 // Get some data from the model for convenient access.
900 storm::storage::SparseMatrix<T> const& transitionMatrix = model.getTransitionMatrix();
901 storm::storage::SparseMatrix<T> backwardTransitions = model.getBackwardTransitions();
902
903 // First, we add the formulas that encode
904 // (1) if an incoming transition is chosen, an outgoing one is chosen as well (for non-initial states)
905 // (2) an outgoing transition out of the initial states is taken.
906 storm::expressions::Expression initialStateExpression = variableInformation.manager->boolean(false);
907 for (auto relevantState : relevancyInformation.relevantStates) {
908 if (!model.getInitialStates().get(relevantState)) {
909 // Assert the constraints (1).
910 storm::storage::FlatSet<uint_fast64_t> relevantPredecessors;
911 for (auto const& predecessorEntry : backwardTransitions.getRow(relevantState)) {
912 if (relevantState != predecessorEntry.getColumn() && relevancyInformation.relevantStates.get(predecessorEntry.getColumn())) {
913 relevantPredecessors.insert(predecessorEntry.getColumn());
914 }
915 }
916
917 storm::storage::FlatSet<uint_fast64_t> relevantSuccessors;
918 for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(relevantState)) {
919 for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
920 if (relevantState != successorEntry.getColumn() &&
921 (relevancyInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn()))) {
922 relevantSuccessors.insert(successorEntry.getColumn());
923 }
924 }
925 }
926
927 storm::expressions::Expression expression = variableInformation.manager->boolean(true);
928 for (auto predecessor : relevantPredecessors) {
929 expression = expression && !variableInformation.statePairVariables.at(
930 variableInformation.statePairToIndexMap.at(std::make_pair(predecessor, relevantState)));
931 }
932 for (auto successor : relevantSuccessors) {
933 expression = expression || variableInformation.statePairVariables.at(
934 variableInformation.statePairToIndexMap.at(std::make_pair(relevantState, successor)));
935 }
936
937 solver.add(expression);
938 } else {
939 // Assert the constraints (2).
940 storm::storage::FlatSet<uint_fast64_t> relevantSuccessors;
941 for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(relevantState)) {
942 for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
943 if (relevantState != successorEntry.getColumn() &&
944 (relevancyInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn()))) {
945 relevantSuccessors.insert(successorEntry.getColumn());
946 }
947 }
948 }
949
950 for (auto successor : relevantSuccessors) {
951 initialStateExpression =
952 initialStateExpression ||
953 variableInformation.statePairVariables.at(variableInformation.statePairToIndexMap.at(std::make_pair(relevantState, successor)));
954 }
955 }
956 }
957 solver.add(initialStateExpression);
958
959 // Finally, add constraints that
960 // (1) if a transition is selected, a valid labeling is selected as well.
961 // (2) enforce that if a transition from s to s' is selected, the ordering variables become strictly larger.
962 for (auto const& statePairIndexPair : variableInformation.statePairToIndexMap) {
963 uint_fast64_t sourceState = statePairIndexPair.first.first;
964 uint_fast64_t targetState = statePairIndexPair.first.second;
965
966 // Assert constraint for (1).
967 storm::storage::FlatSet<uint_fast64_t> choicesForStatePair;
968 for (auto const& relevantChoice : relevancyInformation.relevantChoicesForRelevantStates.at(sourceState)) {
969 for (auto const& successorEntry : transitionMatrix.getRow(relevantChoice)) {
970 if (successorEntry.getColumn() == targetState) {
971 choicesForStatePair.insert(relevantChoice);
972 }
973 }
974 }
975 storm::expressions::Expression labelExpression = !variableInformation.statePairVariables.at(statePairIndexPair.second);
976 for (auto choice : choicesForStatePair) {
977 storm::expressions::Expression choiceExpression = variableInformation.manager->boolean(true);
978 for (auto element : labelSets.at(choice)) {
979 if (relevancyInformation.knownLabels.find(element) == relevancyInformation.knownLabels.end()) {
980 choiceExpression = choiceExpression && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(element));
981 }
982 }
983 labelExpression = labelExpression || choiceExpression;
984 }
985 solver.add(labelExpression);
986
987 // Assert constraint for (2).
988 storm::expressions::Expression orderExpression =
989 !variableInformation.statePairVariables.at(statePairIndexPair.second) ||
990 variableInformation.stateOrderVariables.at(variableInformation.relevantStatesToOrderVariableIndexMap.at(sourceState)).getExpression() <
991 variableInformation.stateOrderVariables.at(variableInformation.relevantStatesToOrderVariableIndexMap.at(targetState)).getExpression();
992 solver.add(orderExpression);
993 }
994 }

1022 static std::pair<storm::expressions::Expression, storm::expressions::Expression> createFullAdder(storm::expressions::Expression in1,
1023 storm::expressions::Expression in2,
1024 storm::expressions::Expression carryIn) {
1025 storm::expressions::Expression resultBit;
1026 storm::expressions::Expression carryBit;
1027
1028 if (carryIn.isFalse()) {
1029 resultBit = (in1 && !in2) || (!in1 && in2);
1030 carryBit = in1 && in2;
1031 } else {
1032 resultBit = (in1 && !in2 && !carryIn) || (!in1 && in2 && !carryIn) || (!in1 && !in2 && carryIn) || (in1 && in2 && carryIn);
1033 carryBit = in1 && in2 || in1 && carryIn || in2 && carryIn;
1034 }
1035
1036 return std::make_pair(carryBit, resultBit);
1037 }

1048 static std::vector<storm::expressions::Expression> createAdder(VariableInformation const& variableInformation,
1049 std::vector<storm::expressions::Expression> const& in1,
1050 std::vector<storm::expressions::Expression> const& in2) {
1051 // Sanity check for sizes of input.
1052 if (in1.size() != in2.size() || in1.size() == 0) {
1053 STORM_LOG_ERROR("Illegal input to adder (" << in1.size() << ", " << in2.size() << ").");
1054 throw storm::exceptions::InvalidArgumentException() << "Illegal input to adder.";
1055 }
1056
1057 // Prepare result.
1058 std::vector<storm::expressions::Expression> result;
1059 result.reserve(in1.size() + 1);
1060
1061 // Add all bits individually and pass on carry bit appropriately.
1062 storm::expressions::Expression carryBit = variableInformation.manager->boolean(false);
1063 for (uint_fast64_t currentBit = 0; currentBit < in1.size(); ++currentBit) {
1064 std::pair<storm::expressions::Expression, storm::expressions::Expression> localResult = createFullAdder(in1[currentBit], in2[currentBit], carryBit);
1065
1066 result.push_back(localResult.second);
1067 carryBit = localResult.first;
1068 }
1069 result.push_back(carryBit);
1070
1071 return result;
1072 }

1110 std::vector<storm::expressions::Variable> const& literals) {

1112 return std::vector<storm::expressions::Expression>();

1125
1126 STORM_LOG_DEBUG("Created counter circuit for " << literals.size() << " literals.");
1127

1253 static boost::optional<storm::storage::FlatSet<uint_fast64_t>> findSmallestCommandSet(storm::solver::SmtSolver& solver,
1254 VariableInformation& variableInformation,
1255 uint_fast64_t& currentBound) {
1256 // Check if we can find a solution with the current bound.
1257 storm::expressions::Expression assumption = !variableInformation.auxiliaryVariables.back();
1258
1259 // As long as the constraints are unsatisfiable, we need to relax the last at-most-k constraint and
1260 // try with an increased bound.
1261 while (solver.checkWithAssumptions({assumption}) == storm::solver::SmtSolver::CheckResult::Unsat) {
1262 STORM_LOG_DEBUG("Constraint system is unsatisfiable with at most " << currentBound << " taken commands; increasing bound.");
1263 solver.add(variableInformation.auxiliaryVariables.back());
1264 variableInformation.auxiliaryVariables.push_back(assertLessOrEqualKRelaxed(solver, variableInformation, ++currentBound));
1265 assumption = !variableInformation.auxiliaryVariables.back();
1266 if (currentBound > variableInformation.minimalityLabelVariables.size()) {
1267 STORM_LOG_DEBUG("Constraint system fully explored: Bound exceeds maximum of " << variableInformation.minimalityLabelVariables.size());
1268 return boost::none;
1269 }
1270 }
1271
1272 // At this point we know that the constraint system was satisfiable, so compute the induced label
1273 // set and return it.
1274 return getUsedLabelSet(*solver.getModel(), variableInformation);

1335 static void analyzeZeroProbabilitySolution(storm::solver::SmtSolver& solver, storm::models::sparse::Model<T> const& subModel,
1336 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& subLabelSets,
1337 storm::models::sparse::Model<T> const& originalModel,
1338 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& originalLabelSets,
1339 storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates,
1340 storm::storage::FlatSet<uint_fast64_t> const& commandSet, VariableInformation& variableInformation,
1341 RelevancyInformation const& relevancyInformation) {
1342 storm::storage::BitVector reachableStates(subModel.getNumberOfStates());
1343
1344 STORM_LOG_DEBUG("Analyzing solution with zero probability.");
1345
1346 // Initialize the stack for the DFS.
1347 bool targetStateIsReachable = false;
1348 std::vector<uint_fast64_t> stack;
1349 stack.reserve(subModel.getNumberOfStates());
1350 for (auto initialState : subModel.getInitialStates()) {
1351 stack.push_back(initialState);
1352 reachableStates.set(initialState, true);
1353 }
1354
1355 storm::storage::SparseMatrix<T> const& transitionMatrix = subModel.getTransitionMatrix();
1356 std::vector<uint_fast64_t> const& nondeterministicChoiceIndices = transitionMatrix.getRowGroupIndices();
1357
1358 // Now determine which states and labels are actually reachable.
1359 storm::storage::FlatSet<uint_fast64_t> reachableLabels;
1360 while (!stack.empty()) {
1361 uint_fast64_t currentState = stack.back();
1362 stack.pop_back();
1363
1364 for (uint_fast64_t currentChoice = nondeterministicChoiceIndices[currentState]; currentChoice < nondeterministicChoiceIndices[currentState + 1];
1365 ++currentChoice) {
1366 bool choiceTargetsRelevantState = false;
1367
1368 for (auto const& successorEntry : transitionMatrix.getRow(currentChoice)) {
1369 if (relevancyInformation.relevantStates.get(successorEntry.getColumn()) && currentState != successorEntry.getColumn()) {
1370 choiceTargetsRelevantState = true;
1371 if (!reachableStates.get(successorEntry.getColumn())) {
1372 reachableStates.set(successorEntry.getColumn());
1373 stack.push_back(successorEntry.getColumn());
1374 }
1375 } else if (psiStates.get(successorEntry.getColumn())) {
1376 targetStateIsReachable = true;
1377 }
1378 }
1379
1380 if (choiceTargetsRelevantState) {
1381 for (auto label : subLabelSets[currentChoice]) {
1382 reachableLabels.insert(label);
1383 }
1384 }
1385 }
1386 }
1387
1388 STORM_LOG_DEBUG("Successfully performed reachability analysis.");
1389
1390 if (targetStateIsReachable) {
1391 STORM_LOG_ERROR("Target must be unreachable for this analysis.");
1392 throw storm::exceptions::InvalidStateException() << "Target must be unreachable for this analysis.";
1393 }
1394
1395 storm::storage::BitVector unreachableRelevantStates = ~reachableStates & relevancyInformation.relevantStates;
1396 storm::storage::BitVector statesThatCanReachTargetStates =
1397 storm::utility::graph::performProbGreater0E(subModel.getBackwardTransitions(), phiStates, psiStates);
1398
1399 storm::storage::FlatSet<uint_fast64_t> locallyRelevantLabels;
1400 std::set_difference(relevancyInformation.relevantLabels.begin(), relevancyInformation.relevantLabels.end(), commandSet.begin(), commandSet.end(),
1401 std::inserter(locallyRelevantLabels, locallyRelevantLabels.begin()));
1402
1403 std::vector<storm::storage::FlatSet<uint_fast64_t>> guaranteedLabelSets =
1404 storm::counterexamples::getGuaranteedLabelSets(originalModel, originalLabelSets, statesThatCanReachTargetStates, locallyRelevantLabels);
1405 STORM_LOG_DEBUG("Found " << reachableLabels.size() << " reachable labels and " << reachableStates.getNumberOfSetBits() << " reachable states.");
1406
1407 // Search for states on the border of the reachable state space, i.e. states that are still reachable
1408 // and possess a (disabled) option to leave the reachable part of the state space.
1409 std::set<storm::storage::FlatSet<uint_fast64_t>> cutLabels;
1410 for (auto state : reachableStates) {
1411 for (auto currentChoice : relevancyInformation.relevantChoicesForRelevantStates.at(state)) {
1412 if (!std::includes(commandSet.begin(), commandSet.end(), originalLabelSets[currentChoice].begin(), originalLabelSets[currentChoice].end())) {
1413 bool isBorderChoice = false;
1414
1415 // Determine whether the state has the option to leave the reachable state space and go to the unreachable relevant states.
1416 for (auto const& successorEntry : originalModel.getTransitionMatrix().getRow(currentChoice)) {
1417 if (unreachableRelevantStates.get(successorEntry.getColumn())) {
1418 isBorderChoice = true;
1419 }
1420 }
1421
1422 if (isBorderChoice) {
1423 storm::storage::FlatSet<uint_fast64_t> currentLabelSet;
1424 std::set_difference(originalLabelSets[currentChoice].begin(), originalLabelSets[currentChoice].end(), commandSet.begin(),
1425 commandSet.end(), std::inserter(currentLabelSet, currentLabelSet.begin()));
1426 std::set_difference(guaranteedLabelSets[state].begin(), guaranteedLabelSets[state].end(), commandSet.begin(), commandSet.end(),
1427 std::inserter(currentLabelSet, currentLabelSet.end()));
1428
1429 cutLabels.insert(currentLabelSet);
1430 }
1431 }
1432 }
1433 }
1434
1435 // Given the results of the previous analysis, we construct the implications.
1436 std::vector<storm::expressions::Expression> formulae;
1437 storm::storage::FlatSet<uint_fast64_t> unknownReachableLabels;
1438 std::set_difference(reachableLabels.begin(), reachableLabels.end(), relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end(),
1439 std::inserter(unknownReachableLabels, unknownReachableLabels.end()));
1440 storm::storage::FlatSet<uint64_t> unknownReachableMinimalityLabels;
1441 std::set_intersection(unknownReachableLabels.begin(), unknownReachableLabels.end(), relevancyInformation.minimalityLabels.begin(),
1442 relevancyInformation.minimalityLabels.end(),
1443 std::inserter(unknownReachableMinimalityLabels, unknownReachableMinimalityLabels.end()));
1444
1445 for (auto label : unknownReachableMinimalityLabels) {
1446 formulae.push_back(!variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(label)));
1447 }
1448 for (auto const& cutLabelSet : cutLabels) {
1449 storm::expressions::Expression cube = variableInformation.manager->boolean(true);
1450 for (auto cutLabel : cutLabelSet) {
1451 cube = cube && variableInformation.labelVariables.at(variableInformation.labelToIndexMap.at(cutLabel));
1452 }
1453
1454 formulae.push_back(cube);
1455 }
1456
1457 STORM_LOG_DEBUG("Asserting reachability implications.");
1458 assertDisjunction(solver, formulae, *variableInformation.manager);
1459 }

1584 static std::pair<std::shared_ptr<storm::models::sparse::Model<T>>, std::vector<storm::storage::FlatSet<uint_fast64_t>>> restrictModelToLabelSet(
1585 storm::models::sparse::Model<T> const& model, storm::storage::FlatSet<uint_fast64_t> const& filterLabelSet,
1586 boost::optional<uint64_t> absorbState = boost::none) {
1587 bool customRowGrouping = model.isOfType(storm::models::ModelType::Mdp);
1588 STORM_LOG_TRACE("Restrict model to label set " << storm::storage::toString(filterLabelSet));
1589 STORM_LOG_TRACE("Absorb state = " << (absorbState == boost::none ? "none" : std::to_string(absorbState.get())));
1590 std::vector<storm::storage::FlatSet<uint_fast64_t>> resultLabelSet;
1591 storm::storage::SparseMatrixBuilder<T> transitionMatrixBuilder(0, model.getTransitionMatrix().getColumnCount(), 0, true, customRowGrouping,
1592 model.getTransitionMatrix().getRowGroupCount());
1593
1594 // Check for each choice of each state, whether the choice commands are fully contained in the given command set.
1595 uint_fast64_t currentRow = 0;
1596 for (uint_fast64_t state = 0; state < model.getNumberOfStates(); ++state) {
1597 bool stateHasValidChoice = false;
1598 for (uint_fast64_t choice = model.getTransitionMatrix().getRowGroupIndices()[state];
1599 choice < model.getTransitionMatrix().getRowGroupIndices()[state + 1]; ++choice) {
1600 auto const& choiceLabelSet = model.getChoiceOrigins()->isPrismChoiceOrigins()
1601 ? model.getChoiceOrigins()->asPrismChoiceOrigins().getCommandSet(choice)
1602 : model.getChoiceOrigins()->asJaniChoiceOrigins().getEdgeIndexSet(choice);
1603 bool choiceValid = std::includes(filterLabelSet.begin(), filterLabelSet.end(), choiceLabelSet.begin(), choiceLabelSet.end());
1604
1605 // If the choice is valid, copy over all its elements.
1606 if (choiceValid) {
1607 STORM_LOG_TRACE("Choice " << choice << " has a valid label set " << storm::storage::toString(choiceLabelSet));
1608
1609 if (!stateHasValidChoice && customRowGrouping) {
1610 transitionMatrixBuilder.newRowGroup(currentRow);
1611 }
1612 stateHasValidChoice = true;
1613 for (auto const& entry : model.getTransitionMatrix().getRow(choice)) {
1614 transitionMatrixBuilder.addNextValue(currentRow, entry.getColumn(), entry.getValue());
1615 }
1616 resultLabelSet.push_back(choiceLabelSet);
1617 ++currentRow;
1618 }
1619 }
1620
1621 // If no choice of the current state may be taken, we insert a self-loop to the state instead.
1622 if (!stateHasValidChoice) {
1623 if (customRowGrouping) {
1624 transitionMatrixBuilder.newRowGroup(currentRow);
1625 }
1626 uint64_t targetState = (absorbState == boost::none ? state : absorbState.get());
1627 transitionMatrixBuilder.addNextValue(currentRow, targetState, storm::utility::one<T>());
1628 // Insert an empty label set for this choice
1629 resultLabelSet.emplace_back();
1630 ++currentRow;
1631 }
1632 }
1633
1634 std::shared_ptr<storm::models::sparse::Model<T>> resultModel;
1635 if (model.isOfType(storm::models::ModelType::Dtmc)) {
1636 resultModel = std::make_shared<storm::models::sparse::Dtmc<T>>(
1637 transitionMatrixBuilder.build(), storm::models::sparse::StateLabeling(model.getStateLabeling()), model.getRewardModels());
1638 } else {
1639 resultModel = std::make_shared<storm::models::sparse::Mdp<T>>(
1640 transitionMatrixBuilder.build(), storm::models::sparse::StateLabeling(model.getStateLabeling()), model.getRewardModels());
1641 }
1642
1643 return std::make_pair(resultModel, std::move(resultLabelSet));
1644 }

1751 auto setupTimeClock = std::chrono::high_resolution_clock::now();
1752 decltype(std::chrono::high_resolution_clock::now() - setupTimeClock) totalSetupTime(0);
1753
1754 auto solverClock = std::chrono::high_resolution_clock::now();
1755 decltype(std::chrono::high_resolution_clock::now() - solverClock) totalSolverTime(0);
1756
1757 auto modelCheckingClock = std::chrono::high_resolution_clock::now();
1758 decltype(std::chrono::high_resolution_clock::now() - modelCheckingClock) totalModelCheckingTime(0);
1759
1760 auto analysisClock = std::chrono::high_resolution_clock::now();
1761 decltype(std::chrono::high_resolution_clock::now() - analysisClock) totalAnalysisTime(0);
1762
1763 // (0) Obtain the label sets for each choice.
1764 // The label set of a choice corresponds to the set of prism commands that induce the choice.
1765 STORM_LOG_THROW(model.hasChoiceOrigins(), storm::exceptions::InvalidArgumentException,
1766 "Restriction to minimal command set is impossible for model without choice origins.");
1767 STORM_LOG_THROW(model.getChoiceOrigins()->isPrismChoiceOrigins() || model.getChoiceOrigins()->isJaniChoiceOrigins(),
1768 storm::exceptions::InvalidArgumentException, "Restriction to label set is impossible for model without PRISM or JANI choice origins.");
1769
1770 std::vector<storm::storage::FlatSet<uint_fast64_t>> labelSets(model.getNumberOfChoices());
1771 if (model.getChoiceOrigins()->isPrismChoiceOrigins()) {
1772 storm::storage::sparse::PrismChoiceOrigins const& choiceOrigins = model.getChoiceOrigins()->asPrismChoiceOrigins();
1773 for (uint_fast64_t choice = 0; choice < model.getNumberOfChoices(); ++choice) {
1774 labelSets[choice] = choiceOrigins.getCommandSet(choice);
1775 }
1776 } else {
1777 storm::storage::sparse::JaniChoiceOrigins const& choiceOrigins = model.getChoiceOrigins()->asJaniChoiceOrigins();
1778 for (uint_fast64_t choice = 0; choice < model.getNumberOfChoices(); ++choice) {
1779 labelSets[choice] = choiceOrigins.getEdgeIndexSet(choice);
1780 }
1781 }
1782 assert(labelSets.size() == model.getNumberOfChoices());
1783
1784 // (1) Check whether its possible to exceed the threshold if checkThresholdFeasible is set.
1785 std::vector<double> maximalReachabilityProbability;
1786 if (options.checkThresholdFeasible) {
1787 maximalReachabilityProbability = computeMaximalReachabilityProbability(env, model, phiStates, psiStates, rewardName);
1788
1789 for (uint64_t i = 0; i < maximalReachabilityProbability.size(); ++i) {
1790 STORM_LOG_THROW((strictBound && maximalReachabilityProbability[i] >= propertyThreshold[i]) ||
1791 (!strictBound && maximalReachabilityProbability[i] > propertyThreshold[i]),
1792 storm::exceptions::InvalidArgumentException,
1793 "Given probability threshold " << propertyThreshold[i] << " can not be " << (strictBound ? "achieved" : "exceeded")
1794 << " in model with maximal reachability probability of " << maximalReachabilityProbability[i]
1795 << ".");
1796 std::cout << "\nMaximal property value in model is " << maximalReachabilityProbability[i] << ".\n\n";
1797 }
1798 }
1799
1800 // (2) Identify all states and commands that are relevant, because only these need to be considered later.
1801 RelevancyInformation relevancyInformation = determineRelevantStatesAndLabels(model, labelSets, phiStates, psiStates, dontCareLabels);
1802
1803 // (3) Create a solver.
1804 std::shared_ptr<storm::expressions::ExpressionManager> manager = std::make_shared<storm::expressions::ExpressionManager>();
1805 std::unique_ptr<storm::solver::SmtSolver> solver = std::make_unique<storm::solver::Z3SmtSolver>(*manager);
1806
1807 // (4) Create the variables for the relevant commands.
1808 VariableInformation variableInformation = createVariables(manager, model, psiStates, relevancyInformation, options.encodeReachability);
1809 STORM_LOG_DEBUG("Created variables.");
1810
1811 // (5) Now assert an adder whose result variables can later be used to constrain the nummber of label
1812 // variables that were set to true. Initially, we are looking for a solution that has no label enabled
1813 // and subsequently relax that.
1814 variableInformation.adderVariables = assertAdder(*solver, variableInformation);
1815 variableInformation.auxiliaryVariables.push_back(assertLessOrEqualKRelaxed(*solver, variableInformation, 0));
1816
1817 // As we are done with the setup at this point, stop the clock for the setup time.
1818 totalSetupTime = std::chrono::high_resolution_clock::now() - setupTimeClock;
1819
1820 // (6) Add constraints that cut off a lot of suboptimal solutions.
1821 STORM_LOG_DEBUG("Asserting cuts.");
1822 stats.cutTime =
1823 assertCuts(symbolicModel, model, labelSets, psiStates, variableInformation, relevancyInformation, *solver, options.addBackwardImplicationCuts);
1824 STORM_LOG_DEBUG("Asserted cuts.");
1825 if (options.encodeReachability) {
1826 assertReachabilityCuts(model, labelSets, psiStates, variableInformation, relevancyInformation, *solver);
1827 STORM_LOG_DEBUG("Asserted reachability cuts.");
1828 }
1829
1830 // (7) Find the smallest set of commands that satisfies all constraints. If the probability of
1831 // satisfying phi until psi exceeds the given threshold, the set of labels is minimal and can be returned.
1832 // Otherwise, the current solution has to be ruled out and the next smallest solution is retrieved from
1833 // the solver.
1834
1835 storm::storage::FlatSet<uint_fast64_t> commandSet(relevancyInformation.knownLabels);
1836
1837 // If there are no relevant labels, return directly.
1838 if (relevancyInformation.relevantLabels.empty()) {
1839 return {commandSet};
1840 } else if (relevancyInformation.minimalityLabels.empty()) {
1841 commandSet.insert(relevancyInformation.relevantLabels.begin(), relevancyInformation.relevantLabels.end());
1842 return {commandSet};
1843 }
1844
1845 // Set up some variables for the iterations.
1846 bool done = false;
1847 uint_fast64_t lastSize = 0;
1848 uint_fast64_t iterations = 0;
1849 uint_fast64_t currentBound = 0;
1850 uint64_t firstCounterexampleFound = 0; // The value is not queried before being set.
1851 std::vector<double> maximalPropertyValue;
1852 uint_fast64_t zeroProbabilityCount = 0;
1853 size_t smallestCounterexampleSize = model.getNumberOfChoices(); // Definitive upper bound
1854 uint64_t progressDelay = storm::settings::getModule<storm::settings::modules::GeneralSettings>().getShowProgressDelay();
1855 do {
1856 ++iterations;
1857
1858 if (result.size() > 0 && iterations > firstCounterexampleFound + options.maximumExtraIterations) {
1859 break;
1860 }
1861 if (result.size() == 0) {
1862 STORM_LOG_DEBUG("Sanity check to see whether constraint system is still satisfiable.");
1863 STORM_LOG_ASSERT(solver->check() == storm::solver::SmtSolver::CheckResult::Sat, "Constraint system is not satisfiable anymore.");
1864 }
1865 STORM_LOG_DEBUG("Computing minimal command set.");
1866 solverClock = std::chrono::high_resolution_clock::now();
1867 boost::optional<storm::storage::FlatSet<uint_fast64_t>> smallest = findSmallestCommandSet(*solver, variableInformation, currentBound);
1868 totalSolverTime += std::chrono::high_resolution_clock::now() - solverClock;
1869 if (smallest == boost::none) {
1870 STORM_LOG_DEBUG("No further counterexamples.");
1871 break;
1872 } else {
1873 commandSet = smallest.get();
1874 }
1875 STORM_LOG_DEBUG("Computed minimal command with bound " << currentBound << " and set of size "
1876 << commandSet.size() + relevancyInformation.knownLabels.size() << " (" << commandSet.size()
1877 << " + " << relevancyInformation.knownLabels.size() << ") ");
1878
1879 // Restrict the given model to the current set of labels and compute the reachability probability.
1880 modelCheckingClock = std::chrono::high_resolution_clock::now();
1881 commandSet.insert(relevancyInformation.knownLabels.begin(), relevancyInformation.knownLabels.end());
1882 commandSet.insert(relevancyInformation.dontCareLabels.begin(), relevancyInformation.dontCareLabels.end());
1883 if (commandSet.size() > smallestCounterexampleSize + options.continueAfterFirstCounterexampleUntil ||
1884 (result.size() > 1 && commandSet.size() > options.multipleCounterexampleSizeCap)) {
1885 STORM_LOG_DEBUG("No further counterexamples of similar size.");
1886 break;
1887 }
1888
1889 if (commandSet.size() == nrCommands(symbolicModel)) {
1890 result.push_back(commandSet);
1891 break;
1892 }
1893
1894 auto subChoiceOrigins = restrictModelToLabelSet(model, commandSet, rewardName ? boost::make_optional(psiStates.getNextSetIndex(0)) : boost::none);
1895 std::shared_ptr<storm::models::sparse::Model<T>> const& subModel = subChoiceOrigins.first;
1896 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& subLabelSets = subChoiceOrigins.second;
1897
1898 // Now determine the maximal reachability probability in the sub-model.
1899 maximalPropertyValue = computeMaximalReachabilityProbability(env, *subModel, phiStates, psiStates, rewardName);
1900 totalModelCheckingTime += std::chrono::high_resolution_clock::now() - modelCheckingClock;
1901
1902 // Depending on whether the threshold was successfully achieved or not, we proceed by either analyzing the bad solution or stopping the iteration
1903 // process.
1904 analysisClock = std::chrono::high_resolution_clock::now();
1905 bool violation = false;
1906 for (uint64_t i = 0; i < maximalPropertyValue.size(); i++) {
1907 violation |=
1908 (strictBound && maximalPropertyValue[i] < propertyThreshold[i]) || (!strictBound && maximalPropertyValue[i] <= propertyThreshold[i]);
1909 }
1910
1911 if (violation) {
1912 if (!rewardName && maximalPropertyValue.front() == storm::utility::zero<T>()) {
1913 ++zeroProbabilityCount;
1914 }
1915
1916 if (options.useDynamicConstraints) {
1917 // Determine which of the two analysis techniques to call by performing a reachability analysis.
1918 storm::storage::BitVector reachableStates =
1919 storm::utility::graph::getReachableStates(subModel->getTransitionMatrix(), subModel->getInitialStates(), phiStates, psiStates);
1920
1921 if (reachableStates.isDisjointFrom(psiStates)) {
1922 // If there was no target state reachable, analyze the solution and guide the solver into the right direction.
1923 analyzeZeroProbabilitySolution(*solver, *subModel, subLabelSets, model, labelSets, phiStates, psiStates, commandSet,
1924 variableInformation, relevancyInformation);
1925 } else {
1926 // If the reachability probability was greater than zero (i.e. there is a reachable target state), but the probability was insufficient
1927 // to exceed the given threshold, we analyze the solution and try to guide the solver into the right direction.
1928 analyzeInsufficientProbabilitySolution(*solver, *subModel, subLabelSets, model, labelSets, phiStates, psiStates, commandSet,
1929 variableInformation, relevancyInformation);
1930 }
1931
1932 if (relevancyInformation.dontCareLabels.size() > 0) {
1933 ruleOutSingleSolution(*solver, commandSet, variableInformation, relevancyInformation);
1934 }
1935 } else {
1936 // Do not guide solver, just rule out current solution.
1937 ruleOutSingleSolution(*solver, commandSet, variableInformation, relevancyInformation);
1938 }
1939 } else {
1940 STORM_LOG_DEBUG("Found a counterexample.");
1941 if (result.empty()) {
1942 // If this is the first counterexample we find, we store when we found it.
1943 firstCounterexampleFound = iterations;
1944 }
1945 result.push_back(commandSet);
1946 if (options.maximumCounterexamples > result.size()) {
1947 STORM_LOG_DEBUG("Exclude counterexample for future.");
1948 ruleOutBiggerSolutions(*solver, commandSet, variableInformation, relevancyInformation);
1949 } else {
1950 STORM_LOG_DEBUG("Stop searching for further counterexamples.");
1951 done = true;
1952 }
1953 smallestCounterexampleSize = std::min(smallestCounterexampleSize, commandSet.size());
1954 }
1955 totalAnalysisTime += (std::chrono::high_resolution_clock::now() - analysisClock);
1956
1957 auto now = std::chrono::high_resolution_clock::now();
1958 auto durationSinceLastMessage = std::chrono::duration_cast<std::chrono::seconds>(now - timeOfLastMessage).count();
1959 if (static_cast<uint64_t>(durationSinceLastMessage) >= progressDelay || lastSize < commandSet.size()) {
1960 auto milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(now - totalClock).count();
1961 if (lastSize < commandSet.size()) {
1962 STORM_LOG_DEBUG("Improved lower bound to " << currentBound << " after " << milliseconds << "ms.");
1963 lastSize = commandSet.size();
1964 } else {
1965 STORM_LOG_DEBUG("Lower bound on label set size is " << currentBound << " after " << milliseconds << "ms (checked " << iterations
1966 << " models, " << zeroProbabilityCount << " could not reach the target set).");
1967 timeOfLastMessage = std::chrono::high_resolution_clock::now();
1968 }
1969 }
1970 } while (!done);
1971
1972 // Compute and emit the time measurements if the corresponding flag was set.
1973 totalTime = std::chrono::high_resolution_clock::now() - totalClock;
1974
1975 stats.analysisTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalAnalysisTime);
1976 stats.setupTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalSetupTime);
1977 stats.modelCheckingTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalModelCheckingTime);
1978 stats.solverTime = std::chrono::duration_cast<std::chrono::milliseconds>(totalSolverTime);
1979 stats.iterations = iterations;
1980
1981 if (storm::settings::getModule<storm::settings::modules::CoreSettings>().isShowStatisticsSet()) {
1982 storm::storage::FlatSet<uint64_t> allLabels;
1983 for (auto const& e : labelSets) {
1984 allLabels.insert(e.begin(), e.end());
1985 }
1986
1987 std::cout << "Metrics:\n";
1988 std::cout << " * all labels: " << allLabels.size() << '\n';
1989 std::cout << " * known labels: " << relevancyInformation.knownLabels.size() << '\n';
1990 std::cout << " * relevant labels: " << (relevancyInformation.knownLabels.size() + relevancyInformation.relevantLabels.size()) << "\n\n";
1991 std::cout << "Time breakdown:\n";
1992 std::cout << " * time for setup: " << std::chrono::duration_cast<std::chrono::milliseconds>(totalSetupTime).count() << "ms\n";
1993 std::cout << " * time for solving: " << std::chrono::duration_cast<std::chrono::milliseconds>(totalSolverTime).count() << "ms\n";
1994 std::cout << " * time for checking: " << std::chrono::duration_cast<std::chrono::milliseconds>(totalModelCheckingTime).count() << "ms\n";
1995 std::cout << " * time for analysis: " << std::chrono::duration_cast<std::chrono::milliseconds>(totalAnalysisTime).count() << "ms\n";
1996 std::cout << "------------------------------------------\n";
1997 std::cout << " * total time: " << std::chrono::duration_cast<std::chrono::milliseconds>(totalTime).count() << "ms\n\n";
1998 std::cout << "Other:\n";
1999 std::cout << " * number of models checked: " << iterations << '\n';
2000 std::cout << " * number of models that could not reach a target state: " << zeroProbabilityCount << " ("
2001 << 100 * static_cast<double>(zeroProbabilityCount) / iterations << "%)\n\n";
2002 }
2003
2004 return result;
2005#else
2006 throw storm::exceptions::NotImplementedException() << "This functionality is unavailable since storm has been compiled without support for Z3.";
2007#endif
2008 }

2010 static void extendLabelSetLowerBound(storm::models::sparse::Model<T> const& model, storm::storage::FlatSet<uint_fast64_t>& commandSet,
2011 storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates, bool silent = false) {
2012 auto startTime = std::chrono::high_resolution_clock::now();
2013
2014 // Create sub-model that only contains the choices allowed by the given command set.
2015 std::shared_ptr<storm::models::sparse::Model<T>> subModel = restrictModelToLabelSet(model, commandSet).first;
2016
2017 // Then determine all prob0E(psi) states that are reachable in the sub-model.
2018 storm::storage::BitVector reachableProb0EStates =
2019 storm::utility::graph::getReachableStates(subModel->getTransitionMatrix(), subModel->getInitialStates(), phiStates, psiStates);
2020
2021 // Create a queue of reachable prob0E(psi) states so we can check which commands need to be added
2022 // to give them a strategy that avoids psi states.
2023 std::queue<uint_fast64_t> prob0EWorklist;
2024 for (auto e : reachableProb0EStates) {
2025 prob0EWorklist.push(e);
2026 }
2027
2028 // As long as there are reachable prob0E(psi) states, we add commands so they can stay within
2029 // prob0E(states).
2030 while (!prob0EWorklist.empty()) {
2031 uint_fast64_t state = prob0EWorklist.front();
2032 prob0EWorklist.pop();
2033
2034 // Now iterate over the original choices of the prob0E(psi) state and add at least one.
2035 bool hasLabeledChoice = false;
2036 uint64_t smallestCommandSetSize = 0;
2037 uint64_t smallestCommandChoice = model.getTransitionMatrix().getRowGroupIndices()[state];
2038
2039 // Determine the choice with the least amount of commands (bad heuristic).
2040 for (uint64_t choice = smallestCommandChoice; choice < model.getTransitionMatrix().getRowGroupIndices()[state + 1]; ++choice) {
2041 bool onlyProb0ESuccessors = true;
2042 for (auto const& successorEntry : model.getTransitionMatrix().getRow(choice)) {
2043 if (!psiStates.get(successorEntry.getColumn())) {
2044 onlyProb0ESuccessors = false;
2045 break;
2046 }
2047 }
2048
2049 if (onlyProb0ESuccessors) {
2050 uint64_t labelSetSize = 0;
2051 if (model.getChoiceOrigins()->isPrismChoiceOrigins()) {
2052 labelSetSize = model.getChoiceOrigins()->asPrismChoiceOrigins().getCommandSet(choice).size();
2053 } else {
2054 assert(model.getChoiceOrigins()->isJaniChoiceOrigins());
2055 labelSetSize = model.getChoiceOrigins()->asJaniChoiceOrigins().getEdgeIndexSet(choice).size();
2056 }
2057 hasLabeledChoice |= (labelSetSize != 0);
2058
2059 if (smallestCommandChoice == 0 || labelSetSize < smallestCommandSetSize) {
2060 smallestCommandSetSize = labelSetSize;
2061 smallestCommandChoice = choice;
2062 }
2063 }
2064 }
2065
2066 if (hasLabeledChoice) {
2067 // Take all labels of the selected choice.
2068 if (model.getChoiceOrigins()->isPrismChoiceOrigins()) {
2069 auto const& labelSet = model.getChoiceOrigins()->asPrismChoiceOrigins().getCommandSet(smallestCommandChoice);
2070 commandSet.insert(labelSet.begin(), labelSet.end());
2071 } else {
2072 assert(model.getChoiceOrigins()->isJaniChoiceOrigins());
2073 auto const& labelSet = model.getChoiceOrigins()->asJaniChoiceOrigins().getEdgeIndexSet(smallestCommandChoice);
2074 commandSet.insert(labelSet.begin(), labelSet.end());
2075 }
2076
2077 // Check for which successor states choices need to be added
2078 for (auto const& successorEntry : model.getTransitionMatrix().getRow(smallestCommandChoice)) {
2079 if (!storm::utility::isZero(successorEntry.getValue())) {
2080 if (!reachableProb0EStates.get(successorEntry.getColumn())) {
2081 reachableProb0EStates.set(successorEntry.getColumn());
2082 prob0EWorklist.push(successorEntry.getColumn());
2083 }
2084 }
2085 }
2086 }
2087 }
2088
2089 auto endTime = std::chrono::high_resolution_clock::now();
2090 if (!silent) {
2091 std::cout << "\nExtended command for lower bounded property to size " << commandSet.size() << " in "
2092 << std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime).count() << "ms.\n";
2093 }
2094 }

2112 static CexInput precompute(Environment const& env, storm::storage::SymbolicModelDescription const& symbolicModel,
2113 storm::models::sparse::Model<T> const& model, std::shared_ptr<storm::logic::Formula const> const& formula) {
2114 CexInput result;
2115 STORM_LOG_THROW(formula->isProbabilityOperatorFormula() || formula->isRewardOperatorFormula(), storm::exceptions::InvalidPropertyException,
2116 "Counterexample generation does not support this kind of formula. Expecting a probability operator as the outermost formula element.");
2117 if (formula->isProbabilityOperatorFormula()) {
2118 storm::logic::ProbabilityOperatorFormula const& probabilityOperator = formula->asProbabilityOperatorFormula();
2119 STORM_LOG_THROW(probabilityOperator.hasBound(), storm::exceptions::InvalidPropertyException,
2120 "Counterexample generation only supports bounded formulas.");
2121 STORM_LOG_THROW(probabilityOperator.getSubformula().isUntilFormula() || probabilityOperator.getSubformula().isEventuallyFormula(),
2122 storm::exceptions::InvalidPropertyException,
2123 "Path formula is required to be of the form 'phi U psi' for counterexample generation.");
2124
2125 result.comparisonType = probabilityOperator.getComparisonType();
2126 result.threshold.push_back(probabilityOperator.getThresholdAs<T>());
2127 } else {
2128 assert(formula->isRewardOperatorFormula());
2129 storm::logic::RewardOperatorFormula const& rewardOperator = formula->asRewardOperatorFormula();
2130 STORM_LOG_THROW(rewardOperator.hasBound(), storm::exceptions::InvalidPropertyException,
2131 "Counterexample generation only supports bounded formulas.");
2132 STORM_LOG_THROW(rewardOperator.getSubformula().isEventuallyFormula(), storm::exceptions::InvalidPropertyException,
2133 "Path formula is required to be of the form 'F phi' for counterexample generation.");
2134
2135 result.comparisonType = rewardOperator.getComparisonType();
2136 result.threshold.push_back(rewardOperator.getThresholdAs<T>());
2137 result.rewardName = std::vector<std::string>();
2138 result.rewardName.get().push_back(rewardOperator.getRewardModelName());
2139
2140 STORM_LOG_THROW(!storm::logic::isLowerBound(result.comparisonType), storm::exceptions::NotSupportedException,
2141 "Lower bounds in counterexamples are only supported for probability formulas.");
2142 STORM_LOG_THROW(model.hasRewardModel(result.rewardName.get().front()), storm::exceptions::InvalidPropertyException,
2143 "Property refers to reward " << result.rewardName.get().front() << " but model does not contain such a reward model.");
2144 STORM_LOG_THROW(model.getRewardModel(result.rewardName.get().front()).hasOnlyStateRewards(), storm::exceptions::NotSupportedException,
2145 "We only support state-based rewards at the moment.");
2146 }
2147 result.strictBound = result.comparisonType == storm::logic::ComparisonType::Less;
2148 storm::logic::Formula const& subformula = formula->asOperatorFormula().getSubformula();
2149
2150 storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Model<T>> modelchecker(model);
2151
2152 if (subformula.isUntilFormula()) {
2153 STORM_LOG_THROW(!storm::logic::isLowerBound(result.comparisonType), storm::exceptions::NotSupportedException,
2154 "Lower bounds in counterexamples are only supported for eventually formulas.");
2155 storm::logic::UntilFormula const& untilFormula = subformula.asUntilFormula();
2156
2157 std::unique_ptr<storm::modelchecker::CheckResult> leftResult = modelchecker.check(env, untilFormula.getLeftSubformula());
2158 std::unique_ptr<storm::modelchecker::CheckResult> rightResult = modelchecker.check(env, untilFormula.getRightSubformula());
2159
2160 storm::modelchecker::ExplicitQualitativeCheckResult const& leftQualitativeResult = leftResult->asExplicitQualitativeCheckResult();
2161 storm::modelchecker::ExplicitQualitativeCheckResult const& rightQualitativeResult = rightResult->asExplicitQualitativeCheckResult();
2162
2163 result.phiStates = leftQualitativeResult.getTruthValuesVector();
2164 result.psiStates = rightQualitativeResult.getTruthValuesVector();
2165 } else if (subformula.isEventuallyFormula()) {
2166 storm::logic::EventuallyFormula const& eventuallyFormula = subformula.asEventuallyFormula();
2167
2168 std::unique_ptr<storm::modelchecker::CheckResult> subResult = modelchecker.check(env, eventuallyFormula.getSubformula());
2169
2170 storm::modelchecker::ExplicitQualitativeCheckResult const& subQualitativeResult = subResult->asExplicitQualitativeCheckResult();
2171
2172 result.phiStates = storm::storage::BitVector(model.getNumberOfStates(), true);
2173 result.psiStates = subQualitativeResult.getTruthValuesVector();
2174 }
2175
2176 if (storm::logic::isLowerBound(result.comparisonType)) {
2177 STORM_LOG_DEBUG("Computing counterexample for a lowerbound.");
2178 // If the formula specifies a lower bound, we need to modify the phi and psi states.
2179 // More concretely, we convert P(min)>lambda(F psi) to P(max)<(1-lambda)(G !psi) = P(max)<(1-lambda)(!psi U prob0E(psi))
2180 // where prob0E(psi) is the set of states for which there exists a strategy \sigma_0 that avoids
2181 // reaching psi states completely.
2182
2183 // This means that from all states in prob0E(psi) we need to include labels such that \sigma_0
2184 // is actually included in the resulting model. This prevents us from guaranteeing the minimality of
2185 // the returned counterexample, so we warn about that.
2186
2187 // Modify bound appropriately.
2188 result.comparisonType = storm::logic::invertPreserveStrictness(result.comparisonType);
2189 result.threshold.back() = storm::utility::one<T>() - result.threshold.back();
2190
2191 // Modify the phi and psi states appropriately.
2192 storm::storage::BitVector statesWithProbability0E =
2193 storm::utility::graph::performProb0E(model.getTransitionMatrix(), model.getTransitionMatrix().getRowGroupIndices(),
2194 model.getBackwardTransitions(), result.phiStates, result.psiStates);
2195 result.phiStates = ~result.psiStates;
2196 result.psiStates = std::move(statesWithProbability0E);
2197
2198 // Remember our transformation so we can add commands to guarantee that the prob0E(a) states actually
2199 // have a strategy that voids a states.
2200 result.lowerBoundedFormula = true;
2201 }
2202 return result;
