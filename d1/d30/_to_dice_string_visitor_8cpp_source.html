70boost::any ToDiceStringVisitor::visit(BinaryNumericalFunctionExpression const& expression, boost::any const& data) {
71 switch (expression.getOperatorType()) {
72 case BinaryNumericalFunctionExpression::OperatorType::Plus:
73 stream << "(";
74 expression.getFirstOperand()->accept(*this, data);
75 stream << "+";
76 expression.getSecondOperand()->accept(*this, data);
77 stream << ")";
78 break;
79 case BinaryNumericalFunctionExpression::OperatorType::Minus:
80 stream << "(";
81 expression.getFirstOperand()->accept(*this, data);
82 stream << "-";
83 expression.getSecondOperand()->accept(*this, data);
84 stream << ")";
85 break;
86 case BinaryNumericalFunctionExpression::OperatorType::Times:
87 stream << "(";
88 expression.getFirstOperand()->accept(*this, data);
89 stream << "*";
90 expression.getSecondOperand()->accept(*this, data);
91 stream << ")";
92 break;
93 case BinaryNumericalFunctionExpression::OperatorType::Divide: {
94 STORM_LOG_THROW(expression.getSecondOperand()->isIntegerLiteralExpression(), storm::exceptions::NotSupportedException,
95 "Dice does not support modulo with nonconst rhs");
96 uint64_t denominator = expression.getSecondOperand()->evaluateAsInt();
97 int shifts = 0;
98 while (denominator % 2 == 0) {
99 denominator = denominator >> 1;
100 shifts++;
101 }
102 denominator = denominator >> 1;
103 if (denominator > 0) {
104 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Dice does not support division with non-powers of two");
105 }
106 if (shifts > 0) {
107 stream << "(";
108 expression.getFirstOperand()->accept(*this, data);
109 stream << " >> " << shifts;
110 stream << ")";
111 } else {
112 expression.getFirstOperand()->accept(*this, data);
113 }
114
115 } break;
116 case BinaryNumericalFunctionExpression::OperatorType::Power:
117 stream << "(";
118 expression.getFirstOperand()->accept(*this, data);
119 stream << "^";
120 expression.getSecondOperand()->accept(*this, data);
121 stream << ")";
122 break;
123 case BinaryNumericalFunctionExpression::OperatorType::Modulo:
124 STORM_LOG_THROW(expression.getSecondOperand()->isIntegerLiteralExpression(), storm::exceptions::NotSupportedException,
125 "Dice does not support modulo with nonconst rhs");
126 STORM_LOG_THROW(expression.getSecondOperand()->evaluateAsInt() == 2, storm::exceptions::NotSupportedException,
127 "Dice does not support modulo with rhs != 2");
128
129 stream << "( nth_bit(int(" << nrBits << "," << nrBits - 1 << "), ";
130 expression.getFirstOperand()->accept(*this, data);
131 stream << "))";
132 break;
133 case BinaryNumericalFunctionExpression::OperatorType::Logarithm:
134 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Dice translation not supported for log expressions.");
135 case BinaryNumericalFunctionExpression::OperatorType::Max:
136 stream << "max(";
137 expression.getFirstOperand()->accept(*this, data);
138 stream << ",";
139 expression.getSecondOperand()->accept(*this, data);
140 stream << ")";
141 break;
142 case BinaryNumericalFunctionExpression::OperatorType::Min:
143 stream << "min(";
144 expression.getFirstOperand()->accept(*this, data);
145 stream << ",";
146 expression.getSecondOperand()->accept(*this, data);
147 stream << ")";
148 break;
149 }
150 return boost::any();
151}

241boost::any ToDiceStringVisitor::visit(PredicateExpression const& expression, boost::any const& data) {
242 auto const& pdt = expression.getPredicateType();
243 STORM_LOG_ASSERT(pdt == PredicateExpression::PredicateType::ExactlyOneOf || pdt == PredicateExpression::PredicateType::AtLeastOneOf ||
244 pdt == PredicateExpression::PredicateType::AtMostOneOf,
245 "Only some predicate types are supported.");
246 stream << "(";
247 if (pdt == PredicateExpression::PredicateType::ExactlyOneOf || pdt == PredicateExpression::PredicateType::AtMostOneOf) {
248 stream << "(true ";
249 for (uint64_t operandi = 0; operandi < expression.getArity(); ++operandi) {
250 for (uint64_t operandj = operandi + 1; operandj < expression.getArity(); ++operandj) {
251 stream << "&& !(";
252 expression.getOperand(operandi)->accept(*this, data);
253 stream << " && ";
254 expression.getOperand(operandj)->accept(*this, data);
255 stream << ")";
256 }
257 }
258 stream << ")";
259 }
260 if (pdt == PredicateExpression::PredicateType::ExactlyOneOf) {
261 stream << " && ";
262 }
263 if (pdt == PredicateExpression::PredicateType::ExactlyOneOf || pdt == PredicateExpression::PredicateType::AtLeastOneOf) {
264 stream << "( false";
265 for (uint64_t operandj = 0; operandj < expression.getArity(); ++operandj) {
266 stream << "|| ";
267 expression.getOperand(operandj)->accept(*this, data);
268 }
269 stream << ")";
270 }
271 stream << ")";
272 return boost::any();
