101 template<typename ValueType>
102 void DFT<ValueType>::setDynamicBehaviorInfo() {
103  std::vector<bool> dynamicBehaviorVector(mElements.size(), false);
104 
105  std::queue<DFTElementPointer> elementQueue;
106 
107  // deal with all dynamic elements
108  for (auto const& element : mElements) {
109  switch (element->type()) {
110  case storm::dft::storage::elements::DFTElementType::PAND:
111  case storm::dft::storage::elements::DFTElementType::POR:
112  case storm::dft::storage::elements::DFTElementType::MUTEX: {
113  auto gate = std::static_pointer_cast<storm::dft::storage::elements::DFTChildren<ValueType>>(element);
114  dynamicBehaviorVector[gate->id()] = true;
115  for (auto const& child : gate->children()) {
116  // only enqueue static children
117  if (!dynamicBehaviorVector.at(child->id())) {
118  elementQueue.push(child);
119  }
120  }
121  break;
122  }
123  // TODO different cases
124  case storm::dft::storage::elements::DFTElementType::SPARE: {
125  auto spare = std::static_pointer_cast<storm::dft::storage::elements::DFTSpare<ValueType>>(element);
126 
127  // Iterate over all children (representatives of spare modules)
128  for (auto const& child : spare->children()) {
129  // Case 1: Shared Module
130  // If child only has one parent, it is this SPARE -> nothing to check
131  if (child->nrParents() > 1) {
132  // TODO make more efficient by directly setting ALL spares which share a module to be dynamic
133  for (auto const& parent : child->parents()) {
134  if (parent->isSpareGate() and parent->id() != spare->id()) {
135  dynamicBehaviorVector[spare->id()] = true;
136  break; // inner loop
137  }
138  }
139  }
140  // Case 2: Triggering outside events
141  // If the SPARE was already detected to have dynamic behavior, do not proceed
142  if (!dynamicBehaviorVector[spare->id()]) {
143  for (auto const& memberID : module(child->id()).getElements()) {
144  // Iterate over all members of the module child represents
145  auto member = getElement(memberID);
146  for (auto const& dep : member->outgoingDependencies()) {
147  // If the member has outgoing dependencies, check if those trigger something outside the module
148  for (auto const& depEvent : dep->dependentEvents()) {
149  // If a dependent event is not found in the module, SPARE is dynamic
150  auto const& childModuleElements = module(child->id()).getElements();
151  if (std::find(childModuleElements.begin(), childModuleElements.end(), depEvent->id()) == childModuleElements.end()) {
152  dynamicBehaviorVector[spare->id()] = true;
153  break; // depEvent-loop
154  }
155  }
156  if (dynamicBehaviorVector[spare->id()]) {
157  break;
158  } // dependency-loop
159  }
160  if (dynamicBehaviorVector[spare->id()]) {
161  break;
162  } // module-loop
163  }
164  }
165  if (dynamicBehaviorVector[spare->id()]) {
166  break;
167  } // child-loop
168  }
169  // if during the computation, dynamic behavior was detected, add children to queue
170  if (dynamicBehaviorVector[spare->id()]) {
171  for (auto const& child : spare->children()) {
172  // only enqueue static children
173  if (!dynamicBehaviorVector.at(child->id())) {
174  elementQueue.push(child);
175  }
176  }
177  }
178  break;
179  }
180  case storm::dft::storage::elements::DFTElementType::SEQ: {
181  auto seq = std::static_pointer_cast<storm::dft::storage::elements::DFTSeq<ValueType>>(element);
182  // A SEQ only has dynamic behavior if not all children are BEs
183  if (!seq->allChildrenBEs()) {
184  dynamicBehaviorVector[seq->id()] = true;
185  for (auto const& child : seq->children()) {
186  // only enqueue static children
187  if (!dynamicBehaviorVector.at(child->id())) {
188  elementQueue.push(child);
189  }
190  }
191  }
192  break;
193  }
194  default: {
195  break;
196  }
197  }
198  }
199  // propagate dynamic behavior
200  while (!elementQueue.empty()) {
201  DFTElementPointer currentElement = elementQueue.front();
202  elementQueue.pop();
203  switch (currentElement->type()) {
204  // Static Gates
205  case storm::dft::storage::elements::DFTElementType::AND:
206  case storm::dft::storage::elements::DFTElementType::OR:
207  case storm::dft::storage::elements::DFTElementType::VOT: {
208  // check all parents and if one has dynamic behavior, propagate it
209  dynamicBehaviorVector[currentElement->id()] = true;
210  auto gate = std::static_pointer_cast<storm::dft::storage::elements::DFTGate<ValueType>>(currentElement);
211  for (auto const& child : gate->children()) {
212  // only enqueue static children
213  if (!dynamicBehaviorVector.at(child->id())) {
214  elementQueue.push(child);
215  }
216  }
217  break;
218  }
219  // BEs
220  case storm::dft::storage::elements::DFTElementType::BE: {
221  auto be = std::static_pointer_cast<storm::dft::storage::elements::DFTBE<ValueType>>(currentElement);
222  dynamicBehaviorVector[be->id()] = true;
223  // add all ingoing dependencies to queue
224  for (auto const& dep : be->ingoingDependencies()) {
225  if (!dynamicBehaviorVector.at(dep->id())) {
226  elementQueue.push(dep);
227  }
228  }
229  break;
230  }
231  case storm::dft::storage::elements::DFTElementType::PDEP: {
232  auto dep = std::static_pointer_cast<storm::dft::storage::elements::DFTDependency<ValueType>>(currentElement);
233  dynamicBehaviorVector[dep->id()] = true;
234  // add all ingoing dependencies to queue
235  auto trigger = dep->triggerEvent();
236  if (!dynamicBehaviorVector.at(trigger->id())) {
237  elementQueue.push(trigger);
238  }
239  break;
240  }
241  default:
242  break;
243  }
244  }
245  mDynamicBehavior = dynamicBehaviorVector;
246 }

248 template<typename ValueType>
249 DFTStateGenerationInfo DFT<ValueType>::buildStateGenerationInfo(storm::dft::storage::DftSymmetries const& symmetries) const {
250  DFTStateGenerationInfo generationInfo(nrElements(), mNrOfSpares, mNrRepresentatives, mMaxSpareChildCount);
251 
252  for (auto const& elem : mElements) {
253  // Check if BEs are immediately failed
254  if (elem->isBasicElement()) {
255  auto be = std::static_pointer_cast<storm::dft::storage::elements::DFTBE<ValueType> const>(elem);
256  if ((be->beType() == storm::dft::storage::elements::BEType::CONSTANT && be->canFail()) ||
257  be->beType() == storm::dft::storage::elements::BEType::PROBABILITY) {
258  generationInfo.addImmediateFailedBE(be->id());
259  }
260  }
261  // Generate pre- and post-set info for restrictions, and mutexes
262  if (!elem->isDependency() && !elem->isRestriction()) {
263  // Ids of elements which are the direct predecessor in the list of children of a restriction
264  std::vector<size_t> seqRestrictionPres;
265  // Ids of elements which are the direct successor in the list of children of a restriction
266  std::vector<size_t> seqRestrictionPosts;
267  // Ids of elements which are under the same mutex
268  std::vector<size_t> mutexRestrictionElements;
269 
270  for (auto const& restr : elem->restrictions()) {
271  if (restr->isSeqEnforcer()) {
272  auto it = restr->children().cbegin();
273  for (; it != restr->children().cend(); ++it) {
274  if ((*it)->id() == elem->id()) {
275  break;
276  }
277  }
278  STORM_LOG_ASSERT(it != restr->children().cend(), "Child " << elem->id() << " not found in restriction " << *restr);
279  // Add child following element in SEQ
280  ++it;
281  if (it != restr->children().cend()) {
282  seqRestrictionPosts.push_back((*it)->id());
283  }
284  // Add child before element in SEQ
285  --it;
286  if (it != restr->children().cbegin()) {
287  --it;
288  seqRestrictionPres.push_back((*it)->id());
289  }
290  } else {
291  STORM_LOG_ASSERT(restr->isMutex(), "Restriction " << *restr << " is neither SEQ nor MUTEX.");
292  bool found = false;
293  for (auto it = restr->children().cbegin(); it != restr->children().cend(); ++it) {
294  if ((*it)->id() != elem->id()) {
295  mutexRestrictionElements.push_back((*it)->id());
296  } else {
297  found = true;
298  }
299  }
300  STORM_LOG_ASSERT(found, "Child " << elem->id() << " is not included in restriction " << *restr);
301  }
302  }
303 
304  generationInfo.setRestrictionPreElements(elem->id(), seqRestrictionPres);
305  generationInfo.setRestrictionPostElements(elem->id(), seqRestrictionPosts);
306  generationInfo.setMutexElements(elem->id(), mutexRestrictionElements);
307  }
308  }
309 
310  size_t stateIndex = 0;
311  std::queue<size_t> visitQueue;
312  storm::storage::BitVector visited(nrElements(), false);
313 
314  if (symmetries.nrSymmetries() == 0) {
315  // Perform DFS for whole tree
316  visitQueue.push(mTopLevelIndex);
317  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
318  } else {
319  // Generate information according to symmetries
320  for (size_t symmetryIndex : symmetries) {
321  STORM_LOG_ASSERT(!visited[symmetryIndex], "Element already considered for symmetry.");
322  auto const& symmetryGroup = symmetries.getSymmetryGroup(symmetryIndex);
323  STORM_LOG_ASSERT(!symmetryGroup.empty(), "No symmetry available.");
324 
325  // Insert all elements of first subtree of each symmetry
326  size_t groupIndex = stateIndex;
327  for (std::vector<size_t> const& symmetryElement : symmetryGroup) {
328  if (visited[symmetryElement[0]]) {
329  groupIndex = std::min(groupIndex, generationInfo.getStateIndex(symmetryElement[0]));
330  } else {
331  stateIndex = generateStateInfo(generationInfo, symmetryElement[0], visited, stateIndex);
332  }
333  }
334  size_t offset = stateIndex - groupIndex;
335 
336  // Mirror symmetries
337  size_t noSymmetricElements = symmetryGroup.front().size();
338  STORM_LOG_ASSERT(noSymmetricElements > 1, "No symmetry available.");
339 
340  for (std::vector<size_t> symmetricElements : symmetryGroup) {
341  STORM_LOG_ASSERT(symmetricElements.size() == noSymmetricElements, "No. of symmetric elements do not coincide.");
342  if (visited[symmetricElements[1]]) {
343  // Elements already mirrored
344  for (size_t index : symmetricElements) {
345  STORM_LOG_ASSERT(visited[index], "Element not mirrored.");
346  }
347  continue;
348  }
349 
350  // Initialize for original element
351  size_t originalElement = symmetricElements[0];
352  size_t index = generationInfo.getStateIndex(originalElement);
353  size_t activationIndex = isRepresentative(originalElement) ? generationInfo.getSpareActivationIndex(originalElement) : 0;
354  size_t usageIndex = mElements[originalElement]->isSpareGate() ? generationInfo.getSpareUsageIndex(originalElement) : 0;
355 
356  // Mirror symmetry for each element
357  for (size_t i = 1; i < symmetricElements.size(); ++i) {
358  size_t symmetricElement = symmetricElements[i];
359  STORM_LOG_ASSERT(!visited[symmetricElement], "Symmetric element " << symmetricElement << " already considered before.");
360  visited.set(symmetricElement);
361 
362  generationInfo.addStateIndex(symmetricElement, index + offset * i);
363  stateIndex += 2;
364 
365  STORM_LOG_ASSERT((activationIndex > 0) == isRepresentative(symmetricElement), "Bits for representative incorrect.");
366  if (activationIndex > 0) {
367  generationInfo.addSpareActivationIndex(symmetricElement, activationIndex + offset * i);
368  ++stateIndex;
369  }
370 
371  STORM_LOG_ASSERT((usageIndex > 0) == mElements[symmetricElement]->isSpareGate(), "Bits for usage incorrect.");
372  if (usageIndex > 0) {
373  generationInfo.addSpareUsageIndex(symmetricElement, usageIndex + offset * i);
374  stateIndex += generationInfo.usageInfoBits();
375  }
376  }
377  }
378 
379  // Store starting indices of symmetry groups
380  std::vector<size_t> symmetryIndices;
381  for (size_t i = 0; i < noSymmetricElements; ++i) {
382  symmetryIndices.push_back(groupIndex + i * offset);
383  }
384  generationInfo.addSymmetry(offset, symmetryIndices);
385  }
386  }
387 
388  // TODO symmetries in dependencies?
389  // Consider dependencies
390  for (size_t idDependency : getDependencies()) {
391  std::shared_ptr<storm::dft::storage::elements::DFTDependency<ValueType> const> dependency = getDependency(idDependency);
392  visitQueue.push(dependency->id());
393  visitQueue.push(dependency->triggerEvent()->id());
394  STORM_LOG_THROW(dependency->dependentEvents().size() == 1, storm::exceptions::NotSupportedException,
395  "Direct state generation does not support n-ary dependencies. Consider rewriting them by setting the binary dependency flag.");
396  visitQueue.push(dependency->dependentEvents()[0]->id());
397  }
398  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
399  STORM_LOG_ASSERT(visitQueue.empty(), "VisitQueue not empty");
400 
401  // Visit all remaining states
402  for (size_t i = 0; i < visited.size(); ++i) {
403  if (!visited[i]) {
404  visitQueue.push(i);
405  stateIndex = performStateGenerationInfoDFS(generationInfo, visitQueue, visited, stateIndex);
406  }
407  }
408 
409  generationInfo.generateSymmetries();
410 
411  STORM_LOG_TRACE(generationInfo);
412  STORM_LOG_ASSERT(stateIndex == mStateVectorSize, "Id incorrect.");
413  STORM_LOG_ASSERT(visited.full(), "Not all elements considered.");
414  generationInfo.checkSymmetries();
415 
416  return generationInfo;
417 }

464 std::vector<DFT<ValueType>> DFT<ValueType>::topModularisation() const {
465  STORM_LOG_ASSERT(isGate(mTopLevelIndex), "Top level element is no gate.");
466  auto const& children = getGate(mTopLevelIndex)->children();
467  std::map<size_t, std::vector<size_t>> subdfts;
468  for (auto const& child : children) {
469  std::vector<size_t> isubdft;
470  if (child->nrParents() > 1 || child->hasOutgoingDependencies() || child->hasRestrictions()) {
471  STORM_LOG_TRACE("child " << child->name() << " does not allow modularisation.");
472  return {*this};
473  }
474  if (isGate(child->id())) {
475  isubdft = getGate(child->id())->independentSubDft(false);
476  } else {
477  STORM_LOG_ASSERT(isBasicElement(child->id()), "Child is no BE.");
478  if (getBasicElement(child->id())->hasIngoingDependencies()) {
479  STORM_LOG_TRACE("child " << child->name() << " does not allow modularisation.");
480  return {*this};
481  } else {
482  isubdft = {child->id()};
483  }
484  }
485  if (isubdft.empty()) {
486  return {*this};
487  } else {
488  subdfts[child->id()] = isubdft;
489  }
490  }
491 
492  std::vector<DFT<ValueType>> res;
493  for (auto const& subdft : subdfts) {
494  storm::dft::builder::DFTBuilder<ValueType> builder;
495 
496  for (size_t id : subdft.second) {
497  builder.cloneElement(mElements[id]);
498  }
499  builder.setTopLevel(mElements[subdft.first]->name());
500  res.push_back(builder.build());
501  }
502  return res;

516 template<typename ValueType>
517 DFT<ValueType> DFT<ValueType>::optimize() const {
518  std::vector<size_t> modIdea = findModularisationRewrite();
519  STORM_LOG_DEBUG("Modularisation idea: " << storm::utility::vector::toString(modIdea));
520 
521  if (modIdea.empty()) {
522  // No rewrite needed
523  return *this;
524  }
525 
526  std::vector<std::vector<size_t>> rewriteIds;
527  rewriteIds.push_back(modIdea);
528 
529  storm::dft::builder::DFTBuilder<ValueType> builder;
530 
531  // Accumulate elements which must be rewritten
532  std::set<size_t> rewriteSet;
533  for (std::vector<size_t> rewrites : rewriteIds) {
534  rewriteSet.insert(rewrites.front());
535  }
536  // Copy all other elements which do not change
537  for (auto elem : mElements) {
538  if (rewriteSet.count(elem->id()) == 0) {
539  builder.cloneElement(elem);
540  }
541  }
542 
543  size_t uniqueIndex = 0; // Counter to ensure unique names
544  // Add rewritten elements
545  for (std::vector<size_t> rewrites : rewriteIds) {
546  STORM_LOG_ASSERT(rewrites.size() > 1, "No rewritten elements.");
547  STORM_LOG_ASSERT(mElements[rewrites[1]]->hasParents(), "Rewritten elements has no parents.");
548  STORM_LOG_ASSERT(mElements[rewrites[1]]->parents().front()->isGate(), "Rewritten element has no parent gate.");
549  DFTGatePointer originalParent = std::static_pointer_cast<storm::dft::storage::elements::DFTGate<ValueType>>(mElements[rewrites[0]]);
550  STORM_LOG_ASSERT(std::find_if(mElements[rewrites[1]]->parents().begin(), mElements[rewrites[1]]->parents().end(),
551  [&originalParent](std::shared_ptr<storm::dft::storage::elements::DFTElement<ValueType>> const& p) {
552  return p->id() == originalParent->id();
553  }) != mElements[rewrites[1]]->parents().end(),
554  "Rewritten element has not the same parent");
555  std::string newParentName = originalParent->name() + "_" + std::to_string(++uniqueIndex);
556 
557  // Accumulate children names
558  std::vector<std::string> childrenNames;
559  for (size_t i = 1; i < rewrites.size(); ++i) {
560  STORM_LOG_ASSERT(std::find_if(mElements[rewrites[i]]->parents().begin(), mElements[rewrites[i]]->parents().end(),
561  [&originalParent](std::shared_ptr<storm::dft::storage::elements::DFTElement<ValueType>> const& p) {
562  return p->id() == originalParent->id();
563  }) != mElements[rewrites[i]]->parents().end(),
564  "Children have not the same father for rewrite " << mElements[rewrites[i]]->name());
565 
566  childrenNames.push_back(mElements[rewrites[i]]->name());
567  }
568 
569  // Add element in-between parent and children
570  switch (originalParent->type()) {
571  case storm::dft::storage::elements::DFTElementType::AND:
572  builder.addAndGate(newParentName, childrenNames);
573  break;
574  case storm::dft::storage::elements::DFTElementType::OR:
575  builder.addOrGate(newParentName, childrenNames);
576  break;
577  default:
578  // Other elements are not supported
579  STORM_LOG_ASSERT(false, "Dft type " << originalParent->type() << " can not be rewritten.");
580  break;
581  }
582 
583  // Add parent with the new child newParent and all its remaining children
584  childrenNames.clear();
585  childrenNames.push_back(newParentName);
586  for (auto const& child : originalParent->children()) {
587  if (std::find(rewrites.begin() + 1, rewrites.end(), child->id()) == rewrites.end()) {
588  // Child was not rewritten and must be kept
589  childrenNames.push_back(child->name());
590  }
591  }
592  builder.cloneElementWithNewChildren(originalParent, childrenNames);
593  }
594 
595  builder.setTopLevel(mElements[mTopLevelIndex]->name());
596  // TODO use reference?
597  DFT<ValueType> newDft = builder.build();
598  STORM_LOG_TRACE(newDft.getElementsString());
599  return newDft.optimize();

780 template<typename ValueType>
781 std::vector<size_t> DFT<ValueType>::findModularisationRewrite() const {
782  for (auto const& e : mElements) {
783  if (e->isGate() &&
784  (e->type() == storm::dft::storage::elements::DFTElementType::AND || e->type() == storm::dft::storage::elements::DFTElementType::OR)) {
785  // suitable parent gate! - Lets check the independent submodules of the children
786  auto const& children = std::static_pointer_cast<storm::dft::storage::elements::DFTGate<ValueType>>(e)->children();
787  for (auto const& child : children) {
788  auto ISD = std::static_pointer_cast<storm::dft::storage::elements::DFTGate<ValueType>>(child)->independentSubDft(true);
789  // In the ISD, check for other children:
790 
791  std::vector<size_t> rewrite = {e->id(), child->id()};
792  for (size_t isdElemId : ISD) {
793  if (isdElemId == child->id())
794  continue;
795  if (std::find_if(children.begin(), children.end(),
796  [&isdElemId](std::shared_ptr<storm::dft::storage::elements::DFTElement<ValueType>> const& e) {
797  return e->id() == isdElemId;
798  }) != children.end()) {
799  // element in subtree is also child
800  rewrite.push_back(isdElemId);
801  }
802  }
803  if (rewrite.size() > 2 && rewrite.size() < children.size() - 1) {
804  return rewrite;
805  }
806  }
807  }
808  }
809  return {};
810 }
