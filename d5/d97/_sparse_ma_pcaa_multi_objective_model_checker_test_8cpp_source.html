258template<typename ValueType>
259bool expectPointConained(std::vector<std::vector<ValueType>> const& pointset, std::vector<ValueType> const& point, ValueType precision) {
260 for (auto const& p : pointset) {
261 EXPECT_EQ(p.size(), point.size()) << "Missmatch in point dimension.";
262 bool found = true;
263 for (uint64_t i = 0; i < p.size(); ++i) {
264 if (storm::utility::abs<ValueType>(p[i] - point[i]) > precision) {
265 found = false;
266 break;
267 }
268 }
269 if (found) {
270 return true;
271 }
272 }
273 // prepare failure message:
274 std::stringstream errstr;
275 errstr << "Point [";
276 bool firstPi = true;
277 for (auto const& pi : point) {
278 if (firstPi) {
279 firstPi = false;
280 } else {
281 errstr << ", ";
282 }
283 errstr << pi;
284 }
285 errstr << "] is not contained in point set {";
286 bool firstP = true;
287 for (auto const& p : pointset) {
288 if (firstP) {
289 firstP = false;
290 } else {
291 errstr << ", \t";
292 }
293 errstr << "[";
294 firstPi = true;
295 for (auto const& pi : p) {
296 if (firstPi) {
297 firstPi = false;
298 } else {
299 errstr << ", ";
300 }
301 errstr << pi;
302 }
303 errstr << "]";
304 }
305 errstr << "}.";
306 ADD_FAILURE() << errstr.str();
307 return false;
308}

310template<typename ValueType>
311bool expectSubset(std::vector<std::vector<ValueType>> const& lhs, std::vector<std::vector<ValueType>> const& rhs, ValueType precision) {
312 for (auto const& p : lhs) {
313 if (!expectPointConained(rhs, p, precision)) {
314 return false;
315 }
316 }
317 return true;
318}

320template<typename ValueType>
321std::vector<std::vector<ValueType>> convertPointset(std::vector<std::vector<std::string>> const& in) {
322 std::vector<std::vector<ValueType>> out;
323 for (auto const& point_str : in) {
324 out.emplace_back();
325 for (auto const& pi_str : point_str) {
326 out.back().push_back(storm::utility::convertNumber<ValueType>(pi_str));
327 }
328 }
329 return out;
330}
