22template<typename ValueType>
23bool expectPointConained(std::vector<std::vector<ValueType>> const& pointset, std::vector<ValueType> const& point, ValueType precision) {
24 for (auto const& p : pointset) {
25 EXPECT_EQ(p.size(), point.size()) << "Missmatch in point dimension.";
26 bool found = true;
27 for (uint64_t i = 0; i < p.size(); ++i) {
28 if (storm::utility::abs<ValueType>(p[i] - point[i]) > precision) {
29 found = false;
30 break;
31 }
32 }
33 if (found) {
34 return true;
35 }
36 }
37 // prepare failure message:
38 std::stringstream errstr;
39 errstr << "Point [";
40 bool firstPi = true;
41 for (auto const& pi : point) {
42 if (firstPi) {
43 firstPi = false;
44 } else {
45 errstr << ", ";
46 }
47 errstr << pi;
48 }
49 errstr << "] is not contained in point set {";
50 bool firstP = true;
51 for (auto const& p : pointset) {
52 if (firstP) {
53 firstP = false;
54 } else {
55 errstr << ", \t";
56 }
57 errstr << "[";
58 firstPi = true;
59 for (auto const& pi : p) {
60 if (firstPi) {
61 firstPi = false;
62 } else {
63 errstr << ", ";
64 }
65 errstr << pi;
66 }
67 errstr << "]";
68 }
69 errstr << "}.";
70 ADD_FAILURE() << errstr.str();
71 return false;
72}

74template<typename ValueType>
75bool expectSubset(std::vector<std::vector<ValueType>> const& lhs, std::vector<std::vector<ValueType>> const& rhs, ValueType precision) {
76 for (auto const& p : lhs) {
77 if (!expectPointConained(rhs, p, precision)) {
78 return false;
79 }
80 }
81 return true;
82}

84template<typename ValueType>
85std::vector<std::vector<ValueType>> convertPointset(std::vector<std::vector<std::string>> const& in) {
86 std::vector<std::vector<ValueType>> out;
87 for (auto const& point_str : in) {
88 out.emplace_back();
89 for (auto const& pi_str : point_str) {
90 out.back().push_back(storm::utility::convertNumber<ValueType>(pi_str));
91 }
92 }
93 return out;
94}

96template<typename ValueType>
97void evaluateScheduler(storm::Environment const& env, storm::models::sparse::Mdp<ValueType> const& mdp, storm::storage::Scheduler<ValueType> const& scheduler,
98 storm::logic::MultiObjectiveFormula const& formula, std::vector<ValueType>& result) {
99 ASSERT_FALSE(scheduler.isPartialScheduler());
100 ASSERT_EQ(mdp.getNumberOfStates(), scheduler.getNumberOfModelStates());
101 auto inducedModel = mdp.applyScheduler(scheduler);
102 ASSERT_TRUE(inducedModel->isOfType(storm::models::ModelType::Dtmc));
103 auto const dtmcPtr = inducedModel->template as<storm::models::sparse::Dtmc<ValueType>>();
104 ASSERT_TRUE(dtmcPtr != nullptr);
105 auto const& dtmc = *dtmcPtr;
106 ASSERT_TRUE(dtmc.getInitialStates().hasUniqueSetBit());
107 storm::modelchecker::SparseDtmcPrctlModelChecker<storm::models::sparse::Dtmc<ValueType>> modelChecker(dtmc);
108
109 for (auto const& obj : formula.getSubformulas()) {
110 auto objRes = modelChecker.check(env, *obj);
111 EXPECT_TRUE(objRes->isExplicitQuantitativeCheckResult()) << "Objective " << *obj << " did not produce a quantitative result.";
112 auto const& quantitativeResult = objRes->template asExplicitQuantitativeCheckResult<ValueType>();
113 result.push_back(quantitativeResult[*dtmc.getInitialStates().begin()]);
114 }
115}

117template<typename ValueType>
118void assertParetoResult(storm::Environment const& env, storm::models::sparse::Mdp<ValueType> const& mdp, storm::logic::MultiObjectiveFormula const& formula,
119 storm::modelchecker::CheckResult const& result, std::vector<std::vector<std::string>> const& expectedPoints, bool expectSchedulers) {
120 ASSERT_TRUE(result.isExplicitParetoCurveCheckResult()) << "Result is not an explicit Pareto curve check result.";
121 ValueType const eps = storm::utility::convertNumber<ValueType>(1e-4);
122 auto const& paretoResult = result.asExplicitParetoCurveCheckResult<ValueType>();
123 EXPECT_TRUE(expectSubset(paretoResult.getPoints(), convertPointset<ValueType>(expectedPoints), eps)) << "Non-Pareto point found.";
124 EXPECT_TRUE(expectSubset(convertPointset<ValueType>(expectedPoints), paretoResult.getPoints(), eps)) << "Pareto point missing.";
125 ASSERT_EQ(expectSchedulers, paretoResult.hasScheduler());
126 if (expectSchedulers) {
127 ASSERT_EQ(paretoResult.getPoints().size(), paretoResult.getSchedulers().size());
128 for (uint64_t i = 0; i < paretoResult.getPoints().size(); ++i) {
129 auto const& point = paretoResult.getPoints()[i];
130 std::vector<ValueType> inducedPoint;
131 evaluateScheduler<ValueType>(env, mdp, paretoResult.getSchedulers()[i], formula, inducedPoint);
132 ASSERT_EQ(point.size(), inducedPoint.size()) << "Unable to evaluate scheduler for Pareto point " << i;
133 for (uint64_t objDim = 0; objDim < point.size(); ++objDim) {
134 EXPECT_NEAR(point[objDim], inducedPoint[objDim], eps)
135 << "Scheduler for Pareto point " << i << " does not yield expected value for objective dimension " << objDim << ".";
136 }
137 }
138 }
