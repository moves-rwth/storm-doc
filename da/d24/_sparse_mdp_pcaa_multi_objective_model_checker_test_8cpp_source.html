24template<typename ValueType>
25bool expectPointConained(std::vector<std::vector<ValueType>> const& pointset, std::vector<ValueType> const& point, ValueType precision) {
26 for (auto const& p : pointset) {
27 EXPECT_EQ(p.size(), point.size()) << "Missmatch in point dimension.";
28 bool found = true;
29 for (uint64_t i = 0; i < p.size(); ++i) {
30 if (storm::utility::abs<ValueType>(p[i] - point[i]) > precision) {
31 found = false;
32 break;
33 }
34 }
35 if (found) {
36 return true;
37 }
38 }
39 // prepare failure message:
40 std::stringstream errstr;
41 errstr << "Point [";
42 bool firstPi = true;
43 for (auto const& pi : point) {
44 if (firstPi) {
45 firstPi = false;
46 } else {
47 errstr << ", ";
48 }
49 errstr << pi;
50 }
51 errstr << "] is not contained in point set {";
52 bool firstP = true;
53 for (auto const& p : pointset) {
54 if (firstP) {
55 firstP = false;
56 } else {
57 errstr << ", \t";
58 }
59 errstr << "[";
60 firstPi = true;
61 for (auto const& pi : p) {
62 if (firstPi) {
63 firstPi = false;
64 } else {
65 errstr << ", ";
66 }
67 errstr << pi;
68 }
69 errstr << "]";
70 }
71 errstr << "}.";
72 ADD_FAILURE() << errstr.str();
73 return false;
74}

76template<typename ValueType>
77bool expectSubset(std::vector<std::vector<ValueType>> const& lhs, std::vector<std::vector<ValueType>> const& rhs, ValueType precision) {
78 for (auto const& p : lhs) {
79 if (!expectPointConained(rhs, p, precision)) {
80 return false;
81 }
82 }
83 return true;
84}

86template<typename ValueType>
87std::vector<std::vector<ValueType>> convertPointset(std::vector<std::vector<std::string>> const& in) {
88 std::vector<std::vector<ValueType>> out;
89 for (auto const& point_str : in) {
90 out.emplace_back();
91 for (auto const& pi_str : point_str) {
92 out.back().push_back(storm::utility::convertNumber<ValueType>(pi_str));
93 }
94 }
95 return out;
96}

98template<typename ValueType>
99void evaluateScheduler(storm::Environment const& env, storm::models::sparse::Mdp<ValueType> const& mdp, storm::storage::Scheduler<ValueType> const& scheduler,
100 storm::logic::MultiObjectiveFormula const& formula, std::vector<ValueType>& result) {
101 ASSERT_FALSE(scheduler.isPartialScheduler());
102 ASSERT_EQ(mdp.getNumberOfStates(), scheduler.getNumberOfModelStates());
103 auto inducedModel = mdp.applyScheduler(scheduler);
104 ASSERT_TRUE(inducedModel->isOfType(storm::models::ModelType::Dtmc));
105 auto const dtmcPtr = inducedModel->template as<storm::models::sparse::Dtmc<ValueType>>();
106 ASSERT_TRUE(dtmcPtr != nullptr);
107 auto const& dtmc = *dtmcPtr;
108 ASSERT_TRUE(dtmc.getInitialStates().hasUniqueSetBit());
109 storm::modelchecker::SparseDtmcPrctlModelChecker<storm::models::sparse::Dtmc<ValueType>> modelChecker(dtmc);
110
111 for (auto const& obj : formula.getSubformulas()) {
112 auto objRes = modelChecker.check(env, *obj);
113 EXPECT_TRUE(objRes->isExplicitQuantitativeCheckResult()) << "Objective " << *obj << " did not produce a quantitative result.";
114 auto const& quantitativeResult = objRes->template asExplicitQuantitativeCheckResult<ValueType>();
115 result.push_back(quantitativeResult[*dtmc.getInitialStates().begin()]);
116 }
117}

119template<typename ValueType>
120void assertParetoResult(storm::Environment const& env, storm::models::sparse::Mdp<ValueType> const& mdp, storm::logic::MultiObjectiveFormula const& formula,
121 storm::modelchecker::CheckResult const& result, std::vector<std::vector<std::string>> const& expectedPoints, bool expectSchedulers) {
122 ASSERT_TRUE(result.isExplicitParetoCurveCheckResult()) << "Result is not an explicit Pareto curve check result.";
123 ValueType const eps = storm::utility::convertNumber<ValueType>(1e-4);
124 auto const& paretoResult = result.asExplicitParetoCurveCheckResult<ValueType>();
125 EXPECT_TRUE(expectSubset(paretoResult.getPoints(), convertPointset<ValueType>(expectedPoints), eps)) << "Non-Pareto point found.";
126 EXPECT_TRUE(expectSubset(convertPointset<ValueType>(expectedPoints), paretoResult.getPoints(), eps)) << "Pareto point missing.";
127 ASSERT_EQ(expectSchedulers, paretoResult.hasScheduler());
128 if (expectSchedulers) {
129 ASSERT_EQ(paretoResult.getPoints().size(), paretoResult.getSchedulers().size());
130 for (uint64_t i = 0; i < paretoResult.getPoints().size(); ++i) {
131 auto const& point = paretoResult.getPoints()[i];
132 std::vector<ValueType> inducedPoint;
133 evaluateScheduler<ValueType>(env, mdp, paretoResult.getSchedulers()[i], formula, inducedPoint);
134 ASSERT_EQ(point.size(), inducedPoint.size()) << "Unable to evaluate scheduler for Pareto point " << i;
135 for (uint64_t objDim = 0; objDim < point.size(); ++objDim) {
136 EXPECT_NEAR(point[objDim], inducedPoint[objDim], eps)
137 << "Scheduler for Pareto point " << i << " does not yield expected value for objective dimension " << objDim << ".";
138 }
139 }
140 }
