147 template <typename ValueType>
148 bool expectPointConained(std::vector<std::vector<ValueType>> const& pointset, std::vector<ValueType> const& point, ValueType precision) {
149  for (auto const& p : pointset) {
150  EXPECT_EQ(p.size(), point.size()) << "Missmatch in point dimension.";
151  bool found = true;
152  for (uint64_t i = 0; i < p.size(); ++i) {
153  if (storm::utility::abs<ValueType>(p[i] - point[i]) > precision) {
154  found = false;
155  break;
156  }
157  }
158  if (found) {
159  return true;
160  }
161  }
162  // prepare failure message:
163  std::stringstream errstr;
164  errstr << "Point [";
165  bool firstPi = true;
166  for (auto const& pi : point) {
167  if (firstPi) {
168  firstPi = false;
169  } else {
170  errstr << ", ";
171  }
172  errstr << pi;
173  }
174  errstr << "] is not contained in point set {";
175  bool firstP = true;
176  for (auto const& p : pointset) {
177  if (firstP) {
178  firstP = false;
179  } else {
180  errstr << ", \t";
181  }
182  errstr << "[";
183  firstPi = true;
184  for (auto const& pi : p) {
185  if (firstPi) {
186  firstPi = false;
187  } else {
188  errstr << ", ";
189  }
190  errstr << pi;
191  }
192  errstr << "]";
193  }
194  errstr << "}.";
195  ADD_FAILURE() << errstr.str();
196  return false;
197 }

200 template <typename ValueType>
201 bool expectSubset(std::vector<std::vector<ValueType>> const& lhs, std::vector<std::vector<ValueType>> const& rhs, ValueType precision) {
202  for (auto const& p : lhs) {
203  if (!expectPointConained(rhs, p, precision)) {
204  return false;
205  }
206  }
207  return true;
208 }

210 template <typename ValueType>
211 std::vector<std::vector<ValueType>> convertPointset(std::vector<std::vector<std::string>> const& in) {
212  std::vector<std::vector<ValueType>> out;
213  for (auto const& point_str : in) {
214  out.emplace_back();

216  out.back().push_back(storm::utility::convertNumber<ValueType>(pi_str));
217  }
