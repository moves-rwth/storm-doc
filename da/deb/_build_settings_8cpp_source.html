44BuildSettings::BuildSettings() : ModuleSettings(moduleName) {
45 this->addOption(storm::settings::OptionBuilder(moduleName, prismCompatibilityOptionName, false,
46 "Enables PRISM compatibility. This may be necessary to process some PRISM models.")
47 .setShortName(prismCompatibilityOptionShortName)
48 .build());
49 this->addOption(storm::settings::OptionBuilder(moduleName, dontFixDeadlockOptionName, false,
50 "If the model contains deadlock states, they are `fixed` by attaching self-loops. This option allows "
51 "disabling this operation, which may then violate prerequisites of various verification routines.")
52 .setShortName(dontFixDeadlockOptionShortName)
53 .setIsAdvanced()
54 .build());
55 this->addOption(storm::settings::OptionBuilder(moduleName, fullModelBuildOptionName, false, "If set, include all rewards and labels.").build());
56 this->addOption(storm::settings::OptionBuilder(moduleName, applyNoMaxProgAssumptionOptionName, false,
57 "If set, the maximum progress assumption is not applied while building the model (relevant for MAs)")
58 .setIsAdvanced()
59 .build());
60 this->addOption(
61 storm::settings::OptionBuilder(moduleName, buildChoiceLabelOptionName, false, "If set, also build the choice labels").setIsAdvanced().build());
62 this->addOption(
63 storm::settings::OptionBuilder(moduleName, buildChoiceOriginsOptionName, false,
64 "If set, also build information that for each choice indicates the part(s) of the input that yielded the choice.")
65 .setIsAdvanced()
66 .build());
67 this->addOption(
68 storm::settings::OptionBuilder(moduleName, buildStateValuationsOptionName, false, "If set, also build the state valuations").setIsAdvanced().build());
69 this->addOption(storm::settings::OptionBuilder(moduleName, buildObservationValuationsOptionName, false,
70 "If set, also build the observation valuations (only relevant for POMDPs)")
71 .setIsAdvanced()
72 .build());
73 this->addOption(storm::settings::OptionBuilder(moduleName, buildAllLabelsOptionName, false, "If set, build all labels").setIsAdvanced().build());
74 this->addOption(storm::settings::OptionBuilder(moduleName, noBuildOptionName, false, "If set, do not build the model.").setIsAdvanced().build());
75
76 std::vector<std::string> explorationOrders = {"dfs", "bfs"};
77 this->addOption(storm::settings::OptionBuilder(moduleName, explorationOrderOptionName, false, "Sets which exploration order to use.")
78 .setShortName(explorationOrderOptionShortName)
79 .setIsAdvanced()
80 .addArgument(storm::settings::ArgumentBuilder::createStringArgument("name", "The name of the exploration order to choose.")
81 .addValidatorString(ArgumentValidatorFactory::createMultipleChoiceValidator(explorationOrders))
82 .setDefaultValueString("bfs")
83 .build())
84 .build());
85 this->addOption(storm::settings::OptionBuilder(moduleName, explorationChecksOptionName, false,
86 "If set, additional checks (if available) are performed during model exploration to debug the model.")
87 .setShortName(explorationChecksOptionShortName)
88 .build());
89 this->addOption(storm::settings::OptionBuilder(moduleName, buildOutOfBoundsStateOptionName, false, "If set, a state for out-of-bounds valuations is added")
90 .setIsAdvanced()
91 .build());
92 this->addOption(storm::settings::OptionBuilder(moduleName, buildOverlappingGuardsLabelOptionName, false,
93 "For states where multiple guards are enabled, we add a label (for debugging DTMCs)")
94 .setIsAdvanced()
95 .build());
96 this->addOption(
97 storm::settings::OptionBuilder(moduleName, noSimplifyOptionName, false, "If set, simplification PRISM input is disabled.").setIsAdvanced().build());
98 this->addOption(storm::settings::OptionBuilder(moduleName, bitsForUnboundedVariablesOptionName, false,
99 "Sets the number of bits that is used for unbounded integer variables.")
100 .setIsAdvanced()
101 .addArgument(storm::settings::ArgumentBuilder::createUnsignedIntegerArgument("number", "The number of bits.")
102 .addValidatorUnsignedInteger(ArgumentValidatorFactory::createUnsignedRangeValidatorExcluding(0, 63))
103 .setDefaultValueUnsignedInteger(32)
104 .build())
105 .build());
106 this->addOption(storm::settings::OptionBuilder(moduleName, performLocationElimination, false,
107 "If set, location elimination will be performed before the model is built.")
108 .setIsAdvanced()
109 .addArgument(storm::settings::ArgumentBuilder::createUnsignedIntegerArgument(
110 "location-heuristic", "If this number of locations is reached, no further unfolding will be performed")
111 .setDefaultValueUnsignedInteger(10)
112 .makeOptional()
113 .build())
114 .addArgument(storm::settings::ArgumentBuilder::createUnsignedIntegerArgument(
115 "edges-heuristic", "Determines how many new edges may be created by a single elimination")
116 .setDefaultValueUnsignedInteger(10000)
117 .makeOptional()
118 .build())
119 .build());
120
121 this->addOption(storm::settings::OptionBuilder(moduleName, explorationStateLimitOptionName, false,
122 "Potentially stopped exploration once the specified number of states is exceeded.")
123 .setIsAdvanced()
124 .addArgument(storm::settings::ArgumentBuilder::createUnsignedIntegerArgument("number", "states to explore before stopping.").build())
125 .build());
