34template<typename SparseDtmcModelType>
35bool SparseDtmcPrctlModelChecker<SparseDtmcModelType>::canHandleStatic(CheckTask<storm::logic::Formula, ValueType> const& checkTask,
36 bool* requiresSingleInitialState) {
37 storm::logic::Formula const& formula = checkTask.getFormula();
38 if (formula.isInFragment(storm::logic::prctlstar()
39 .setLongRunAverageRewardFormulasAllowed(true)
40 .setLongRunAverageProbabilitiesAllowed(true)
41 .setConditionalProbabilityFormulasAllowed(true)
42 .setConditionalRewardFormulasAllowed(true)
43 .setTotalRewardFormulasAllowed(true)
44 .setOnlyEventuallyFormuluasInConditionalFormulasAllowed(true)
45 .setRewardBoundedUntilFormulasAllowed(true)
46 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
47 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
48 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
49 .setTimeOperatorsAllowed(true)
50 .setReachbilityTimeFormulasAllowed(true)
51 .setRewardAccumulationAllowed(true)
52 .setHOAPathFormulasAllowed(true)
53 .setDiscountedTotalRewardFormulasAllowed(true)
54 .setDiscountedCumulativeRewardFormulasAllowed(true))) {
55 return true;
56 } else if (checkTask.isOnlyInitialStatesRelevantSet() && formula.isInFragment(storm::logic::quantiles())) {
57 if (requiresSingleInitialState) {
58 *requiresSingleInitialState = true;
59 }
60 return true;
61 }
62 return false;
63}

76std::unique_ptr<CheckResult> SparseDtmcPrctlModelChecker<SparseDtmcModelType>::computeBoundedUntilProbabilities(
77 Environment const& env, CheckTask<storm::logic::BoundedUntilFormula, ValueType> const& checkTask) {
78 storm::logic::BoundedUntilFormula const& pathFormula = checkTask.getFormula();
79 if (pathFormula.isMultiDimensional() || pathFormula.getTimeBoundReference().isRewardBound()) {
80 STORM_LOG_THROW(checkTask.isOnlyInitialStatesRelevantSet(), storm::exceptions::InvalidOperationException,
81 "Checking non-trivial bounded until probabilities can only be computed for the initial states of the model.");
82 storm::logic::OperatorInformation opInfo;
83 if (checkTask.isBoundSet()) {
84 opInfo.bound = checkTask.getBound();
85 }
86 auto formula = std::make_shared<storm::logic::ProbabilityOperatorFormula>(checkTask.getFormula().asSharedPointer(), opInfo);
87 auto numericResult = storm::modelchecker::helper::SparseDtmcPrctlHelper<ValueType>::computeRewardBoundedValues(env, this->getModel(), formula);
88 return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(numericResult)));
89 } else {
90 STORM_LOG_THROW(pathFormula.hasUpperBound(), storm::exceptions::InvalidPropertyException, "Formula needs to have (a single) upper step bound.");
91 STORM_LOG_THROW(pathFormula.hasIntegerLowerBound(), storm::exceptions::InvalidPropertyException, "Formula lower step bound must be discrete/integral.");
92 STORM_LOG_THROW(pathFormula.hasIntegerUpperBound(), storm::exceptions::InvalidPropertyException, "Formula needs to have discrete upper step bound.");
93 std::unique_ptr<CheckResult> leftResultPointer = this->check(env, pathFormula.getLeftSubformula());
94 std::unique_ptr<CheckResult> rightResultPointer = this->check(env, pathFormula.getRightSubformula());
95 ExplicitQualitativeCheckResult const& leftResult = leftResultPointer->asExplicitQualitativeCheckResult();
96 ExplicitQualitativeCheckResult const& rightResult = rightResultPointer->asExplicitQualitativeCheckResult();
97 storm::modelchecker::helper::SparseDeterministicStepBoundedHorizonHelper<ValueType> helper;
98 std::vector<ValueType> numericResult =
99 helper.compute(env, storm::solver::SolveGoal<ValueType>(this->getModel(), checkTask), this->getModel().getTransitionMatrix(),
100 this->getModel().getBackwardTransitions(), leftResult.getTruthValuesVector(), rightResult.getTruthValuesVector(),
101 pathFormula.getNonStrictLowerBound<uint64_t>(), pathFormula.getNonStrictUpperBound<uint64_t>(), checkTask.getHint());
102 std::unique_ptr<CheckResult> result = std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(numericResult)));
103 return result;
104 }

364 "Computing quantiles is only supported for the initial states of a model.");
365 STORM_LOG_THROW(this->getModel().getInitialStates().getNumberOfSetBits() == 1, storm::exceptions::InvalidOperationException,
366 "Quantiles not supported on models with multiple initial states.");
367 uint64_t initialState = *this->getModel().getInitialStates().begin();
368
369 helper::rewardbounded::QuantileHelper<SparseDtmcModelType> qHelper(this->getModel(), checkTask.getFormula());
370 auto res = qHelper.computeQuantile(env);
371
372 if (res.size() == 1 && res.front().size() == 1) {
373 return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(initialState, std::move(res.front().front())));
374 } else {
375 return std::unique_ptr<CheckResult>(new ExplicitParetoCurveCheckResult<ValueType>(initialState, std::move(res)));
376 }
377}

379template<typename SparseDtmcModelType>
380std::unique_ptr<CheckResult> SparseDtmcPrctlModelChecker<SparseDtmcModelType>::computeSteadyStateDistribution(Environment const& env) {
381 // Initialize helper
382 storm::modelchecker::helper::SparseDeterministicInfiniteHorizonHelper<ValueType> helper(this->getModel().getTransitionMatrix());
383
384 // Compute result
385 std::vector<ValueType> result;
386 auto const& initialStates = this->getModel().getInitialStates();
387 uint64_t numInitStates = initialStates.getNumberOfSetBits();
388 if (numInitStates == 1) {
389 result = helper.computeLongRunAverageStateDistribution(env, *initialStates.begin());
390 } else {
391 STORM_LOG_WARN("Multiple initial states found. A uniform distribution over initial states is assumed.");
392 ValueType initProb = storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(numInitStates);
393 result = helper.computeLongRunAverageStateDistribution(env, [&initialStates, &initProb](uint64_t const& stateIndex) {
394 return initialStates.get(stateIndex) ? initProb : storm::utility::zero<ValueType>();
395 });
396 }
397
398 // Return CheckResult
399 return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(result)));
400}

404 // Initialize helper

406
407 // Compute result
408 std::vector<ValueType> result;

414
415 // Return CheckResult
