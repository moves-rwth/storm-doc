35bool SparseDtmcPrctlModelChecker<SparseDtmcModelType>::canHandleStatic(CheckTask<storm::logic::Formula, SolutionType> const& checkTask,
36 bool* requiresSingleInitialState) {
37 storm::logic::Formula const& formula = checkTask.getFormula();
38 if constexpr (storm::IsIntervalType<ValueType>) {
39 if (formula.isInFragment(storm::logic::propositional())) {
40 return true;
41 }
42 if (formula.isInFragment(storm::logic::reachability().setReachabilityRewardFormulasAllowed(true).setRewardOperatorsAllowed(true))) {
43 return true;
44 }
45 } else {
46 if (formula.isInFragment(storm::logic::prctlstar()
47 .setLongRunAverageRewardFormulasAllowed(true)
48 .setLongRunAverageProbabilitiesAllowed(true)
49 .setConditionalProbabilityFormulasAllowed(true)
50 .setConditionalRewardFormulasAllowed(true)
51 .setTotalRewardFormulasAllowed(true)
52 .setOnlyEventuallyFormuluasInConditionalFormulasAllowed(true)
53 .setRewardBoundedUntilFormulasAllowed(true)
54 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
55 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
56 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
57 .setTimeOperatorsAllowed(true)
58 .setReachbilityTimeFormulasAllowed(true)
59 .setRewardAccumulationAllowed(true)
60 .setHOAPathFormulasAllowed(true)
61 .setDiscountedTotalRewardFormulasAllowed(true)
62 .setDiscountedCumulativeRewardFormulasAllowed(true))) {
63 return true;
64 } else if (checkTask.isOnlyInitialStatesRelevantSet() && formula.isInFragment(storm::logic::quantiles())) {
65 if (requiresSingleInitialState) {
66 *requiresSingleInitialState = true;
67 }
68 return true;
69 }
70 }
71
72 return false;
73}
