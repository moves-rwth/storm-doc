41bool SparseDtmcPrctlModelChecker<SparseDtmcModelType>::canHandleStatic(CheckTask<storm::logic::Formula, ValueType> const& checkTask,
42 bool* requiresSingleInitialState) {
43 storm::logic::Formula const& formula = checkTask.getFormula();
44 if (formula.isInFragment(storm::logic::prctlstar()
45 .setLongRunAverageRewardFormulasAllowed(true)
46 .setLongRunAverageProbabilitiesAllowed(true)
47 .setConditionalProbabilityFormulasAllowed(true)
48 .setConditionalRewardFormulasAllowed(true)
49 .setTotalRewardFormulasAllowed(true)
50 .setOnlyEventuallyFormuluasInConditionalFormulasAllowed(true)
51 .setRewardBoundedUntilFormulasAllowed(true)
52 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
53 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
54 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
55 .setTimeOperatorsAllowed(true)
56 .setReachbilityTimeFormulasAllowed(true)
57 .setRewardAccumulationAllowed(true)
58 .setHOAPathFormulasAllowed(true))) {
59 return true;
60 } else if (checkTask.isOnlyInitialStatesRelevantSet() && formula.isInFragment(storm::logic::quantiles())) {
61 if (requiresSingleInitialState) {
62 *requiresSingleInitialState = true;
63 }
64 return true;
65 }
66 return false;
67}

347template<typename SparseDtmcModelType>
348std::unique_ptr<CheckResult> SparseDtmcPrctlModelChecker<SparseDtmcModelType>::computeSteadyStateDistribution(Environment const& env) {
349 // Initialize helper
350 storm::modelchecker::helper::SparseDeterministicInfiniteHorizonHelper<ValueType> helper(this->getModel().getTransitionMatrix());
351
352 // Compute result
353 std::vector<ValueType> result;
354 auto const& initialStates = this->getModel().getInitialStates();
355 uint64_t numInitStates = initialStates.getNumberOfSetBits();
356 if (numInitStates == 1) {
357 result = helper.computeLongRunAverageStateDistribution(env, *initialStates.begin());
358 } else {
359 STORM_LOG_WARN("Multiple initial states found. A uniform distribution over initial states is assumed.");
360 ValueType initProb = storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(numInitStates);
361 result = helper.computeLongRunAverageStateDistribution(env, [&initialStates, &initProb](uint64_t const& stateIndex) {
362 return initialStates.get(stateIndex) ? initProb : storm::utility::zero<ValueType>();
363 });
364 }
365
366 // Return CheckResult
367 return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(result)));
368}
