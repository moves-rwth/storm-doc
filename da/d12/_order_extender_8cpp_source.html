297  template <typename ValueType, typename ConstantType>
298  std::tuple<std::shared_ptr<Order>, uint_fast64_t, uint_fast64_t> OrderExtender<ValueType, ConstantType>::extendOrder(std::shared_ptr<Order> order, std::shared_ptr<MonotonicityResult<VariableType>> monRes, std::shared_ptr<expressions::BinaryRelationExpression> assumption) {
299  if (assumption != nullptr) {
300  STORM_LOG_INFO("Handling assumption " << *assumption << std::endl);
301  handleAssumption(order, assumption);
302  }
303 
304  auto currentStateMode = getNextState(order, numberOfStates, false);
305  while (currentStateMode.first != numberOfStates) {
306  assert (currentStateMode.first < numberOfStates);
307  auto& currentState = currentStateMode.first;
308  auto& successors = stateMap[currentState];
309  std::pair<uint_fast64_t, uint_fast64_t> result = {numberOfStates, numberOfStates};
310 
311  if (successors.size() == 1) {
312  assert (order->contains(successors[0]));
313  handleOneSuccessor(order, currentState, successors[0]);
314  } else if (!successors.empty()) {
315  if (order->isOnlyBottomTopOrder()) {
316  order->add(currentState);
317  if (!order->isTrivial(currentState)) {
318  // This state is part of an scc, therefore, we could do forward reasoning here
319  result = extendByForwardReasoning(order, currentState, successors, assumption!=nullptr);
320  } else {
321  result = {numberOfStates, numberOfStates};
322  }
323  } else {
324  result = extendNormal(order, currentState, successors, assumption != nullptr);
325  }
326  }
327 
328  if (result.first == numberOfStates) {
329  // We did extend the order
330  assert (result.second == numberOfStates);
331  assert (order->sortStates(&successors).size() == successors.size());
332  assert (order->contains(currentState) && order->getNode(currentState) != nullptr);
333 
334  if (monRes != nullptr && currentStateMode.second != -1) {
335  for (auto& param : occuringVariablesAtState[currentState]) {
336  checkParOnStateMonRes(currentState, order, param, monRes);
337  }
338  }
339  // Get the next state
340  currentStateMode = getNextState(order, currentState, true);
341  } else {
342  assert (result.first < numberOfStates);
343  assert (result.second < numberOfStates);
344  assert (order->compare(result.first, result.second) == Order::UNKNOWN);
345  assert (order->compare(result.second, result.first) == Order::UNKNOWN);
346  // Try to add states based on min/max and assumptions, only if we are not in statesToHandle mode
347  if (currentStateMode.second && extendByAssumption(order, currentState, result.first, result.second)) {
348  continue;
349  }
350  // We couldn't extend the order
351  if (nonParametricStates.find(currentState) != nonParametricStates.end()) {
352  if (!order->contains(currentState)) {
353  // State is not parametric, so we hope that just adding it between =) and =( will help us
354  order->add(currentState);
355  }
356  currentStateMode = getNextState(order, currentState, true);
357  continue;
358  } else {
359  if (!currentStateMode.second) {
360  // The state was based on statesToHandle, so it is not bad if we cannot continue with this.
361  currentStateMode = getNextState(order, currentState, false);
362  continue;
363  } else {
364  // The state was based on the topological sorting, so we need to return, but first add this state to the states Sorted as we are not done with it
365  order->addStateSorted(currentState);
366  continueExtending[order] = false;
367  return {order, result.first, result.second};
368  }
369  }
370  }
371  assert (order->sortStates(&successors).size() == successors.size());
372  }
373 
374  assert (order->getDoneBuilding());
375  if (monRes != nullptr) {
376  // monotonicity result for the in-build checking of monotonicity
377  monRes->setDone();
378  }
379  return std::make_tuple(order, numberOfStates, numberOfStates);
380  }

545  }
546 
547  if (s1 == numberOfStates) {
548  assert (statesSorted.size() == successors.size() + 1);
549  // all could be sorted, no need to do anything
550  } else if (s2 == numberOfStates) {
551  if (!order->contains(s1)) {
552  order->add(s1);
553  }
554 
555  if (statesSorted[0] == currentState) {
556  order->addRelation(s1, statesSorted[0], allowMerge);
557  auto res = order->compare(s1, statesSorted[0]);
558  assert ((order->compare(s1, statesSorted[0]) == Order::ABOVE) || (allowMerge && (order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::SAME)));
559  order->addRelation(s1, statesSorted[statesSorted.size() - 1], allowMerge);
560  assert ((order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::ABOVE) || (allowMerge && (order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::SAME)));
561  order->addStateToHandle(s1);
562  } else if (statesSorted[statesSorted.size() - 1] == currentState) {
563  order->addRelation(statesSorted[0], s1, allowMerge);
564  assert ((order->compare(s1, statesSorted[0]) == Order::BELOW) || (allowMerge && (order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::SAME)));
565  order->addRelation(statesSorted[statesSorted.size() - 1], s1, allowMerge);
566  assert ((order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::BELOW) || (allowMerge && (order->compare(s1, statesSorted[statesSorted.size() - 1]) == Order::SAME)));
567  order->addStateToHandle(s1);
568  } else {
569  bool continueSearch = true;
570  for (auto& entry : matrix.getRow(currentState)) {
571  if (entry.getColumn() == s1) {
572  if (entry.getValue().isConstant()) {
573  continueSearch = false;
574  }
575  }
576  }
577  if (continueSearch) {
578  for (auto& i : statesSorted) {
579  if (order->compare(i, s1) == Order::UNKNOWN) {
580  return {i, s1};
581  }
582  }
583  }
584  }
585  } else {
586  return {s1, s2};
587  }
588  assert (order->contains(currentState) && order->compare(order->getNode(currentState), order->getBottom()) == Order::ABOVE && order->compare(order->getNode(currentState), order->getTop()) == Order::BELOW);
589  return {numberOfStates, numberOfStates};
590  }

605  template <typename ValueType, typename ConstantType>
606  Order::NodeComparison OrderExtender<ValueType, ConstantType>::addStatesBasedOnMinMax(std::shared_ptr<Order> order, uint_fast64_t state1, uint_fast64_t state2) const {
607  assert (order->compareFast(state1, state2) == Order::UNKNOWN);
608  assert (minValues.find(order) != minValues.end());
609  std::vector<ConstantType> const& mins = minValues.at(order);
610  std::vector<ConstantType> const& maxs = maxValues.at(order);
611  if (mins[state1] == maxs[state1]
612  && mins[state2] == maxs[state2]
613  && mins[state1] == mins[state2]) {
614  if (order->contains(state1)) {
615  if (order->contains(state2)) {
616  order->merge(state1, state2);
617  assert (!order->isInvalid());
618  } else {
619  order->addToNode(state2, order->getNode(state1));
620  }
621  }
622  return Order::SAME;
623  } else if (mins[state1] > maxs[state2]) {
624  // state 1 will always be larger than state2
625  if (!order->contains(state1)) {
626  order->add(state1);
627  }
628  if (!order->contains(state2)) {
629  order->add(state2);
630  }
631  assert (order->compare(state1, state2) != Order::BELOW);
632  assert (order->compare(state1, state2) != Order::SAME);
633  order->addRelation(state1, state2);
634 
635  return Order::ABOVE;
636  } else if (mins[state2] > maxs[state1]) {
637  // state2 will always be larger than state1
638  if (!order->contains(state1)) {
639  order->add(state1);
640  }
641  if (!order->contains(state2)) {
642  order->add(state2);
643  }
644  assert (order->compare(state2, state1) != Order::BELOW);
645  assert (order->compare(state2, state1) != Order::SAME);
646  order->addRelation(state2, state1);
647  return Order::BELOW;
648  } else {
649  // Couldn't add relation between state1 and state 2 based on min/max values;
650  return Order::UNKNOWN;
651  }
652  }
