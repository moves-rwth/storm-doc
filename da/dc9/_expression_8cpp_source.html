577  // Balance the syntax tree if there are enough operands
578  if (expressions.size() >= 4) {
579  // we treat operands and literals seperately so that a subsequent call to simplify() is more successful (see, e.g., (a + 1) + (b + 1))
580  std::vector<std::vector<storm::expressions::Expression>> operandTypes(2);
581  auto& nonliterals = operandTypes[0];
582  auto& literals = operandTypes[1];
583  for (auto const& expression : expressions) {
584  if (expression.isLiteral()) {
585  literals.push_back(expression);
586  } else {
587  nonliterals.push_back(expression);
588  }
589  }
590 
591  for (auto& operands : operandTypes) {
592  auto opIt = operands.begin();
593  while (operands.size() > 1) {
594  if (opIt == operands.end() || opIt == operands.end() - 1) {
595  opIt = operands.begin();
596  }
597  *opIt = function(*opIt, operands.back());
598  operands.pop_back();
599  ++opIt;
600  }
601  }
602  if (nonliterals.empty()) {
603  return literals.front();
604  } else if (literals.empty()) {
605  return nonliterals.front();
606  } else {
607  return function(literals.front(), nonliterals.front());
608  }
609  } else {
610  return apply(expressions, function);
611  }
612 }
613 } // namespace expressions
614 } // namespace storm
