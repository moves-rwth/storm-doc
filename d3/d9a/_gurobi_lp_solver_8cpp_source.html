190 template<typename ValueType, bool RawMode>
191 GurobiConstraint createConstraint(typename GurobiLpSolver<ValueType, RawMode>::Constraint const& constraint,
192  std::map<storm::expressions::Variable, int> const& variableToIndexMap) {
193  GurobiConstraint gurobiConstraint;
194  storm::expressions::RelationType relationType;
195  if constexpr (RawMode) {
196  gurobiConstraint.rhs = storm::utility::convertNumber<double>(constraint.rhs);
197  relationType = constraint.relationType;
198  gurobiConstraint.variableIndices.insert(gurobiConstraint.variableIndices.end(), constraint.lhsVariableIndices.begin(),
199  constraint.lhsVariableIndices.end());
200  gurobiConstraint.coefficients.reserve(constraint.lhsCoefficients.size());
201  for (auto const& coef : constraint.lhsCoefficients) {
202  gurobiConstraint.coefficients.push_back(storm::utility::convertNumber<double>(coef));
203  }
204  } else {
205  STORM_LOG_THROW(constraint.isRelationalExpression(), storm::exceptions::InvalidArgumentException, "Illegal constraint is not a relational expression.");
206  storm::expressions::LinearCoefficientVisitor::VariableCoefficients leftCoefficients =
207  storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(0));
208  storm::expressions::LinearCoefficientVisitor::VariableCoefficients rightCoefficients =
209  storm::expressions::LinearCoefficientVisitor().getLinearCoefficients(constraint.getOperand(1));
210  leftCoefficients.separateVariablesFromConstantPart(rightCoefficients);
211  gurobiConstraint.rhs = rightCoefficients.getConstantPart();
212  relationType = constraint.getBaseExpression().asBinaryRelationExpression().getRelationType();
213  int len = std::distance(leftCoefficients.begin(), leftCoefficients.end());
214  gurobiConstraint.variableIndices.reserve(len);
215  gurobiConstraint.coefficients.reserve(len);
216  for (auto const& variableCoefficientPair : leftCoefficients) {
217  auto variableIndexPair = variableToIndexMap.find(variableCoefficientPair.first);
218  gurobiConstraint.variableIndices.push_back(variableIndexPair->second);
219  gurobiConstraint.coefficients.push_back(variableCoefficientPair.second);
220  }
221  }
222  // Determine the type of the constraint and add it properly.
223  switch (relationType) {
224  case storm::expressions::RelationType::Less:
225  gurobiConstraint.sense = GRB_LESS_EQUAL;
226  gurobiConstraint.rhs -= storm::settings::getModule<storm::settings::modules::GurobiSettings>().getIntegerTolerance();
227  break;
228  case storm::expressions::RelationType::LessOrEqual:
229  gurobiConstraint.sense = GRB_LESS_EQUAL;
230  break;
231  case storm::expressions::RelationType::Greater:
232  gurobiConstraint.sense = GRB_GREATER_EQUAL;
233  gurobiConstraint.rhs += storm::settings::getModule<storm::settings::modules::GurobiSettings>().getIntegerTolerance();
234  break;
235  case storm::expressions::RelationType::GreaterOrEqual:
236  gurobiConstraint.sense = GRB_GREATER_EQUAL;
237  break;
238  case storm::expressions::RelationType::Equal:
239  gurobiConstraint.sense = GRB_EQUAL;
240  break;
241  default:
242  STORM_LOG_ASSERT(false, "Illegal operator in LP solver constraint.");
243  }
244  return gurobiConstraint;
245 }

264 void GurobiLpSolver<ValueType, RawMode>::addIndicatorConstraint(std::string const& name, Variable indicatorVariable, bool indicatorValue,
265  Constraint const& constraint) {
266  int indVar;
267  // Extract constraint data
268  if constexpr (RawMode) {
269  indVar = indicatorVariable;
270  } else {

282  grbConstr.variableIndices.data(), grbConstr.coefficients.data(), grbConstr.sense, grbConstr.rhs);
283  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException,
284  "Could not assert constraint (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");

322  "Unable to set Gurobi parameter (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");
323 
324  this->optimize();
325 
326  error = GRBgetintattr(model, GRB_INT_ATTR_STATUS, &optimalityStatus);
327  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException,
328  "Unable to retrieve optimization status of Gurobi model (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");
329

511  lvl.firstConstraintIndex = num;

517 template<typename ValueType, bool RawMode>
518 void GurobiLpSolver<ValueType, RawMode>::pop() {
519  if (incrementalData.empty()) {
520  STORM_LOG_ERROR("Tried to pop from a solver without pushing before.");
521  } else {
522  IncrementalLevel const& lvl = incrementalData.back();
523  int num;
524  GRBgetintattr(model, GRB_INT_ATTR_NUMCONSTRS, &num);
525  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(lvl.firstConstraintIndex, num);
526  int error = GRBdelconstrs(model, indicesToBeRemoved.size(), indicesToBeRemoved.data());
527  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException,
528  "Unable to delete constraints (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");
529  GRBgetintattr(model, GRB_INT_ATTR_NUMGENCONSTRS, &num);
530  indicesToBeRemoved = storm::utility::vector::buildVectorForRange(lvl.firstGenConstraintIndex, num);
531  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException,
532  "Unable to delete general constraints (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");
533  indicesToBeRemoved.clear();
534 
535  if (!lvl.variables.empty()) {
536  int firstIndex = -1;
537  bool first = true;
538  for (auto const& var : lvl.variables) {
539  if (first) {
540  auto it = variableToIndexMap.find(var);
541  firstIndex = it->second;
542  variableToIndexMap.erase(it);
543  first = false;
544  } else {
545  variableToIndexMap.erase(var);
546  }
547  }
548  std::vector<int> indicesToBeRemoved = storm::utility::vector::buildVectorForRange(firstIndex, nextVariableIndex);
549  error = GRBdelvars(model, indicesToBeRemoved.size(), indicesToBeRemoved.data());
550  STORM_LOG_THROW(error == 0, storm::exceptions::InvalidStateException,
551  "Unable to delete variables (" << GRBgeterrormsg(**environment) << ", error code " << error << ").");
552  nextVariableIndex = firstIndex;
553  }
554  update();
555  // Assert that indices are as expected
556  GRBgetintattr(model, GRB_INT_ATTR_NUMCONSTRS, &num);
557  STORM_LOG_THROW(lvl.firstConstraintIndex == num, storm::exceptions::InvalidStateException, "Unexpected number of constraints after deletion.");
558  GRBgetintattr(model, GRB_INT_ATTR_NUMGENCONSTRS, &num);
559  STORM_LOG_THROW(lvl.firstGenConstraintIndex == num, storm::exceptions::InvalidStateException,
560  "Unexpected number of general constraints after deletion.");
561  GRBgetintattr(model, GRB_INT_ATTR_NUMVARS, &num);
562  STORM_LOG_THROW(nextVariableIndex == num, storm::exceptions::InvalidStateException, "Unexpected number ofvariables after deletion.");
563 
564  incrementalData.pop_back();
565  }
566 }
