26template<class SparseMdpModelType>
27RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::RewardBoundedMdpPcaaWeightVectorChecker(
28 preprocessing::SparseMultiObjectivePreprocessorResult<SparseMdpModelType> const& preprocessorResult)
29 : PcaaWeightVectorChecker<SparseMdpModelType>(preprocessorResult.objectives),
30 swAll(true),
31 rewardUnfolding(*preprocessorResult.preprocessedModel, preprocessorResult.objectives) {
32 auto rewardAnalysis = preprocessing::SparseMultiObjectiveRewardAnalysis<SparseMdpModelType>::analyze(preprocessorResult);
33 STORM_LOG_THROW(rewardAnalysis.rewardFinitenessType == preprocessing::RewardFinitenessType::AllFinite, storm::exceptions::NotSupportedException,
34 "There is a scheduler that yields infinite reward for one objective. This is not supported.");
35 STORM_LOG_THROW(preprocessorResult.preprocessedModel->getInitialStates().getNumberOfSetBits() == 1, storm::exceptions::NotSupportedException,
36 "The model has multiple initial states.");
37
38 // Update the objective bounds with what the reward unfolding can compute
39 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
40 this->objectives[objIndex].lowerResultBound = rewardUnfolding.getLowerObjectiveBound(objIndex);
41 this->objectives[objIndex].upperResultBound = rewardUnfolding.getUpperObjectiveBound(objIndex);
42 }
43
44 numCheckedEpochs = 0;
45 numChecks = 0;

128template<class SparseMdpModelType>
129void RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::computeEpochSolution(
130 Environment const& env, typename helper::rewardbounded::MultiDimensionalRewardUnfolding<ValueType, false>::Epoch const& epoch,
131 std::vector<ValueType> const& weightVector, EpochCheckingData& cachedData) {
132 ++numCheckedEpochs;
133 swEpochModelBuild.start();
134 auto& epochModel = rewardUnfolding.setCurrentEpoch(epoch);
135 swEpochModelBuild.stop();
136 swEpochModelAnalysis.start();
137 std::vector<typename helper::rewardbounded::MultiDimensionalRewardUnfolding<ValueType, false>::SolutionType> result;
138 result.reserve(epochModel.epochInStates.getNumberOfSetBits());
139 uint64_t solutionSize = this->objectives.size() + 1;
140
141 // If the epoch matrix is empty we do not need to solve linear equation systems
142 if (epochModel.epochMatrix.getEntryCount() == 0) {
143 std::vector<ValueType> weights = weightVector;
144 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
145 if (storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType())) {
146 weights[objIndex] *= -storm::utility::one<ValueType>();
147 }
148 }
149
150 auto stepSolutionIt = epochModel.stepSolutions.begin();
151 auto stepChoiceIt = epochModel.stepChoices.begin();
152 for (auto state : epochModel.epochInStates) {
153 // Obtain the best choice for this state according to the weighted combination of objectives
154 ValueType bestValue;
155 uint64_t bestChoice = std::numeric_limits<uint64_t>::max();
156 auto bestChoiceStepSolutionIt = epochModel.stepSolutions.end();
157 uint64_t lastChoice = epochModel.epochMatrix.getRowGroupIndices()[state + 1];
158 bool firstChoice = true;
159 for (uint64_t choice = epochModel.epochMatrix.getRowGroupIndices()[state]; choice < lastChoice; ++choice) {
160 ValueType choiceValue = storm::utility::zero<ValueType>();
161 // Obtain the (weighted) objective rewards
162 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
163 if (epochModel.objectiveRewardFilter[objIndex].get(choice)) {
164 choiceValue += weights[objIndex] * epochModel.objectiveRewards[objIndex][choice];
165 }
166 }
167
168 // Obtain the step solution if this is a step choice
169 while (*stepChoiceIt < choice) {
170 ++stepChoiceIt;
171 ++stepSolutionIt;
172 }
173 if (*stepChoiceIt == choice) {
174 choiceValue += stepSolutionIt->front();
175 // Check if this choice is better
176 if (firstChoice || choiceValue > bestValue) {
177 bestValue = std::move(choiceValue);
178 bestChoice = choice;
179 bestChoiceStepSolutionIt = stepSolutionIt;
180 }
181 } else if (firstChoice || choiceValue > bestValue) {
182 bestValue = std::move(choiceValue);
183 bestChoice = choice;
184 bestChoiceStepSolutionIt = epochModel.stepSolutions.end();
185 }
186 firstChoice = false;
187 }
188
189 // Insert the solution w.r.t. this choice
190 result.emplace_back();
191 result.back().reserve(solutionSize);
192 result.back().push_back(std::move(bestValue));
193
194 if (bestChoiceStepSolutionIt != epochModel.stepSolutions.end()) {
195 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
196 if (epochModel.objectiveRewardFilter[objIndex].get(bestChoice)) {
197 result.back().push_back((epochModel.objectiveRewards[objIndex][bestChoice] + (*bestChoiceStepSolutionIt)[objIndex + 1]));
198 } else {
199 result.back().push_back((*bestChoiceStepSolutionIt)[objIndex + 1]);
200 }
201 }
202 } else {
203 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
204 if (epochModel.objectiveRewardFilter[objIndex].get(bestChoice)) {
205 result.back().push_back((epochModel.objectiveRewards[objIndex][bestChoice]));
206 } else {
207 result.back().push_back(storm::utility::zero<ValueType>());
208 }
209 }
210 }
211 }
212 } else {
213 updateCachedData(env, epochModel, cachedData, weightVector);
214
215 // Formulate a min-max equation system max(A*x+b)=x for the weighted sum of the objectives
216 assert(cachedData.bMinMax.capacity() >= epochModel.epochMatrix.getRowCount());
217 assert(cachedData.xMinMax.size() == epochModel.epochMatrix.getRowGroupCount());
218 cachedData.bMinMax.assign(epochModel.epochMatrix.getRowCount(), storm::utility::zero<ValueType>());
219 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
220 ValueType weight =
221 storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType()) ? -weightVector[objIndex] : weightVector[objIndex];
222 if (!storm::utility::isZero(weight)) {
223 std::vector<ValueType> const& objectiveReward = epochModel.objectiveRewards[objIndex];
224 for (auto choice : epochModel.objectiveRewardFilter[objIndex]) {
225 cachedData.bMinMax[choice] += weight * objectiveReward[choice];
226 }
227 }
228 }
229 auto stepSolutionIt = epochModel.stepSolutions.begin();
230 for (auto choice : epochModel.stepChoices) {
231 cachedData.bMinMax[choice] += stepSolutionIt->front();
232 ++stepSolutionIt;
233 }
234
235 // Invoke the min max solver
236 cachedData.minMaxSolver->solveEquations(env, cachedData.xMinMax, cachedData.bMinMax);
237 for (auto state : epochModel.epochInStates) {
238 result.emplace_back();
239 result.back().reserve(solutionSize);
240 result.back().push_back(cachedData.xMinMax[state]);
241 }
242
243 // Check whether the linear equation solver needs to be updated
244 auto const& choices = cachedData.minMaxSolver->getSchedulerChoices();
245 if (cachedData.schedulerChoices != choices) {
246 std::vector<uint64_t> choicesTmp = choices;
247 cachedData.minMaxSolver->setInitialScheduler(std::move(choicesTmp));
248 cachedData.schedulerChoices = choices;
249 storm::solver::GeneralLinearEquationSolverFactory<ValueType> linEqSolverFactory;
250 bool needEquationSystem = linEqSolverFactory.getEquationProblemFormat(env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
251 storm::storage::SparseMatrix<ValueType> subMatrix = epochModel.epochMatrix.selectRowsFromRowGroups(choices, needEquationSystem);
252 if (needEquationSystem) {
253 subMatrix.convertToEquationSystem();
254 }
255 cachedData.linEqSolver = linEqSolverFactory.create(env, std::move(subMatrix));
256 cachedData.linEqSolver->setCachingEnabled(true);
257 }
258
259 // Formulate for each objective the linear equation system induced by the performed choices
260 assert(cachedData.bLinEq.size() == choices.size());
261 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
262 auto const& obj = this->objectives[objIndex];
263 std::vector<ValueType> const& objectiveReward = epochModel.objectiveRewards[objIndex];
264 auto rowGroupIndexIt = epochModel.epochMatrix.getRowGroupIndices().begin();
265 auto choiceIt = choices.begin();
266 auto stepChoiceIt = epochModel.stepChoices.begin();
267 auto stepSolutionIt = epochModel.stepSolutions.begin();
268 std::vector<ValueType>& x = cachedData.xLinEq[objIndex];
269 auto xIt = x.begin();
270 for (auto& b_i : cachedData.bLinEq) {
271 uint64_t i = *rowGroupIndexIt + *choiceIt;
272 if (epochModel.objectiveRewardFilter[objIndex].get(i)) {
273 b_i = objectiveReward[i];
274 } else {
275 b_i = storm::utility::zero<ValueType>();
276 }
277 while (*stepChoiceIt < i) {
278 ++stepChoiceIt;
279 ++stepSolutionIt;
280 }
281 if (i == *stepChoiceIt) {
282 b_i += (*stepSolutionIt)[objIndex + 1];
283 ++stepChoiceIt;
284 ++stepSolutionIt;
285 }
286 // We can already set x_i correctly if row i is empty.
287 // Appearingly, some linear equation solvers struggle to converge otherwise.
288 if (epochModel.epochMatrix.getRow(i).getNumberOfEntries() == 0) {
289 *xIt = b_i;
290 }
291 ++xIt;
292 ++rowGroupIndexIt;
293 ++choiceIt;
294 }
295 assert(x.size() == choices.size());
296 auto req = cachedData.linEqSolver->getRequirements(env);
297 cachedData.linEqSolver->clearBounds();
298 if (obj.lowerResultBound) {
299 req.clearLowerBounds();
300 cachedData.linEqSolver->setLowerBound(*obj.lowerResultBound);
301 }
302 if (obj.upperResultBound) {
303 cachedData.linEqSolver->setUpperBound(*obj.upperResultBound);
304 req.clearUpperBounds();
305 }
306 STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException,
307 "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
308 cachedData.linEqSolver->solveEquations(env, x, cachedData.bLinEq);
309 auto resultIt = result.begin();
310 for (auto state : epochModel.epochInStates) {
311 resultIt->push_back(x[state]);
312 ++resultIt;
313 }
314 }
315 }
316 rewardUnfolding.setSolutionForCurrentEpoch(std::move(result));
317 swEpochModelAnalysis.stop();
318}

320template<class SparseMdpModelType>
321void RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::updateCachedData(Environment const& env,
322 helper::rewardbounded::EpochModel<ValueType, false> const& epochModel,
323 EpochCheckingData& cachedData, std::vector<ValueType> const& weightVector) {
324 if (epochModel.epochMatrixChanged) {
325 // Update the cached MinMaxSolver data
326 cachedData.bMinMax.resize(epochModel.epochMatrix.getRowCount());
327 cachedData.xMinMax.assign(epochModel.epochMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
328 storm::solver::GeneralMinMaxLinearEquationSolverFactory<ValueType> minMaxSolverFactory;
329 cachedData.minMaxSolver = minMaxSolverFactory.create(env, epochModel.epochMatrix);
330 cachedData.minMaxSolver->setHasUniqueSolution();
331 cachedData.minMaxSolver->setHasNoEndComponents();
332 cachedData.minMaxSolver->setTrackScheduler(true);
333 cachedData.minMaxSolver->setCachingEnabled(true);
334 auto req = cachedData.minMaxSolver->getRequirements(env);
335 boost::optional<ValueType> lowerBound = this->computeWeightedResultBound(true, weightVector, storm::storage::BitVector(weightVector.size(), true));
336 if (lowerBound) {
337 cachedData.minMaxSolver->setLowerBound(lowerBound.get());
338 req.clearLowerBounds();
339 }
340 boost::optional<ValueType> upperBound = this->computeWeightedResultBound(false, weightVector, storm::storage::BitVector(weightVector.size(), true));
341 if (upperBound) {
342 cachedData.minMaxSolver->setUpperBound(upperBound.get());
343 req.clearUpperBounds();
344 }
345 STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException,
346 "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
347 cachedData.minMaxSolver->setRequirementsChecked(true);
348 cachedData.minMaxSolver->setOptimizationDirection(storm::solver::OptimizationDirection::Maximize);
349
350 // Clear the scheduler choices so that an update of the linEqSolver is enforced
351 cachedData.schedulerChoices.clear();
352 cachedData.schedulerChoices.reserve(epochModel.epochMatrix.getRowGroupCount());
353
354 // Update data for linear equation solving
355 cachedData.bLinEq.resize(epochModel.epochMatrix.getRowGroupCount());
356 cachedData.xLinEq.resize(this->objectives.size());
357 for (auto& x_o : cachedData.xLinEq) {
358 x_o.assign(epochModel.epochMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
359 }
360 }
