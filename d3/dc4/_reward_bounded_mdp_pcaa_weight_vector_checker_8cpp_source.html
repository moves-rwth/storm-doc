31template<class SparseMdpModelType>
32RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::RewardBoundedMdpPcaaWeightVectorChecker(
33 preprocessing::SparseMultiObjectivePreprocessorResult<SparseMdpModelType> const& preprocessorResult)
34 : PcaaWeightVectorChecker<SparseMdpModelType>(preprocessorResult.objectives),
35 swAll(true),
36 rewardUnfolding(*preprocessorResult.preprocessedModel, preprocessorResult.objectives) {
37 auto rewardAnalysis = preprocessing::SparseMultiObjectiveRewardAnalysis<SparseMdpModelType>::analyze(preprocessorResult);
38 STORM_LOG_THROW(rewardAnalysis.rewardFinitenessType == preprocessing::RewardFinitenessType::AllFinite, storm::exceptions::NotSupportedException,
39 "There is a scheduler that yields infinite reward for one objective. This is not supported.");
40 STORM_LOG_THROW(preprocessorResult.preprocessedModel->getInitialStates().getNumberOfSetBits() == 1, storm::exceptions::NotSupportedException,
41 "The model has multiple initial states.");
42
43 // Update the objective bounds with what the reward unfolding can compute
44 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
45 this->objectives[objIndex].lowerResultBound = rewardUnfolding.getLowerObjectiveBound(objIndex);
46 this->objectives[objIndex].upperResultBound = rewardUnfolding.getUpperObjectiveBound(objIndex);
47 }
48
49 numCheckedEpochs = 0;
50 numChecks = 0;

133template<class SparseMdpModelType>
134void RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::computeEpochSolution(
135 Environment const& env, typename helper::rewardbounded::MultiDimensionalRewardUnfolding<ValueType, false>::Epoch const& epoch,
136 std::vector<ValueType> const& weightVector, EpochCheckingData& cachedData) {
137 ++numCheckedEpochs;
138 swEpochModelBuild.start();
139 auto& epochModel = rewardUnfolding.setCurrentEpoch(epoch);
140 swEpochModelBuild.stop();
141 swEpochModelAnalysis.start();
142 std::vector<typename helper::rewardbounded::MultiDimensionalRewardUnfolding<ValueType, false>::SolutionType> result;
143 result.reserve(epochModel.epochInStates.getNumberOfSetBits());
144 uint64_t solutionSize = this->objectives.size() + 1;
145
146 // If the epoch matrix is empty we do not need to solve linear equation systems
147 if (epochModel.epochMatrix.getEntryCount() == 0) {
148 std::vector<ValueType> weights = weightVector;
149 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
150 if (storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType())) {
151 weights[objIndex] *= -storm::utility::one<ValueType>();
152 }
153 }
154
155 auto stepSolutionIt = epochModel.stepSolutions.begin();
156 auto stepChoiceIt = epochModel.stepChoices.begin();
157 for (auto state : epochModel.epochInStates) {
158 // Obtain the best choice for this state according to the weighted combination of objectives
159 ValueType bestValue;
160 uint64_t bestChoice = std::numeric_limits<uint64_t>::max();
161 auto bestChoiceStepSolutionIt = epochModel.stepSolutions.end();
162 uint64_t lastChoice = epochModel.epochMatrix.getRowGroupIndices()[state + 1];
163 bool firstChoice = true;
164 for (uint64_t choice = epochModel.epochMatrix.getRowGroupIndices()[state]; choice < lastChoice; ++choice) {
165 ValueType choiceValue = storm::utility::zero<ValueType>();
166 // Obtain the (weighted) objective rewards
167 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
168 if (epochModel.objectiveRewardFilter[objIndex].get(choice)) {
169 choiceValue += weights[objIndex] * epochModel.objectiveRewards[objIndex][choice];
170 }
171 }
172
173 // Obtain the step solution if this is a step choice
174 while (*stepChoiceIt < choice) {
175 ++stepChoiceIt;
176 ++stepSolutionIt;
177 }
178 if (*stepChoiceIt == choice) {
179 choiceValue += stepSolutionIt->front();
180 // Check if this choice is better
181 if (firstChoice || choiceValue > bestValue) {
182 bestValue = std::move(choiceValue);
183 bestChoice = choice;
184 bestChoiceStepSolutionIt = stepSolutionIt;
185 }
186 } else if (firstChoice || choiceValue > bestValue) {
187 bestValue = std::move(choiceValue);
188 bestChoice = choice;
189 bestChoiceStepSolutionIt = epochModel.stepSolutions.end();
190 }
191 firstChoice = false;
192 }
193
194 // Insert the solution w.r.t. this choice
195 result.emplace_back();
196 result.back().reserve(solutionSize);
197 result.back().push_back(std::move(bestValue));
198
199 if (bestChoiceStepSolutionIt != epochModel.stepSolutions.end()) {
200 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
201 if (epochModel.objectiveRewardFilter[objIndex].get(bestChoice)) {
202 result.back().push_back((epochModel.objectiveRewards[objIndex][bestChoice] + (*bestChoiceStepSolutionIt)[objIndex + 1]));
203 } else {
204 result.back().push_back((*bestChoiceStepSolutionIt)[objIndex + 1]);
205 }
206 }
207 } else {
208 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
209 if (epochModel.objectiveRewardFilter[objIndex].get(bestChoice)) {
210 result.back().push_back((epochModel.objectiveRewards[objIndex][bestChoice]));
211 } else {
212 result.back().push_back(storm::utility::zero<ValueType>());
213 }
214 }
215 }
216 }
217 } else {
218 updateCachedData(env, epochModel, cachedData, weightVector);
219
220 // Formulate a min-max equation system max(A*x+b)=x for the weighted sum of the objectives
221 assert(cachedData.bMinMax.capacity() >= epochModel.epochMatrix.getRowCount());
222 assert(cachedData.xMinMax.size() == epochModel.epochMatrix.getRowGroupCount());
223 cachedData.bMinMax.assign(epochModel.epochMatrix.getRowCount(), storm::utility::zero<ValueType>());
224 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
225 ValueType weight =
226 storm::solver::minimize(this->objectives[objIndex].formula->getOptimalityType()) ? -weightVector[objIndex] : weightVector[objIndex];
227 if (!storm::utility::isZero(weight)) {
228 std::vector<ValueType> const& objectiveReward = epochModel.objectiveRewards[objIndex];
229 for (auto choice : epochModel.objectiveRewardFilter[objIndex]) {
230 cachedData.bMinMax[choice] += weight * objectiveReward[choice];
231 }
232 }
233 }
234 auto stepSolutionIt = epochModel.stepSolutions.begin();
235 for (auto choice : epochModel.stepChoices) {
236 cachedData.bMinMax[choice] += stepSolutionIt->front();
237 ++stepSolutionIt;
238 }
239
240 // Invoke the min max solver
241 cachedData.minMaxSolver->solveEquations(env, cachedData.xMinMax, cachedData.bMinMax);
242 for (auto state : epochModel.epochInStates) {
243 result.emplace_back();
244 result.back().reserve(solutionSize);
245 result.back().push_back(cachedData.xMinMax[state]);
246 }
247
248 // Check whether the linear equation solver needs to be updated
249 auto const& choices = cachedData.minMaxSolver->getSchedulerChoices();
250 if (cachedData.schedulerChoices != choices) {
251 std::vector<uint64_t> choicesTmp = choices;
252 cachedData.minMaxSolver->setInitialScheduler(std::move(choicesTmp));
253 cachedData.schedulerChoices = choices;
254 storm::solver::GeneralLinearEquationSolverFactory<ValueType> linEqSolverFactory;
255 bool needEquationSystem = linEqSolverFactory.getEquationProblemFormat(env) == storm::solver::LinearEquationSolverProblemFormat::EquationSystem;
256 storm::storage::SparseMatrix<ValueType> subMatrix = epochModel.epochMatrix.selectRowsFromRowGroups(choices, needEquationSystem);
257 if (needEquationSystem) {
258 subMatrix.convertToEquationSystem();
259 }
260 cachedData.linEqSolver = linEqSolverFactory.create(env, std::move(subMatrix));
261 cachedData.linEqSolver->setCachingEnabled(true);
262 }
263
264 // Formulate for each objective the linear equation system induced by the performed choices
265 assert(cachedData.bLinEq.size() == choices.size());
266 for (uint64_t objIndex = 0; objIndex < this->objectives.size(); ++objIndex) {
267 auto const& obj = this->objectives[objIndex];
268 std::vector<ValueType> const& objectiveReward = epochModel.objectiveRewards[objIndex];
269 auto rowGroupIndexIt = epochModel.epochMatrix.getRowGroupIndices().begin();
270 auto choiceIt = choices.begin();
271 auto stepChoiceIt = epochModel.stepChoices.begin();
272 auto stepSolutionIt = epochModel.stepSolutions.begin();
273 std::vector<ValueType>& x = cachedData.xLinEq[objIndex];
274 auto xIt = x.begin();
275 for (auto& b_i : cachedData.bLinEq) {
276 uint64_t i = *rowGroupIndexIt + *choiceIt;
277 if (epochModel.objectiveRewardFilter[objIndex].get(i)) {
278 b_i = objectiveReward[i];
279 } else {
280 b_i = storm::utility::zero<ValueType>();
281 }
282 while (*stepChoiceIt < i) {
283 ++stepChoiceIt;
284 ++stepSolutionIt;
285 }
286 if (i == *stepChoiceIt) {
287 b_i += (*stepSolutionIt)[objIndex + 1];
288 ++stepChoiceIt;
289 ++stepSolutionIt;
290 }
291 // We can already set x_i correctly if row i is empty.
292 // Appearingly, some linear equation solvers struggle to converge otherwise.
293 if (epochModel.epochMatrix.getRow(i).getNumberOfEntries() == 0) {
294 *xIt = b_i;
295 }
296 ++xIt;
297 ++rowGroupIndexIt;
298 ++choiceIt;
299 }
300 assert(x.size() == choices.size());
301 auto req = cachedData.linEqSolver->getRequirements(env);
302 cachedData.linEqSolver->clearBounds();
303 if (obj.lowerResultBound) {
304 req.clearLowerBounds();
305 cachedData.linEqSolver->setLowerBound(*obj.lowerResultBound);
306 }
307 if (obj.upperResultBound) {
308 cachedData.linEqSolver->setUpperBound(*obj.upperResultBound);
309 req.clearUpperBounds();
310 }
311 STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException,
312 "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
313 cachedData.linEqSolver->solveEquations(env, x, cachedData.bLinEq);
314 auto resultIt = result.begin();
315 for (auto state : epochModel.epochInStates) {
316 resultIt->push_back(x[state]);
317 ++resultIt;
318 }
319 }
320 }
321 rewardUnfolding.setSolutionForCurrentEpoch(std::move(result));
322 swEpochModelAnalysis.stop();
323}

326void RewardBoundedMdpPcaaWeightVectorChecker<SparseMdpModelType>::updateCachedData(Environment const& env,
327 helper::rewardbounded::EpochModel<ValueType, false> const& epochModel,
328 EpochCheckingData& cachedData, std::vector<ValueType> const& weightVector) {
329 if (epochModel.epochMatrixChanged) {
330 // Update the cached MinMaxSolver data
331 cachedData.bMinMax.resize(epochModel.epochMatrix.getRowCount());
332 cachedData.xMinMax.assign(epochModel.epochMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
333 storm::solver::GeneralMinMaxLinearEquationSolverFactory<ValueType> minMaxSolverFactory;
334 cachedData.minMaxSolver = minMaxSolverFactory.create(env, epochModel.epochMatrix);
335 cachedData.minMaxSolver->setHasUniqueSolution();
336 cachedData.minMaxSolver->setHasNoEndComponents();
337 cachedData.minMaxSolver->setTrackScheduler(true);
338 cachedData.minMaxSolver->setCachingEnabled(true);
339 auto req = cachedData.minMaxSolver->getRequirements(env);
340 boost::optional<ValueType> lowerBound = this->computeWeightedResultBound(true, weightVector, storm::storage::BitVector(weightVector.size(), true));
341 if (lowerBound) {
342 cachedData.minMaxSolver->setLowerBound(lowerBound.get());
343 req.clearLowerBounds();
344 }
345 boost::optional<ValueType> upperBound = this->computeWeightedResultBound(false, weightVector, storm::storage::BitVector(weightVector.size(), true));
346 if (upperBound) {
347 cachedData.minMaxSolver->setUpperBound(upperBound.get());
348 req.clearUpperBounds();
349 }
350 STORM_LOG_THROW(!req.hasEnabledCriticalRequirement(), storm::exceptions::UncheckedRequirementException,
351 "Solver requirements " + req.getEnabledRequirementsAsString() + " not checked.");
352 cachedData.minMaxSolver->setRequirementsChecked(true);
353 cachedData.minMaxSolver->setOptimizationDirection(storm::solver::OptimizationDirection::Maximize);
354
355 // Clear the scheduler choices so that an update of the linEqSolver is enforced
356 cachedData.schedulerChoices.clear();
357 cachedData.schedulerChoices.reserve(epochModel.epochMatrix.getRowGroupCount());
358
359 // Update data for linear equation solving
360 cachedData.bLinEq.resize(epochModel.epochMatrix.getRowGroupCount());
361 cachedData.xLinEq.resize(this->objectives.size());
362 for (auto& x_o : cachedData.xLinEq) {
363 x_o.assign(epochModel.epochMatrix.getRowGroupCount(), storm::utility::zero<ValueType>());
364 }
365 }
