622  storm::expressions::Expression varElimHelper(typename ArrayEliminatorData::Replacement const& replacement, ArrayAccessIndices const& indices,
623  uint64_t const& pos) {
624  if (pos == 0) {
625  STORM_LOG_ASSERT(replacement.isVariable(), "Replacement is not a variable. Is this a full array access?");
626  return replacement.getVariable().getExpressionVariable().getExpression();
627  } else {
628  STORM_LOG_ASSERT(!replacement.isVariable(), "Are there too many nested array accesses?");
629  auto indexExpr = indices[pos - 1];
630  storm::expressions::Expression result;
631  if (indexExpr.containsVariables()) {
632  for (uint64_t index = 0; index < replacement.size(); ++index) {
633  auto child = varElimHelper(replacement.at(index), indices, pos - 1);
634  if (child.isInitialized()) { // i.e. there is no out-of-bounds situation for the child
635  if (result.isInitialized()) {
636  auto currIndexExpr = (indexExpr == indexExpr.getManager().integer(index));
637  result = storm::expressions::ite(currIndexExpr, child, result);
638  } else {
639  result = child;
640  }
641  }
642  }
643  // At this point result remains uninitialized iff all childs are uninitialized (i.e. out-of-bounds).
644  // The underlying assumption here is that indexExpr will never evaluate to an index where the access is out-of-bounds.
645  return result;
646  } else {
647  auto index = static_cast<uint64_t>(indexExpr.evaluateAsInt());
648  if (index < replacement.size()) {
649  return varElimHelper(replacement.at(index), indices, pos - 1);
650  } else {
651  // We have an out-of-bounds situation
652  return storm::expressions::Expression();
653  }
654  }
655  }

733  boost::any arrayExprHelper(storm::expressions::ArrayExpression const& expression, boost::any const& data) {
734  auto& indices = getArrayAccessIndices(data);
735  STORM_LOG_THROW(!indices.empty(), storm::exceptions::NotSupportedException,
736  "Unable to translate ValueArrayExpression to element expression since it does not seem to be within an array access expression.");
737  auto indexExpr = indices.back();
738  auto childIndices = indices;
739  childIndices.pop_back();
740  if (indexExpr.containsVariables()) {
741  STORM_LOG_THROW(!expression.size()->containsVariables(), storm::exceptions::NotSupportedException,
742  "Unable to eliminate array expression of unknown size.");
743  auto exprSize = static_cast<uint64_t>(expression.size()->evaluateAsInt());
744  storm::expressions::Expression result;
745  for (uint64_t index = 0; index < exprSize; ++index) {
746  auto child = boost::any_cast<ResultType>(expression.at(index)->accept(*this, &childIndices));
747  if (!child.isArrayOutOfBounds()) {
748  if (result.isInitialized()) {
749  auto currIndexExpr = (indexExpr == indexExpr.getManager().integer(index));
750  result = storm::expressions::ite(currIndexExpr, child.expr()->toExpression(), result);
751  } else {
752  result = child.expr()->toExpression();
753  }
754  }
755  }
756  if (result.isInitialized()) {
757  return ResultType(result.getBaseExpressionPointer());
758  } else {
759  return ResultType();
760  }
761  } else {
762  auto index = indexExpr.evaluateAsInt();
763  if (expression.size()->containsVariables()) {
764  STORM_LOG_WARN("ignoring size of array expression " << expression << " as it is not constant.");
765  } else {
766  if (index >= expression.size()->evaluateAsInt()) {
767  return ResultType();
768  }
769  }
770  return expression.at(index)->accept(*this, &childIndices);
771  }
772  }

918  virtual void traverse(OrderedAssignments& orderedAssignments, boost::any const& data) override {
919  // Replace array occurrences in LValues and assigned expressions.
920  std::vector<Assignment> newAssignments;
921  if (!orderedAssignments.empty()) {
922  int64_t level = orderedAssignments.getLowestLevel();
923  // Collect for each array variable all kinds of assignments to that variable on the current level
924  std::unordered_map<storm::expressions::Variable, std::vector<Assignment const*>> collectedArrayAssignments;
925 
926  auto processCollectedAssignments = [&]() {
927  for (auto& arrayAssignments : collectedArrayAssignments) {
928  handleArrayAssignments(arrayAssignments.second, level, elimData.replacements.at(arrayAssignments.first), newAssignments);
929  }
930  collectedArrayAssignments.clear();
931  };
932 
933  for (Assignment const& assignment : orderedAssignments) {
934  if (assignment.getLevel() != level) {
935  STORM_LOG_ASSERT(assignment.getLevel() > level, "Ordered Assignment does not have the expected order.");
936  processCollectedAssignments();
937  level = assignment.getLevel();
938  }
939 
940  if (assignment.getLValue().isArray()) {
941  auto insertionRes = collectedArrayAssignments.emplace(assignment.getVariable().getExpressionVariable(), std::vector<Assignment const*>());
942  insertionRes.first->second.push_back(&assignment);
943  } else {
944  // In this case we have a non-array variable, we don't need to eliminate anything in the lhs
945  newAssignments.emplace_back(assignment.getLValue(), arrayExprEliminator.eliminate(assignment.getAssignedExpression()),
946  assignment.getLevel());
947  }
948  }
949  processCollectedAssignments();
950 
951  // Now insert new assignments
952  orderedAssignments.clear();
953  for (auto const& assignment : newAssignments) {
954  orderedAssignments.add(assignment);
955  STORM_LOG_ASSERT(!storm::jani::containsArrayExpression(assignment.getAssignedExpression()),
956  "Assignment still contains array expression: " << assignment);
957  }
958  }

961  private:
962  template<class InsertionCallback>
963  void insertArrayAssignmentReplacements(std::vector<storm::expressions::Expression> const& aaIndices, uint64_t const& currDepth,
964  typename ArrayEliminatorData::Replacement const& currReplacement, storm::expressions::Expression const& currRhs,
965  storm::expressions::Expression const& currCondition, InsertionCallback const& insert) {
966  if (currDepth < aaIndices.size()) {
967  STORM_LOG_ASSERT(!currReplacement.isVariable(), "Did not expect a variable replacement at this depth.");
968  auto currIndexExpr = arrayExprEliminator.eliminate(aaIndices[currDepth]);
969  if (currIndexExpr.containsVariables()) {
970  for (uint64_t index = 0; index < currReplacement.size(); ++index) {
971  auto indexExpr = currIndexExpr.getManager().integer(index);
972  auto childCondition = (currIndexExpr == indexExpr);
973  if (currCondition.isInitialized()) {
974  childCondition = currCondition && childCondition;
975  }
976  // if the current index expression is a single variable, we can substitute it by the concrete value in the rhs
977  if (indexExpr.isVariable()) {
978  auto sub = std::map<storm::expressions::Variable, storm::expressions::Expression>(
979  {{indexExpr.getBaseExpression().asVariableExpression().getVariable(), indexExpr}});
980  auto childRhs = currRhs.substitute(sub);
981  insertArrayAssignmentReplacements(aaIndices, currDepth + 1, currReplacement.at(index), childRhs, childCondition, insert);
982  } else {
983  insertArrayAssignmentReplacements(aaIndices, currDepth + 1, currReplacement.at(index), currRhs, childCondition, insert);
984  }
985  }
986  } else {
987  insertArrayAssignmentReplacements(aaIndices, currDepth + 1, currReplacement.at(currIndexExpr.evaluateAsInt()), currRhs, currCondition, insert);
988  }
989  } else if (currReplacement.isVariable()) {
990  auto eliminationRes = arrayExprEliminator.eliminateChkSizes(currRhs);
991  if (eliminationRes.isArrayOutOfBounds()) {
992  // We have an out-of-bounds situation. This is expected if we have an incomplete array access
993  // For example, when we set a:=b where b is smaller than a
994  STORM_LOG_THROW(currDepth > aaIndices.size(), storm::exceptions::InvalidOperationException,
995  "Detected out of bounds array access in assignment to " << currReplacement.getVariable().getName() << ".");
996  insert(currReplacement.getVariable(), {}, currCondition);
997  } else {
998  insert(currReplacement.getVariable(), eliminationRes.expr(), currCondition);
999  }
1000  } else {
1001  for (uint64_t index = 0; index < currReplacement.size(); ++index) {
1002  auto childRhs = std::make_shared<storm::expressions::ArrayAccessExpression>(currRhs.getManager(), currRhs.getType().getElementType(),
1003  currRhs.getBaseExpressionPointer(),
1004  currRhs.getManager().integer(index).getBaseExpressionPointer())
1005  ->toExpression();
1006  insertArrayAssignmentReplacements(aaIndices, currDepth + 1, currReplacement.at(index), childRhs, currCondition, insert);
1007  }
1008  }

1018  void handleArrayAssignments(std::vector<Assignment const*> const& arrayAssignments, int64_t level,
1019  typename ArrayEliminatorData::Replacement const& replacements, std::vector<Assignment>& newAssignments) {
1020  // We keep array access assignments if we are requested to do so, if all array accesses are full, and if at least one index is not constant
1021  if (keepNonTrivialArrayAccess &&
1022  std::all_of(arrayAssignments.begin(), arrayAssignments.end(), [](Assignment const* a) { return a->getLValue().isFullArrayAccess(); }) &&
1023  std::any_of(arrayAssignments.begin(), arrayAssignments.end(), [](Assignment const* a) { return a->getLValue().arrayIndexContainsVariable(); })) {
1024  for (auto aa : arrayAssignments) {
1025  // Eliminate array expressions in array access indices and the assigned expression
1026  auto const& lValue = aa->getLValue();
1027  STORM_LOG_ASSERT(lValue.isFullArrayAccess(), "unexpected type of lValue");
1028  std::vector<storm::expressions::Expression> newAaIndices;
1029  newAaIndices.reserve(lValue.getArrayIndexVector().size());
1030  for (auto const& i : lValue.getArrayIndexVector()) {
1031  newAaIndices.push_back(arrayExprEliminator.eliminate(i));
1032  }
1033  newAssignments.emplace_back(storm::jani::LValue(lValue.getVariable(), newAaIndices), arrayExprEliminator.eliminate(aa->getAssignedExpression()),
1034  level);
1035  }
1036  } else {
1037  convertToNonArrayAssignments(arrayAssignments, level, replacements, newAssignments);
1038  }
1039  }

1048  void convertToNonArrayAssignments(std::vector<Assignment const*> const& arrayAssignments, int64_t level,
1049  typename ArrayEliminatorData::Replacement const& replacements, std::vector<Assignment>& newAssignments) {
1050  std::map<storm::expressions::Variable, Assignment> collectedAssignments;
1051  auto insert = [&](storm::jani::Variable const& var, storm::expressions::Expression const& newRhs, storm::expressions::Expression const& condition) {
1052  STORM_LOG_ASSERT(!condition.isInitialized() || !storm::jani::containsArrayExpression(condition),
1053  "condition " << condition << " still contains array expressions.");
1054  auto rhs = newRhs.isInitialized() ? newRhs : getOutOfBoundsValue(var);
1055  auto findIt = collectedAssignments.find(var.getExpressionVariable());
1056  if (findIt == collectedAssignments.end()) {
1057  if (condition.isInitialized()) {
1058  rhs = storm::expressions::ite(condition, rhs, var.getExpressionVariable().getExpression());
1059  }
1060  collectedAssignments.emplace(var.getExpressionVariable(), Assignment(LValue(var), rhs, level));
1061  } else {
1062  auto otherRhs = findIt->second.getAssignedExpression();
1063  // Deal with cases like 'a[0]:=x; a:=b' or 'a:=b; a[0]:=x'
1064  STORM_LOG_THROW(condition.isInitialized(), storm::exceptions::InvalidOperationException,
1065  "Found conflicting array assignments to " << var.getName());
