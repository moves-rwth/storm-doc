36template<typename SparseMdpModelType>
37bool SparseMdpPrctlModelChecker<SparseMdpModelType>::canHandleStatic(CheckTask<storm::logic::Formula, SolutionType> const& checkTask,
38 bool* requiresSingleInitialState) {
39 storm::logic::Formula const& formula = checkTask.getFormula();
40 if constexpr (std::is_same_v<ValueType, storm::Interval>) {
41 if (formula.isInFragment(storm::logic::propositional())) {
42 return true;
43 }
44 if (formula.isInFragment(storm::logic::reachability())) {
45 return true;
46 }
47 } else {
48 if (formula.isInFragment(storm::logic::prctlstar()
49 .setLongRunAverageRewardFormulasAllowed(true)
50 .setLongRunAverageProbabilitiesAllowed(true)
51 .setConditionalProbabilityFormulasAllowed(true)
52 .setOnlyEventuallyFormuluasInConditionalFormulasAllowed(true)
53 .setTotalRewardFormulasAllowed(true)
54 .setRewardBoundedUntilFormulasAllowed(true)
55 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
56 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
57 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
58 .setTimeOperatorsAllowed(true)
59 .setReachbilityTimeFormulasAllowed(true)
60 .setRewardAccumulationAllowed(true)
61 .setDiscountedTotalRewardFormulasAllowed(true)
62 .setDiscountedCumulativeRewardFormulasAllowed(true))) {
63 return true;
64 } else if (checkTask.isOnlyInitialStatesRelevantSet()) {
65 auto multiObjectiveFragment = storm::logic::multiObjective()
66 .setTimeAllowed(true)
67 .setCumulativeRewardFormulasAllowed(true)
68 .setTimeBoundedCumulativeRewardFormulasAllowed(true)
69 .setStepBoundedCumulativeRewardFormulasAllowed(true)
70 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
71 .setTimeBoundedUntilFormulasAllowed(true)
72 .setStepBoundedUntilFormulasAllowed(true)
73 .setRewardBoundedUntilFormulasAllowed(true)
74 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
75 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
76 .setRewardAccumulationAllowed(true);
77 auto lexObjectiveFragment = storm::logic::lexObjective()
78 .setHOAPathFormulasAllowed(true)
79 .setCumulativeRewardFormulasAllowed(true)
80 .setTimeBoundedCumulativeRewardFormulasAllowed(true)
81 .setStepBoundedCumulativeRewardFormulasAllowed(true)
82 .setRewardBoundedCumulativeRewardFormulasAllowed(true)
83 .setTimeBoundedUntilFormulasAllowed(true)
84 .setStepBoundedUntilFormulasAllowed(true)
85 .setRewardBoundedUntilFormulasAllowed(true)
86 .setMultiDimensionalBoundedUntilFormulasAllowed(true)
87 .setMultiDimensionalCumulativeRewardFormulasAllowed(true)
88 .setRewardAccumulationAllowed(true);
89
90 if (formula.isInFragment(multiObjectiveFragment) || formula.isInFragment(storm::logic::quantiles()) || formula.isInFragment(lexObjectiveFragment)) {
91 if (requiresSingleInitialState) {
92 *requiresSingleInitialState = true;
93 }
94 return true;
95 }
96 }
97 }
98 return false;
99}
