34 storm::dd::DdType const ddType = storm::dd::DdType::Sylvan;

38 }
39 STORM_LOG_THROW(modelDescription.isPrismProgram() || modelDescription.isJaniModel(), storm::exceptions::UnexpectedException,

41 switch (builderType) {
42 case BuilderType::Explicit:

44 return storm::generator::PrismNextStateGenerator<ValueType>::canHandle(modelDescription.asPrismProgram());

47 }
48 case BuilderType::Dd:

51 } else {

53 }
54 }
55 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Unhandled builderType.");
