91 void ImcaParserGrammar<ValueType, StateType>::addChoiceToStateBehavior(StateType const& state, std::string const& label,
92  std::vector<std::pair<StateType, ValueType>> const& transitions,
93  boost::optional<ValueType> const& reward) {
94  bool isMarkovian = label == "!";
95  storm::generator::Choice<ValueType, StateType> choice(0, isMarkovian);
96  STORM_LOG_THROW(!transitions.empty(), storm::exceptions::WrongFormatException, "Empty choice defined for state s" << state << ".");
97  if (buildChoiceLabels && !isMarkovian) {
98  choice.addLabel(label);
99  }
100  if (reward && !isMarkovian) {
101  hasActionReward = true;
102  choice.addReward(reward.get());
103  }
104  for (auto const& t : transitions) {
105  STORM_LOG_THROW(t.second > storm::utility::zero<ValueType>(), storm::exceptions::WrongFormatException,
106  "Probabilities and rates have to be positive. got " << t.second << " at state s" << state << ".");
107  choice.addProbability(t.first, t.second);
108  }
109  STORM_LOG_THROW(isMarkovian || storm::utility::isOne(choice.getTotalMass()), storm::exceptions::WrongFormatException,
110  "Probability for choice " << label << " on state s" << state << " does not sum up to one.");
111 
112  ++numChoices;
113  numTransitions += choice.size();
114  auto& behavior = stateBehaviors[state];
115  behavior.setExpanded(true);
116  behavior.addChoice(std::move(choice));
117  if (isMarkovian) {
118  markovianStates.set(state);
119  if (reward) {
120  hasStateReward = true;
121  behavior.addStateReward(reward.get());
122  }
123  }

126 template<typename ValueType, typename StateType>
127 storm::storage::sparse::ModelComponents<ValueType> ImcaParserGrammar<ValueType, StateType>::createModelComponents() {
128  // Prepare the statelabeling
129  initialStates.resize(numStates);
130  goalStates.resize(numStates);
131  markovianStates.resize(numStates);
132  storm::models::sparse::StateLabeling stateLabeling(numStates);
133  stateLabeling.addLabel("init", std::move(initialStates));
134  stateLabeling.addLabel("goal", std::move(goalStates));
135 
136  // Fix deadlocks (if required)
137  assert(stateBehaviors.size() == numStates);
138  if (!storm::settings::getModule<storm::settings::modules::BuildSettings>().isDontFixDeadlocksSet()) {
139  StateType state = 0;
140  for (auto& behavior : stateBehaviors) {
141  if (!behavior.wasExpanded()) {
142  storm::generator::Choice<ValueType, StateType> choice(0, true);
143  choice.addProbability(state, storm::utility::one<ValueType>());
144  behavior.setExpanded(true);
145  behavior.addChoice(std::move(choice));
146  markovianStates.set(state);
147  ++numChoices;
148  ++numTransitions;
149  }
150  ++state;
151  }
152  }
153 
154  // Build the transition matrix together with exit rates, reward models, and choice labeling
155  storm::storage::SparseMatrixBuilder<ValueType> matrixBuilder(numChoices, numStates, numTransitions, true, true, numStates);
156  std::vector<ValueType> exitRates;
157  exitRates.reserve(numStates);
158  std::optional<std::vector<ValueType>> stateRewards, actionRewards;
159  if (hasStateReward) {
160  stateRewards = std::vector<ValueType>(numStates, storm::utility::zero<ValueType>());
161  }
162  if (hasActionReward) {
163  actionRewards = std::vector<ValueType>(numChoices, storm::utility::zero<ValueType>());
164  }
165  std::optional<storm::models::sparse::ChoiceLabeling> choiceLabeling;
166  if (buildChoiceLabels) {
167  choiceLabeling = storm::models::sparse::ChoiceLabeling(numChoices);
168  }
169  StateType state = 0;
170  StateType row = 0;
171  for (auto const& behavior : stateBehaviors) {
172  matrixBuilder.newRowGroup(row);
173  if (!behavior.getStateRewards().empty()) {
174  assert(behavior.getStateRewards().size() == 1);
175  stateRewards.value()[state] = behavior.getStateRewards().front();
176  }
177  if (markovianStates.get(state)) {
178  // For Markovian states, the Markovian choice has to be the first one in the resulting transition matrix.
179  bool markovianChoiceFound = false;
180  for (auto const& choice : behavior) {
181  if (choice.isMarkovian()) {
182  STORM_LOG_THROW(!markovianChoiceFound, storm::exceptions::WrongFormatException,
183  "Multiple Markovian choices defined for state " << state << ".");
184  markovianChoiceFound = true;
185  if (!choice.getRewards().empty()) {
186  assert(choice.getRewards().size() == 1);
187  actionRewards.value()[row] = choice.getRewards().front();
188  }
189  if (buildChoiceLabels && choice.hasLabels()) {
190  assert(choice.getLabels().size() == 1);
191  std::string const& label = *choice.getLabels().begin();
192  if (!choiceLabeling->containsLabel(label)) {
193  choiceLabeling->addLabel(label);
194  }
195  choiceLabeling->addLabelToChoice(label, row);
196  }
197  exitRates.push_back(choice.getTotalMass());
198  for (auto const& transition : choice) {
199  matrixBuilder.addNextValue(row, transition.first, static_cast<ValueType>(transition.second / exitRates.back()));
200  }
201  ++row;
202  }
203  }
204  } else {
205  exitRates.push_back(storm::utility::zero<ValueType>());
206  }
207  // Now add all probabilistic choices.
208  for (auto const& choice : behavior) {
209  if (!choice.isMarkovian()) {
210  if (!choice.getRewards().empty()) {
211  assert(choice.getRewards().size() == 1);
212  actionRewards.value()[row] = choice.getRewards().front();
213  }
214  if (buildChoiceLabels && choice.hasLabels()) {
215  assert(choice.getLabels().size() == 1);
216  std::string const& label = *choice.getLabels().begin();
217  if (!choiceLabeling->containsLabel(label)) {
218  choiceLabeling->addLabel(label);
219  }
220  choiceLabeling->addLabelToChoice(label, row);
221  }
222  for (auto const& transition : choice) {
223  matrixBuilder.addNextValue(row, transition.first, transition.second);
224  }
225  ++row;
226  }
227  }
228  ++state;
229  }
230 
231  // Finalize the model components
232  std::unordered_map<std::string, storm::models::sparse::StandardRewardModel<ValueType>> rewardModel;
233  if (hasStateReward || hasActionReward) {
234  rewardModel.insert(std::make_pair("", storm::models::sparse::StandardRewardModel<ValueType>(stateRewards, actionRewards)));
235  }
236  storm::storage::sparse::ModelComponents<ValueType> components(matrixBuilder.build(), std::move(stateLabeling), std::move(rewardModel), false,
237  std::move(markovianStates));
238  components.exitRates = std::move(exitRates);
239  components.choiceLabeling = std::move(choiceLabeling);
240 
241  return components;
242 }
