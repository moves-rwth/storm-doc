631  std::vector<boost::optional<storm::logic::TimeBound>> lowerBounds, upperBounds;
632  std::vector<storm::logic::TimeBoundReference> timeBoundReferences;
633  for (auto const& subformula : subformulas) {

643  } else {

648  } else {
