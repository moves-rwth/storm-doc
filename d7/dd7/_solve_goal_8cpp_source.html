55 if (optimizationDirection) {
56 if (optimizationDirection == storm::solver::OptimizationDirection::Minimize) {
57 optimizationDirection = storm::solver::OptimizationDirection::Maximize;
58 } else {
59 optimizationDirection = storm::solver::OptimizationDirection::Minimize;
60 }
61 }
62 if (threshold) {
63 this->threshold = storm::utility::one<SolutionType>() - this->threshold.get();
64 }
65 if (comparisonType) {
66 switch (comparisonType.get()) {
67 case storm::logic::ComparisonType::Less:
68 comparisonType = storm::logic::ComparisonType::GreaterEqual;
69 break;
70 case storm::logic::ComparisonType::LessEqual:
71 comparisonType = storm::logic::ComparisonType::Greater;
72 break;
73 case storm::logic::ComparisonType::Greater:
74 comparisonType = storm::logic::ComparisonType::LessEqual;
75 break;
76 case storm::logic::ComparisonType::GreaterEqual:
77 comparisonType = storm::logic::ComparisonType::Less;
78 break;
79 }
