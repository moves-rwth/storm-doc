74template<typename RewardModelType>
75FilteredRewardModel<RewardModelType> createFilteredRewardModel(RewardModelType const& baseRewardModel, storm::logic::RewardAccumulation const& acc,
76 bool isDiscreteTimeModel) {
77 STORM_LOG_THROW(isDiscreteTimeModel || !acc.isExitSet() || !baseRewardModel.hasStateRewards(), storm::exceptions::NotSupportedException,
78 "Exit rewards for continuous time models are not supported.");
79 // Check which of the available reward types are allowed.
80 bool hasStateRewards = isDiscreteTimeModel ? acc.isExitSet() : acc.isTimeSet();
81 bool hasStateActionRewards = acc.isStepsSet();
82 bool hasTransitionRewards = acc.isStepsSet();
83 return FilteredRewardModel<RewardModelType>(baseRewardModel, hasStateRewards, hasStateActionRewards, hasTransitionRewards);
84}
