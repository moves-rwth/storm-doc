18 struct GateGroupToHash {
19  static constexpr uint_fast64_t fivebitmask = (1 << 6) - 1;
20  static constexpr uint_fast64_t eightbitmask = (1 << 9) - 1;
21 
25  uint_fast64_t operator()(storm::dft::storage::elements::DFTElementType type, size_t nrChildren, size_t nrParents, size_t nrPDEPs, size_t nrRestrictions,
26  size_t rank) const {
27  // Sets first bit to 1
28  uint_fast64_t groupHash = static_cast<uint_fast64_t>(1) << 63;
29  // Assumes 5 bits for the rank,
30  groupHash |= (static_cast<uint_fast64_t>(rank) & fivebitmask) << (62 - 5);
31  // 8 bits for the nrChildren
32  groupHash |= (static_cast<uint_fast64_t>(nrChildren) & eightbitmask) << (62 - 5 - 8);
33  // 5 bits for nrParents
34  groupHash |= (static_cast<uint_fast64_t>(nrParents) & fivebitmask) << (62 - 5 - 8 - 5);
35  // 5 bits for nrPDEPs
36  groupHash |= (static_cast<uint_fast64_t>(nrPDEPs) & fivebitmask) << (62 - 5 - 8 - 5 - 5);
37  // 5 bits for nrRestrictions
38  groupHash |= (static_cast<uint_fast64_t>(nrPDEPs) & fivebitmask) << (62 - 5 - 8 - 5 - 5 - 5);
39  // 5 bits for the type
40  groupHash |= (static_cast<uint_fast64_t>(type) & fivebitmask) << (62 - 5 - 8 - 5 - 5 - 5 - 5);
41  return groupHash;
42  }
43 };

84 template<typename ValueType>
85 bool operator==(BEColourClass<ValueType> const& lhs, BEColourClass<ValueType> const& rhs) {
86  if (lhs.type != rhs.type) {
87  return false;
88  }
89  if (lhs.nrParents != rhs.nrParents || lhs.nrOutDep != rhs.nrOutDep || lhs.nrInDep != rhs.nrInDep || lhs.nrRestrictions != rhs.nrRestrictions) {
90  return false;
91  }
92  switch (lhs.type) {
93  case storm::dft::storage::elements::BEType::CONSTANT:
94  return lhs.failed == rhs.failed;
95  case storm::dft::storage::elements::BEType::PROBABILITY:
96  case storm::dft::storage::elements::BEType::EXPONENTIAL:
97  case storm::dft::storage::elements::BEType::LOGNORMAL:
98  case storm::dft::storage::elements::BEType::WEIBULL:
99  return lhs.valueA == rhs.valueA && lhs.valueB == rhs.valueB;
100  case storm::dft::storage::elements::BEType::ERLANG:
101  return lhs.valueA == rhs.valueA && lhs.valueA == rhs.valueB && lhs.phases == rhs.phases;
102  case storm::dft::storage::elements::BEType::SAMPLES:
103  // Samples are not compared and always assumed to be non-symmetric
104  return false;
105  default:
106  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "BE of type '" << lhs.type << "' is not known.");
107  break;
108  }

409  if (!candidatesCompatible) {

411  }
412  if (bijection.empty()) {

414  } else {

433  void constructInitialBijection() {
434  STORM_LOG_ASSERT(candidatesCompatible, "Candidates are not compatible.");
435  // We first construct the currentPermutations, which helps to determine the current state of the check.
436  initializePermutationsAndTreatTrivialGroups(bleft.beCandidates, bright.beCandidates, currentPermutations.beCandidates);
437  initializePermutationsAndTreatTrivialGroups(bleft.gateCandidates, bright.gateCandidates, currentPermutations.gateCandidates);
438  initializePermutationsAndTreatTrivialGroups(bleft.pdepCandidates, bright.pdepCandidates, currentPermutations.pdepCandidates);
439  initializePermutationsAndTreatTrivialGroups(bleft.restrictionCandidates, bright.restrictionCandidates, currentPermutations.restrictionCandidates);
440  STORM_LOG_TRACE(bijection.size() << " vs. " << bleft.size() << " vs. " << bright.size());
441  STORM_LOG_ASSERT(bijection.size() == bleft.size(), "No. of bijection elements do not match.");

450  bool foundNext = false;
451  if (!currentPermutations.beCandidates.empty()) {
452  auto it = currentPermutations.beCandidates.begin();

454  foundNext = std::next_permutation(it->second.begin(), it->second.end());

462  ++it;

464  }

470  ++it;

472  }

475  auto it = currentPermutations.restrictionCandidates.begin();
476  while (!foundNext && it != currentPermutations.restrictionCandidates.end()) {

478  ++it;

501  }
502  }

516  bool check() const {
517  // Perform additional checks not yet covered by the colouring
518  STORM_LOG_ASSERT(bijection.size() == bleft.size(), "No. of bijection elements do not match.");
519  for (auto const& indexpair : bijection) {
520  if (!dft.getElement(indexpair.first)->isTypeEqualTo(*dft.getElement(indexpair.second))) {
521  // In-depth type check failed, e.g. voting thresholds do not match
522  return false;
523  }
524  if (dft.getElement(indexpair.first)->isRelevant() || dft.getElement(indexpair.second)->isRelevant()) {
525  // Relevant events are not symmetric
526  return false;
527  }
528  if (dft.isGate(indexpair.first)) {
529  STORM_LOG_ASSERT(dft.isGate(indexpair.second), "Element is no gate.");
530  auto const& lGate = dft.getGate(indexpair.first);
531  auto const& rGate = dft.getGate(indexpair.second);
532  // Compare children for gates
533  if (!checkChildren(lGate->children(), rGate->children(), !lGate->isStaticElement())) {
534  return false;
535  }
536  } else if (dft.isDependency(indexpair.first)) {
537  STORM_LOG_ASSERT(dft.isDependency(indexpair.second), "Element is no dependency.");
538  auto const& lDep = dft.getDependency(indexpair.first);
539  auto const& rDep = dft.getDependency(indexpair.second);
540 
541  if (bijection.at(lDep->triggerEvent()->id()) != rDep->triggerEvent()->id()) {
542  // Symmetric dependencies must have the same trigger event
543  return false;
544  }
545  if (!checkChildren(lDep->dependentEvents(), rDep->dependentEvents(), false)) {
546  // Symmetric dependencies must have the same dependent events
547  return false;
548  }
549  } else if (dft.isRestriction(indexpair.first)) {
550  STORM_LOG_ASSERT(dft.isRestriction(indexpair.second), "Element is no restriction.");

552  auto const& rRestr = dft.getRestriction(indexpair.second);

560  STORM_LOG_ASSERT(dft.isBasicElement(indexpair.second), "Element is no BE.");
561  // No operations required.
562  }
563  }
564  return true;

622  void initializePermutationsAndTreatTrivialGroups(std::unordered_map<ColourType, std::vector<size_t>> const& left,
623  std::unordered_map<ColourType, std::vector<size_t>> const& right,
624  std::unordered_map<ColourType, std::vector<size_t>>& permutations) {

629  std::sort(it.first->second.begin(), it.first->second.end());
630  zipVectorsIntoMap(left.at(colour.first), it.first->second, bijection);
631  } else {

633  STORM_LOG_ASSERT(bijection.count(left.at(colour.first).front()) == 0, "Element already contained.");
634  bijection[left.at(colour.first).front()] = colour.second.front();
635  }

652  bool checkChildren(std::vector<std::shared_ptr<storm::dft::storage::elements::DFTElement<ValueType>>> const& left,
653  std::vector<std::shared_ptr<storm::dft::storage::elements::DFTElement<ValueType>>> const& right, bool ensureOrder) const {
654  if (ensureOrder) {
655  // Observe order of entries
656  size_t mappedId;
657  size_t rightId;
658  for (size_t i = 0; i < left.size(); ++i) {
659  if (bleft.has(left[i]->id())) {
660  mappedId = bijection.at(left[i]->id());
661  } else {
662  mappedId = -1;
663  }
664  if (bright.has(right[i]->id())) {
665  rightId = right[i]->id();
666  } else {
667  rightId = -1;
668  }
669  if (mappedId != rightId) {
670  return false;
671  }
672  }
673  return true;
674  } else {
675  // Order is irrelevant
676  std::set<size_t> mappedIds;
677  std::set<size_t> rightIds;
678  for (auto const& l : left) {
679  if (bleft.has(l->id())) {
680  mappedIds.insert(l->id());
681  }
682  }
683  for (auto const& r : right) {
684  if (bright.has(r->id())) {
685  rightIds.insert(r->id());
686  }
687  }
688  return mappedIds != rightIds;
689  }
690  }

699  size_t operator()(storm::dft::storage::BEColourClass<ValueType> const& bcc) const {
700  constexpr uint_fast64_t fivebitmask = (1ul << 6) - 1ul;
701  constexpr uint_fast64_t eightbitmask = (1ul << 9) - 1ul;
702  constexpr uint_fast64_t fortybitmask = (1ul << 41) - 1ul;
703  std::hash<ValueType> hasher;
704  uint_fast64_t groupHash = static_cast<uint_fast64_t>(1) << 63;
705  groupHash |= (static_cast<uint_fast64_t>(bcc.type) & fivebitmask) << (62 - 5);
706 
707  switch (bcc.type) {
708  case storm::dft::storage::elements::BEType::CONSTANT:
709  groupHash |= (static_cast<uint_fast64_t>(bcc.failed) & fortybitmask) << 8;
710  break;
711  case storm::dft::storage::elements::BEType::PROBABILITY:
712  case storm::dft::storage::elements::BEType::EXPONENTIAL:
713  case storm::dft::storage::elements::BEType::LOGNORMAL:
714  case storm::dft::storage::elements::BEType::WEIBULL:
715  groupHash |= ((hasher(bcc.valueA) ^ hasher(bcc.valueB)) & fortybitmask) << 8;
716  break;
717  case storm::dft::storage::elements::BEType::ERLANG:
718  groupHash |= ((hasher(bcc.valueA) ^ hasher(bcc.valueB) ^ static_cast<uint_fast64_t>(bcc.phases)) & fortybitmask) << 8;
719  break;
720  case storm::dft::storage::elements::BEType::SAMPLES:
721  // Samples have no dedicated hashing
722  break;
723  }
724  groupHash |= static_cast<uint_fast64_t>(bcc.nrParents) & eightbitmask;
725  return groupHash;
726  }
