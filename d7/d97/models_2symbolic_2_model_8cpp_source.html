117  // First check whether we have a BDD for this label.
118  auto bddIt = labelToBddMap.find(label);

138 template<storm::dd::DdType Type, typename ValueType>
139 storm::dd::Bdd<Type> Model<Type, ValueType>::getStates(storm::expressions::Expression const& expression) const {
140  if (expression.isTrue()) {
141  return this->getReachableStates();
142  } else if (expression.isFalse()) {
143  return manager->getBddZero();
144  }
145 
146  // Look up the string equivalent of the expression.
147  std::stringstream stream;
148  stream << expression;
149  auto bddIt = labelToBddMap.find(stream.str());
150  if (bddIt != labelToBddMap.end()) {
151  return bddIt->second;
152  }
153 
154  // Finally try to translate the expression with an adapter.
155  STORM_LOG_THROW(rowExpressionAdapter != nullptr, storm::exceptions::InvalidOperationException,
156  "Cannot create BDD for expression without expression adapter.");
157  return rowExpressionAdapter->translateExpression(expression).toBdd() && this->reachableStates;
158 }

456  } else if (this->getType() == storm::models::ModelType::S2pg) {
