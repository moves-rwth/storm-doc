22RewardModelInformation::RewardModelInformation(Model const& model, storm::expressions::Expression const& rewardModelExpression)
23 : stateRewards(false), actionRewards(false), transitionRewards(false) {
24 auto variablesInRewardExpression = rewardModelExpression.getVariables();
25 std::map<storm::expressions::Variable, storm::expressions::Expression> initialSubstitution;
26 bool containsNonTransientVariable = false;
27 for (auto const& v : variablesInRewardExpression) {
28 if (model.hasConstant(v.getName())) {
29 auto const& janiConst = model.getConstant(v.getName());
30 if (janiConst.isDefined()) {
31 initialSubstitution.emplace(v, janiConst.getExpression());
32 }
33 } else {
34 STORM_LOG_ASSERT(model.hasGlobalVariable(v.getName()), "Unable to find global variable " << v.getName() << " occurring in a reward expression.");
35 auto const& janiVar = model.getGlobalVariable(v.getName());
36 if (janiVar.hasInitExpression()) {
37 initialSubstitution.emplace(v, janiVar.getInitExpression());
38 }
39 if (!janiVar.isTransient()) {
40 containsNonTransientVariable = true;
41 }
42 }
43 }
44 auto initExpr = storm::jani::substituteJaniExpression(rewardModelExpression, initialSubstitution, true).simplify();
45 if (containsNonTransientVariable || initExpr.containsVariables() || !storm::utility::isZero(initExpr.evaluateAsRational())) {
46 stateRewards = true;
47 actionRewards = true;
48 transitionRewards = true;
49 }
50 traverse(model, &variablesInRewardExpression);
51}

54 auto const& vars = *boost::any_cast<std::set<storm::expressions::Variable>*>(data);
