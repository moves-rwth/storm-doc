11bool checkDefiningExpression(storm::expressions::Variable const& var, storm::expressions::Expression const& definingExpression,
12 storm::expressions::Expression const& constaintExpression) {

14 auto smtSolver = storm::utility::solver::getSmtSolver(*manager);
