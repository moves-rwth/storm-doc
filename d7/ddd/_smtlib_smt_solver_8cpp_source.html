129SmtSolver::CheckResult SmtlibSmtSolver::check() {
130 writeCommand("( check-sat )", false);
131
132 if (processIdOfSolver != 0) {
133 auto solverOutput = readSolverOutput();
134 STORM_LOG_THROW(
135 solverOutput.size() == 1, storm::exceptions::UnexpectedException,
136 "expected a single line of output after smt2 command ( check-sat ). Got " + std::to_string(solverOutput.size()) + " lines of output instead.");
137 solverOutput[0].erase(std::remove_if(solverOutput[0].begin(), solverOutput[0].end(), ::isspace), solverOutput[0].end()); // remove spaces
138 if (solverOutput[0] == "sat")
139 return SmtSolver::CheckResult::Sat;
140 if (solverOutput[0] == "unsat")
141 return SmtSolver::CheckResult::Unsat;
142 if (solverOutput[0] == "unknown")
143 return SmtSolver::CheckResult::Unknown;
144 // if we reach this point, something unexpected happened. Lets return unknown and print some debug output
145 STORM_LOG_DEBUG("unexpected solver output: " << solverOutput[0] << ". Returning result 'unknown'");
146 return SmtSolver::CheckResult::Unknown;
147 } else {
148 STORM_LOG_WARN("No SMT-LIBv2 Solver Command specified, which means that no actual SMT solving is done... Assume that the result is \"unknown\"");
149 return SmtSolver::CheckResult::Unknown;
150 }

161void SmtlibSmtSolver::init() {
162 if (storm::settings::getModule<storm::settings::modules::Smt2SmtSolverSettings>().isSolverCommandSet()) {
163 signal(SIGPIPE, SIG_IGN);
164 this->needsRestart = false;
165
166 // for executing the solver we will need the path to the executable file as well as the arguments as a char* array
167 // where the first argument is the path to the executable file and the last is NULL
168 const std::string cmdString = storm::settings::getModule<storm::settings::modules::Smt2SmtSolverSettings>().getSolverCommand();
169 std::vector<std::string> solverCommandVec;
170 boost::split(solverCommandVec, cmdString, boost::is_any_of("\t "));
171 char** solverArgs = new char*[solverCommandVec.size() + 1];
172 std::string solverPath = solverCommandVec[0].substr(0, cmdString.rfind('/') + 1);
173 solverArgs[0] = const_cast<char*>(solverPath.c_str());
174 for (uint_fast64_t argumentIndex = 1; argumentIndex < solverCommandVec.size(); ++argumentIndex) {
175 solverArgs[argumentIndex] = const_cast<char*>(solverCommandVec[argumentIndex].c_str());
176 }
177 solverArgs[solverCommandVec.size()] = NULL;
178
179 // get the pipes started
180 int pipeIn[2];
181 int pipeOut[2];
182 const int READ = 0;
183 const int WRITE = 1;
184 STORM_LOG_THROW(pipe(pipeIn) == 0 && pipe(pipeOut) == 0, storm::exceptions::UnexpectedException, "Could not open pipe to new process");
185
186 // now start the child process, i.e., the solver
187 pid_t pid = fork();
188 STORM_LOG_THROW(pid >= 0, storm::exceptions::UnexpectedException, "Could not start new process for the smt solver");
189 if (pid == 0) {
190 // Child process
191 // duplicate the fd so that standard input and output will be send to our pipes
192 dup2(pipeIn[READ], STDIN_FILENO);
193 dup2(pipeOut[WRITE], STDOUT_FILENO);
194 dup2(pipeOut[WRITE], STDERR_FILENO);
195 // we can now close everything since our child process will use std in and out to address the pipes
196 close(pipeIn[READ]);
197 close(pipeIn[WRITE]);
198 close(pipeOut[READ]);
199 close(pipeOut[WRITE]);
200
201 execv(solverCommandVec[0].c_str(), solverArgs); //"-smt2 -in"
202 // if we reach this point, execl was not successful
203 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "Could not execute the solver correctly");
204 }
205 // Parent Process
206 toSolver = pipeIn[WRITE];
207 fromSolver = pipeOut[READ];
208 close(pipeOut[WRITE]);
209 close(pipeIn[READ]);
210 processIdOfSolver = pid;
211
212 } else {
213 STORM_LOG_WARN("No SMT-LIBv2 Solver Command specified, which means that no actual SMT solving can be done");
214 }
215
216 if (storm::settings::getModule<storm::settings::modules::Smt2SmtSolverSettings>().isExportSmtLibScriptSet()) {
217 STORM_LOG_DEBUG("The SMT-LIBv2 commands are exportet to the given file");
218 storm::io::openFile(storm::settings::getModule<storm::settings::modules::Smt2SmtSolverSettings>().getExportSmtLibScriptPath(), commandFile);
219 isCommandFileOpen = true;
220 // TODO also close file
221 }
222
223 // some initial commands
224 writeCommand("( set-option :print-success true )", true);
225 writeCommand("( set-logic QF_NRA )", true);
226 // writeCommand("( get-info :name )");
227}

254std::vector<std::string> SmtlibSmtSolver::readSolverOutput(bool waitForOutput) {
255 if (processIdOfSolver == 0) {
256 STORM_LOG_DEBUG("failed to read solver output as the solver is not running");
257 return std::vector<std::string>();
258 }
259 int bytesReadable;
260 if (waitForOutput) {
261 bytesReadable = 1; // just assume that there are bytes readable
262 } else if (ioctl(fromSolver, FIONREAD, &bytesReadable) < 0) { // actually obtain the readable bytes
263 STORM_LOG_ERROR("Could not check if the solver has output");
264 return std::vector<std::string>();
265 }
266 std::string solverOutput = "";
267 const ssize_t MAX_CHUNK_SIZE = 256;
268 char chunk[MAX_CHUNK_SIZE];
269 while (bytesReadable > 0) {
270 ssize_t chunkSize = read(fromSolver, chunk, MAX_CHUNK_SIZE);
271 STORM_LOG_THROW(chunkSize >= 0, storm::exceptions::UnexpectedException, "failed to read solver output");
272 solverOutput += std::string(chunk, chunkSize);
273 if (ioctl(fromSolver, FIONREAD, &bytesReadable) < 0) { // obtain the new amount of readable bytes
274 STORM_LOG_ERROR("Could not check if the solver has output");
275 return std::vector<std::string>();
276 }
277 if (bytesReadable == 0 && solverOutput.back() != '\n') {
278 STORM_LOG_DEBUG(
279 "Solver Output '"
280 << solverOutput
281 << "' did not end with newline symbol (\\n). Since we assume that this should be the case, we will wait for more output from the solver");
282 bytesReadable = 1; // we expect more output!
283 }
284 if (bytesReadable > 0) {
285 pid_t w = waitpid(processIdOfSolver, nullptr, WNOHANG);
286 if (w != 0) {
287 STORM_LOG_WARN_COND(w > 0, "Error when checking whether the solver is still running. Will assume that solver has terminated.");
288 STORM_LOG_WARN("The solver exited unexpectedly when reading output: " << solverOutput);
289 solverOutput += "terminated";
290 this->needsRestart = true;
291 this->processIdOfSolver = 0;
292 bytesReadable = 0;
293 }
294 }
295 }
296 checkForErrorMessage(solverOutput);
297 std::vector<std::string> solverOutputAsVector;
298 if (solverOutput.length() > 0) {
299 solverOutput = solverOutput.substr(0, solverOutput.length() - 1);
300 boost::split(solverOutputAsVector, solverOutput, boost::is_any_of("\n"));
301 // note: this is a little bit unsafe as \n can be contained within a solver response
302 }
303 return solverOutputAsVector;
304}

306void SmtlibSmtSolver::checkForErrorMessage(const std::string message) {
307 size_t errorOccurrance = message.find("error");
308 // do not throw an exception for timeout or memout errors
309 if (message.find("timeout") != std::string::npos) {
310 STORM_LOG_INFO("SMT solver answered: '" << message << "' and I am interpreting this as timeout ");
311 this->needsRestart = true;
312 this->processIdOfSolver = 0;
313 } else if (message.find("memory") != std::string::npos) {
314 STORM_LOG_INFO("SMT solver answered: '" << message << "' and I am interpreting this as out of memory ");
315 this->needsRestart = true;
316 this->processIdOfSolver = 0;
317 } else if (errorOccurrance != std::string::npos) {
318 this->needsRestart = true;
319 std::string errorMsg = "An error was detected while checking the solver output. ";
320 STORM_LOG_DEBUG("Detected an error message in the solver response:\n" + message);
321 size_t firstQuoteSign = message.find('\"', errorOccurrance);
322 if (firstQuoteSign != std::string::npos && message.find("\\\"", firstQuoteSign - 1) != firstQuoteSign - 1) {
323 size_t secondQuoteSign = message.find('\"', firstQuoteSign + 1);
324 while (secondQuoteSign != std::string::npos && message.find("\\\"", secondQuoteSign - 1) == secondQuoteSign - 1) {
325 secondQuoteSign = message.find('\"', secondQuoteSign + 1);
326 }
327 if (secondQuoteSign != std::string::npos) {
328 errorMsg += "The error message was: <<" + message.substr(errorOccurrance, secondQuoteSign - errorOccurrance + 1) + ">>.";
329 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, errorMsg);
330 return;
331 }
332 }
333 errorMsg += "The error message could not be parsed correctly. Snippet:\n" + message.substr(errorOccurrance, 200);
334 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, errorMsg);
335 }
