35 bool SparseMarkovAutomatonCslModelChecker<ModelType>::canHandleStatic(CheckTask<storm::logic::Formula, ValueType> const& checkTask,
36  bool* requiresSingleInitialState) {
37  auto singleObjectiveFragment = storm::logic::csrlstar()
38  .setRewardOperatorsAllowed(true)
39  .setReachabilityRewardFormulasAllowed(true)
40  .setTotalRewardFormulasAllowed(true)
41  .setTimeAllowed(true)
42  .setLongRunAverageProbabilitiesAllowed(true)
43  .setLongRunAverageRewardFormulasAllowed(true)
44  .setRewardAccumulationAllowed(true)
45  .setInstantaneousFormulasAllowed(false);
46  auto multiObjectiveFragment =
47  storm::logic::multiObjective().setTimeAllowed(true).setTimeBoundedUntilFormulasAllowed(true).setRewardAccumulationAllowed(true);
48  if (!storm::NumberTraits<ValueType>::SupportsExponential) {
49  singleObjectiveFragment.setBoundedUntilFormulasAllowed(false).setCumulativeRewardFormulasAllowed(false);
50  multiObjectiveFragment.setTimeBoundedUntilFormulasAllowed(false).setCumulativeRewardFormulasAllowed(false);
51  }
52  if (checkTask.getFormula().isInFragment(singleObjectiveFragment)) {
53  return true;
54  } else if (checkTask.isOnlyInitialStatesRelevantSet() && checkTask.getFormula().isInFragment(multiObjectiveFragment)) {
55  if (requiresSingleInitialState) {
56  *requiresSingleInitialState = true;
57  }
58  return true;
59  }
60  return false;
61 }
