21template<typename SparseModelType>
22std::shared_ptr<SparseModelType> SparseModelNondeterministicMemoryProduct<SparseModelType>::build() const {
23 // For Markov automata, we can not introduce nondeterminism at Markovian states.
24 // We could introduce new intermediate probabilistic states, however that is not supported yet.
25 STORM_LOG_THROW(!model.isOfType(storm::models::ModelType::MarkovAutomaton), storm::exceptions::NotSupportedException,
26 "Nondeterministic memory product not supported for Markov automata as this would introduce nondeterminism at Markovian states.");
27 // For simplicity we first build the 'full' product of model and memory (with model.numStates * memory.numStates states).
28 storm::storage::sparse::ModelComponents<ValueType> components;
29 components.transitionMatrix = buildTransitions();
30 components.stateLabeling = buildStateLabeling();
31
32 // Now delete unreachable states.
33 storm::storage::BitVector allStates(components.transitionMatrix.getRowGroupCount(), true);
34 auto reachableStates =
35 storm::utility::graph::getReachableStates(components.transitionMatrix, components.stateLabeling.getStates("init"), allStates, ~allStates);
36 components.transitionMatrix = components.transitionMatrix.getSubmatrix(true, reachableStates, reachableStates);
37 components.stateLabeling = components.stateLabeling.getSubLabeling(reachableStates);
38
39 // build the remaining components
40 for (auto const& rewModel : model.getRewardModels()) {
41 components.rewardModels.emplace(rewModel.first, buildRewardModel(rewModel.second, reachableStates));
42 }
43 /* Markov automata are not supported (This code would introduce nondeterminism at Markovian states...)
44 if (model.isOfType(storm::models::ModelType::MarkovAutomaton)) {
45 STORM_LOG_ASSERT((std::is_same<SparseModelType, storm::models::sparse::MarkovAutomaton<ValueType>>::value), "Model has unexpected type.");
46 auto ma = model.template as<storm::models::sparse::MarkovAutomaton<ValueType>>();
47 components.exitRates = buildExitRateVector(*ma, reachableStates);
48 components.markovianStates = buildMarkovianStateLabeling(*ma, reachableStates);
49 } */
50 return std::make_shared<SparseModelType>(std::move(components));
51}

53template<typename SparseModelType>
54storm::storage::SparseMatrix<typename SparseModelNondeterministicMemoryProduct<SparseModelType>::ValueType>
55SparseModelNondeterministicMemoryProduct<SparseModelType>::buildTransitions() const {
56 storm::storage::SparseMatrix<ValueType> const& origTransitions = model.getTransitionMatrix();
57 uint64_t numRows = 0;
58 uint64_t numEntries = 0;
59 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
60 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
61 numRows += origTransitions.getRowGroupSize(modelState) * memory.getNumberOfOutgoingTransitions(memState);
62 numEntries += origTransitions.getRowGroup(modelState).getNumberOfEntries() * memory.getNumberOfOutgoingTransitions(memState);
63 }
64 }
65 storm::storage::SparseMatrixBuilder<ValueType> builder(numRows, model.getNumberOfStates() * memory.getNumberOfStates(), numEntries, true, true,
66 model.getNumberOfStates() * memory.getNumberOfStates());
67
68 uint64_t row = 0;
69 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
70 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
71 builder.newRowGroup(row);
72 for (uint64_t origRow = origTransitions.getRowGroupIndices()[modelState]; origRow < origTransitions.getRowGroupIndices()[modelState + 1];
73 ++origRow) {
74 for (auto memStatePrime : memory.getTransitions(memState)) {
75 for (auto const& entry : origTransitions.getRow(origRow)) {
76 builder.addNextValue(row, getProductState(entry.getColumn(), memStatePrime), entry.getValue());
77 }
78 ++row;
79 }
80 }
81 }
82 }
83 return builder.build();
84}

86template<typename SparseModelType>
87storm::models::sparse::StateLabeling SparseModelNondeterministicMemoryProduct<SparseModelType>::buildStateLabeling() const {
88 storm::models::sparse::StateLabeling labeling(model.getNumberOfStates() * memory.getNumberOfStates());
89 for (auto const& labelName : model.getStateLabeling().getLabels()) {
90 storm::storage::BitVector newStates(model.getNumberOfStates() * memory.getNumberOfStates(), false);
91
92 // The init label is only assigned to Product states with the initial memory state
93 if (labelName == "init") {
94 for (auto modelState : model.getStateLabeling().getStates(labelName)) {
95 newStates.set(getProductState(modelState, memory.getInitialState()));
96 }
97 } else {
98 for (auto modelState : model.getStateLabeling().getStates(labelName)) {
99 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
100 newStates.set(getProductState(modelState, memState));
101 }
102 }
103 }
104 labeling.addLabel(labelName, std::move(newStates));
105 }
106 return labeling;
107}

109template<typename SparseModelType>
110storm::models::sparse::StandardRewardModel<typename SparseModelNondeterministicMemoryProduct<SparseModelType>::ValueType>
111SparseModelNondeterministicMemoryProduct<SparseModelType>::buildRewardModel(storm::models::sparse::StandardRewardModel<ValueType> const& rewardModel,
112 storm::storage::BitVector const& reachableStates) const {
113 std::optional<std::vector<ValueType>> stateRewards, actionRewards;
114 if (rewardModel.hasStateRewards()) {
115 stateRewards = std::vector<ValueType>();
116 stateRewards->reserve(model.getNumberOfStates() * memory.getNumberOfStates());
117 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
118 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
119 if (reachableStates.get(getProductState(modelState, memState))) {
120 stateRewards->push_back(rewardModel.getStateReward(modelState));
121 }
122 }
123 }
124 }
125 if (rewardModel.hasStateActionRewards()) {
126 actionRewards = std::vector<ValueType>();
127 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
128 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
129 if (reachableStates.get(getProductState(modelState, memState))) {
130 for (uint64_t origRow = model.getTransitionMatrix().getRowGroupIndices()[modelState];
131 origRow < model.getTransitionMatrix().getRowGroupIndices()[modelState + 1]; ++origRow) {
132 ValueType const& actionReward = rewardModel.getStateActionReward(origRow);
133 actionRewards->insert(actionRewards->end(), memory.getNumberOfOutgoingTransitions(memState), actionReward);
134 }
135 }
136 }
137 }
138 }
139 STORM_LOG_THROW(!rewardModel.hasTransitionRewards(), storm::exceptions::NotSupportedException, "Transition rewards are currently not supported.");
140 return storm::models::sparse::StandardRewardModel<ValueType>(std::move(stateRewards), std::move(actionRewards));
141}
