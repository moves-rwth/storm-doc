23template<typename SparseModelType>
24std::shared_ptr<SparseModelType> SparseModelNondeterministicMemoryProduct<SparseModelType>::build() const {
25 // For Markov automata, we can not introduce nondeterminism at Markovian states.
26 // We could introduce new intermediate probabilistic states, however that is not supported yet.
27 STORM_LOG_THROW(!model.isOfType(storm::models::ModelType::MarkovAutomaton), storm::exceptions::NotSupportedException,
28 "Nondeterministic memory product not supported for Markov automata as this would introduce nondeterminism at Markovian states.");
29 // For simplicity we first build the 'full' product of model and memory (with model.numStates * memory.numStates states).
30 storm::storage::sparse::ModelComponents<ValueType> components;
31 components.transitionMatrix = buildTransitions();
32 components.stateLabeling = buildStateLabeling();
33
34 // Now delete unreachable states.
35 storm::storage::BitVector allStates(components.transitionMatrix.getRowGroupCount(), true);
36 auto reachableStates =
37 storm::utility::graph::getReachableStates(components.transitionMatrix, components.stateLabeling.getStates("init"), allStates, ~allStates);
38 components.transitionMatrix = components.transitionMatrix.getSubmatrix(true, reachableStates, reachableStates);
39 components.stateLabeling = components.stateLabeling.getSubLabeling(reachableStates);
40
41 // build the remaining components
42 for (auto const& rewModel : model.getRewardModels()) {
43 components.rewardModels.emplace(rewModel.first, buildRewardModel(rewModel.second, reachableStates));
44 }
45 /* Markov automata are not supported (This code would introduce nondeterminism at Markovian states...)
46 if (model.isOfType(storm::models::ModelType::MarkovAutomaton)) {
47 STORM_LOG_ASSERT((std::is_same<SparseModelType, storm::models::sparse::MarkovAutomaton<ValueType>>::value), "Model has unexpected type.");
48 auto ma = model.template as<storm::models::sparse::MarkovAutomaton<ValueType>>();
49 components.exitRates = buildExitRateVector(*ma, reachableStates);
50 components.markovianStates = buildMarkovianStateLabeling(*ma, reachableStates);
51 } */
52 return std::make_shared<SparseModelType>(std::move(components));
53}

55template<typename SparseModelType>
56storm::storage::SparseMatrix<typename SparseModelNondeterministicMemoryProduct<SparseModelType>::ValueType>
57SparseModelNondeterministicMemoryProduct<SparseModelType>::buildTransitions() const {
58 storm::storage::SparseMatrix<ValueType> const& origTransitions = model.getTransitionMatrix();
59 uint64_t numRows = 0;
60 uint64_t numEntries = 0;
61 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
62 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
63 numRows += origTransitions.getRowGroupSize(modelState) * memory.getNumberOfOutgoingTransitions(memState);
64 numEntries += origTransitions.getRowGroup(modelState).getNumberOfEntries() * memory.getNumberOfOutgoingTransitions(memState);
65 }
66 }
67 storm::storage::SparseMatrixBuilder<ValueType> builder(numRows, model.getNumberOfStates() * memory.getNumberOfStates(), numEntries, true, true,
68 model.getNumberOfStates() * memory.getNumberOfStates());
69
70 uint64_t row = 0;
71 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
72 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
73 builder.newRowGroup(row);
74 for (uint64_t origRow = origTransitions.getRowGroupIndices()[modelState]; origRow < origTransitions.getRowGroupIndices()[modelState + 1];
75 ++origRow) {
76 for (auto memStatePrime : memory.getTransitions(memState)) {
77 for (auto const& entry : origTransitions.getRow(origRow)) {
78 builder.addNextValue(row, getProductState(entry.getColumn(), memStatePrime), entry.getValue());
79 }
80 ++row;
81 }
82 }
83 }
84 }
85 return builder.build();
86}

88template<typename SparseModelType>
89storm::models::sparse::StateLabeling SparseModelNondeterministicMemoryProduct<SparseModelType>::buildStateLabeling() const {
90 storm::models::sparse::StateLabeling labeling(model.getNumberOfStates() * memory.getNumberOfStates());
91 for (auto const& labelName : model.getStateLabeling().getLabels()) {
92 storm::storage::BitVector newStates(model.getNumberOfStates() * memory.getNumberOfStates(), false);
93
94 // The init label is only assigned to Product states with the initial memory state
95 if (labelName == "init") {
96 for (auto modelState : model.getStateLabeling().getStates(labelName)) {
97 newStates.set(getProductState(modelState, memory.getInitialState()));
98 }
99 } else {
100 for (auto modelState : model.getStateLabeling().getStates(labelName)) {
101 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
102 newStates.set(getProductState(modelState, memState));
103 }
104 }
105 }
106 labeling.addLabel(labelName, std::move(newStates));
107 }
108 return labeling;
109}

111template<typename SparseModelType>
112storm::models::sparse::StandardRewardModel<typename SparseModelNondeterministicMemoryProduct<SparseModelType>::ValueType>
113SparseModelNondeterministicMemoryProduct<SparseModelType>::buildRewardModel(storm::models::sparse::StandardRewardModel<ValueType> const& rewardModel,
114 storm::storage::BitVector const& reachableStates) const {
115 std::optional<std::vector<ValueType>> stateRewards, actionRewards;
116 if (rewardModel.hasStateRewards()) {
117 stateRewards = std::vector<ValueType>();
118 stateRewards->reserve(model.getNumberOfStates() * memory.getNumberOfStates());
119 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
120 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
121 if (reachableStates.get(getProductState(modelState, memState))) {
122 stateRewards->push_back(rewardModel.getStateReward(modelState));
123 }
124 }
125 }
126 }
127 if (rewardModel.hasStateActionRewards()) {
128 actionRewards = std::vector<ValueType>();
129 for (uint64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
130 for (uint64_t memState = 0; memState < memory.getNumberOfStates(); ++memState) {
131 if (reachableStates.get(getProductState(modelState, memState))) {
132 for (uint64_t origRow = model.getTransitionMatrix().getRowGroupIndices()[modelState];
133 origRow < model.getTransitionMatrix().getRowGroupIndices()[modelState + 1]; ++origRow) {
134 ValueType const& actionReward = rewardModel.getStateActionReward(origRow);
135 actionRewards->insert(actionRewards->end(), memory.getNumberOfOutgoingTransitions(memState), actionReward);
136 }
137 }
138 }
139 }
140 }
141 STORM_LOG_THROW(!rewardModel.hasTransitionRewards(), storm::exceptions::NotSupportedException, "Transition rewards are currently not supported.");
142 return storm::models::sparse::StandardRewardModel<ValueType>(std::move(stateRewards), std::move(actionRewards));
143}
