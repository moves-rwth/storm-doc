50template<typename ValueType>
51bool EliminationLinearEquationSolver<ValueType>::internalSolveEquations(Environment const& env, std::vector<ValueType>& x,
52 std::vector<ValueType> const& b) const {
53 // FIXME: This solver will not work for all input systems. More concretely, the current implementation will
54 // not work for systems that have a 1 on the diagonal. This is not a restriction of this technique in general
55 // but arbitrary matrices require pivoting, which is not currently implemented.
56
57 STORM_LOG_INFO("Solving linear equation system (" << x.size() << " rows) with elimination");
58
59 storm::storage::SparseMatrix<ValueType> const& transitionMatrix = localA ? *localA : *A;
60 storm::storage::SparseMatrix<ValueType> backwardTransitions = transitionMatrix.transpose();
61
62 // Initialize the solution to the right-hand side of the equation system.
63 x = b;
64
65 // Translate the matrix and its transpose into the flexible format.
66 storm::storage::FlexibleSparseMatrix<ValueType> flexibleMatrix(transitionMatrix, false);
67 storm::storage::FlexibleSparseMatrix<ValueType> flexibleBackwardTransitions(backwardTransitions, true);
68
69 boost::optional<std::vector<uint_fast64_t>> distanceBasedPriorities;
70
71 // TODO: get the order from the environment
72 storm::settings::modules::EliminationSettings::EliminationOrder order =
73 storm::settings::getModule<storm::settings::modules::EliminationSettings>().getEliminationOrder();
74
75 if (eliminationOrderNeedsDistances(order)) {
76 // Since we have no initial states at this point, we determine a representative of every BSCC regarding
77 // the backward transitions, because this means that every row is reachable from this set of rows, which
78 // we require to make sure we cover every row.
79 storm::storage::BitVector initialRows = storm::utility::graph::getBsccCover(backwardTransitions);
80 distanceBasedPriorities = getDistanceBasedPriorities(transitionMatrix, backwardTransitions, initialRows, b,
81 eliminationOrderNeedsForwardDistances(order), eliminationOrderNeedsReversedDistances(order));
82 }
83
84 std::shared_ptr<StatePriorityQueue> priorityQueue =
85 createStatePriorityQueue<ValueType>(distanceBasedPriorities, flexibleMatrix, flexibleBackwardTransitions, b, storm::storage::BitVector(x.size(), true));
86
87 // Create a state eliminator to perform the actual elimination.
88 PrioritizedStateEliminator<ValueType> eliminator(flexibleMatrix, flexibleBackwardTransitions, priorityQueue, x);
89
90 // Eliminate all states.
91 while (priorityQueue->hasNext()) {
92 auto state = priorityQueue->pop();
93 eliminator.eliminateState(state, false);
94 }
95
96 return true;
97}
