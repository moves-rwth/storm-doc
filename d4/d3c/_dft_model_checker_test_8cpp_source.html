99 
100  // Create relevant names

102  if (!config.useDC) {

106 
107  // Perform model checking

109  storm::dft::api::analyzeDFT<double>(*dft, properties, config.useSR, config.useMod, relevantEvents, false);
110  return boost::get<double>(results[0]);
111  }
