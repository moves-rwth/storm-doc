78 std::set<VariableType> const& consideredVariables) const {
79 std::set<VariableType> nonMonotoneVariables;
80 std::set<VariableType> monotoneVariables;

83 auto res = getMonotonicity(var);
84 if (res == Monotonicity::Not || res == Monotonicity::Unknown) {
85 nonMonotoneVariables.insert(var);

87 monotoneVariables.insert(var);
