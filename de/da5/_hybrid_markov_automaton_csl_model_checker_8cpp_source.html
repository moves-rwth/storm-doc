26bool HybridMarkovAutomatonCslModelChecker<ModelType>::canHandleStatic(CheckTask<storm::logic::Formula, ValueType> const& checkTask) {
27 auto singleObjectiveFragment = storm::logic::csl()
28 .setGloballyFormulasAllowed(false)
29 .setNextFormulasAllowed(false)
30 .setRewardOperatorsAllowed(true)
31 .setReachabilityRewardFormulasAllowed(true)
32 .setTotalRewardFormulasAllowed(false)
33 .setTimeAllowed(true)
34 .setLongRunAverageProbabilitiesAllowed(true)
35 .setLongRunAverageRewardFormulasAllowed(true)
36 .setRewardAccumulationAllowed(true)
37 .setInstantaneousFormulasAllowed(false)
38 .setCumulativeRewardFormulasAllowed(false);
39 if (!storm::NumberTraits<ValueType>::SupportsExponential) {
40 singleObjectiveFragment.setBoundedUntilFormulasAllowed(false);
41 }
42 return checkTask.getFormula().isInFragment(singleObjectiveFragment);
43}

97template<typename ModelType>
98std::unique_ptr<CheckResult> HybridMarkovAutomatonCslModelChecker<ModelType>::computeBoundedUntilProbabilities(
99 Environment const& env, CheckTask<storm::logic::BoundedUntilFormula, ValueType> const& checkTask) {
100 storm::logic::BoundedUntilFormula const& pathFormula = checkTask.getFormula();
101 std::unique_ptr<CheckResult> leftResultPointer = this->check(env, pathFormula.getLeftSubformula());
102 SymbolicQualitativeCheckResult<DdType> const& leftResult = leftResultPointer->asSymbolicQualitativeCheckResult<DdType>();
103
104 std::unique_ptr<CheckResult> rightResultPointer = this->check(env, pathFormula.getRightSubformula());
105 SymbolicQualitativeCheckResult<DdType> const& rightResult = rightResultPointer->asSymbolicQualitativeCheckResult<DdType>();
106
107 STORM_LOG_THROW(checkTask.isOptimizationDirectionSet(), storm::exceptions::InvalidPropertyException,
108 "Formula needs to specify whether minimal or maximal values are to be computed on nondeterministic model.");
109 STORM_LOG_THROW(pathFormula.getTimeBoundReference().isTimeBound(), storm::exceptions::NotImplementedException,
110 "Currently step-bounded and reward-bounded properties on MarkovAutomatons are not supported.");
111 double lowerBound = 0;
112 double upperBound = 0;
113 if (pathFormula.hasLowerBound()) {
114 lowerBound = pathFormula.getLowerBound<double>();
115 }
116 if (pathFormula.hasUpperBound()) {
117 upperBound = pathFormula.getNonStrictUpperBound<double>();
118 } else {
119 upperBound = storm::utility::infinity<double>();
120 }
121
122 return storm::modelchecker::helper::HybridMarkovAutomatonCslHelper::computeBoundedUntilProbabilities<DdType, ValueType>(
123 env, checkTask.getOptimizationDirection(), this->getModel(), this->getModel().getTransitionMatrix(), this->getModel().getMarkovianStates(),
124 this->getModel().getExitRateVector(), leftResult.getTruthValuesVector(), rightResult.getTruthValuesVector(), checkTask.isQualitativeSet(), lowerBound,
125 upperBound);
126}
