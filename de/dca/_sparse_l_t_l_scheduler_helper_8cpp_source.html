55 template<typename ValueType, bool Nondeterministic>
56 void SparseLTLSchedulerHelper<ValueType, Nondeterministic>::saveProductEcChoices(
57  automata::AcceptanceCondition const& acceptance, storm::storage::MaximalEndComponent const& mec,
58  std::vector<automata::AcceptanceCondition::acceptance_expr::ptr> const& conjunction, typename transformer::DAProduct<productModelType>::ptr product) {
59  // Save all states contained in this MEC and find out whether there is some overlap with another, already processed accepting mec
60  storm::storage::BitVector mecStates(product->getProductModel().getNumberOfStates(), false);
61  storm::storage::BitVector overlapStates;
62 
63  for (auto const& stateChoicePair : mec) {
64  if (_accInfSets[stateChoicePair.first].is_initialized()) {
65  overlapStates.resize(product->getProductModel().getNumberOfStates(), false);
66  overlapStates.set(stateChoicePair.first);
67  } else {
68  mecStates.set(stateChoicePair.first);
69  }
70  }
71 
72  if (!overlapStates.empty()) {
73  // If all the states in mec are overlapping, we are done already.
74  if (!mecStates.empty()) {
75  // Simply Reach the overlapStates almost surely
76  // set inf sets
77  for (auto mecState : mecStates) {
78  STORM_LOG_ASSERT(!_accInfSets[mecState].is_initialized(), "accepting inf sets were already defined for a MEC state which is not expected.");
79  _accInfSets[mecState] = std::set<uint_fast64_t>({DEFAULT_INFSET});
80  }
81 
82  // Define scheduler choices for the states in this MEC (that are not in any other MEC)
83  // Compute a scheduler that, with prob=1 reaches the overlap states
84  storm::storage::Scheduler<ValueType> mecScheduler(product->getProductModel().getNumberOfStates());
85  storm::utility::graph::computeSchedulerProb1E<ValueType>(mecStates, product->getProductModel().getTransitionMatrix(),
86  product->getProductModel().getBackwardTransitions(), mecStates, overlapStates,
87  mecScheduler);
88 
89  // Extract scheduler choices
90  for (auto pState : mecStates) {
91  this->_producedChoices.insert(
92  {std::make_tuple(product->getModelState(pState), product->getAutomatonState(pState), DEFAULT_INFSET), mecScheduler.getChoice(pState)});
93  }
94  }
95  } else {
96  // No overlap! Let's do actual work.
97 
98  // We know the MEC satisfied the conjunction: Save InfSets.
99  std::set<uint_fast64_t> infSetIds;
100  for (auto const& literal : conjunction) {
101  if (literal->isAtom()) {
102  const cpphoafparser::AtomAcceptance& atom = literal->getAtom();
103  if (atom.getType() == cpphoafparser::AtomAcceptance::TEMPORAL_INF) {
104  storm::storage::BitVector infSet;
105  if (atom.isNegated()) {
106  infSet = ~acceptance.getAcceptanceSet(atom.getAcceptanceSet());
107  } else {
108  infSet = acceptance.getAcceptanceSet(atom.getAcceptanceSet());
109  }
110  // Save new InfSet
111  infSetIds.insert(_infSets.getOrCreateIndex(std::move(infSet)));
112  }
113  // A TEMPORAL_FIN atom can be ignored at this point since the mec is already known to only contain "allowed" states
114  }
115  // TRUE literals can be ignored since those will be satisfied anyway
116  // FALSE literals are not possible here since the MEC is known to be accepting.
117  }
118  if (infSetIds.empty()) {
119  // There might not be any infSet at this point (e.g. because all literals are TEMPORAL_FIN atoms). We need at least one infset, though
120  infSetIds.insert(_infSets.getOrCreateIndex(storm::storage::BitVector(product->getProductModel().getNumberOfStates(), true)));
121  }
122 
123  // Save the InfSets into the _accInfSets for states in this MEC
124  for (auto const& mecState : mecStates) {
125  STORM_LOG_ASSERT(!_accInfSets[mecState].is_initialized(), "accepting inf sets were already defined for a MEC state which is not expected.");
126  _accInfSets[mecState].emplace(infSetIds);
127  }
128 
129  // Define scheduler choices for the states in this MEC (that are not in any other MEC)
130  // The resulting scheduler will visit each InfSet inf often
131  for (uint_fast64_t id : infSetIds) {
132  // Scheduler that satisfies the MEC acceptance condition
133  storm::storage::Scheduler<ValueType> mecScheduler(product->getProductModel().getNumberOfStates());
134 
135  storm::storage::BitVector infStatesWithinMec = _infSets.get(id) & mecStates;
136  // States not in InfSet: Compute a scheduler that, with prob=1, reaches the infSet via mecStates
137  storm::utility::graph::computeSchedulerProb1E<ValueType>(mecStates, product->getProductModel().getTransitionMatrix(),
138  product->getProductModel().getBackwardTransitions(), mecStates, infStatesWithinMec,
139  mecScheduler);
140 
141  // States that already reached the InfSet
142  for (auto pState : infStatesWithinMec) {
143  // Prob1E sets an arbitrary choice for the psi states, but we want to stay in this accepting MEC.
144  mecScheduler.setChoice(*mec.getChoicesForState(pState).begin() - product->getProductModel().getTransitionMatrix().getRowGroupIndices()[pState],
145  pState);
146  }
147 
148  // Extract scheduler choices
149  for (auto pState : mecStates) {
150  // We want to reach the InfSet, save choice: <s, q, InfSetID> ---> choice
151  this->_producedChoices.insert(
152  {std::make_tuple(product->getModelState(pState), product->getAutomatonState(pState), id), mecScheduler.getChoice(pState)});
153  }
154  }
155  }

158 template<typename ValueType, bool Nondeterministic>
159 void SparseLTLSchedulerHelper<ValueType, Nondeterministic>::prepareScheduler(uint_fast64_t numDaStates, storm::storage::BitVector const& acceptingProductStates,
160  std::unique_ptr<storm::storage::Scheduler<ValueType>> reachScheduler,
161  transformer::DAProductBuilder const& productBuilder,
162  typename transformer::DAProduct<productModelType>::ptr product,
163  storm::storage::BitVector const& modelStatesOfInterest,
164  storm::storage::SparseMatrix<ValueType> const& transitionMatrix) {
165  STORM_LOG_ASSERT(_infSets.size() > 0, "There is no inf set. Were the accepting ECs processed before?");
166 
167  // Compute size of the resulting memory structure: A state <q, infSet> is encoded as (q* (|infSets|))+ |infSet|
168  uint64_t numMemoryStates = (numDaStates) * (_infSets.size());
169  _dontCareStates = std::vector<storm::storage::BitVector>(numMemoryStates, storm::storage::BitVector(transitionMatrix.getRowGroupCount(), false));
170 
171  // Set choices for states or consider them "dontCare"
172  for (storm::storage::sparse::state_type automatonState = 0; automatonState < numDaStates; ++automatonState) {
173  for (storm::storage::sparse::state_type modelState = 0; modelState < transitionMatrix.getRowGroupCount(); ++modelState) {
174  if (!product->isValidProductState(modelState, automatonState)) {
175  // If the state <s,q> does not occur in the product model, all the infSet combinations are irrelevant for the scheduler.
176  for (uint_fast64_t infSet = 0; infSet < _infSets.size(); ++infSet) {
177  _dontCareStates[getMemoryState(automatonState, infSet)].set(modelState, true);
178  }
179  } else {
180  auto pState = product->getProductStateIndex(modelState, automatonState);
181  if (acceptingProductStates.get(pState)) {
182  // For states in accepting ECs set the MEC-scheduler. Missing combinations are "dontCare", they are not reachable using the scheduler
183  // choices.
184  for (uint_fast64_t infSet = 0; infSet < _infSets.size(); ++infSet) {
185  if (_producedChoices.count(std::make_tuple(product->getModelState(pState), product->getAutomatonState(pState), infSet)) == 0) {
186  _dontCareStates[getMemoryState(product->getAutomatonState(pState), infSet)].set(product->getModelState(pState), true);
187  }
188  }
189 
190  } else {
191  // Extract the choices of the REACH-scheduler (choices to reach an acc. MEC) for the MDP-DA product: <s,q> -> choice. The memory structure
192  // corresponds to the "0th" copy of the DA (DEFAULT_INFSET).
193  this->_accInfSets[pState] = std::set<uint_fast64_t>({DEFAULT_INFSET});
194  if (reachScheduler->isDontCare(pState)) {
195  // Mark the maybe States of the untilProbability scheduler as "dontCare"
196  _dontCareStates[getMemoryState(product->getAutomatonState(pState), DEFAULT_INFSET)].set(product->getModelState(pState), true);
197  } else {
198  // Set choice For non-accepting states that are not in any accepting EC
199  this->_producedChoices.insert({std::make_tuple(product->getModelState(pState), product->getAutomatonState(pState), DEFAULT_INFSET),
200  reachScheduler->getChoice(pState)});
201  };
202  // All other InfSet combinations are unreachable (dontCare)
203  static_assert(DEFAULT_INFSET == 0, "This code assumes that the default infset is 0");
204  for (uint_fast64_t infSet = 1; infSet < _infSets.size(); ++infSet) {
205  _dontCareStates[getMemoryState(product->getAutomatonState(pState), infSet)].set(product->getModelState(pState), true);
206  }
207  }
208  }
209  }
210  }
211 
212  // Prepare the memory structure. For that, we need: transitions, initialMemoryStates (and memoryStateLabeling)
213 
214  // The next move function of the memory, will be build based on the transitions of the DA and jumps between InfSets.
215  _memoryTransitions = std::vector<std::vector<storm::storage::BitVector>>(
216  numMemoryStates, std::vector<storm::storage::BitVector>(numMemoryStates, storm::storage::BitVector(transitionMatrix.getRowGroupCount(), false)));
217  for (storm::storage::sparse::state_type automatonFrom = 0; automatonFrom < numDaStates; ++automatonFrom) {
218  for (storm::storage::sparse::state_type modelState = 0; modelState < transitionMatrix.getRowGroupCount(); ++modelState) {
219  uint_fast64_t automatonTo = productBuilder.getSuccessor(automatonFrom, modelState);
220 
221  if (product->isValidProductState(modelState, automatonTo)) {
222  uint_fast64_t daProductState = product->getProductStateIndex(modelState, automatonTo);
223  STORM_LOG_ASSERT(_accInfSets[daProductState] != boost::none,
224  "The list of InfSets for the product state <" << modelState << ", " << automatonTo << "> is undefined.");
225  std::set<uint_fast64_t> const& daProductStateInfSets = _accInfSets[daProductState].get();
226  // Add the modelState to one outgoing transition of all states of the form <automatonFrom, InfSet>
227  // For non-accepting states that are not in any accepting EC and for overlapping accepting ECs we always use the '0th' copy of the DA.
228  // For the states in accepting ECs that do not overlap we cycle through copies 0,1,2, ... k of the DA (skipping copies whose inf-sets are not
229  // needed for the accepting EC).
230  for (uint_fast64_t currentInfSet = 0; currentInfSet < _infSets.size(); ++currentInfSet) {
231  uint_fast64_t newInfSet;
232  // Check if we need to switch the inf set (i.e. the DA copy)
233  if (daProductStateInfSets.count(currentInfSet) == 0) {
234  // This infSet is not relevant for the daProductState. We need to switch to a copy representing a relevant infset.
235  newInfSet = *daProductStateInfSets.begin();
236  } else if (daProductStateInfSets.size() > 1 && (_infSets.get(currentInfSet).get(daProductState))) {
237  // We have reached a state from the current infSet and thus need to move on to the next infSet in the list.
238  // Note that if the list contains just a single item, the switch would have no effect.
239  // In particular, this is the case for states that are not in an accepting MEC as those only have DEFAULT_INFSET in their list
240  auto nextInfSetIt = daProductStateInfSets.find(currentInfSet);
241  STORM_LOG_ASSERT(nextInfSetIt != daProductStateInfSets.end(), "The list of InfSets for the product state <"
242  << modelState << ", " << automatonTo
243  << "> does not contain the infSet " << currentInfSet);
244  nextInfSetIt++;
245  if (nextInfSetIt == daProductStateInfSets.end()) {
246  // Start again.
247  nextInfSetIt = daProductStateInfSets.begin();
248  }
249  newInfSet = *nextInfSetIt;
250  } else {
251  // In all other cases we can keep the current inf set (i.e. the DA copy)
252  newInfSet = currentInfSet;
253  }
254  _memoryTransitions[getMemoryState(automatonFrom, currentInfSet)][getMemoryState(automatonTo, newInfSet)].set(modelState);
255  }
256  }
257  }
258  }
259  // Finished creation of transitions.
260 
261  // Find initial memory states
262  this->_memoryInitialStates = std::vector<uint_fast64_t>(transitionMatrix.getRowGroupCount());
263  // Save for each relevant model state its initial memory state (get the s-successor q of q0)
264  for (storm::storage::sparse::state_type modelState : modelStatesOfInterest) {
265  storm::storage::sparse::state_type automatonState = productBuilder.getInitialState(modelState);
266  STORM_LOG_ASSERT(product->isValidProductState(modelState, automatonState),
267  "The memory successor state for the model state " << modelState << "does not exist in the DA-Model Product.");
268  STORM_LOG_ASSERT(_accInfSets[product->getProductStateIndex(modelState, automatonState)] != boost::none,
269  "The list of InfSets for the product state <" << modelState << ", " << automatonState << "> is undefined.");
270  // Start in the first InfSet of <s, q>
271  auto infSet = _accInfSets[product->getProductStateIndex(modelState, automatonState)].get().begin();
272  _memoryInitialStates[modelState] = getMemoryState(automatonState, *infSet);
273  }
274 }

276 template<typename ValueType, bool Nondeterministic>
277 storm::storage::Scheduler<ValueType> SparseLTLSchedulerHelper<ValueType, Nondeterministic>::SparseLTLSchedulerHelper::extractScheduler(
278  storm::models::sparse::Model<ValueType> const& model, bool onlyInitialStatesRelevant) {
279  if (_randomScheduler) {
280  storm::storage::Scheduler<ValueType> scheduler(model.getNumberOfStates());
281  for (storm::storage::sparse::state_type state = 0; state < model.getNumberOfStates(); ++state) {
282  scheduler.setChoice(0, state);
283  }
284  return scheduler;
285  }
286 
287  // Otherwise, we compute a scheduler with memory.
288 
289  // Create a memory structure for the MDP scheduler with memory. If hasRelevantStates is set, we only consider initial model states relevant.
290  auto memoryBuilder = storm::storage::MemoryStructureBuilder<ValueType>(this->_memoryTransitions.size(), model, onlyInitialStatesRelevant);
291 
292  // Build the transitions between the memory states: startState to goalState using modelStates (transitionVector).
293  for (storm::storage::sparse::state_type startState = 0; startState < this->_memoryTransitions.size(); ++startState) {
294  for (storm::storage::sparse::state_type goalState = 0; goalState < this->_memoryTransitions.size(); ++goalState) {
295  // Bitvector that represents modelStates the model states that trigger this transition.
296  memoryBuilder.setTransition(startState, goalState, this->_memoryTransitions[startState][goalState]);
297  }
298  }
299 
300  // InitialMemoryStates: Assign an initial memory state model states
301  if (onlyInitialStatesRelevant) {
302  // Only consider initial model states
303  for (uint_fast64_t modelState : model.getInitialStates()) {
304  memoryBuilder.setInitialMemoryState(modelState, this->_memoryInitialStates[modelState]);
305  }
306  } else {
307  // All model states are relevant
308  for (uint_fast64_t modelState = 0; modelState < model.getNumberOfStates(); ++modelState) {
309  memoryBuilder.setInitialMemoryState(modelState, this->_memoryInitialStates[modelState]);
310  }
311  }
312 
313  // Build the memoryStructure.
314  storm::storage::MemoryStructure memoryStructure = memoryBuilder.build();
315 
316  // Create a scheduler (with memory) for the model from the REACH and MEC scheduler of the MDP-DA-product model.
317  storm::storage::Scheduler<ValueType> scheduler(model.getNumberOfStates(), memoryStructure);
318 
319  // Use choices in the product model to create a choice based on model state and memory state
320  for (const auto& choice : this->_producedChoices) {
321  // <s, q, InfSet> -> choice
322  storm::storage::sparse::state_type modelState = std::get<0>(choice.first);
323  storm::storage::sparse::state_type daState = std::get<1>(choice.first);
324  uint_fast64_t infSet = std::get<2>(choice.first);
325  STORM_LOG_ASSERT(!this->_dontCareStates[getMemoryState(daState, infSet)].get(modelState), "Tried to set choice for dontCare state.");
326  scheduler.setChoice(choice.second, modelState, getMemoryState(daState, infSet));
327  }
328 
329  // Set "dontCare" states
330  for (uint_fast64_t memoryState = 0; memoryState < this->_dontCareStates.size(); ++memoryState) {
331  for (auto state : this->_dontCareStates[memoryState]) {
332  scheduler.setDontCare(state, memoryState);
333  }
334  }
335 
336  // Sanity check for created scheduler.
337  STORM_LOG_ASSERT(scheduler.isDeterministicScheduler(), "Expected a deterministic scheduler");
338  STORM_LOG_ASSERT(!scheduler.isPartialScheduler(), "Expected a fully defined scheduler");
339 
340  return scheduler;
341 }
