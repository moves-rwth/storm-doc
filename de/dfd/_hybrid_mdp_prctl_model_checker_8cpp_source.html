31template<typename ModelType>
32bool HybridMdpPrctlModelChecker<ModelType>::canHandleStatic(CheckTask<storm::logic::Formula, ValueType> const& checkTask, bool* requiresSingleInitialState) {
33 storm::logic::Formula const& formula = checkTask.getFormula();
34 if (formula.isInFragment(storm::logic::prctl()
35 .setLongRunAverageRewardFormulasAllowed(false)
36 .setTimeOperatorsAllowed(true)
37 .setReachbilityTimeFormulasAllowed(true)
38 .setRewardAccumulationAllowed(true))) {
39 return true;
40 } else if (checkTask.isOnlyInitialStatesRelevantSet() && formula.isInFragment(storm::logic::multiObjective().setCumulativeRewardFormulasAllowed(true))) {
41 if (requiresSingleInitialState) {
42 *requiresSingleInitialState = true;
43 }
44 return true;
45 }
46 return false;
47}

170template<typename ModelType>
171std::unique_ptr<CheckResult> HybridMdpPrctlModelChecker<ModelType>::checkMultiObjectiveFormula(
172 Environment const& env, CheckTask<storm::logic::MultiObjectiveFormula, ValueType> const& checkTask) {
173 auto sparseModel = storm::transformer::SymbolicMdpToSparseMdpTransformer<DdType, ValueType>::translate(this->getModel());
174 std::unique_ptr<CheckResult> explicitResult =
175 multiobjective::performMultiObjectiveModelChecking(env, *sparseModel, checkTask.getFormula(), checkTask.isProduceSchedulersSet());
176
177 // Convert the explicit result
178 if (explicitResult->isExplicitQualitativeCheckResult()) {
179 if (explicitResult->asExplicitQualitativeCheckResult()[*sparseModel->getInitialStates().begin()]) {
180 return std::unique_ptr<CheckResult>(new storm::modelchecker::SymbolicQualitativeCheckResult<DdType>(
181 this->getModel().getReachableStates(), this->getModel().getInitialStates(), this->getModel().getManager().getBddOne()));
182 } else {
183 return std::unique_ptr<CheckResult>(new storm::modelchecker::SymbolicQualitativeCheckResult<DdType>(
184 this->getModel().getReachableStates(), this->getModel().getInitialStates(), this->getModel().getManager().getBddZero()));
185 }
186 } else if (explicitResult->isExplicitQuantitativeCheckResult()) {
187 ValueType const& res = explicitResult->template asExplicitQuantitativeCheckResult<ValueType>()[*sparseModel->getInitialStates().begin()];
188 return std::unique_ptr<CheckResult>(new storm::modelchecker::SymbolicQuantitativeCheckResult<DdType, ValueType>(
189 this->getModel().getReachableStates(), this->getModel().getInitialStates(), this->getModel().getManager().template getConstant<ValueType>(res)));
190 } else if (explicitResult->isExplicitParetoCurveCheckResult()) {
191 ExplicitParetoCurveCheckResult<ValueType> const& paretoResult = explicitResult->template asExplicitParetoCurveCheckResult<ValueType>();
192 return std::unique_ptr<CheckResult>(new storm::modelchecker::SymbolicParetoCurveCheckResult<DdType, ValueType>(
193 this->getModel().getInitialStates(), paretoResult.getPoints(), paretoResult.getUnderApproximation(), paretoResult.getOverApproximation()));
194 } else {
195 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException, "The obtained checkresult has an unexpected type.");
196 return nullptr;
197 }
