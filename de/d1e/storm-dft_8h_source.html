86  bool exponentialDistributions) {
87  std::shared_ptr<storm::dft::storage::DFT<ValueType>> transformedDft = std::make_shared<storm::dft::storage::DFT<ValueType>>(dft);
88  if (exponentialDistributions && !storm::dft::transformations::DftTransformer<ValueType>::hasOnlyExponentialDistributions(*transformedDft)) {
89  transformedDft = storm::dft::transformations::DftTransformer<ValueType>::transformExponentialDistributions(*transformedDft);
90  }
91  if (uniqueBE && !storm::dft::transformations::DftTransformer<ValueType>::hasUniqueFailedBE(*transformedDft)) {
92  transformedDft = storm::dft::transformations::DftTransformer<ValueType>::transformUniqueFailedBE(*transformedDft);
93  }

95  transformedDft = storm::dft::transformations::DftTransformer<ValueType>::transformBinaryDependencies(*transformedDft);
96  }

118 template<typename ValueType>
119 bool computeDependencyConflicts(storm::dft::storage::DFT<ValueType>& dft, bool useSMT, double solverTimeout) {
120  // Initialize which DFT elements have dynamic behavior
121  dft.setDynamicBehaviorInfo();
122 
123  std::vector<std::pair<uint64_t, uint64_t>> fdepConflicts =
124  storm::dft::utility::FDEPConflictFinder<ValueType>::getDependencyConflicts(dft, useSMT, solverTimeout);
125 
126  for (auto const& pair : fdepConflicts) {
127  STORM_LOG_DEBUG("Conflict between " << dft.getElement(pair.first)->name() << " and " << dft.getElement(pair.second)->name());
128  }
129 
130  // Set the conflict map of the dft
131  std::set<size_t> conflict_set;
132  for (auto const& conflict : fdepConflicts) {
133  conflict_set.insert(size_t(conflict.first));
134  conflict_set.insert(size_t(conflict.second));
135  }
136  for (size_t depId : dft.getDependencies()) {
137  if (!conflict_set.count(depId)) {
138  dft.setDependencyNotInConflict(depId);
139  }
140  }
141  return !fdepConflicts.empty();
142 }
