176  result = quantitativeResult->template asExplicitQuantitativeCheckResult<ConstantType>().compareAgainstBound(

178  this->currentCheckTask->getFormula().asOperatorFormula().template getThresholdAs<ConstantType>());
179  hint.setResultHint(std::move(quantitativeResult->template asExplicitQuantitativeCheckResult<ConstantType>().getValueVector()));
