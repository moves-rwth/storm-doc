12template<typename ValueType>
13bool GlobalPOMDPSelfLoopEliminator<ValueType>::preservesFormula(storm::logic::Formula const& formula) const {
14 if (formula.isOperatorFormula() && formula.asOperatorFormula().hasOptimalityType()) {
15 bool maxProb = formula.isProbabilityOperatorFormula() && storm::solver::maximize(formula.asOperatorFormula().getOptimalityType());
16 bool minRew = formula.isRewardOperatorFormula() && storm::solver::minimize(formula.asOperatorFormula().getOptimalityType());
17
18 auto const& subformula = formula.asOperatorFormula().getSubformula();
19 if (subformula.isEventuallyFormula()) {
20 if (subformula.asEventuallyFormula().getSubformula().isInFragment(storm::logic::propositional())) {
21 return maxProb || minRew;
22 }
23 } else if (subformula.isUntilFormula()) {
24 if (subformula.asUntilFormula().getLeftSubformula().isInFragment(storm::logic::propositional()) &&
25 subformula.asUntilFormula().getRightSubformula().isInFragment(storm::logic::propositional())) {
26 return maxProb;
27 }
28 }
29 }
30 return false;
31}

34std::shared_ptr<storm::models::sparse::Pomdp<ValueType>> GlobalPOMDPSelfLoopEliminator<ValueType>::transform() const {
35 uint64_t nrStates = pomdp.getNumberOfStates();
36
37 // For each observation, get the choice indices that have a selfloop at every state with that observation
38 std::vector<storm::storage::BitVector> observationSelfLoopMasks(pomdp.getNrObservations());
39 for (uint64_t state = 0; state < nrStates; ++state) {
40 uint32_t observation = pomdp.getObservation(state);
41 assert(pomdp.getNumberOfChoices(state) != 0);
42
43 STORM_LOG_ASSERT(observation < observationSelfLoopMasks.size(),
44 "Observation index (" << observation << ") should be less than number of observations (" << observationSelfLoopMasks.size() << "). ");
45 auto& observationSelfLoopMask = observationSelfLoopMasks[observation];
46 // If we see this observation for the first time, add the corresponding amount of bits to the observation mask
47
48 if (observationSelfLoopMask.size() == 0) {
49 observationSelfLoopMask.resize(pomdp.getNumberOfChoices(state), true);
50 }
51 STORM_LOG_ASSERT(observationSelfLoopMask.size() == pomdp.getNumberOfChoices(state),
52 "State " << state << " has " << pomdp.getNumberOfChoices(state)
53 << " actions, different from other state(s) with same observation, which have " << observationSelfLoopMask.size()
54 << " actions.");
55
56 // Iterate over all set bits in observationSelfLoopMask
57 // We use `getNextSetIndex` (instead of the more concise `for each`-syntax) to safely unset bits in the BitVector while iterating over it.
58 for (uint64_t localChoiceIndex = observationSelfLoopMask.getNextSetIndex(0); localChoiceIndex < observationSelfLoopMask.size();
59 localChoiceIndex = observationSelfLoopMask.getNextSetIndex(localChoiceIndex + 1)) {
60 // We just look at the first entry.
61 auto row = pomdp.getTransitionMatrix().getRow(state, localChoiceIndex);
62 STORM_LOG_ASSERT(row.getNumberOfEntries() > 0, "Unexpected empty row.");
63 if (row.getNumberOfEntries() > 1 || row.begin()->getColumn() != state) {
64 observationSelfLoopMask.set(localChoiceIndex, false);
65 }
66 }
67 }
68
69 // For each observation, make sure that we keep at least one choice
70 for (auto& obsMask : observationSelfLoopMasks) {
71 if (obsMask.size() > 0 && obsMask.full()) {
72 obsMask.set(0, false);
73 }
74 }
75
76 // Finally create a filter that is 1 for every choice we can keep.
77 storm::storage::BitVector filter(pomdp.getNumberOfChoices(), true);
78 uint64_t offset = 0;
79 for (uint64_t state = 0; state < nrStates; ++state) {
80 storm::storage::BitVector& observationSelfLoopMask = observationSelfLoopMasks[pomdp.getObservation(state)];
81 assert(!observationSelfLoopMask.full());
82 for (auto const localChoiceIndex : observationSelfLoopMask) {
83 filter.set(offset + localChoiceIndex, false);
84 }
85 offset += pomdp.getNumberOfChoices(state);
86 }
87
88 // We do not consider rewards right now, since this reduction only preserves maximizing probabilities and minimizing expected rewards, anyway.
89 // In both cases, the reward at a selfloop has no effect (assuming not reaching goal means infinite reward)
90
91 STORM_LOG_INFO("Selfloop reduction eliminates " << (filter.size() - filter.getNumberOfSetBits()) << " choices.");
92
93 ChoiceSelector<ValueType> cs(pomdp);
94 auto res = cs.transform(filter)->template as<storm::models::sparse::Pomdp<ValueType>>();
95
96 // The transformation preserves canonicity.
97 res->setIsCanonic(pomdp.isCanonic());
98 return res;
99}
