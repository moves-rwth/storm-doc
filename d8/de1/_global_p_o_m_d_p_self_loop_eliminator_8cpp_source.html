14template<typename ValueType>
15bool GlobalPOMDPSelfLoopEliminator<ValueType>::preservesFormula(storm::logic::Formula const& formula) const {
16 if (formula.isOperatorFormula() && formula.asOperatorFormula().hasOptimalityType()) {
17 bool maxProb = formula.isProbabilityOperatorFormula() && storm::solver::maximize(formula.asOperatorFormula().getOptimalityType());
18 bool minRew = formula.isRewardOperatorFormula() && storm::solver::minimize(formula.asOperatorFormula().getOptimalityType());
19
20 auto const& subformula = formula.asOperatorFormula().getSubformula();
21 if (subformula.isEventuallyFormula()) {
22 if (subformula.asEventuallyFormula().getSubformula().isInFragment(storm::logic::propositional())) {
23 return maxProb || minRew;
24 }
25 } else if (subformula.isUntilFormula()) {
26 if (subformula.asUntilFormula().getLeftSubformula().isInFragment(storm::logic::propositional()) &&
27 subformula.asUntilFormula().getRightSubformula().isInFragment(storm::logic::propositional())) {
28 return maxProb;
29 }
30 }
31 }
32 return false;
33}

36std::shared_ptr<storm::models::sparse::Pomdp<ValueType>> GlobalPOMDPSelfLoopEliminator<ValueType>::transform() const {
37 uint64_t nrStates = pomdp.getNumberOfStates();
38
39 // For each observation, get the choice indices that have a selfloop at every state with that observation
40 std::vector<storm::storage::BitVector> observationSelfLoopMasks(pomdp.getNrObservations());
41 for (uint64_t state = 0; state < nrStates; ++state) {
42 uint32_t observation = pomdp.getObservation(state);
43 assert(pomdp.getNumberOfChoices(state) != 0);
44
45 STORM_LOG_ASSERT(observation < observationSelfLoopMasks.size(),
46 "Observation index (" << observation << ") should be less than number of observations (" << observationSelfLoopMasks.size() << "). ");
47 auto& observationSelfLoopMask = observationSelfLoopMasks[observation];
48 // If we see this observation for the first time, add the corresponding amount of bits to the observation mask
49
50 if (observationSelfLoopMask.size() == 0) {
51 observationSelfLoopMask.resize(pomdp.getNumberOfChoices(state), true);
52 }
53 STORM_LOG_ASSERT(observationSelfLoopMask.size() == pomdp.getNumberOfChoices(state),
54 "State " << state << " has " << pomdp.getNumberOfChoices(state)
55 << " actions, different from other state(s) with same observation, which have " << observationSelfLoopMask.size()
56 << " actions.");
57
58 // Iterate over all set bits in observationSelfLoopMask
59 // We use `getNextSetIndex` (instead of the more concise `for each`-syntax) to safely unset bits in the BitVector while iterating over it.
60 for (uint64_t localChoiceIndex = observationSelfLoopMask.getNextSetIndex(0); localChoiceIndex < observationSelfLoopMask.size();
61 localChoiceIndex = observationSelfLoopMask.getNextSetIndex(localChoiceIndex + 1)) {
62 // We just look at the first entry.
63 auto row = pomdp.getTransitionMatrix().getRow(state, localChoiceIndex);
64 STORM_LOG_ASSERT(row.getNumberOfEntries() > 0, "Unexpected empty row.");
65 if (row.getNumberOfEntries() > 1 || row.begin()->getColumn() != state) {
66 observationSelfLoopMask.set(localChoiceIndex, false);
67 }
68 }
69 }
70
71 // For each observation, make sure that we keep at least one choice
72 for (auto& obsMask : observationSelfLoopMasks) {
73 if (obsMask.size() > 0 && obsMask.full()) {
74 obsMask.set(0, false);
75 }
76 }
77
78 // Finally create a filter that is 1 for every choice we can keep.
79 storm::storage::BitVector filter(pomdp.getNumberOfChoices(), true);
80 uint64_t offset = 0;
81 for (uint64_t state = 0; state < nrStates; ++state) {
82 storm::storage::BitVector& observationSelfLoopMask = observationSelfLoopMasks[pomdp.getObservation(state)];
83 assert(!observationSelfLoopMask.full());
84 for (auto const localChoiceIndex : observationSelfLoopMask) {
85 filter.set(offset + localChoiceIndex, false);
86 }
87 offset += pomdp.getNumberOfChoices(state);
88 }
89
90 // We do not consider rewards right now, since this reduction only preserves maximizing probabilities and minimizing expected rewards, anyway.
91 // In both cases, the reward at a selfloop has no effect (assuming not reaching goal means infinite reward)
92
93 STORM_LOG_INFO("Selfloop reduction eliminates " << (filter.size() - filter.getNumberOfSetBits()) << " choices.");
94
95 ChoiceSelector<ValueType> cs(pomdp);
96 auto res = cs.transform(filter)->template as<storm::models::sparse::Pomdp<ValueType>>();
97
98 // The transformation preserves canonicity.
99 res->setIsCanonic(pomdp.isCanonic());
100 return res;
101}
