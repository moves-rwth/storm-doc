21bool parseOptions(const int argc, const char* argv[]) {
22 try {
23 storm::settings::mutableManager().setFromCommandLine(argc, argv);
24 } catch (storm::exceptions::OptionParserException&) {
25 STORM_LOG_ERROR("Unable to parse command line options. Type '" + std::string(argv[0]) + " --help' or '" + std::string(argv[0]) +
26 " --help all' for help.");
27 return false;
28 }
29
30 storm::settings::modules::GeneralSettings const& general = storm::settings::getModule<storm::settings::modules::GeneralSettings>();
31 bool terminate = false;
32 if (general.isHelpSet()) {
33 storm::settings::manager().printHelp(general.getHelpFilterExpression());
34 terminate = true;
35 }
36
37 if (general.isVersionSet()) {
38 storm::cli::printVersion();
39 terminate = true;
40 }
41 if (terminate) {
42 exit(0); // Terminate after help and version output with success.
43 // TODO: Issue 674 discusses that this may not be ideal.
44 }
45
46 return true;
47}

84int process(std::string const& name, std::string const& executableName, std::function<void(std::string const&, std::string const&)> initSettingsFunc,
85 std::function<void(void)> processOptionsFunc, const int argc, const char** argv) {
86 storm::utility::setUp();
87 storm::cli::printHeader(name, argc, argv);
88
89 // Initialize settings
90 initSettingsFunc(name, executableName);
91
92 storm::utility::Stopwatch totalTimer(true);
93 if (!parseOptions(argc, argv)) {
94 return -1;
95 }
96
97 // Start by setting some urgent options (log levels, resources, etc.)
98 setResourceLimits();
99 setLogLevel();
100 setFileLogging();
101 // Set output precision
102 storm::utility::setOutputDigitsFromGeneralPrecision(storm::settings::getModule<storm::settings::modules::GeneralSettings>().getPrecision());
103
104 // Process options and start computations
105 processOptionsFunc();
106
107 totalTimer.stop();
108 if (storm::settings::getModule<storm::settings::modules::ResourceSettings>().isPrintTimeAndMemorySet()) {
109 storm::cli::printTimeAndMemoryStatistics(totalTimer.getTimeInMilliseconds());
110 }
111
112 // All operations have been performed, so we clean up everything and terminate.
113 storm::utility::cleanUp();
114 return 0;
115}
