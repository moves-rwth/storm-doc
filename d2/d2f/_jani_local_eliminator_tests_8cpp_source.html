41 // As many tests rely on building and checking a model (which requires quite a few lines of code), we use this common
42 // helper function to do the model checking. Apart from this, the tests don't share any helper functions.
43 void checkModel(storm::jani::Model model, std::vector<storm::jani::Property> properties,
44  std::map<storm::expressions::Variable, storm::expressions::Expression> consts, double expectedValue) {
45  model = model.defineUndefinedConstants(consts);
46  properties[0] = properties[0].substitute(consts);
47 
48  auto formulae = storm::api::extractFormulasFromProperties(properties);
49  storm::builder::BuilderOptions options(formulae, model);
50  options.setBuildAllLabels(true);
51  auto explicitModel = storm::api::buildSparseModel<double>(model, options)->template as<Dtmc>();
52 
53  auto task = storm::modelchecker::CheckTask<>(*(formulae[0]), true);
54  storm::Environment env;
55  auto checkResult = storm::api::verifyWithSparseEngine<double>(env, explicitModel, task);
56  auto quantResult = checkResult->asExplicitQuantitativeCheckResult<double>();
57 
58  auto initialStates = explicitModel->getInitialStates();
59  EXPECT_EQ(1u, initialStates.getNumberOfSetBits());
60  for (auto state = initialStates.begin(); state != initialStates.end(); ++state) {
61  EXPECT_NEAR(expectedValue, quantResult[*state], storm::settings::getModule<storm::settings::modules::GeneralSettings>().getPrecision());
62  }
63 }
