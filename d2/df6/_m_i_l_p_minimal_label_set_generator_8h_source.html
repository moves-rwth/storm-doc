125 static struct ChoiceInformation determineRelevantAndProblematicChoices(storm::models::sparse::Mdp<T> const& mdp,
126 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
127 StateInformation const& stateInformation,
128 storm::storage::BitVector const& psiStates) {
129 // Create result and shortcuts to needed data for convenience.
130 ChoiceInformation result;
131 storm::storage::SparseMatrix<T> const& transitionMatrix = mdp.getTransitionMatrix();
132 std::vector<uint_fast64_t> const& nondeterministicChoiceIndices = mdp.getNondeterministicChoiceIndices();
133
134 // Now traverse all choices of all relevant states and check whether there is a relevant target state.
135 // If so, the associated labels become relevant. Also, if a choice of relevant state has at least one
136 // relevant successor, the choice is considered to be relevant.
137 for (auto state : stateInformation.relevantStates) {
138 result.relevantChoicesForRelevantStates.emplace(state, std::list<uint_fast64_t>());
139 if (stateInformation.problematicStates.get(state)) {
140 result.problematicChoicesForProblematicStates.emplace(state, std::list<uint_fast64_t>());
141 }
142 for (uint_fast64_t row = nondeterministicChoiceIndices[state]; row < nondeterministicChoiceIndices[state + 1]; ++row) {
143 bool currentChoiceRelevant = false;
144 bool allSuccessorsProblematic = true;
145 for (auto const& successorEntry : transitionMatrix.getRow(row)) {
146 // If there is a relevant successor, we need to add the labels of the current choice.
147 if (stateInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn())) {
148 for (auto const& label : labelSets[row]) {
149 result.allRelevantLabels.insert(label);
150 }
151 if (!currentChoiceRelevant) {
152 currentChoiceRelevant = true;
153 result.relevantChoicesForRelevantStates[state].push_back(row);
154 }
155 }
156 if (!stateInformation.problematicStates.get(successorEntry.getColumn())) {
157 allSuccessorsProblematic = false;
158 }
159 }
160
161 // If all successors of a problematic state are problematic themselves, we record this choice
162 // as being problematic.
163 if (stateInformation.problematicStates.get(state) && allSuccessorsProblematic) {
164 result.problematicChoicesForProblematicStates[state].push_back(row);
165 }
166 }
167 }
168
169 // Finally, determine the set of labels that are known to be taken.
170 result.knownLabels = storm::counterexamples::getGuaranteedLabelSet(mdp, labelSets, psiStates, result.allRelevantLabels);
171 STORM_LOG_DEBUG("Found " << result.allRelevantLabels.size() << " relevant labels and " << result.knownLabels.size() << " known labels.");
172
173 return result;
174 }

296 static std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> createProblematicStateVariables(
297 storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp, StateInformation const& stateInformation,
298 ChoiceInformation const& choiceInformation) {
299 std::stringstream variableNameBuffer;
300 uint_fast64_t numberOfVariablesCreated = 0;
301 std::unordered_map<uint_fast64_t, storm::expressions::Variable> resultingMap;
302
303 for (auto state : stateInformation.problematicStates) {
304 // First check whether there is not already a variable for this state and advance to the next state
305 // in this case.
306 if (resultingMap.find(state) == resultingMap.end()) {
307 variableNameBuffer.str("");
308 variableNameBuffer.clear();
309 variableNameBuffer << "r" << state;
310 resultingMap[state] = solver.addBoundedContinuousVariable(variableNameBuffer.str(), 0, 1);
311 ++numberOfVariablesCreated;
312 }
313
314 std::list<uint_fast64_t> const& relevantChoicesForState = choiceInformation.relevantChoicesForRelevantStates.at(state);
315 for (uint_fast64_t row : relevantChoicesForState) {
316 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(row)) {
317 if (stateInformation.relevantStates.get(successorEntry.getColumn())) {
318 if (resultingMap.find(successorEntry.getColumn()) == resultingMap.end()) {
319 variableNameBuffer.str("");
320 variableNameBuffer.clear();
321 variableNameBuffer << "r" << successorEntry.getColumn();
322 resultingMap[successorEntry.getColumn()] = solver.addBoundedContinuousVariable(variableNameBuffer.str(), 0, 1);
323 ++numberOfVariablesCreated;
324 }
325 }
326 }
327 }
328 }
329 return std::make_pair(resultingMap, numberOfVariablesCreated);
330 }

376 static VariableInformation createVariables(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
377 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation) {
378 // Create a struct that stores all information about variables.
379 VariableInformation result;
380
381 // Create variables for involved labels.
382 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> labelVariableResult =
383 createLabelVariables(solver, choiceInformation.allRelevantLabels);
384 result.labelToVariableMap = std::move(labelVariableResult.first);
385 result.numberOfVariables += labelVariableResult.second;
386 STORM_LOG_DEBUG("Created variables for labels.");
387
388 // Create scheduler variables for relevant states and their actions.
389 std::pair<std::unordered_map<uint_fast64_t, std::list<storm::expressions::Variable>>, uint_fast64_t> schedulerVariableResult =
390 createSchedulerVariables(solver, stateInformation, choiceInformation);
391 result.stateToChoiceVariablesMap = std::move(schedulerVariableResult.first);
392 result.numberOfVariables += schedulerVariableResult.second;
393 STORM_LOG_DEBUG("Created variables for nondeterministic choices.");
394
395 // Create scheduler variables for nondeterministically choosing an initial state.
396 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> initialChoiceVariableResult =
397 createInitialChoiceVariables(solver, mdp, stateInformation);
398 result.initialStateToChoiceVariableMap = std::move(initialChoiceVariableResult.first);
399 result.numberOfVariables += initialChoiceVariableResult.second;
400 STORM_LOG_DEBUG("Created variables for the nondeterministic choice of the initial state.");
401
402 // Create variables for probabilities for all relevant states.
403 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> probabilityVariableResult =
404 createProbabilityVariables(solver, stateInformation);
405 result.stateToProbabilityVariableMap = std::move(probabilityVariableResult.first);
406 result.numberOfVariables += probabilityVariableResult.second;
407 STORM_LOG_DEBUG("Created variables for the reachability probabilities.");
408
409 // Create a probability variable for a virtual initial state that nondeterministically chooses one of the system's real initial states as its target
410 // state.
411 std::pair<storm::expressions::Variable, uint_fast64_t> virtualInitialStateVariableResult = createVirtualInitialStateVariable(solver);
412 result.virtualInitialStateVariable = virtualInitialStateVariableResult.first;
413 result.numberOfVariables += virtualInitialStateVariableResult.second;
414 STORM_LOG_DEBUG("Created variables for the virtual initial state.");
415
416 // Create variables for problematic states.
417 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> problematicStateVariableResult =
418 createProblematicStateVariables(solver, mdp, stateInformation, choiceInformation);
419 result.problematicStateToVariableMap = std::move(problematicStateVariableResult.first);
420 result.numberOfVariables += problematicStateVariableResult.second;
421 STORM_LOG_DEBUG("Created variables for the problematic states.");
422
423 // Create variables for problematic choices.
424 std::pair<std::unordered_map<std::pair<uint_fast64_t, uint_fast64_t>, storm::expressions::Variable, PairHash>, uint_fast64_t>
425 problematicTransitionVariableResult = createProblematicChoiceVariables(solver, mdp, stateInformation, choiceInformation);
426 result.problematicTransitionToVariableMap = problematicTransitionVariableResult.first;
427 result.numberOfVariables += problematicTransitionVariableResult.second;
428 STORM_LOG_DEBUG("Created variables for the problematic choices.");
429
430 // Finally, we need to update the model to make the new variables usable.
431 solver.update();
432 STORM_LOG_INFO("Successfully created " << result.numberOfVariables << " MILP variables.");
433
434 // Finally, return variable information struct.
435 return result;
436 }

468 static uint_fast64_t assertValidPolicy(storm::solver::LpSolver<double>& solver, StateInformation const& stateInformation,
469 VariableInformation const& variableInformation) {
470 // Assert that the policy chooses at most one action in each state of the system.
471 uint_fast64_t numberOfConstraintsCreated = 0;
472 for (auto state : stateInformation.relevantStates) {
473 std::list<storm::expressions::Variable> const& choiceVariableIndices = variableInformation.stateToChoiceVariablesMap.at(state);
474 storm::expressions::Expression constraint = solver.getConstant(0);
475
476 for (auto const& choiceVariable : choiceVariableIndices) {
477 constraint = constraint + choiceVariable;
478 }
479
480 constraint = constraint <= solver.getConstant(1);
481
482 solver.addConstraint("ValidPolicy" + std::to_string(numberOfConstraintsCreated), constraint);
483 ++numberOfConstraintsCreated;
484 }
485
486 // Now assert that the virtual initial state picks exactly one initial state from the system as its
487 // successor state.
488 storm::expressions::Expression constraint = solver.getConstant(0);
489 for (auto const& initialStateVariablePair : variableInformation.initialStateToChoiceVariableMap) {
490 constraint = constraint + initialStateVariablePair.second;
491 }
492 constraint = constraint == solver.getConstant(1);
493
494 solver.addConstraint("VirtualInitialStateChoosesOneInitialState", constraint);
495 ++numberOfConstraintsCreated;
496
497 return numberOfConstraintsCreated;
498 }

568 static uint_fast64_t assertReachabilityProbabilities(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
569 storm::storage::BitVector const& psiStates, StateInformation const& stateInformation,
570 ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
571 uint_fast64_t numberOfConstraintsCreated = 0;
572 for (auto state : stateInformation.relevantStates) {
573 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(state).begin();
574 for (auto choice : choiceInformation.relevantChoicesForRelevantStates.at(state)) {
575 storm::expressions::Expression constraint = variableInformation.stateToProbabilityVariableMap.at(state);
576
577 double rightHandSide = 1;
578 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
579 if (stateInformation.relevantStates.get(successorEntry.getColumn())) {
580 constraint = constraint - solver.getConstant(successorEntry.getValue()) *
581 variableInformation.stateToProbabilityVariableMap.at(successorEntry.getColumn());
582 } else if (psiStates.get(successorEntry.getColumn())) {
583 rightHandSide += successorEntry.getValue();
584 }
585 }
586
587 constraint = constraint + *choiceVariableIterator <= solver.getConstant(rightHandSide);
588 solver.addConstraint("ReachabilityProbabilities" + std::to_string(numberOfConstraintsCreated), constraint);
589
590 ++numberOfConstraintsCreated;
591 ++choiceVariableIterator;
592 }
593 }
594
595 // Make sure that the virtual initial state is being assigned the probability from the initial state
596 // that it selected as a successor state.
597 for (auto const& initialStateVariablePair : variableInformation.initialStateToChoiceVariableMap) {
598 storm::expressions::Expression constraint = variableInformation.virtualInitialStateVariable -
599 variableInformation.stateToProbabilityVariableMap.at(initialStateVariablePair.first) +
600 initialStateVariablePair.second <=
601 solver.getConstant(1);
602 solver.addConstraint("VirtualInitialStateHasCorrectProbability" + std::to_string(numberOfConstraintsCreated), constraint);
603 ++numberOfConstraintsCreated;
604 }
605
606 return numberOfConstraintsCreated;
607 }

619 static uint_fast64_t assertUnproblematicStateReachable(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
620 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation,
621 VariableInformation const& variableInformation) {
622 uint_fast64_t numberOfConstraintsCreated = 0;
623
624 for (auto stateListPair : choiceInformation.problematicChoicesForProblematicStates) {
625 for (auto problematicChoice : stateListPair.second) {
626 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
627 variableInformation.stateToChoiceVariablesMap.at(stateListPair.first).begin();
628 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(stateListPair.first)) {
629 if (relevantChoice == problematicChoice) {
630 break;
631 }
632 ++choiceVariableIterator;
633 }
634
635 storm::expressions::Expression constraint = *choiceVariableIterator;
636 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(problematicChoice)) {
637 constraint =
638 constraint - variableInformation.problematicTransitionToVariableMap.at(std::make_pair(stateListPair.first, successorEntry.getColumn()));
639 }
640 constraint = constraint <= solver.getConstant(0);
641
642 solver.addConstraint("UnproblematicStateReachable" + std::to_string(numberOfConstraintsCreated), constraint);
643 ++numberOfConstraintsCreated;
644 }
645 }
646
647 for (auto state : stateInformation.problematicStates) {
648 for (auto problematicChoice : choiceInformation.problematicChoicesForProblematicStates.at(state)) {
649 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(problematicChoice)) {
650 storm::expressions::Expression constraint = variableInformation.problematicStateToVariableMap.at(state);
651 constraint = constraint - variableInformation.problematicStateToVariableMap.at(successorEntry.getColumn());
652 constraint = constraint + variableInformation.problematicTransitionToVariableMap.at(std::make_pair(state, successorEntry.getColumn()));
653 constraint = constraint < solver.getConstant(1);
654
655 solver.addConstraint("UnproblematicStateReachable" + std::to_string(numberOfConstraintsCreated), constraint);
656 ++numberOfConstraintsCreated;
657 }
658 }
659 }
660 return numberOfConstraintsCreated;
661 }

695 static uint_fast64_t assertSchedulerCuts(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
696 storm::storage::BitVector const& psiStates, StateInformation const& stateInformation,
697 ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
698 storm::storage::SparseMatrix<T> backwardTransitions = mdp.getBackwardTransitions();
699 uint_fast64_t numberOfConstraintsCreated = 0;
700
701 for (auto state : stateInformation.relevantStates) {
702 // Assert that all states, that select an action, this action either has a non-zero probability to
703 // go to a psi state directly, or in the successor states, at least one action is selected as well.
704 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(state).begin();
705 for (auto choice : choiceInformation.relevantChoicesForRelevantStates.at(state)) {
706 bool psiStateReachableInOneStep = false;
707 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
708 if (psiStates.get(successorEntry.getColumn())) {
709 psiStateReachableInOneStep = true;
710 }
711 }
712
713 if (!psiStateReachableInOneStep) {
714 storm::expressions::Expression constraint = *choiceVariableIterator;
715 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
716 if (state != successorEntry.getColumn() && stateInformation.relevantStates.get(successorEntry.getColumn())) {
717 std::list<storm::expressions::Variable> const& successorChoiceVariableIndices =
718 variableInformation.stateToChoiceVariablesMap.at(successorEntry.getColumn());
719
720 for (auto const& choiceVariable : successorChoiceVariableIndices) {
721 constraint = constraint - choiceVariable;
722 }
723 }
724 }
725 constraint = constraint <= solver.getConstant(1);
726
727 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
728 ++numberOfConstraintsCreated;
729 }
730
731 ++choiceVariableIterator;
732 }
733
734 // For all states assert that there is either a selected incoming transition in the subsystem or the
735 // state is the chosen initial state if there is one selected action in the current state.
736 storm::expressions::Expression constraint = solver.getConstant(0);
737
738 for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(state)) {
739 constraint = constraint + choiceVariable;
740 }
741
742 // Compute the set of predecessors.
743 std::unordered_set<uint_fast64_t> predecessors;
744 for (auto const& predecessorEntry : backwardTransitions.getRow(state)) {
745 if (state != predecessorEntry.getColumn()) {
746 predecessors.insert(predecessorEntry.getColumn());
747 }
748 }
749
750 for (auto predecessor : predecessors) {
751 // If the predecessor is not a relevant state, we need to skip it.
752 if (!stateInformation.relevantStates.get(predecessor)) {
753 continue;
754 }
755
756 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
757 variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
758 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
759 bool choiceTargetsCurrentState = false;
760
761 // Check if the current choice targets the current state.
762 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
763 if (state == successorEntry.getColumn()) {
764 choiceTargetsCurrentState = true;
765 break;
766 }
767 }
768
769 // If it does, we can add the choice to the sum.
770 if (choiceTargetsCurrentState) {
771 constraint = constraint - *choiceVariableIterator;
772 }
773 ++choiceVariableIterator;
774 }
775 }
776
777 // If the current state is an initial state and is selected as a successor state by the virtual
778 // initial state, then this also justifies making a choice in the current state.
779 if (mdp.getStates("init").get(state)) {
780 constraint = constraint - variableInformation.initialStateToChoiceVariableMap.at(state);
781 }
782 constraint = constraint <= solver.getConstant(0);
783
784 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
785 ++numberOfConstraintsCreated;
786 }
787
788 // Assert that at least one initial state selects at least one action.
789 storm::expressions::Expression constraint = solver.getConstant(0);
790 for (auto initialState : mdp.getStates("init")) {
791 for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(initialState)) {
792 constraint = constraint + choiceVariable;
793 }
794 }
795 constraint = constraint >= solver.getConstant(1);
796 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
797 ++numberOfConstraintsCreated;
798
799 // Add constraints that ensure at least one choice is selected that targets a psi state.
800 constraint = solver.getConstant(0);
801 std::unordered_set<uint_fast64_t> predecessors;
802 for (auto psiState : psiStates) {
803 // Compute the set of predecessors.
804 for (auto const& predecessorEntry : backwardTransitions.getRow(psiState)) {
805 if (psiState != predecessorEntry.getColumn()) {
806 predecessors.insert(predecessorEntry.getColumn());
807 }
808 }
809 }
810
811 for (auto predecessor : predecessors) {
812 // If the predecessor is not a relevant state, we need to skip it.
813 if (!stateInformation.relevantStates.get(predecessor)) {
814 continue;
815 }
816
817 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
818 variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
819 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
820 bool choiceTargetsPsiState = false;
821
822 // Check if the current choice targets the current state.
823 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
824 if (psiStates.get(successorEntry.getColumn())) {
825 choiceTargetsPsiState = true;
826 break;
827 }
828 }
829
830 // If it does, we can add the choice to the sum.
831 if (choiceTargetsPsiState) {
832 constraint = constraint + *choiceVariableIterator;
833 }
834 ++choiceVariableIterator;
835 }
836 }
837 constraint = constraint >= solver.getConstant(1);
838
839 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
840 ++numberOfConstraintsCreated;
841
842 return numberOfConstraintsCreated;
843 }

860 static void buildConstraintSystem(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
861 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, storm::storage::BitVector const& psiStates,
862 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation,
863 VariableInformation const& variableInformation, double probabilityThreshold, bool strictBound,
864 bool includeSchedulerCuts = false) {
865 // Assert that the reachability probability in the subsystem exceeds the given threshold.
866 uint_fast64_t numberOfConstraints = assertProbabilityGreaterThanThreshold(solver, variableInformation, probabilityThreshold, strictBound);
867 STORM_LOG_DEBUG("Asserted that reachability probability exceeds threshold.");
868
869 // Add constraints that assert the policy takes at most one action in each state.
870 numberOfConstraints += assertValidPolicy(solver, stateInformation, variableInformation);
871 STORM_LOG_DEBUG("Asserted that policy is valid.");
872
873 // Add constraints that assert the labels that belong to some taken choices are taken as well.
874 numberOfConstraints += assertChoicesImplyLabels(solver, mdp, labelSets, stateInformation, choiceInformation, variableInformation);
875 STORM_LOG_DEBUG("Asserted that labels implied by choices are taken.");
876
877 // Add constraints that encode that the reachability probability from states which do not pick any action
878 // is zero.
879 numberOfConstraints += assertZeroProbabilityWithoutChoice(solver, stateInformation, variableInformation);
880 STORM_LOG_DEBUG("Asserted that reachability probability is zero if no choice is taken.");
881
882 // Add constraints that encode the reachability probabilities for states.
883 numberOfConstraints += assertReachabilityProbabilities(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
884 STORM_LOG_DEBUG("Asserted constraints for reachability probabilities.");
885
886 // Add constraints that ensure the reachability of an unproblematic state from each problematic state.
887 numberOfConstraints += assertUnproblematicStateReachable(solver, mdp, stateInformation, choiceInformation, variableInformation);
888 STORM_LOG_DEBUG("Asserted that unproblematic state reachable from problematic states.");
889
890 // Add constraints that express that certain labels are already known to be taken.
891 numberOfConstraints += assertKnownLabels(solver, choiceInformation, variableInformation);
892 STORM_LOG_DEBUG("Asserted known labels are taken.");
893
894 // If required, assert additional constraints that reduce the number of possible policies.
895 if (includeSchedulerCuts) {
896 numberOfConstraints += assertSchedulerCuts(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
897 STORM_LOG_DEBUG("Asserted scheduler cuts.");
898 }
899
900 // Finally, we can tell the solver to incorporate the latest changes.
901 solver.update();
902
903 STORM_LOG_INFO("Successfully created " << numberOfConstraints << " MILP constraints.");
904 }

912 static storm::storage::FlatSet<uint_fast64_t> getUsedLabelsInSolution(storm::solver::LpSolver<double> const& solver,
913 VariableInformation const& variableInformation) {
914 storm::storage::FlatSet<uint_fast64_t> result;
915
916 for (auto const& labelVariablePair : variableInformation.labelToVariableMap) {
917 bool labelTaken = solver.getBinaryValue(labelVariablePair.second);
918
919 if (labelTaken) {
920 result.insert(labelVariablePair.first);
921 }
922 }
923 return result;

927 static storm::storage::FlatSet<uint_fast64_t> getMinimalLabelSet(Environment const& env, storm::models::sparse::Mdp<T> const& mdp,
928 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
929 storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates,
930 double probabilityThreshold, bool strictBound, bool checkThresholdFeasible = false,
931 bool includeSchedulerCuts = false) {
932 // (0) Check whether the label sets are valid
933 STORM_LOG_THROW(mdp.getNumberOfChoices() == labelSets.size(), storm::exceptions::InvalidArgumentException,
934 "The given number of labels does not match the number of choices.");
935
936 // (1) Check whether its possible to exceed the threshold if checkThresholdFeasible is set.
937 double maximalReachabilityProbability = 0;
938 if (checkThresholdFeasible) {
939 storm::modelchecker::helper::SparseMdpPrctlHelper<T> modelcheckerHelper;
940 std::vector<T> result = std::move(
941 modelcheckerHelper
942 .computeUntilProbabilities(env, false, mdp.getTransitionMatrix(), mdp.getBackwardTransitions(), phiStates, psiStates, false, false)
943 .values);
944 for (auto state : mdp.getInitialStates()) {
945 maximalReachabilityProbability = std::max(maximalReachabilityProbability, result[state]);
946 }
947 STORM_LOG_THROW((strictBound && maximalReachabilityProbability >= probabilityThreshold) ||
948 (!strictBound && maximalReachabilityProbability > probabilityThreshold),
949 storm::exceptions::InvalidArgumentException,
950 "Given probability threshold " << probabilityThreshold << " can not be " << (strictBound ? "achieved" : "exceeded")
951 << " in model with maximal reachability probability of " << maximalReachabilityProbability << ".");
952 std::cout << "\nMaximal reachability in model is " << maximalReachabilityProbability << ".\n\n";
953 }
954
955 // (2) Identify relevant and problematic states.
956 StateInformation stateInformation = determineRelevantAndProblematicStates(mdp, phiStates, psiStates);
957
958 // (3) Determine sets of relevant labels and problematic choices.
959 ChoiceInformation choiceInformation = determineRelevantAndProblematicChoices(mdp, labelSets, stateInformation, psiStates);
960
961 // (4) Encode resulting system as MILP problem.
962 std::shared_ptr<storm::solver::LpSolver<double>> solver = storm::utility::solver::getLpSolver<double>("MinimalLabelSetCounterexample");
963
964 // (4.1) Create variables.
965 VariableInformation variableInformation = createVariables(*solver, mdp, stateInformation, choiceInformation);
966
967 // (4.2) Construct constraint system.
968 buildConstraintSystem(*solver, mdp, labelSets, psiStates, stateInformation, choiceInformation, variableInformation, probabilityThreshold, strictBound,
969 includeSchedulerCuts);
970
971 // (4.3) Optimize the model.
972 solver->optimize();
973
974 // (4.4) Read off result from variables.
975 storm::storage::FlatSet<uint_fast64_t> usedLabelSet = getUsedLabelsInSolution(*solver, variableInformation);
976 usedLabelSet.insert(choiceInformation.knownLabels.begin(), choiceInformation.knownLabels.end());
977
978 // (5) Return result.
979 return usedLabelSet;
980 }

990 static std::shared_ptr<HighLevelCounterexample> computeCounterexample(Environment const& env, storm::storage::SymbolicModelDescription const& symbolicModel,
991 storm::models::sparse::Mdp<T> const& mdp,
992 std::shared_ptr<storm::logic::Formula const> const& formula) {
993 std::cout << "\nGenerating minimal label counterexample for formula " << *formula << '\n';
994
995 // Check whether there are choice origins available
996 STORM_LOG_THROW(mdp.hasChoiceOrigins(), storm::exceptions::InvalidArgumentException,
997 "Restriction to minimal command set is impossible for model without choice origns.");
998 STORM_LOG_THROW(mdp.getChoiceOrigins()->isPrismChoiceOrigins() || mdp.getChoiceOrigins()->isJaniChoiceOrigins(),
999 storm::exceptions::InvalidArgumentException,
1000 "Restriction to command set is impossible for model without PRISM or JANI choice origins.");
1001
1002 STORM_LOG_THROW(formula->isProbabilityOperatorFormula(), storm::exceptions::InvalidPropertyException,
1003 "Counterexample generation does not support this kind of formula. Expecting a probability operator as the outermost formula element.");
1004 storm::logic::ProbabilityOperatorFormula const& probabilityOperator = formula->asProbabilityOperatorFormula();
1005 STORM_LOG_THROW(probabilityOperator.hasBound(), storm::exceptions::InvalidPropertyException,
1006 "Counterexample generation only supports bounded formulas.");
1007 storm::logic::ComparisonType comparisonType = probabilityOperator.getComparisonType();
1008 STORM_LOG_THROW(comparisonType == storm::logic::ComparisonType::Less || comparisonType == storm::logic::ComparisonType::LessEqual,
1009 storm::exceptions::InvalidPropertyException, "Counterexample generation only supports formulas with an upper probability bound.");
1010 STORM_LOG_THROW(probabilityOperator.getSubformula().isUntilFormula() || probabilityOperator.getSubformula().isEventuallyFormula(),
1011 storm::exceptions::InvalidPropertyException, "Path formula is required to be of the form 'phi U psi' for counterexample generation.");
1012
1013 bool strictBound = comparisonType == storm::logic::ComparisonType::Less;
1014 double threshold = probabilityOperator.getThresholdAs<double>();
1015
1016 storm::storage::BitVector phiStates;
1017 storm::storage::BitVector psiStates;
1018 storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Mdp<T>> modelchecker(mdp);
1019
1020 if (probabilityOperator.getSubformula().isUntilFormula()) {
1021 storm::logic::UntilFormula const& untilFormula = probabilityOperator.getSubformula().asUntilFormula();
1022
1023 std::unique_ptr<storm::modelchecker::CheckResult> leftResult = modelchecker.check(env, untilFormula.getLeftSubformula());
1024 std::unique_ptr<storm::modelchecker::CheckResult> rightResult = modelchecker.check(env, untilFormula.getRightSubformula());
1025
1026 storm::modelchecker::ExplicitQualitativeCheckResult const& leftQualitativeResult = leftResult->asExplicitQualitativeCheckResult();
1027 storm::modelchecker::ExplicitQualitativeCheckResult const& rightQualitativeResult = rightResult->asExplicitQualitativeCheckResult();
1028
1029 phiStates = leftQualitativeResult.getTruthValuesVector();
1030 psiStates = rightQualitativeResult.getTruthValuesVector();
1031 } else if (probabilityOperator.getSubformula().isEventuallyFormula()) {
1032 storm::logic::EventuallyFormula const& eventuallyFormula = probabilityOperator.getSubformula().asEventuallyFormula();
1033
1034 std::unique_ptr<storm::modelchecker::CheckResult> subResult = modelchecker.check(env, eventuallyFormula.getSubformula());
1035
1036 storm::modelchecker::ExplicitQualitativeCheckResult const& subQualitativeResult = subResult->asExplicitQualitativeCheckResult();
1037
1038 phiStates = storm::storage::BitVector(mdp.getNumberOfStates(), true);
1039 psiStates = subQualitativeResult.getTruthValuesVector();
1040 }
1041
1042 // Obtain the label sets for each choice.
1043 std::vector<storm::storage::FlatSet<uint_fast64_t>> labelSets(mdp.getNumberOfChoices());
1044 if (mdp.getChoiceOrigins()->isPrismChoiceOrigins()) {
1045 storm::storage::sparse::PrismChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asPrismChoiceOrigins();
1046 for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1047 labelSets[choice] = choiceOrigins.getCommandSet(choice);
1048 }
1049 } else {
1050 storm::storage::sparse::JaniChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asJaniChoiceOrigins();
1051
1052 // The choice origins are known to be JANI ones at this point.
1053 for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1054 labelSets[choice] = choiceOrigins.getEdgeIndexSet(choice);
1055 }
1056 }
1057
1058 // Delegate the actual computation work to the function of equal name.
1059 auto startTime = std::chrono::high_resolution_clock::now();
1060 storm::storage::FlatSet<uint_fast64_t> usedLabelSet =
1061 getMinimalLabelSet(env, mdp, labelSets, phiStates, psiStates, threshold, strictBound, true,
1062 storm::settings::getModule<storm::settings::modules::CounterexampleGeneratorSettings>().isUseSchedulerCutsSet());
1063 auto endTime = std::chrono::high_resolution_clock::now();
1064 std::cout << "\nComputed minimal command set of size " << usedLabelSet.size() << " in "
1065 << std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime).count() << "ms.\n";
1066
1067 if (symbolicModel.isPrismProgram()) {
1068 return std::make_shared<HighLevelCounterexample>(symbolicModel.asPrismProgram().restrictCommands(usedLabelSet));
1069 } else {
1070 STORM_LOG_ASSERT(symbolicModel.isJaniModel(), "Unknown symbolic model description type.");
1071 return std::make_shared<HighLevelCounterexample>(symbolicModel.asJaniModel().restrictEdges(usedLabelSet));
1072 }
1073 }
1074};
