125 static ChoiceInformation determineRelevantAndProblematicChoices(storm::models::sparse::Mdp<T> const& mdp,
126 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
127 StateInformation const& stateInformation, storm::storage::BitVector const& psiStates) {
128 // Create result and shortcuts to needed data for convenience.
129 ChoiceInformation result;
130 storm::storage::SparseMatrix<T> const& transitionMatrix = mdp.getTransitionMatrix();
131 std::vector<uint_fast64_t> const& nondeterministicChoiceIndices = mdp.getNondeterministicChoiceIndices();
132
133 // Now traverse all choices of all relevant states and check whether there is a relevant target state.
134 // If so, the associated labels become relevant. Also, if a choice of relevant state has at least one
135 // relevant successor, the choice is considered to be relevant.
136 for (auto state : stateInformation.relevantStates) {
137 result.relevantChoicesForRelevantStates.emplace(state, std::list<uint_fast64_t>());
138 if (stateInformation.problematicStates.get(state)) {
139 result.problematicChoicesForProblematicStates.emplace(state, std::list<uint_fast64_t>());
140 }
141 for (uint_fast64_t row = nondeterministicChoiceIndices[state]; row < nondeterministicChoiceIndices[state + 1]; ++row) {
142 bool currentChoiceRelevant = false;
143 bool allSuccessorsProblematic = true;
144 for (auto const& successorEntry : transitionMatrix.getRow(row)) {
145 // If there is a relevant successor, we need to add the labels of the current choice.
146 if (stateInformation.relevantStates.get(successorEntry.getColumn()) || psiStates.get(successorEntry.getColumn())) {
147 for (auto const& label : labelSets[row]) {
148 result.allRelevantLabels.insert(label);
149 }
150 if (!currentChoiceRelevant) {
151 currentChoiceRelevant = true;
152 result.relevantChoicesForRelevantStates[state].push_back(row);
153 }
154 }
155 if (!stateInformation.problematicStates.get(successorEntry.getColumn())) {
156 allSuccessorsProblematic = false;
157 }
158 }
159
160 // If all successors of a problematic state are problematic themselves, we record this choice
161 // as being problematic.
162 if (stateInformation.problematicStates.get(state) && allSuccessorsProblematic) {
163 result.problematicChoicesForProblematicStates[state].push_back(row);
164 }
165 }
166 }
167
168 // Finally, determine the set of labels that are known to be taken.
169 result.knownLabels = storm::counterexamples::getGuaranteedLabelSet(mdp, labelSets, psiStates, result.allRelevantLabels);
170 STORM_LOG_DEBUG("Found " << result.allRelevantLabels.size() << " relevant labels and " << result.knownLabels.size() << " known labels.");
171
172 return result;
173 }

295 static std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> createProblematicStateVariables(
296 storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp, StateInformation const& stateInformation,
297 ChoiceInformation const& choiceInformation) {
298 std::stringstream variableNameBuffer;
299 uint_fast64_t numberOfVariablesCreated = 0;
300 std::unordered_map<uint_fast64_t, storm::expressions::Variable> resultingMap;
301
302 for (auto state : stateInformation.problematicStates) {
303 // First check whether there is not already a variable for this state and advance to the next state
304 // in this case.
305 if (resultingMap.find(state) == resultingMap.end()) {
306 variableNameBuffer.str("");
307 variableNameBuffer.clear();
308 variableNameBuffer << "r" << state;
309 resultingMap[state] = solver.addBoundedContinuousVariable(variableNameBuffer.str(), 0, 1);
310 ++numberOfVariablesCreated;
311 }
312
313 std::list<uint_fast64_t> const& relevantChoicesForState = choiceInformation.relevantChoicesForRelevantStates.at(state);
314 for (uint_fast64_t row : relevantChoicesForState) {
315 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(row)) {
316 if (stateInformation.relevantStates.get(successorEntry.getColumn())) {
317 if (resultingMap.find(successorEntry.getColumn()) == resultingMap.end()) {
318 variableNameBuffer.str("");
319 variableNameBuffer.clear();
320 variableNameBuffer << "r" << successorEntry.getColumn();
321 resultingMap[successorEntry.getColumn()] = solver.addBoundedContinuousVariable(variableNameBuffer.str(), 0, 1);
322 ++numberOfVariablesCreated;
323 }
324 }
325 }
326 }
327 }
328 return std::make_pair(resultingMap, numberOfVariablesCreated);

375 static VariableInformation createVariables(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
376 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation) {
377 // Create a struct that stores all information about variables.
378 VariableInformation result;
379
380 // Create variables for involved labels.
381 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> labelVariableResult =
382 createLabelVariables(solver, choiceInformation.allRelevantLabels);
383 result.labelToVariableMap = std::move(labelVariableResult.first);
384 result.numberOfVariables += labelVariableResult.second;
385 STORM_LOG_DEBUG("Created variables for labels.");
386
387 // Create scheduler variables for relevant states and their actions.
388 std::pair<std::unordered_map<uint_fast64_t, std::list<storm::expressions::Variable>>, uint_fast64_t> schedulerVariableResult =
389 createSchedulerVariables(solver, stateInformation, choiceInformation);
390 result.stateToChoiceVariablesMap = std::move(schedulerVariableResult.first);
391 result.numberOfVariables += schedulerVariableResult.second;
392 STORM_LOG_DEBUG("Created variables for nondeterministic choices.");
393
394 // Create scheduler variables for nondeterministically choosing an initial state.
395 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> initialChoiceVariableResult =
396 createInitialChoiceVariables(solver, mdp, stateInformation);
397 result.initialStateToChoiceVariableMap = std::move(initialChoiceVariableResult.first);
398 result.numberOfVariables += initialChoiceVariableResult.second;
399 STORM_LOG_DEBUG("Created variables for the nondeterministic choice of the initial state.");
400
401 // Create variables for probabilities for all relevant states.
402 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> probabilityVariableResult =
403 createProbabilityVariables(solver, stateInformation);
404 result.stateToProbabilityVariableMap = std::move(probabilityVariableResult.first);
405 result.numberOfVariables += probabilityVariableResult.second;
406 STORM_LOG_DEBUG("Created variables for the reachability probabilities.");
407
408 // Create a probability variable for a virtual initial state that nondeterministically chooses one of the system's real initial states as its target
409 // state.
410 std::pair<storm::expressions::Variable, uint_fast64_t> virtualInitialStateVariableResult = createVirtualInitialStateVariable(solver);
411 result.virtualInitialStateVariable = virtualInitialStateVariableResult.first;
412 result.numberOfVariables += virtualInitialStateVariableResult.second;
413 STORM_LOG_DEBUG("Created variables for the virtual initial state.");
414
415 // Create variables for problematic states.
416 std::pair<std::unordered_map<uint_fast64_t, storm::expressions::Variable>, uint_fast64_t> problematicStateVariableResult =
417 createProblematicStateVariables(solver, mdp, stateInformation, choiceInformation);
418 result.problematicStateToVariableMap = std::move(problematicStateVariableResult.first);
419 result.numberOfVariables += problematicStateVariableResult.second;
420 STORM_LOG_DEBUG("Created variables for the problematic states.");
421
422 // Create variables for problematic choices.
423 std::pair<std::unordered_map<std::pair<uint_fast64_t, uint_fast64_t>, storm::expressions::Variable, PairHash>, uint_fast64_t>
424 problematicTransitionVariableResult = createProblematicChoiceVariables(solver, mdp, stateInformation, choiceInformation);
425 result.problematicTransitionToVariableMap = problematicTransitionVariableResult.first;
426 result.numberOfVariables += problematicTransitionVariableResult.second;
427 STORM_LOG_DEBUG("Created variables for the problematic choices.");
428
429 // Finally, we need to update the model to make the new variables usable.
430 solver.update();
431 STORM_LOG_INFO("Successfully created " << result.numberOfVariables << " MILP variables.");
432
433 // Finally, return variable information struct.
434 return result;
435 }

467 static uint_fast64_t assertValidPolicy(storm::solver::LpSolver<double>& solver, StateInformation const& stateInformation,
468 VariableInformation const& variableInformation) {
469 // Assert that the policy chooses at most one action in each state of the system.
470 uint_fast64_t numberOfConstraintsCreated = 0;
471 for (auto state : stateInformation.relevantStates) {
472 std::list<storm::expressions::Variable> const& choiceVariableIndices = variableInformation.stateToChoiceVariablesMap.at(state);
473 storm::expressions::Expression constraint = solver.getConstant(0);
474
475 for (auto const& choiceVariable : choiceVariableIndices) {
476 constraint = constraint + choiceVariable;
477 }
478
479 constraint = constraint <= solver.getConstant(1);
480
481 solver.addConstraint("ValidPolicy" + std::to_string(numberOfConstraintsCreated), constraint);
482 ++numberOfConstraintsCreated;
483 }
484
485 // Now assert that the virtual initial state picks exactly one initial state from the system as its
486 // successor state.
487 storm::expressions::Expression constraint = solver.getConstant(0);
488 for (auto const& initialStateVariablePair : variableInformation.initialStateToChoiceVariableMap) {
489 constraint = constraint + initialStateVariablePair.second;
490 }
491 constraint = constraint == solver.getConstant(1);
492
493 solver.addConstraint("VirtualInitialStateChoosesOneInitialState", constraint);
494 ++numberOfConstraintsCreated;
495
496 return numberOfConstraintsCreated;
497 }

567 static uint_fast64_t assertReachabilityProbabilities(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
568 storm::storage::BitVector const& psiStates, StateInformation const& stateInformation,
569 ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
570 uint_fast64_t numberOfConstraintsCreated = 0;
571 for (auto state : stateInformation.relevantStates) {
572 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(state).begin();
573 for (auto choice : choiceInformation.relevantChoicesForRelevantStates.at(state)) {
574 storm::expressions::Expression constraint = variableInformation.stateToProbabilityVariableMap.at(state);
575
576 double rightHandSide = 1;
577 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
578 if (stateInformation.relevantStates.get(successorEntry.getColumn())) {
579 constraint = constraint - solver.getConstant(successorEntry.getValue()) *
580 variableInformation.stateToProbabilityVariableMap.at(successorEntry.getColumn());
581 } else if (psiStates.get(successorEntry.getColumn())) {
582 rightHandSide += successorEntry.getValue();
583 }
584 }
585
586 constraint = constraint + *choiceVariableIterator <= solver.getConstant(rightHandSide);
587 solver.addConstraint("ReachabilityProbabilities" + std::to_string(numberOfConstraintsCreated), constraint);
588
589 ++numberOfConstraintsCreated;
590 ++choiceVariableIterator;
591 }
592 }
593
594 // Make sure that the virtual initial state is being assigned the probability from the initial state
595 // that it selected as a successor state.
596 for (auto const& initialStateVariablePair : variableInformation.initialStateToChoiceVariableMap) {
597 storm::expressions::Expression constraint = variableInformation.virtualInitialStateVariable -
598 variableInformation.stateToProbabilityVariableMap.at(initialStateVariablePair.first) +
599 initialStateVariablePair.second <=
600 solver.getConstant(1);
601 solver.addConstraint("VirtualInitialStateHasCorrectProbability" + std::to_string(numberOfConstraintsCreated), constraint);
602 ++numberOfConstraintsCreated;
603 }
604
605 return numberOfConstraintsCreated;
606 }

618 static uint_fast64_t assertUnproblematicStateReachable(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
619 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation,
620 VariableInformation const& variableInformation) {
621 uint_fast64_t numberOfConstraintsCreated = 0;
622
623 for (auto stateListPair : choiceInformation.problematicChoicesForProblematicStates) {
624 for (auto problematicChoice : stateListPair.second) {
625 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
626 variableInformation.stateToChoiceVariablesMap.at(stateListPair.first).begin();
627 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(stateListPair.first)) {
628 if (relevantChoice == problematicChoice) {
629 break;
630 }
631 ++choiceVariableIterator;
632 }
633
634 storm::expressions::Expression constraint = *choiceVariableIterator;
635 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(problematicChoice)) {
636 constraint =
637 constraint - variableInformation.problematicTransitionToVariableMap.at(std::make_pair(stateListPair.first, successorEntry.getColumn()));
638 }
639 constraint = constraint <= solver.getConstant(0);
640
641 solver.addConstraint("UnproblematicStateReachable" + std::to_string(numberOfConstraintsCreated), constraint);
642 ++numberOfConstraintsCreated;
643 }
644 }
645
646 for (auto state : stateInformation.problematicStates) {
647 for (auto problematicChoice : choiceInformation.problematicChoicesForProblematicStates.at(state)) {
648 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(problematicChoice)) {
649 storm::expressions::Expression constraint = variableInformation.problematicStateToVariableMap.at(state);
650 constraint = constraint - variableInformation.problematicStateToVariableMap.at(successorEntry.getColumn());
651 constraint = constraint + variableInformation.problematicTransitionToVariableMap.at(std::make_pair(state, successorEntry.getColumn()));
652 constraint = constraint < solver.getConstant(1);
653
654 solver.addConstraint("UnproblematicStateReachable" + std::to_string(numberOfConstraintsCreated), constraint);
655 ++numberOfConstraintsCreated;
656 }
657 }
658 }
659 return numberOfConstraintsCreated;
660 }

694 static uint_fast64_t assertSchedulerCuts(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
695 storm::storage::BitVector const& psiStates, StateInformation const& stateInformation,
696 ChoiceInformation const& choiceInformation, VariableInformation const& variableInformation) {
697 storm::storage::SparseMatrix<T> backwardTransitions = mdp.getBackwardTransitions();
698 uint_fast64_t numberOfConstraintsCreated = 0;
699
700 for (auto state : stateInformation.relevantStates) {
701 // Assert that all states, that select an action, this action either has a non-zero probability to
702 // go to a psi state directly, or in the successor states, at least one action is selected as well.
703 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator = variableInformation.stateToChoiceVariablesMap.at(state).begin();
704 for (auto choice : choiceInformation.relevantChoicesForRelevantStates.at(state)) {
705 bool psiStateReachableInOneStep = false;
706 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
707 if (psiStates.get(successorEntry.getColumn())) {
708 psiStateReachableInOneStep = true;
709 }
710 }
711
712 if (!psiStateReachableInOneStep) {
713 storm::expressions::Expression constraint = *choiceVariableIterator;
714 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(choice)) {
715 if (state != successorEntry.getColumn() && stateInformation.relevantStates.get(successorEntry.getColumn())) {
716 std::list<storm::expressions::Variable> const& successorChoiceVariableIndices =
717 variableInformation.stateToChoiceVariablesMap.at(successorEntry.getColumn());
718
719 for (auto const& choiceVariable : successorChoiceVariableIndices) {
720 constraint = constraint - choiceVariable;
721 }
722 }
723 }
724 constraint = constraint <= solver.getConstant(1);
725
726 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
727 ++numberOfConstraintsCreated;
728 }
729
730 ++choiceVariableIterator;
731 }
732
733 // For all states assert that there is either a selected incoming transition in the subsystem or the
734 // state is the chosen initial state if there is one selected action in the current state.
735 storm::expressions::Expression constraint = solver.getConstant(0);
736
737 for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(state)) {
738 constraint = constraint + choiceVariable;
739 }
740
741 // Compute the set of predecessors.
742 std::unordered_set<uint_fast64_t> predecessors;
743 for (auto const& predecessorEntry : backwardTransitions.getRow(state)) {
744 if (state != predecessorEntry.getColumn()) {
745 predecessors.insert(predecessorEntry.getColumn());
746 }
747 }
748
749 for (auto predecessor : predecessors) {
750 // If the predecessor is not a relevant state, we need to skip it.
751 if (!stateInformation.relevantStates.get(predecessor)) {
752 continue;
753 }
754
755 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
756 variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
757 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
758 bool choiceTargetsCurrentState = false;
759
760 // Check if the current choice targets the current state.
761 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
762 if (state == successorEntry.getColumn()) {
763 choiceTargetsCurrentState = true;
764 break;
765 }
766 }
767
768 // If it does, we can add the choice to the sum.
769 if (choiceTargetsCurrentState) {
770 constraint = constraint - *choiceVariableIterator;
771 }
772 ++choiceVariableIterator;
773 }
774 }
775
776 // If the current state is an initial state and is selected as a successor state by the virtual
777 // initial state, then this also justifies making a choice in the current state.
778 if (mdp.getStates("init").get(state)) {
779 constraint = constraint - variableInformation.initialStateToChoiceVariableMap.at(state);
780 }
781 constraint = constraint <= solver.getConstant(0);
782
783 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
784 ++numberOfConstraintsCreated;
785 }
786
787 // Assert that at least one initial state selects at least one action.
788 storm::expressions::Expression constraint = solver.getConstant(0);
789 for (auto initialState : mdp.getStates("init")) {
790 for (auto const& choiceVariable : variableInformation.stateToChoiceVariablesMap.at(initialState)) {
791 constraint = constraint + choiceVariable;
792 }
793 }
794 constraint = constraint >= solver.getConstant(1);
795 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
796 ++numberOfConstraintsCreated;
797
798 // Add constraints that ensure at least one choice is selected that targets a psi state.
799 constraint = solver.getConstant(0);
800 std::unordered_set<uint_fast64_t> predecessors;
801 for (auto psiState : psiStates) {
802 // Compute the set of predecessors.
803 for (auto const& predecessorEntry : backwardTransitions.getRow(psiState)) {
804 if (psiState != predecessorEntry.getColumn()) {
805 predecessors.insert(predecessorEntry.getColumn());
806 }
807 }
808 }
809
810 for (auto predecessor : predecessors) {
811 // If the predecessor is not a relevant state, we need to skip it.
812 if (!stateInformation.relevantStates.get(predecessor)) {
813 continue;
814 }
815
816 std::list<storm::expressions::Variable>::const_iterator choiceVariableIterator =
817 variableInformation.stateToChoiceVariablesMap.at(predecessor).begin();
818 for (auto relevantChoice : choiceInformation.relevantChoicesForRelevantStates.at(predecessor)) {
819 bool choiceTargetsPsiState = false;
820
821 // Check if the current choice targets the current state.
822 for (auto const& successorEntry : mdp.getTransitionMatrix().getRow(relevantChoice)) {
823 if (psiStates.get(successorEntry.getColumn())) {
824 choiceTargetsPsiState = true;
825 break;
826 }
827 }
828
829 // If it does, we can add the choice to the sum.
830 if (choiceTargetsPsiState) {
831 constraint = constraint + *choiceVariableIterator;
832 }
833 ++choiceVariableIterator;
834 }
835 }
836 constraint = constraint >= solver.getConstant(1);
837
838 solver.addConstraint("SchedulerCuts" + std::to_string(numberOfConstraintsCreated), constraint);
839 ++numberOfConstraintsCreated;
840
841 return numberOfConstraintsCreated;
842 }

859 static void buildConstraintSystem(storm::solver::LpSolver<double>& solver, storm::models::sparse::Mdp<T> const& mdp,
860 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets, storm::storage::BitVector const& psiStates,
861 StateInformation const& stateInformation, ChoiceInformation const& choiceInformation,
862 VariableInformation const& variableInformation, double probabilityThreshold, bool strictBound,
863 bool includeSchedulerCuts = false) {
864 // Assert that the reachability probability in the subsystem exceeds the given threshold.
865 uint_fast64_t numberOfConstraints = assertProbabilityGreaterThanThreshold(solver, variableInformation, probabilityThreshold, strictBound);
866 STORM_LOG_DEBUG("Asserted that reachability probability exceeds threshold.");
867
868 // Add constraints that assert the policy takes at most one action in each state.
869 numberOfConstraints += assertValidPolicy(solver, stateInformation, variableInformation);
870 STORM_LOG_DEBUG("Asserted that policy is valid.");
871
872 // Add constraints that assert the labels that belong to some taken choices are taken as well.
873 numberOfConstraints += assertChoicesImplyLabels(solver, mdp, labelSets, stateInformation, choiceInformation, variableInformation);
874 STORM_LOG_DEBUG("Asserted that labels implied by choices are taken.");
875
876 // Add constraints that encode that the reachability probability from states which do not pick any action
877 // is zero.
878 numberOfConstraints += assertZeroProbabilityWithoutChoice(solver, stateInformation, variableInformation);
879 STORM_LOG_DEBUG("Asserted that reachability probability is zero if no choice is taken.");
880
881 // Add constraints that encode the reachability probabilities for states.
882 numberOfConstraints += assertReachabilityProbabilities(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
883 STORM_LOG_DEBUG("Asserted constraints for reachability probabilities.");
884
885 // Add constraints that ensure the reachability of an unproblematic state from each problematic state.
886 numberOfConstraints += assertUnproblematicStateReachable(solver, mdp, stateInformation, choiceInformation, variableInformation);
887 STORM_LOG_DEBUG("Asserted that unproblematic state reachable from problematic states.");
888
889 // Add constraints that express that certain labels are already known to be taken.
890 numberOfConstraints += assertKnownLabels(solver, choiceInformation, variableInformation);
891 STORM_LOG_DEBUG("Asserted known labels are taken.");
892
893 // If required, assert additional constraints that reduce the number of possible policies.
894 if (includeSchedulerCuts) {
895 numberOfConstraints += assertSchedulerCuts(solver, mdp, psiStates, stateInformation, choiceInformation, variableInformation);
896 STORM_LOG_DEBUG("Asserted scheduler cuts.");
897 }
898
899 // Finally, we can tell the solver to incorporate the latest changes.
900 solver.update();
901
902 STORM_LOG_INFO("Successfully created " << numberOfConstraints << " MILP constraints.");
903 }

911 static storm::storage::FlatSet<uint_fast64_t> getUsedLabelsInSolution(storm::solver::LpSolver<double> const& solver,
912 VariableInformation const& variableInformation) {
913 storm::storage::FlatSet<uint_fast64_t> result;
914
915 for (auto const& labelVariablePair : variableInformation.labelToVariableMap) {
916 bool labelTaken = solver.getBinaryValue(labelVariablePair.second);
917
918 if (labelTaken) {
919 result.insert(labelVariablePair.first);
920 }
921 }
922 return result;

926 static storm::storage::FlatSet<uint_fast64_t> getMinimalLabelSet(Environment const& env, storm::models::sparse::Mdp<T> const& mdp,
927 std::vector<storm::storage::FlatSet<uint_fast64_t>> const& labelSets,
928 storm::storage::BitVector const& phiStates, storm::storage::BitVector const& psiStates,
929 double probabilityThreshold, bool strictBound, bool checkThresholdFeasible = false,
930 bool includeSchedulerCuts = false) {
931 // (0) Check whether the label sets are valid
932 STORM_LOG_THROW(mdp.getNumberOfChoices() == labelSets.size(), storm::exceptions::InvalidArgumentException,
933 "The given number of labels does not match the number of choices.");
934
935 // (1) Check whether its possible to exceed the threshold if checkThresholdFeasible is set.
936 double maximalReachabilityProbability = 0;
937 if (checkThresholdFeasible) {
938 storm::modelchecker::helper::SparseMdpPrctlHelper<T> modelcheckerHelper;
939 std::vector<T> result = std::move(
940 modelcheckerHelper
941 .computeUntilProbabilities(env, false, mdp.getTransitionMatrix(), mdp.getBackwardTransitions(), phiStates, psiStates, false, false)
942 .values);
943 for (auto state : mdp.getInitialStates()) {
944 maximalReachabilityProbability = std::max(maximalReachabilityProbability, result[state]);
945 }
946 STORM_LOG_THROW((strictBound && maximalReachabilityProbability >= probabilityThreshold) ||
947 (!strictBound && maximalReachabilityProbability > probabilityThreshold),
948 storm::exceptions::InvalidArgumentException,
949 "Given probability threshold " << probabilityThreshold << " can not be " << (strictBound ? "achieved" : "exceeded")
950 << " in model with maximal reachability probability of " << maximalReachabilityProbability << ".");
951 std::cout << "\nMaximal reachability in model is " << maximalReachabilityProbability << ".\n\n";
952 }
953
954 // (2) Identify relevant and problematic states.
955 StateInformation stateInformation = determineRelevantAndProblematicStates(mdp, phiStates, psiStates);
956
957 // (3) Determine sets of relevant labels and problematic choices.
958 ChoiceInformation choiceInformation = determineRelevantAndProblematicChoices(mdp, labelSets, stateInformation, psiStates);
959
960 // (4) Encode resulting system as MILP problem.
961 std::shared_ptr<storm::solver::LpSolver<double>> solver = storm::utility::solver::getLpSolver<double>("MinimalLabelSetCounterexample");
962
963 // (4.1) Create variables.
964 VariableInformation variableInformation = createVariables(*solver, mdp, stateInformation, choiceInformation);
965
966 // (4.2) Construct constraint system.
967 buildConstraintSystem(*solver, mdp, labelSets, psiStates, stateInformation, choiceInformation, variableInformation, probabilityThreshold, strictBound,
968 includeSchedulerCuts);
969
970 // (4.3) Optimize the model.
971 solver->optimize();
972
973 // (4.4) Read off result from variables.
974 storm::storage::FlatSet<uint_fast64_t> usedLabelSet = getUsedLabelsInSolution(*solver, variableInformation);
975 usedLabelSet.insert(choiceInformation.knownLabels.begin(), choiceInformation.knownLabels.end());
976
977 // (5) Return result.
978 return usedLabelSet;
979 }

989 static std::shared_ptr<HighLevelCounterexample> computeCounterexample(Environment const& env, storm::storage::SymbolicModelDescription const& symbolicModel,
990 storm::models::sparse::Mdp<T> const& mdp,
991 std::shared_ptr<storm::logic::Formula const> const& formula) {
992 std::cout << "\nGenerating minimal label counterexample for formula " << *formula << '\n';
993
994 // Check whether there are choice origins available
995 STORM_LOG_THROW(mdp.hasChoiceOrigins(), storm::exceptions::InvalidArgumentException,
996 "Restriction to minimal command set is impossible for model without choice origns.");
997 STORM_LOG_THROW(mdp.getChoiceOrigins()->isPrismChoiceOrigins() || mdp.getChoiceOrigins()->isJaniChoiceOrigins(),
998 storm::exceptions::InvalidArgumentException,
999 "Restriction to command set is impossible for model without PRISM or JANI choice origins.");
1000
1001 STORM_LOG_THROW(formula->isProbabilityOperatorFormula(), storm::exceptions::InvalidPropertyException,
1002 "Counterexample generation does not support this kind of formula. Expecting a probability operator as the outermost formula element.");
1003 storm::logic::ProbabilityOperatorFormula const& probabilityOperator = formula->asProbabilityOperatorFormula();
1004 STORM_LOG_THROW(probabilityOperator.hasBound(), storm::exceptions::InvalidPropertyException,
1005 "Counterexample generation only supports bounded formulas.");
1006 storm::logic::ComparisonType comparisonType = probabilityOperator.getComparisonType();
1007 STORM_LOG_THROW(comparisonType == storm::logic::ComparisonType::Less || comparisonType == storm::logic::ComparisonType::LessEqual,
1008 storm::exceptions::InvalidPropertyException, "Counterexample generation only supports formulas with an upper probability bound.");
1009 STORM_LOG_THROW(probabilityOperator.getSubformula().isUntilFormula() || probabilityOperator.getSubformula().isEventuallyFormula(),
1010 storm::exceptions::InvalidPropertyException, "Path formula is required to be of the form 'phi U psi' for counterexample generation.");
1011
1012 bool strictBound = comparisonType == storm::logic::ComparisonType::Less;
1013 double threshold = probabilityOperator.getThresholdAs<double>();
1014
1015 storm::storage::BitVector phiStates;
1016 storm::storage::BitVector psiStates;
1017 storm::modelchecker::SparsePropositionalModelChecker<storm::models::sparse::Mdp<T>> modelchecker(mdp);
1018
1019 if (probabilityOperator.getSubformula().isUntilFormula()) {
1020 storm::logic::UntilFormula const& untilFormula = probabilityOperator.getSubformula().asUntilFormula();
1021
1022 std::unique_ptr<storm::modelchecker::CheckResult> leftResult = modelchecker.check(env, untilFormula.getLeftSubformula());
1023 std::unique_ptr<storm::modelchecker::CheckResult> rightResult = modelchecker.check(env, untilFormula.getRightSubformula());
1024
1025 storm::modelchecker::ExplicitQualitativeCheckResult const& leftQualitativeResult = leftResult->asExplicitQualitativeCheckResult();
1026 storm::modelchecker::ExplicitQualitativeCheckResult const& rightQualitativeResult = rightResult->asExplicitQualitativeCheckResult();
1027
1028 phiStates = leftQualitativeResult.getTruthValuesVector();
1029 psiStates = rightQualitativeResult.getTruthValuesVector();
1030 } else if (probabilityOperator.getSubformula().isEventuallyFormula()) {
1031 storm::logic::EventuallyFormula const& eventuallyFormula = probabilityOperator.getSubformula().asEventuallyFormula();
1032
1033 std::unique_ptr<storm::modelchecker::CheckResult> subResult = modelchecker.check(env, eventuallyFormula.getSubformula());
1034
1035 storm::modelchecker::ExplicitQualitativeCheckResult const& subQualitativeResult = subResult->asExplicitQualitativeCheckResult();
1036
1037 phiStates = storm::storage::BitVector(mdp.getNumberOfStates(), true);
1038 psiStates = subQualitativeResult.getTruthValuesVector();
1039 }
1040
1041 // Obtain the label sets for each choice.
1042 std::vector<storm::storage::FlatSet<uint_fast64_t>> labelSets(mdp.getNumberOfChoices());
1043 if (mdp.getChoiceOrigins()->isPrismChoiceOrigins()) {
1044 storm::storage::sparse::PrismChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asPrismChoiceOrigins();
1045 for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1046 labelSets[choice] = choiceOrigins.getCommandSet(choice);
1047 }
1048 } else {
1049 storm::storage::sparse::JaniChoiceOrigins const& choiceOrigins = mdp.getChoiceOrigins()->asJaniChoiceOrigins();
1050
1051 // The choice origins are known to be JANI ones at this point.
1052 for (uint_fast64_t choice = 0; choice < mdp.getNumberOfChoices(); ++choice) {
1053 labelSets[choice] = choiceOrigins.getEdgeIndexSet(choice);
1054 }
1055 }
1056
1057 // Delegate the actual computation work to the function of equal name.
1058 auto startTime = std::chrono::high_resolution_clock::now();
1059 storm::storage::FlatSet<uint_fast64_t> usedLabelSet =
1060 getMinimalLabelSet(env, mdp, labelSets, phiStates, psiStates, threshold, strictBound, true,
1061 storm::settings::getModule<storm::settings::modules::CounterexampleGeneratorSettings>().isUseSchedulerCutsSet());
1062 auto endTime = std::chrono::high_resolution_clock::now();
1063 std::cout << "\nComputed minimal command set of size " << usedLabelSet.size() << " in "
1064 << std::chrono::duration_cast<std::chrono::milliseconds>(endTime - startTime).count() << "ms.\n";
1065
1066 if (symbolicModel.isPrismProgram()) {
1067 return std::make_shared<HighLevelCounterexample>(symbolicModel.asPrismProgram().restrictCommands(usedLabelSet));
1068 } else {
1069 STORM_LOG_ASSERT(symbolicModel.isJaniModel(), "Unknown symbolic model description type.");
1070 return std::make_shared<HighLevelCounterexample>(symbolicModel.asJaniModel().restrictEdges(usedLabelSet));
1071 }
1072 }
1073};
