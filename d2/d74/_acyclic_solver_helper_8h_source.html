18template<typename ValueType>
19boost::optional<std::vector<uint64_t>> computeTopologicalGroupOrdering(storm::storage::SparseMatrix<ValueType> const& matrix) {
20 uint64_t numGroups = matrix.getRowGroupCount();
21 bool orderedMatrixRequired = false;
22 std::vector<uint64_t> result;
23 result.reserve(numGroups);
24 storm::storage::BitVector processed(numGroups, false);
25 storm::storage::BitVector visited(numGroups, false);
26 std::vector<uint64_t> stack;
27
28 // It's more likely that states without a successor are located at the end (due to the way we build the model).
29 // We therefore process the matrix in reverse order.
30 uint64_t startState = numGroups;
31 while (startState > 0) {
32 --startState;
33 // skip already processed states
34 if (processed.get(startState))
35 continue;
36
37 // Now do a dfs from start state.
38 stack.push_back(startState);
39 while (!stack.empty()) {
40 uint64_t current = stack.back();
41 if (visited.get(current)) {
42 // We are backtracking, so add this state now
43 // It might be that we already processed this state before. This can happen, e.g., if this state has two direct predecessors, A and B, and
44 // A is also a predecessor of B. Then, this node gets inserted into the stack two times: when analyzing A and B.
45 if (!processed.get(current)) {
46 result.push_back(current);
47 processed.set(current);
48 }
49 stack.pop_back();
50 } else {
51 visited.set(current);
52 for (auto const& entry : matrix.getRowGroup(current)) {
53 if (!processed.get(entry.getColumn()) && !storm::utility::isZero(entry.getValue())) {
54 orderedMatrixRequired = true;
55 STORM_LOG_THROW(!visited.get(entry.getColumn()), storm::exceptions::UnmetRequirementException, "The model is not acyclic.");
56 stack.push_back(entry.getColumn());
57 }
58 }
59 // If there are no successors to process, we will add the current state to the result in the next iteration.
60 }
61 }
62 }
63 // we will do backwards iterations, so the order has to be reversed
64 if (orderedMatrixRequired) {
65 std::reverse(result.begin(), result.end());
66 STORM_LOG_ASSERT(result.size() == matrix.getRowGroupCount(), "Result vector has an unexpected amount of entries.");
67 return result;
68 } else {
69 return boost::none;
70 }
71}

75template<typename ValueType>
76storm::storage::SparseMatrix<ValueType> createReorderedMatrix(storm::storage::SparseMatrix<ValueType> const& matrix,
77 std::vector<uint64_t> const& newToOrigIndexMap,
78 std::vector<std::pair<uint64_t, ValueType>>& bFactors) {
79 std::vector<uint64_t> origToNewMap(newToOrigIndexMap.size(), std::numeric_limits<uint64_t>::max());
80 for (uint64_t i = 0; i < newToOrigIndexMap.size(); ++i) {
81 origToNewMap[newToOrigIndexMap[i]] = i;
82 }
83
84 bool hasRowGrouping = !matrix.hasTrivialRowGrouping();
85 storm::storage::SparseMatrixBuilder<ValueType> builder(matrix.getRowCount(), matrix.getColumnCount(), matrix.getNonzeroEntryCount(), false, hasRowGrouping,
86 hasRowGrouping ? matrix.getRowGroupCount() : static_cast<uint64_t>(0));
87 uint64_t newRow = 0;
88 for (uint64_t newRowGroup = 0; newRowGroup < newToOrigIndexMap.size(); ++newRowGroup) {
89 auto const& origRowGroup = newToOrigIndexMap[newRowGroup];
90 if (hasRowGrouping) {
91 builder.newRowGroup(newRow);
92 }
93 for (uint64_t origRow = matrix.getRowGroupIndices()[origRowGroup]; origRow < matrix.getRowGroupIndices()[origRowGroup + 1]; ++origRow) {
94 for (auto const& entry : matrix.getRow(origRow)) {
95 if (storm::utility::isZero(entry.getValue())) {
96 continue;
97 }
98 if (entry.getColumn() == origRowGroup) {
99 if (storm::utility::isOne(entry.getValue())) {
100 // a one selfloop can only mean that there is never a non-zero value at the b vector for the current row.
101 // Let's "assert" this by considering infinity. (This is necessary to avoid division by zero)
102 bFactors.emplace_back(newRow, storm::utility::infinity<ValueType>());
103 }
104 ValueType factor = storm::utility::one<ValueType>() / (storm::utility::one<ValueType>() - entry.getValue());
105 bFactors.emplace_back(newRow, factor);
106 }
107 builder.addNextValue(newRow, origToNewMap[entry.getColumn()], entry.getValue());
108 }
109 ++newRow;
110 }
111 }
112 auto result = builder.build(matrix.getRowCount(), matrix.getColumnCount(), matrix.getRowGroupCount());
113 // apply the bFactors to the relevant rows
114 for (auto const& bFactor : bFactors) {
115 STORM_LOG_ASSERT(!storm::utility::isInfinity(bFactor.second) || storm::utility::isZero(result.getRowSum(bFactor.first)),
116 "The input matrix does not seem to be probabilistic.");
117 for (auto& entry : result.getRow(bFactor.first)) {
118 entry.setValue(entry.getValue() * bFactor.second);
119 }
120 }
121 STORM_LOG_DEBUG("Reordered " << matrix.getDimensionsAsString() << " with " << bFactors.size() << " selfloop entries for acyclic solving.");
122 return result;
123}
