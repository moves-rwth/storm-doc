228 template<typename ValueType, typename SolutionType>
229 std::unique_ptr<MinMaxLinearEquationSolver<ValueType, SolutionType>> GeneralMinMaxLinearEquationSolverFactory<ValueType, SolutionType>::create(
230  Environment const& env) const {
231  std::unique_ptr<MinMaxLinearEquationSolver<ValueType, SolutionType>> result;
232  if constexpr (std::is_same_v<ValueType, storm::Interval>) {
233  // TODO: consider robust minMax solver methods and corresponding entries in the environment.
234  return std::make_unique<IterativeMinMaxLinearEquationSolver<ValueType, SolutionType>>();
235  } else {
236  // TODO some minmax linear equation solvers only support SolutionType == ValueType.
237  auto method = env.solver().minMax().getMethod();
238  if (method == MinMaxMethod::ValueIteration || method == MinMaxMethod::PolicyIteration || method == MinMaxMethod::RationalSearch ||
239  method == MinMaxMethod::IntervalIteration || method == MinMaxMethod::SoundValueIteration || method == MinMaxMethod::OptimisticValueIteration ||
240  method == MinMaxMethod::ViToPi) {
241  result = std::make_unique<IterativeMinMaxLinearEquationSolver<ValueType, SolutionType>>(
242  std::make_unique<GeneralLinearEquationSolverFactory<ValueType>>());
243  } else if (method == MinMaxMethod::Topological) {
244  result = std::make_unique<TopologicalMinMaxLinearEquationSolver<ValueType>>();
245  } else if (method == MinMaxMethod::LinearProgramming) {
246  result = std::make_unique<LpMinMaxLinearEquationSolver<ValueType>>(storm::utility::solver::getLpSolverFactory<ValueType>());
247  } else if (method == MinMaxMethod::Acyclic) {
248  result = std::make_unique<AcyclicMinMaxLinearEquationSolver<ValueType>>();
249  } else {
250  STORM_LOG_THROW(false, storm::exceptions::InvalidSettingsException, "Unsupported technique.");
251  }
252  result->setRequirementsChecked(this->isRequirementsCheckedSet());
253  return result;
254  }
