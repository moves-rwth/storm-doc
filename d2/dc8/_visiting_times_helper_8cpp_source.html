12 template<typename ValueType>
13 ValueType VisitingTimesHelper<ValueType>::computeMecTraversalLowerBound(storm::storage::MaximalEndComponent const& mec,
14  storm::storage::SparseMatrix<ValueType> const& transitions,
15  bool assumeOptimalTransitionProbabilities) {
16  STORM_LOG_ASSERT(mec.size() > 0, "empty mec not expected");
17  auto res = storm::utility::one<ValueType>();
18  if (mec.size() == 1) {
19  return res;
20  }
21  // Whenever s' is reachable from s, there is at least one finite path that visits each state of the mec at most once.
22  // We compute a naive lower bound for the probability of such a single path using the product of the smallest probabilities occurring at each state
23  // (Excluding self-loop probabilities)
24  for (auto const& stateChoices : mec) {
25  storm::utility::Minimum<ValueType> v;
26  auto const& s = stateChoices.first;
27  for (auto const& c : stateChoices.second) {
28  auto row = transitions.getRow(c);
29  if (assumeOptimalTransitionProbabilities) {
30  // Count the number of non-selfloop entries and assume a uniform distribution (which gives us the largest minimal probability)
31  auto numEntries = row.getNumberOfEntries();
32  for (auto const& entry : row) {
33  if (entry.getColumn() > s) {
34  if (entry.getColumn() == s) {
35  --numEntries;
36  }
37  break;
38  }
39  }
40  if (numEntries > 0) {
41  v &= storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType>(numEntries);
42  }
43  } else {
44  // actually determine the minimal probability
45  for (auto const& entry : row) {
46  if (entry.getColumn() != s && !storm::utility::isZero(entry.getValue())) {
47  v &= entry.getValue();
48  }
49  }
50  }
51  }
52  STORM_LOG_ASSERT(!v.empty(), "self-loop state in non-singleton mec?");
53  res *= *v;
54  }
55  return res;
56 }

58 template<typename ValueType>
59 ValueType VisitingTimesHelper<ValueType>::computeMecVisitsUpperBound(storm::storage::MaximalEndComponent const& mec,
60  storm::storage::SparseMatrix<ValueType> const& transitions,
61  bool assumeOptimalTransitionProbabilities) {
62  auto const one = storm::utility::one<ValueType>();
63  auto const traversalLowerBound = computeMecTraversalLowerBound(mec, transitions, assumeOptimalTransitionProbabilities);
64  if (assumeOptimalTransitionProbabilities) {
65  // We assume that the probability to go back to the MEC using an exiting choice is zero.
66  return one / traversalLowerBound;
67  } else {
68  // compute the largest probability to go back to the MEC when using an exiting choice
69  storm::utility::Maximum<ValueType> q(storm::utility::zero<ValueType>());
70  for (auto const& stateChoices : mec) {
71  for (auto c : transitions.getRowGroupIndices(stateChoices.first)) {
72  if (stateChoices.second.contains(c)) {
73  continue; // not an exit choice!
74  }
75  auto choiceValue = storm::utility::zero<ValueType>();
76  for (auto const& entry : transitions.getRow(c)) {
77  if (mec.containsState(entry.getColumn())) {
78  choiceValue += entry.getValue();
79  }
80  }
81  q &= choiceValue;
82  }
83  }
84  return one / (traversalLowerBound * (one - *q));
85  }
86 }

134

138  auto const& quotientState = quotientData.oldToNewStateMapping[state];
139  if (quotientState < quotientData.matrix.getRowGroupCount()) {
140  result.push_back(quotientUpperBounds.at(quotientState));
141  } else {
142  result.push_back(-storm::utility::one<ValueType>()); // not in given subsystem;
143  }
144  }
