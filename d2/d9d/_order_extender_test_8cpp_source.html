31  model = simplifier.getSimplifiedModel();
32

42  ASSERT_EQ(model->getNumberOfTransitions(), 195);
43 
44  auto vars = storm::models::sparse::getProbabilityParameters(*model);

46 
47  auto extender = storm::analysis::OrderExtender<storm::RationalFunction, double>(model, formulas[0]);
48  auto monRes = new storm::analysis::MonotonicityResult<typename storm::analysis::OrderExtender<storm::RationalFunction, double>::VariableType>;
