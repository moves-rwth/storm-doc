30template<typename ValueType>
31bool DiscreteTimePrismProgramSimulator<ValueType>::step(uint64_t actionNumber) {
32 uint32_t nextState = behavior.getChoices()[actionNumber].sampleFromDistribution(generator.random());
33 lastActionRewards = behavior.getChoices()[actionNumber].getRewards();
34 STORM_LOG_ASSERT(lastActionRewards.size() == stateGenerator->getNumberOfRewardModels(), "Reward vector should have as many rewards as model.");
35 currentState = idToState[nextState];
36 // TODO we do not need to do this in every step!
37 clearStateCaches();
38 explore();
39 return true;
40}

42template<typename ValueType>
43bool DiscreteTimePrismProgramSimulator<ValueType>::explore() {
44 // Load the current state into the next state generator.
45 stateGenerator->load(currentState);
46 // TODO: This low-level code currently expands all actions, while this is not necessary.
47 // However, using the next state generator ensures compatibliity with the model generator.
48 behavior = stateGenerator->expand(stateToIdCallback);
49 STORM_LOG_ASSERT(behavior.getStateRewards().size() == lastActionRewards.size(), "Reward vectors should have same length.");
50 for (uint64_t i = 0; i < behavior.getStateRewards().size(); i++) {
51 lastActionRewards[i] += behavior.getStateRewards()[i];
52 }
53 return true;
54}

56template<typename ValueType>
57bool DiscreteTimePrismProgramSimulator<ValueType>::isSinkState() const {
58 if (behavior.empty()) {
59 return true;
60 }
61 std::set<uint32_t> successorIds;
62 for (Choice<ValueType, uint32_t> const& choice : behavior.getChoices()) {
63 for (auto it = choice.begin(); it != choice.end(); ++it) {
64 successorIds.insert(it->first);
65 if (successorIds.size() > 1) {
66 return false;
67 }
68 }
69 }
70 if (idToState.at(*(successorIds.begin())) == currentState) {
71 return true;
72 }
73 return false;
74}

155uint32_t DiscreteTimePrismProgramSimulator<ValueType>::getOrAddStateIndex(generator::CompressedState const& state) {
156 uint32_t newIndex = static_cast<uint32_t>(stateToId.size());
157
158 // Check, if the state was already registered.
159 std::pair<uint32_t, std::size_t> actualIndexBucketPair = stateToId.findOrAddAndGetBucket(state, newIndex);
160
161 uint32_t actualIndex = actualIndexBucketPair.first;
162 if (actualIndex == newIndex) {
163 idToState[actualIndex] = state;
164 }
165 return actualIndex;
