44 bool DiscreteTimePrismProgramSimulator<ValueType>::explore() {

57 template<typename ValueType>
58 bool DiscreteTimePrismProgramSimulator<ValueType>::isSinkState() const {
59  if (behavior.empty()) {
60  return true;
61  }
62  std::set<uint32_t> successorIds;
63  for (Choice<ValueType, uint32_t> const& choice : behavior.getChoices()) {
64  for (auto it = choice.begin(); it != choice.end(); ++it) {
65  successorIds.insert(it->first);
66  if (successorIds.size() > 1) {
67  return false;
68  }
69  }
70  }
71  if (idToState.at(*(successorIds.begin())) == currentState) {
72  return true;
73  }
74  return false;
