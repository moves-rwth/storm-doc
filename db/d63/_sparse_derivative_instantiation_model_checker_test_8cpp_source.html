104 void SparseDerivativeInstantiationModelCheckerTest<TestType>::testModel(std::shared_ptr<storm::models::sparse::Dtmc<storm::RationalFunction>> dtmc,
105  std::vector<std::shared_ptr<const storm::logic::Formula>> formulas,
106  storm::RationalFunction reachabilityFunction) {
107  auto formulaWithoutBound = std::make_shared<storm::logic::ProbabilityOperatorFormula>(
108  formulas[0]->asProbabilityOperatorFormula().getSubformula().asSharedPointer(), storm::logic::OperatorInformation(boost::none, boost::none));
109 
110  auto parameters = storm::models::sparse::getProbabilityParameters(*dtmc);
111  storm::derivative::SparseDerivativeInstantiationModelChecker<storm::RationalFunction, typename TestType::ConstantType> derivativeModelChecker(*dtmc);
112 
113  std::map<VariableType<storm::RationalFunction>, storm::RationalFunction> derivatives;
114  for (auto const& parameter : parameters) {
115  derivatives[parameter] = reachabilityFunction.derivative(parameter);
116  }
117 
118  // Generate test cases.
119  std::vector<Instantiation<storm::RationalFunction>> testInstantiations;
120  Instantiation<storm::RationalFunction> emptyInstantiation;
121  testInstantiations.push_back(emptyInstantiation);
122  for (auto const& param : parameters) {
123  std::vector<Instantiation<storm::RationalFunction>> newInstantiations;
124  for (auto point : testInstantiations) {
125  for (typename TestType::ConstantType x = storm::utility::convertNumber<ConstantType>(1e-6); x <= 1;
126  x += (1 - storm::utility::convertNumber<ConstantType>(1e-6)) / 10) {
127  std::map<VariableType<storm::RationalFunction>, CoefficientType<storm::RationalFunction>> newMap(point);
128  newMap[param] = storm::utility::convertNumber<CoefficientType<storm::RationalFunction>>(x);
129  newInstantiations.push_back(newMap);
130  }
131  }
132  testInstantiations = newInstantiations;
133  }
134 
135  // The test cases we are going to study. Left are the actual instantiations, right are the maps
136  // for the results (which happen to share the same type).
137  std::map<Instantiation<storm::RationalFunction>, ResultMap<storm::RationalFunction>> testCases;
138  for (auto const& instantiation : testInstantiations) {
139  ResultMap<storm::RationalFunction> resultMap;
140  for (auto const& entry : instantiation) {
141  auto parameter = entry.first;
142  auto derivativeWrtParameter = derivatives[parameter];
143  typename TestType::ConstantType evaluatedDerivative =
144  storm::utility::convertNumber<typename TestType::ConstantType>(derivativeWrtParameter.evaluate(instantiation));
145  resultMap[parameter] = evaluatedDerivative;
146  }
147  testCases[instantiation] = resultMap;
148  }
149 
150  auto checkTask = storm::modelchecker::CheckTask<storm::logic::Formula, ValueType>(*formulaWithoutBound);
151  derivativeModelChecker.specifyFormula(env(), checkTask);
152 
153  for (auto const& testCase : testCases) {
154  Instantiation<ValueType> instantiation = testCase.first;
155  for (auto const& position : instantiation) {
156  auto parameter = position.first;
157  auto parameterValue = position.second;
158  auto expectedResult = testCase.second.at(parameter);
159 
160  auto derivative = derivativeModelChecker.check(env(), instantiation, parameter);
161  ASSERT_NEAR(storm::utility::convertNumber<double>(derivative->getValueVector()[0]), storm::utility::convertNumber<double>(expectedResult), 1e-6)
162  << instantiation;
163  }
164  }
165 }
