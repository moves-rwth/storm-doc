402  template <typename SparseModelType, typename ConstantType>
403  void SparseDtmcParameterLiftingModelChecker<SparseModelType, ConstantType>::computeRegionSplitEstimates(std::vector<ConstantType> const& quantitativeResult, std::vector<uint_fast64_t> const& schedulerChoices, storm::storage::ParameterRegion<ValueType> const& region, storm::solver::OptimizationDirection const& dirForParameters) {
404  std::map<VariableType, double> deltaLower, deltaUpper;
405  for (auto const& p : region.getVariables()) {
406  deltaLower.insert(std::make_pair(p, 0.0));
407  deltaUpper.insert(std::make_pair(p, 0.0));
408  }
409  auto const& choiceValuations = parameterLifter->getRowLabels();
410  auto const& matrix = parameterLifter->getMatrix();
411  auto const& vector = parameterLifter->getVector();
412 
413  std::vector<ConstantType> stateResults;
414  for (uint64_t state = 0; state < schedulerChoices.size(); ++state) {
415  uint64_t rowOffset = matrix.getRowGroupIndices()[state];
416  uint64_t optimalChoice = schedulerChoices[state];
417  auto const& optimalChoiceVal = choiceValuations[rowOffset + optimalChoice];
418  assert(optimalChoiceVal.getUnspecifiedParameters().empty());
419  stateResults.clear();
420  for (uint64_t row = rowOffset; row < matrix.getRowGroupIndices()[state + 1]; ++row) {
421  stateResults.push_back(matrix.multiplyRowWithVector(row, quantitativeResult) + vector[row]);
422  }
423  // Do this twice, once for upperbound once for lowerbound
424  bool checkUpperParameters = false;
425  do {
426  auto const& consideredParameters = checkUpperParameters ? optimalChoiceVal.getUpperParameters() : optimalChoiceVal.getLowerParameters();
427  for (auto const& p : consideredParameters) {
428  // Find the 'best' choice that assigns the parameter to the other bound
429  ConstantType bestValue = 0;
430  bool foundBestValue = false;
431  for (uint64_t choice = 0; choice < stateResults.size(); ++choice) {
432  if (choice != optimalChoice) {
433  auto const& otherBoundParsOfChoice = checkUpperParameters ? choiceValuations[rowOffset + choice].getLowerParameters() : choiceValuations[rowOffset + choice].getUpperParameters();
434  if (otherBoundParsOfChoice.find(p) != otherBoundParsOfChoice.end()) {
435  ConstantType const& choiceValue = stateResults[choice];
436  if (!foundBestValue || (storm::solver::minimize(dirForParameters) ? choiceValue < bestValue : choiceValue > bestValue)) {
437  foundBestValue = true;
438  bestValue = choiceValue;
439  }
440  }
441  }
442  }
443  auto optimal = storm::utility::convertNumber<double>(stateResults[optimalChoice]);
444  auto diff = optimal - storm::utility::convertNumber<double>(bestValue);
445  if (foundBestValue) {
446  if (checkUpperParameters) {
447  deltaLower[p] += std::abs(diff);
448  } else {
449  deltaUpper[p] += std::abs(diff);
450  }
451  }
452  }
453  checkUpperParameters = !checkUpperParameters;
454  } while (checkUpperParameters);
455  }
456 
457  regionSplitEstimates.clear();
458  useRegionSplitEstimates = false;
459  for (auto const& p : region.getVariables()) {
460  if (this->possibleMonotoneParameters.find(p) != this->possibleMonotoneParameters.end()) {
461  if (deltaLower[p] > deltaUpper[p] && deltaUpper[p] >= 0.0001) {
462  regionSplitEstimates.insert(std::make_pair(p, deltaUpper[p]));
463  useRegionSplitEstimates = true;
464  } else if (deltaLower[p] <= deltaUpper[p] && deltaLower[p] >= 0.0001) {
465  {
466  regionSplitEstimates.insert(std::make_pair(p, deltaLower[p]));
467  useRegionSplitEstimates = true;
468  }
469  }
470  }
471  }
472  // large regionsplitestimate implies that parameter p occurs as p and 1-p at least once
473  }

573  }
574  }
575  }
576  }
577  } else {
578  // Do nothing, we already checked this one
579  }
580  state = order->getNextDoneState(state);
581  }
582  auto const statesAtVariable = parameterLifter->getOccuringStatesAtVariable();
583  bool allDone = true;
584  for (auto const & entry : statesAtVariable) {
585  auto states = entry.second;
