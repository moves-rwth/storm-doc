55  }

64  // Check again for shared spares
65  sharedSpareMode = true;
66  isubdft1 = dft.getGate(index1)->independentSubDft(false, true);
67  isubdft2 = dft.getGate(index2)->independentSubDft(false, true);
68  if (isubdft1.empty() || isubdft2.empty() || isubdft1.size() != isubdft2.size()) {
69  return {};
70  }
71  } else {
72  return {};
73  }
74  }

83  bool bijectionSpareCompatible = true;
84  for (size_t elementId : isubdft1) {

89

91  bijectionSpareCompatible = false;
92  break;
93  }

95  for (size_t i = 0; i < spareLeft->nrChildren(); ++i) {
96  size_t childLeftId = spareLeft->children().at(i)->id();
97  size_t childRightId = spareRight->children().at(i)->id();
98

100  if (childLeftId == childRightId) {
101  // Ignore shared child
102  continue;
103  }
104 
105  // TODO generalize for more than one parent
106  if (spareLeft->children().at(i)->nrParents() != 1 || spareRight->children().at(i)->nrParents() != 1) {
107  bijectionSpareCompatible = false;
108  break;
109  }
110

117  }
118  }

120  break;
121  }
122  }
123  }

126  }
127  } else {
128  return bijection;
129  }
130  } // end while

135 void SymmetryFinder<ValueType>::findSymmetriesHelper(storm::dft::storage::DFT<ValueType> const& dft, std::vector<size_t> const& candidates,
136  storm::dft::storage::DFTColouring<ValueType> const& colouring,
137  std::map<size_t, std::vector<std::vector<size_t>>>& result) {
138  if (candidates.size() <= 0) {
139  return;
140  }
141 
142  std::set<size_t> foundEqClassFor;
143  for (auto it1 = candidates.cbegin(); it1 != candidates.cend(); ++it1) {
144  std::vector<std::vector<size_t>> symClass;
145  if (foundEqClassFor.count(*it1) > 0) {
146  // This item is already in a class.
147  continue;
148  }
149  auto elem1 = dft.getElement(*it1);
150  if (!elem1->hasOnlyStaticParents()) {

158  auto it2 = it1;
159  for (++it2; it2 != candidates.cend(); ++it2) {
160  auto elem2 = dft.getElement(*it2);
161  if (!elem2->hasOnlyStaticParents()) {
162  continue;
163  }
164  if (hasSeqRestriction(elem2)) {
165  continue;
166  }
167 
168  if (influencedElem1Ids == getInfluencedIds(dft, *it2)) {
169  std::map<size_t, size_t> bijection = findBijection(dft, *it1, *it2, colouring, true);
170  if (!bijection.empty()) {

172  foundEqClassFor.insert(*it2);
173  if (symClass.empty()) {
174  for (auto const& i : bijection) {
175  symClass.push_back(std::vector<size_t>({i.first}));
176  }
177  }
178  auto symClassIt = symClass.begin();
179  for (auto const& i : bijection) {
180  symClassIt->emplace_back(i.second);
181  ++symClassIt;
182  }
183  }
184  }
185  }
186
