142boost::optional<derivative::GradientDescentMethod> DerivativeSettings::methodFromString(const std::string &str) const {
143 derivative::GradientDescentMethod method;
144 if (str == "adam") {
145 method = derivative::GradientDescentMethod::ADAM;
146 } else if (str == "radam") {
147 method = derivative::GradientDescentMethod::RADAM;
148 } else if (str == "rmsprop") {
149 method = derivative::GradientDescentMethod::RMSPROP;
150 } else if (str == "plain") {
151 method = derivative::GradientDescentMethod::PLAIN;
152 } else if (str == "plain-sign") {
153 method = derivative::GradientDescentMethod::PLAIN_SIGN;
154 } else if (str == "momentum") {
155 method = derivative::GradientDescentMethod::MOMENTUM;
156 } else if (str == "momentum-sign") {
157 method = derivative::GradientDescentMethod::MOMENTUM_SIGN;
158 } else if (str == "nesterov") {
159 method = derivative::GradientDescentMethod::NESTEROV;
160 } else if (str == "nesterov-sign") {
161 method = derivative::GradientDescentMethod::NESTEROV_SIGN;
162 } else {
163 return boost::none;
164 }
165 return method;
166}

168boost::optional<derivative::GradientDescentConstraintMethod> DerivativeSettings::constraintMethodFromString(const std::string &str) const {
169 derivative::GradientDescentConstraintMethod method;
170 if (str == "project-gradient") {
171 method = derivative::GradientDescentConstraintMethod::PROJECT_WITH_GRADIENT;
172 } else if (str == "project") {
173 method = derivative::GradientDescentConstraintMethod::PROJECT;
174 } else if (str == "penalty-quadratic") {
175 method = derivative::GradientDescentConstraintMethod::PENALTY_QUADRATIC;
176 } else if (str == "barrier-logarithmic") {
177 method = derivative::GradientDescentConstraintMethod::BARRIER_LOGARITHMIC;
178 } else if (str == "barrier-infinity") {
179 method = derivative::GradientDescentConstraintMethod::BARRIER_INFINITY;
180 } else if (str == "logistic-sigmoid") {
181 method = derivative::GradientDescentConstraintMethod::LOGISTIC_SIGMOID;
182 } else {
