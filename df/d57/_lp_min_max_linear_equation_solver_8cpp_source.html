35template<typename ValueType>
36bool LpMinMaxLinearEquationSolver<ValueType>::internalSolveEquations(Environment const& env, OptimizationDirection dir, std::vector<ValueType>& x,
37 std::vector<ValueType> const& b) const {
38 STORM_LOG_THROW(env.solver().minMax().getMethod() == MinMaxMethod::LinearProgramming, storm::exceptions::InvalidEnvironmentException,
39 "This min max solver does not support the selected technique.");
40
41 // Set up the LP solver
42 std::unique_ptr<storm::solver::LpSolver<ValueType>> solver = lpSolverFactory->create("");
43 solver->setOptimizationDirection(invert(dir));
44 // Create a variable for each row group
45 std::vector<storm::expressions::Expression> variableExpressions;
46 variableExpressions.reserve(this->A->getRowGroupCount());
47 for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
48 if (this->hasLowerBound()) {
49 ValueType lowerBound = this->getLowerBound(rowGroup);
50 if (this->hasUpperBound()) {
51 ValueType upperBound = this->getUpperBound(rowGroup);
52 if (lowerBound == upperBound) {
53 // Some solvers (like glpk) don't support variables with bounds [x,x]. We therefore just use a constant instead. This should be more
54 // efficient anyways.
55 variableExpressions.push_back(solver->getConstant(lowerBound));
56 } else {
57 STORM_LOG_ASSERT(lowerBound <= upperBound,
58 "Lower Bound at row group " << rowGroup << " is " << lowerBound << " which exceeds the upper bound " << upperBound << ".");
59 variableExpressions.emplace_back(
60 solver->addBoundedContinuousVariable("x" + std::to_string(rowGroup), lowerBound, upperBound, storm::utility::one<ValueType>()));
61 }
62 } else {
63 variableExpressions.emplace_back(
64 solver->addLowerBoundedContinuousVariable("x" + std::to_string(rowGroup), lowerBound, storm::utility::one<ValueType>()));
65 }
66 } else {
67 if (this->upperBound) {
68 variableExpressions.emplace_back(
69 solver->addUpperBoundedContinuousVariable("x" + std::to_string(rowGroup), this->getUpperBound(rowGroup), storm::utility::one<ValueType>()));
70 } else {
71 variableExpressions.emplace_back(solver->addUnboundedContinuousVariable("x" + std::to_string(rowGroup), storm::utility::one<ValueType>()));
72 }
73 }
74 }
75 solver->update();
76
77 // Add a constraint for each row
78 for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
79 // The rowgroup refers to the state number
80 uint64_t rowIndex, rowGroupEnd;
81 if (this->choiceFixedForRowGroup && this->choiceFixedForRowGroup.get()[rowGroup]) {
82 rowIndex = this->A->getRowGroupIndices()[rowGroup] + this->getInitialScheduler()[rowGroup];
83 rowGroupEnd = rowIndex + 1;
84 } else {
85 rowIndex = this->A->getRowGroupIndices()[rowGroup];
86 rowGroupEnd = this->A->getRowGroupIndices()[rowGroup + 1];
87 }
88 for (; rowIndex < rowGroupEnd; ++rowIndex) {
89 auto row = this->A->getRow(rowIndex);
90 std::vector<storm::expressions::Expression> summands;
91 summands.reserve(1 + row.getNumberOfEntries());
92 summands.push_back(solver->getConstant(b[rowIndex]));
93 for (auto const& entry : row) {
94 summands.push_back(solver->getConstant(entry.getValue()) * variableExpressions[entry.getColumn()]);
95 }
96 storm::expressions::Expression rowConstraint = storm::expressions::sum(summands);
97 if (minimize(dir)) {
98 rowConstraint = variableExpressions[rowGroup] <= rowConstraint;
99 } else {
100 rowConstraint = variableExpressions[rowGroup] >= rowConstraint;
101 }
102 solver->addConstraint("", rowConstraint);
103 }
104 }
105
106 // Invoke optimization
107 solver->optimize();
108 STORM_LOG_THROW(!solver->isInfeasible(), storm::exceptions::UnexpectedException, "The MinMax equation system is infeasible.");
109 STORM_LOG_THROW(!solver->isUnbounded(), storm::exceptions::UnexpectedException, "The MinMax equation system is unbounded.");
110 STORM_LOG_THROW(solver->isOptimal(), storm::exceptions::UnexpectedException, "Unable to find optimal solution for MinMax equation system.");
111
112 // write the solution into the solution vector
113 STORM_LOG_ASSERT(x.size() == variableExpressions.size(), "Dimension of x-vector does not match number of varibales.");
114 auto xIt = x.begin();
115 auto vIt = variableExpressions.begin();
116 for (; xIt != x.end(); ++xIt, ++vIt) {
117 auto const& vBaseExpr = vIt->getBaseExpression();
118 if (vBaseExpr.isVariable()) {
119 *xIt = solver->getContinuousValue(vBaseExpr.asVariableExpression().getVariable());
120 } else {
121 STORM_LOG_ASSERT(vBaseExpr.isRationalLiteralExpression(), "Variable expression has unexpected type.");
122 *xIt = storm::utility::convertNumber<ValueType>(vBaseExpr.asRationalLiteralExpression().getValue());
123 }
124 }
125
126 // If requested, we store the scheduler for retrieval.
127 if (this->isTrackSchedulerSet()) {
128 this->schedulerChoices = std::vector<uint_fast64_t>(this->A->getRowGroupCount());
129 for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
130 if (!this->choiceFixedForRowGroup || !this->choiceFixedForRowGroup.get()[rowGroup]) {
131 // Only update scheduler choice for the states that don't have a fixed choice
132 uint64_t row = this->A->getRowGroupIndices()[rowGroup];
133 uint64_t optimalChoiceIndex = 0;
134 uint64_t currChoice = 0;
135 ValueType optimalGroupValue = this->A->multiplyRowWithVector(row, x) + b[row];
136 for (++row, ++currChoice; row < this->A->getRowGroupIndices()[rowGroup + 1]; ++row, ++currChoice) {
137 ValueType rowValue = this->A->multiplyRowWithVector(row, x) + b[row];
138 if ((minimize(dir) && rowValue < optimalGroupValue) || (maximize(dir) && rowValue > optimalGroupValue)) {
139 optimalGroupValue = rowValue;
140 optimalChoiceIndex = currChoice;
141 }
142 }
143 this->schedulerChoices.get()[rowGroup] = optimalChoiceIndex;
144 }
145 }
146 }
147 return true;
148}
