35 bool LpMinMaxLinearEquationSolver<ValueType>::internalSolveEquations(Environment const& env, OptimizationDirection dir, std::vector<ValueType>& x,
36  std::vector<ValueType> const& b) const {
37  STORM_LOG_THROW(env.solver().minMax().getMethod() == MinMaxMethod::LinearProgramming, storm::exceptions::InvalidEnvironmentException,
38  "This min max solver does not support the selected technique.");
39 
40  // Set up the LP solver
41  std::unique_ptr<storm::solver::LpSolver<ValueType>> solver = lpSolverFactory->create("");
42  solver->setOptimizationDirection(invert(dir));
43  // Create a variable for each row group
44  std::vector<storm::expressions::Expression> variableExpressions;
45  variableExpressions.reserve(this->A->getRowGroupCount());
46  for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
47  if (this->hasLowerBound()) {
48  ValueType lowerBound = this->getLowerBound(rowGroup);
49  if (this->hasUpperBound()) {
50  ValueType upperBound = this->getUpperBound(rowGroup);
51  if (lowerBound == upperBound) {
52  // Some solvers (like glpk) don't support variables with bounds [x,x]. We therefore just use a constant instead. This should be more
53  // efficient anyways.
54  variableExpressions.push_back(solver->getConstant(lowerBound));
55  } else {
56  STORM_LOG_ASSERT(lowerBound <= upperBound,
57  "Lower Bound at row group " << rowGroup << " is " << lowerBound << " which exceeds the upper bound " << upperBound << ".");
58  variableExpressions.emplace_back(
59  solver->addBoundedContinuousVariable("x" + std::to_string(rowGroup), lowerBound, upperBound, storm::utility::one<ValueType>()));
60  }
61  } else {
62  variableExpressions.emplace_back(
63  solver->addLowerBoundedContinuousVariable("x" + std::to_string(rowGroup), lowerBound, storm::utility::one<ValueType>()));
64  }
65  } else {
66  if (this->upperBound) {
67  variableExpressions.emplace_back(
68  solver->addUpperBoundedContinuousVariable("x" + std::to_string(rowGroup), this->getUpperBound(rowGroup), storm::utility::one<ValueType>()));
69  } else {
70  variableExpressions.emplace_back(solver->addUnboundedContinuousVariable("x" + std::to_string(rowGroup), storm::utility::one<ValueType>()));
71  }
72  }
73  }
74  solver->update();
75 
76  // Add a constraint for each row
77  for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
78  // The rowgroup refers to the state number
79  uint64_t rowIndex, rowGroupEnd;
80  if (this->choiceFixedForRowGroup && this->choiceFixedForRowGroup.get()[rowGroup]) {
81  rowIndex = this->A->getRowGroupIndices()[rowGroup] + this->getInitialScheduler()[rowGroup];
82  rowGroupEnd = rowIndex + 1;
83  } else {
84  rowIndex = this->A->getRowGroupIndices()[rowGroup];
85  rowGroupEnd = this->A->getRowGroupIndices()[rowGroup + 1];
86  }
87  for (; rowIndex < rowGroupEnd; ++rowIndex) {
88  auto row = this->A->getRow(rowIndex);
89  std::vector<storm::expressions::Expression> summands;
90  summands.reserve(1 + row.getNumberOfEntries());
91  summands.push_back(solver->getConstant(b[rowIndex]));
92  for (auto const& entry : row) {
93  summands.push_back(solver->getConstant(entry.getValue()) * variableExpressions[entry.getColumn()]);
94  }
95  storm::expressions::Expression rowConstraint = storm::expressions::sum(summands);
96  if (minimize(dir)) {
97  rowConstraint = variableExpressions[rowGroup] <= rowConstraint;
98  } else {
99  rowConstraint = variableExpressions[rowGroup] >= rowConstraint;
100  }
101  solver->addConstraint("", rowConstraint);
102  }
103  }
104 
105  // Invoke optimization
106  solver->optimize();
107  STORM_LOG_THROW(!solver->isInfeasible(), storm::exceptions::UnexpectedException, "The MinMax equation system is infeasible.");
108  STORM_LOG_THROW(!solver->isUnbounded(), storm::exceptions::UnexpectedException, "The MinMax equation system is unbounded.");
109  STORM_LOG_THROW(solver->isOptimal(), storm::exceptions::UnexpectedException, "Unable to find optimal solution for MinMax equation system.");
110 
111  // write the solution into the solution vector
112  STORM_LOG_ASSERT(x.size() == variableExpressions.size(), "Dimension of x-vector does not match number of varibales.");
113  auto xIt = x.begin();
114  auto vIt = variableExpressions.begin();
115  for (; xIt != x.end(); ++xIt, ++vIt) {
116  auto const& vBaseExpr = vIt->getBaseExpression();
117  if (vBaseExpr.isVariable()) {
118  *xIt = solver->getContinuousValue(vBaseExpr.asVariableExpression().getVariable());
119  } else {
120  STORM_LOG_ASSERT(vBaseExpr.isRationalLiteralExpression(), "Variable expression has unexpected type.");
121  *xIt = storm::utility::convertNumber<ValueType>(vBaseExpr.asRationalLiteralExpression().getValue());
122  }
123  }
124 
125  // If requested, we store the scheduler for retrieval.
126  if (this->isTrackSchedulerSet()) {
127  this->schedulerChoices = std::vector<uint_fast64_t>(this->A->getRowGroupCount());
128  for (uint64_t rowGroup = 0; rowGroup < this->A->getRowGroupCount(); ++rowGroup) {
129  if (!this->choiceFixedForRowGroup || !this->choiceFixedForRowGroup.get()[rowGroup]) {
130  // Only update scheduler choice for the states that don't have a fixed choice
131  uint64_t row = this->A->getRowGroupIndices()[rowGroup];
132  uint64_t optimalChoiceIndex = 0;
133  uint64_t currChoice = 0;
134  ValueType optimalGroupValue = this->A->multiplyRowWithVector(row, x) + b[row];
135  for (++row, ++currChoice; row < this->A->getRowGroupIndices()[rowGroup + 1]; ++row, ++currChoice) {
136  ValueType rowValue = this->A->multiplyRowWithVector(row, x) + b[row];
137  if ((minimize(dir) && rowValue < optimalGroupValue) || (maximize(dir) && rowValue > optimalGroupValue)) {
138  optimalGroupValue = rowValue;
139  optimalChoiceIndex = currChoice;
140  }
141  }
142  this->schedulerChoices.get()[rowGroup] = optimalChoiceIndex;
143  }
144  }
145  }
146  return true;
147 }
