238 typename BeliefMdpExplorer<PomdpType, BeliefValueType>::BeliefId BeliefMdpExplorer<PomdpType, BeliefValueType>::exploreNextState() {
239  STORM_LOG_ASSERT(status == Status::Exploring, "Method call is invalid in current status.");
240  // Mark the end of the previously explored row group.
241  if (currentMdpState != noState() && mdpStatesToExplorePrioState.rbegin()->second == exploredChoiceIndices.size()) {
242  internalAddRowGroupIndex();
243  }
244 
245  // Pop from the queue.
246  currentMdpState = mdpStatesToExplorePrioState.rbegin()->second;
247  auto currprio = mdpStatesToExplorePrioState.rbegin()->first;
248  auto range = mdpStatesToExplorePrioState.equal_range(currprio);
249  for (auto i = range.first; i != range.second; ++i) {
250  if (i->second == currentMdpState) {
251  mdpStatesToExplorePrioState.erase(i);
252  break;
253  }
254  }
255  mdpStatesToExploreStatePrio.erase(currentMdpState);
256  if (currentMdpState != nextId && !currentStateHasOldBehavior()) {
257  stateRemapping[currentMdpState] = nextId;
258  STORM_LOG_DEBUG("Explore state " << currentMdpState << " [Bel " << getCurrentBeliefId() << " " << beliefManager->toString(getCurrentBeliefId())
259  << "] as state with ID " << nextId << " (Prio: " << storm::utility::to_string(currprio) << ")");
260  } else {
261  STORM_LOG_DEBUG("Explore state " << currentMdpState << " [Bel " << getCurrentBeliefId() << " " << beliefManager->toString(getCurrentBeliefId()) << "]"
262  << " (Prio: " << storm::utility::to_string(currprio) << ")");
263  }
264 
265  if (!currentStateHasOldBehavior()) {
266  ++nextId;
267  }
268  if (explHeuristic == ExplorationHeuristic::ProbabilityPrio) {
269  probabilityEstimation.push_back(currprio);
270  }
271 
272  return mdpStateToBeliefIdMap[currentMdpState];
273 }

457 void BeliefMdpExplorer<PomdpType, BeliefValueType>::restoreOldBehaviorAtCurrentState(uint64_t const &localActionIndex) {
458  STORM_LOG_ASSERT(currentStateHasOldBehavior(), "Cannot restore old behavior as the current state does not have any.");
459  STORM_LOG_ASSERT(localActionIndex < previousChoiceIndices[currentMdpState + 1] - previousChoiceIndices[currentMdpState],
460  "Action index " << localActionIndex << " was not valid at state " << currentMdpState << " of the previously explored MDP.");
461 
462  if (getCurrentMdpState() == exploredChoiceIndices.size()) {
463  internalAddRowGroupIndex();
464  }

467  assert(getCurrentMdpState() < exploredChoiceIndices.size());

469  uint64_t newChoiceIndex = exploredChoiceIndices.at(getCurrentMdpState()) + localActionIndex;

472  for (auto const &transition : exploredMdp->getTransitionMatrix().getRow(oldChoiceIndex)) {
473  internalAddTransition(newChoiceIndex, transition.getColumn(), transition.getValue());

475  auto beliefId = getBeliefId(transition.getColumn());
476  if (beliefId != beliefManager->noId()) { // Not the extra target or bottom state
477  if (!exploredBeliefIds.get(beliefId)) {

479  exploredBeliefIds.set(beliefId, true);

482  case ExplorationHeuristic::BreadthFirst:
483  currentPrio = prio;

488  break;

494  break;
495  case ExplorationHeuristic::ProbabilityPrio:

500  }
501  break;
502  default:
503  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "Other heuristics not implemented yet");
504  }
505  mdpStatesToExploreStatePrio[transition.getColumn()] = currentPrio;
506  mdpStatesToExplorePrioState.emplace(currentPrio, transition.getColumn());
507  }
508  }

514 template<typename PomdpType, typename BeliefValueType>
515 void BeliefMdpExplorer<PomdpType, BeliefValueType>::finishExploration() {
516  STORM_LOG_ASSERT(status == Status::Exploring, "Method call is invalid in current status.");
517  STORM_LOG_ASSERT(!hasUnexploredState(), "Finishing exploration not possible if there are still unexplored states.");
518 
519  // Complete the exploration
520  // Finish the last row grouping in case the last explored state was new
521  if (!currentStateHasOldBehavior() || exploredChoiceIndices.back() < getCurrentNumberOfMdpChoices()) {
522  internalAddRowGroupIndex();
523  }
524  // Resize state- and choice based vectors to the correct size
525  targetStates.resize(getCurrentNumberOfMdpStates(), false);
526  truncatedStates.resize(getCurrentNumberOfMdpStates(), false);
527  clippedStates.resize(getCurrentNumberOfMdpStates(), false);
528  if (!mdpActionRewards.empty()) {
529  mdpActionRewards.resize(getCurrentNumberOfMdpChoices(), storm::utility::zero<ValueType>());
530  }
531 
532  // We are not exploring anymore
533  currentMdpState = noState();
534 
535  // If this was a restarted exploration, we might still have unexplored states (which were only reachable and explored in a previous build).
536  // We get rid of these before rebuilding the model
537  if (exploredMdp) {
538  dropUnexploredStates();
539  }
540 
541  // The potentially computed optimal choices and the set of states that are reachable under these choices are not valid anymore.
542  optimalChoices = std::nullopt;
543  optimalChoicesReachableMdpStates = std::nullopt;
544 
545  // Apply state remapping to the Belief-State maps
546  if (!stateRemapping.empty()) {
547  std::vector<BeliefId> remappedStateToBeliefIdMap(mdpStateToBeliefIdMap);
548  for (auto const &entry : stateRemapping) {
549  remappedStateToBeliefIdMap[entry.second] = mdpStateToBeliefIdMap[entry.first];
550  }
551  mdpStateToBeliefIdMap = remappedStateToBeliefIdMap;
552  for (auto const &beliefMdpState : beliefIdToMdpStateMap) {
553  if (stateRemapping.find(beliefMdpState.second) != stateRemapping.end()) {
554  beliefIdToMdpStateMap[beliefMdpState.first] = stateRemapping[beliefMdpState.second];
555  }
556  }
557  if (!mdpStateToChoiceLabelsMap.empty()) {
558  std::map<BeliefId, std::map<uint64_t, std::string>> temp(mdpStateToChoiceLabelsMap);
559  for (auto const &entry : stateRemapping) {
560  temp[entry.second] = mdpStateToChoiceLabelsMap[entry.first];
561  }
562  mdpStateToChoiceLabelsMap = temp;
563  }
564  }
565 
566  // Create the transition matrix
567  uint64_t entryCount = 0;
568  for (auto const &row : exploredMdpTransitions) {
569  entryCount += row.size();
570  }
571  storm::storage::SparseMatrixBuilder<ValueType> builder(getCurrentNumberOfMdpChoices(), getCurrentNumberOfMdpStates(), entryCount, true, true,
572  getCurrentNumberOfMdpStates());
573  for (uint64_t groupIndex = 0; groupIndex < exploredChoiceIndices.size() - 1; ++groupIndex) {
574  uint64_t rowIndex = exploredChoiceIndices[groupIndex];
575  uint64_t groupEnd = exploredChoiceIndices[groupIndex + 1];
576  builder.newRowGroup(rowIndex);
577  for (; rowIndex < groupEnd; ++rowIndex) {
578  for (auto const &entry : exploredMdpTransitions[rowIndex]) {
579  if (stateRemapping.find(entry.first) == stateRemapping.end()) {
580  builder.addNextValue(rowIndex, entry.first, entry.second);
581  } else {
582  builder.addNextValue(rowIndex, stateRemapping[entry.first], entry.second);
583  }
584  }
585  }
586  }
587  auto mdpTransitionMatrix = builder.build();
588 
589  // Create a standard labeling
590  storm::models::sparse::StateLabeling mdpLabeling(getCurrentNumberOfMdpStates());
591  mdpLabeling.addLabel("init");
592  mdpLabeling.addLabelToState("init", initialMdpState);
593  targetStates.resize(getCurrentNumberOfMdpStates(), false);
594  mdpLabeling.addLabel("target", std::move(targetStates));
595  truncatedStates.resize(getCurrentNumberOfMdpStates(), false);
596  mdpLabeling.addLabel("truncated", std::move(truncatedStates));
597  clippedStates.resize(getCurrentNumberOfMdpStates(), false);
598  mdpLabeling.addLabel("clipped", std::move(clippedStates));
599 
600  for (uint64_t state = 0; state < getCurrentNumberOfMdpStates(); ++state) {
601  if (state == extraBottomState || state == extraTargetState) {
602  if (!mdpLabeling.containsLabel("__extra")) {
603  mdpLabeling.addLabel("__extra");
604  }
605  mdpLabeling.addLabelToState("__extra", state);
606  } else {
607  STORM_LOG_DEBUG("Observation of MDP state " << state << " : " << beliefManager->getObservationLabel(mdpStateToBeliefIdMap[state]) << "\n");
608  std::string obsLabel = beliefManager->getObservationLabel(mdpStateToBeliefIdMap[state]);
609  uint32_t obsId = beliefManager->getBeliefObservation(mdpStateToBeliefIdMap[state]);
610  if (!obsLabel.empty()) {
611  if (!mdpLabeling.containsLabel(obsLabel)) {
612  mdpLabeling.addLabel(obsLabel);
613  }
614  mdpLabeling.addLabelToState(obsLabel, state);
615  } else if (mdpStateToBeliefIdMap[state] != beliefManager->noId()) {
616  std::string obsIdLabel = "obs_" + std::to_string(obsId);
617  if (!mdpLabeling.containsLabel(obsIdLabel)) {
618  mdpLabeling.addLabel(obsIdLabel);
619  }
620  mdpLabeling.addLabelToState(obsIdLabel, state);
621  }
622  }
623  }
624 
625  // Create a standard reward model (if rewards are available)
626  std::unordered_map<std::string, storm::models::sparse::StandardRewardModel<ValueType>> mdpRewardModels;
627  if (!mdpActionRewards.empty()) {
628  mdpActionRewards.resize(getCurrentNumberOfMdpChoices(), storm::utility::zero<ValueType>());
629  if (!clippingTransitionRewards.empty()) {
630  storm::storage::SparseMatrixBuilder<ValueType> rewardBuilder(getCurrentNumberOfMdpChoices(), getCurrentNumberOfMdpStates(),
631  clippingTransitionRewards.size(), true, true, getCurrentNumberOfMdpStates());
632  for (uint64_t groupIndex = 0; groupIndex < exploredChoiceIndices.size() - 1; ++groupIndex) {
633  uint64_t rowIndex = exploredChoiceIndices[groupIndex];
634  uint64_t groupEnd = exploredChoiceIndices[groupIndex + 1];
635  rewardBuilder.newRowGroup(rowIndex);
636  for (; rowIndex < groupEnd; ++rowIndex) {
637  if (clippingTransitionRewards.find(rowIndex) != clippingTransitionRewards.end()) {
638  STORM_LOG_ASSERT(extraTargetState.has_value(), "Requested a transition to the extra target state but there is none.");
639  rewardBuilder.addNextValue(rowIndex, extraTargetState.value(), clippingTransitionRewards[rowIndex]);
640  }
641  }
642  }
643  auto transitionRewardMatrix = rewardBuilder.build();
644  mdpRewardModels.emplace("default", storm::models::sparse::StandardRewardModel<ValueType>(
645  std::optional<std::vector<ValueType>>(), std::move(mdpActionRewards), std::move(transitionRewardMatrix)));
646  } else {
647  mdpRewardModels.emplace(
648  "default", storm::models::sparse::StandardRewardModel<ValueType>(std::optional<std::vector<ValueType>>(), std::move(mdpActionRewards)));
649  }
650  }
651 
652  // Create model components
653  storm::storage::sparse::ModelComponents<ValueType> modelComponents(std::move(mdpTransitionMatrix), std::move(mdpLabeling), std::move(mdpRewardModels));
654 
655  // Potentially create a choice labeling
656  if (!mdpStateToChoiceLabelsMap.empty()) {
657  modelComponents.choiceLabeling = storm::models::sparse::ChoiceLabeling(getCurrentNumberOfMdpChoices());
658  for (auto const &stateMap : mdpStateToChoiceLabelsMap) {
659  auto rowGroup = stateMap.first;
660  for (auto const &actionLabel : stateMap.second) {
661  if (!modelComponents.choiceLabeling->containsLabel(actionLabel.second)) {
662  modelComponents.choiceLabeling->addLabel(actionLabel.second);
663  }
664  modelComponents.choiceLabeling->addLabelToChoice(actionLabel.second, exploredChoiceIndices.at(rowGroup) + actionLabel.first);
665  }
666  }
667  }
668 
669  if (!delayedExplorationChoices.empty()) {
670  modelComponents.choiceLabeling = storm::models::sparse::ChoiceLabeling(getCurrentNumberOfMdpChoices());
671  delayedExplorationChoices.resize(getCurrentNumberOfMdpChoices(), false);
672  modelComponents.choiceLabeling->addLabel("delayed", std::move(delayedExplorationChoices));
673  }
674 
675  // Create the final model.
676  exploredMdp = std::make_shared<storm::models::sparse::Mdp<ValueType>>(std::move(modelComponents));
677  status = Status::ModelFinished;
678  STORM_LOG_DEBUG("Explored Mdp with " << exploredMdp->getNumberOfStates() << " states (" << clippedStates.getNumberOfSetBits()
679  << " of which were clipped and " << truncatedStates.getNumberOfSetBits() - clippedStates.getNumberOfSetBits()
680  << " of which were flagged as truncated).");
681 }

683 template<typename PomdpType, typename BeliefValueType>
684 void BeliefMdpExplorer<PomdpType, BeliefValueType>::dropUnexploredStates() {
685  STORM_LOG_ASSERT(status == Status::Exploring, "Method call is invalid in current status.");
686  STORM_LOG_ASSERT(!hasUnexploredState(), "Finishing exploration not possible if there are still unexplored states.");
687 
688  STORM_LOG_ASSERT(exploredMdp, "Method called although no 'old' MDP is available.");
689  // Find the states (and corresponding choices) that were not explored.
690  // These correspond to "empty" MDP transitions
691  storm::storage::BitVector relevantMdpStates(getCurrentNumberOfMdpStates(), true), relevantMdpChoices(getCurrentNumberOfMdpChoices(), true);
692  std::vector<MdpStateType> toRelevantStateIndexMap(getCurrentNumberOfMdpStates(), noState());
693  MdpStateType nextRelevantIndex = 0;
694  for (uint64_t groupIndex = 0; groupIndex < exploredChoiceIndices.size() - 1; ++groupIndex) {
695  uint64_t rowIndex = exploredChoiceIndices[groupIndex];
696  // Check first row in group
697  if (exploredMdpTransitions[rowIndex].empty()) {
698  relevantMdpChoices.set(rowIndex, false);
699  relevantMdpStates.set(groupIndex, false);
700  } else {
701  toRelevantStateIndexMap[groupIndex] = nextRelevantIndex;
702  ++nextRelevantIndex;
703  }
704  uint64_t groupEnd = exploredChoiceIndices[groupIndex + 1];
705  // process remaining rows in group
706  for (++rowIndex; rowIndex < groupEnd; ++rowIndex) {
707  // Assert that all actions at the current state were consistently explored or unexplored.
708  STORM_LOG_ASSERT(exploredMdpTransitions[rowIndex].empty() != relevantMdpStates.get(groupIndex),
709  "Actions at 'old' MDP state " << groupIndex << " were only partly explored.");
710  if (exploredMdpTransitions[rowIndex].empty()) {
711  relevantMdpChoices.set(rowIndex, false);
712  }
713  }
714  }
715 
716  if (relevantMdpStates.full()) {
717  // All states are relevant so nothing to do
718  return;
719  }
720 
721  nextId -= (relevantMdpStates.size() - relevantMdpStates.getNumberOfSetBits());
722 
723  // Translate various components to the "new" MDP state set
724  storm::utility::vector::filterVectorInPlace(mdpStateToBeliefIdMap, relevantMdpStates);
725  { // beliefIdToMdpStateMap
726  for (auto belIdToMdpStateIt = beliefIdToMdpStateMap.begin(); belIdToMdpStateIt != beliefIdToMdpStateMap.end();) {
727  if (relevantMdpStates.get(belIdToMdpStateIt->second)) {
728  // Translate current entry and move on to the next one.
729  belIdToMdpStateIt->second = toRelevantStateIndexMap[belIdToMdpStateIt->second];
730  ++belIdToMdpStateIt;
731  } else {
732  STORM_LOG_ASSERT(!exploredBeliefIds.get(belIdToMdpStateIt->first),
733  "Inconsistent exploration information: Unexplored MDPState corresponds to explored beliefId");
734  // Delete current entry and move on to the next one.
735  // This works because std::map::erase does not invalidate other iterators within the map!
736  beliefIdToMdpStateMap.erase(belIdToMdpStateIt++);
737  }
738  }
739  }
740  { // exploredMdpTransitions
741  storm::utility::vector::filterVectorInPlace(exploredMdpTransitions, relevantMdpChoices);
742  // Adjust column indices. Unfortunately, the fastest way seems to be to "rebuild" the map
743  // It might pay off to do this when building the matrix.
744  for (auto &transitions : exploredMdpTransitions) {
745  std::map<MdpStateType, ValueType> newTransitions;
746  for (auto const &entry : transitions) {
747  STORM_LOG_ASSERT(relevantMdpStates.get(entry.first), "Relevant state has transition to irrelevant state.");
748  newTransitions.emplace_hint(newTransitions.end(), toRelevantStateIndexMap[entry.first], entry.second);
749  }
750  transitions = std::move(newTransitions);
751  }
752  }
753  { // exploredChoiceIndices
754  MdpStateType newState = 0;
755  assert(exploredChoiceIndices[0] == 0u);
756  // Loop invariant: all indices up to exploredChoiceIndices[newState] consider the new row indices and all other entries are not touched.
757  for (auto const &oldState : relevantMdpStates) {
758  if (oldState != newState) {
759  assert(oldState > newState);
760  uint64_t groupSize = getRowGroupSizeOfState(oldState);
761  exploredChoiceIndices.at(newState + 1) = exploredChoiceIndices.at(newState) + groupSize;
762  }
763  ++newState;
764  }
765  exploredChoiceIndices.resize(newState + 1);
766  }
767  if (!mdpActionRewards.empty()) {
768  storm::utility::vector::filterVectorInPlace(mdpActionRewards, relevantMdpChoices);
769  }
770  if (extraBottomState) {
771  extraBottomState = toRelevantStateIndexMap[extraBottomState.value()];
772  }
773  if (extraTargetState) {
774  extraTargetState = toRelevantStateIndexMap[extraTargetState.value()];
775  }
776  targetStates = targetStates % relevantMdpStates;
777  truncatedStates = truncatedStates % relevantMdpStates;
778  clippedStates = clippedStates % relevantMdpStates;
779  initialMdpState = toRelevantStateIndexMap[initialMdpState];
780 
781  storm::utility::vector::filterVectorInPlace(lowerValueBounds, relevantMdpStates);
782  storm::utility::vector::filterVectorInPlace(upperValueBounds, relevantMdpStates);
783  storm::utility::vector::filterVectorInPlace(values, relevantMdpStates);
784 
785  { // mdpStateToChoiceLabelsMap
786  if (!mdpStateToChoiceLabelsMap.empty()) {
787  auto temp = std::map<BeliefId, std::map<uint64_t, std::string>>();
788  for (auto const &relevantState : relevantMdpStates) {
789  temp[toRelevantStateIndexMap[relevantState]] = mdpStateToChoiceLabelsMap[relevantState];
790  }
791  mdpStateToChoiceLabelsMap = temp;
792  }
793  }
794 }

1020 template<typename PomdpType, typename BeliefValueType>
1021 void BeliefMdpExplorer<PomdpType, BeliefValueType>::computeOptimalChoicesAndReachableMdpStates(ValueType const &ancillaryChoicesEpsilon,
1022  bool relativeDifference) {
1023  STORM_LOG_ASSERT(status == Status::ModelChecked, "Method call is invalid in current status.");
1024  STORM_LOG_ASSERT(exploredMdp, "Method call is invalid in if no MDP is available.");
1025  STORM_LOG_ASSERT(!optimalChoices.has_value(), "Tried to compute optimal scheduler but this has already been done before.");
1026  STORM_LOG_ASSERT(!optimalChoicesReachableMdpStates.has_value(),
1027  "Tried to compute states that are reachable under an optimal scheduler but this has already been done before.");
1028 
1029  // First find the choices that are optimal
1030  optimalChoices = storm::storage::BitVector(exploredMdp->getNumberOfChoices(), false);
1031  auto const &choiceIndices = exploredMdp->getNondeterministicChoiceIndices();
1032  auto const &transitions = exploredMdp->getTransitionMatrix();
1033  auto const &targetStatesExploredMDP = exploredMdp->getStates("target");
1034  for (uint64_t mdpState = 0; mdpState < exploredMdp->getNumberOfStates(); ++mdpState) {
1035  if (targetStatesExploredMDP.get(mdpState)) {
1036  // Target states can be skipped.
1037  continue;
1038  } else {
1039  auto const &stateValue = values[mdpState];
1040  for (uint64_t globalChoice = choiceIndices[mdpState]; globalChoice < choiceIndices[mdpState + 1]; ++globalChoice) {
1041  ValueType choiceValue = transitions.multiplyRowWithVector(globalChoice, values);
1042  if (exploredMdp->hasRewardModel()) {
1043  choiceValue += exploredMdp->getUniqueRewardModel().getStateActionReward(globalChoice);
1044  }
1045  auto absDiff = storm::utility::abs<ValueType>((choiceValue - stateValue));
1046  if ((relativeDifference && absDiff <= ancillaryChoicesEpsilon * stateValue) || (!relativeDifference && absDiff <= ancillaryChoicesEpsilon)) {
1047  optimalChoices->set(globalChoice, true);
1048  }
1049  }
1050  STORM_LOG_ASSERT(optimalChoices->getNextSetIndex(choiceIndices[mdpState]) < optimalChoices->size(), "Could not find an optimal choice.");
1051  }
1052  }
1053 
1054  // Then, find the states that are reachable via these choices
1055  optimalChoicesReachableMdpStates = storm::utility::graph::getReachableStates(transitions, exploredMdp->getInitialStates(), ~targetStatesExploredMDP,
1056  targetStatesExploredMDP, false, 0, optimalChoices.value());
1057 }

1155 typename BeliefMdpExplorer<PomdpType, BeliefValueType>::MdpStateType BeliefMdpExplorer<PomdpType, BeliefValueType>::getOrAddMdpState(
1156  BeliefId const &beliefId, ValueType const &transitionValue) {
1157  exploredBeliefIds.grow(beliefId + 1, false);
1158  if (exploredBeliefIds.get(beliefId)) {
1159  if (explHeuristic == ExplorationHeuristic::ProbabilityPrio &&
1160  mdpStatesToExploreStatePrio.find(beliefIdToMdpStateMap[beliefId]) != mdpStatesToExploreStatePrio.end()) {
1161  // We check if the value is higher than the current priority and update if necessary
1162  auto newPrio = probabilityEstimation[getCurrentMdpState()] * transitionValue;
1163  if (newPrio > mdpStatesToExploreStatePrio[beliefIdToMdpStateMap[beliefId]]) {
1164  // Erase the state from the "queue" map and re-insert it with the new value
1165  auto range = mdpStatesToExplorePrioState.equal_range(mdpStatesToExploreStatePrio[beliefIdToMdpStateMap[beliefId]]);
1166  for (auto i = range.first; i != range.second; ++i) {
1167  if (i->second == beliefIdToMdpStateMap[beliefId]) {
1168  mdpStatesToExplorePrioState.erase(i);
1169  break;
1170  }
1171  }
1172  mdpStatesToExplorePrioState.emplace(newPrio, beliefIdToMdpStateMap[beliefId]);
1173  mdpStatesToExploreStatePrio[beliefIdToMdpStateMap[beliefId]] = newPrio;
1174  }
1175  }
1176  return beliefIdToMdpStateMap[beliefId];
1177  } else {
1178  // This state needs exploration
1179  exploredBeliefIds.set(beliefId, true);
1180 
1181  // If this is a restart of the exploration, we still might have an MDP state for the belief
1182  if (exploredMdp) {
1183  auto findRes = beliefIdToMdpStateMap.find(beliefId);
1184  if (findRes != beliefIdToMdpStateMap.end()) {
1185  ValueType currentPrio;
1186  switch (explHeuristic) {
1187  case ExplorationHeuristic::BreadthFirst:
1188  currentPrio = prio;

1190  break;

1199  break;

1201  if (getCurrentMdpState() != noState()) {
1202  currentPrio = probabilityEstimation[getCurrentMdpState()] * transitionValue;
1203  } else {
1204  currentPrio = storm::utility::one<ValueType>();
1205  }
1206  break;
1207  default:
1208  STORM_LOG_THROW(false, storm::exceptions::NotImplementedException, "Other heuristics not implemented yet");

1212  return findRes->second;
1213  }

1222  switch (explHeuristic) {
1223  case ExplorationHeuristic::BreadthFirst:
1224  currentPrio = prio;
1225  prio = prio - storm::utility::one<ValueType>();
1226  break;
1227  case ExplorationHeuristic::LowerBoundPrio:
1228  currentPrio = getLowerValueBoundAtCurrentState();
1229  break;
1230  case ExplorationHeuristic::UpperBoundPrio:
1231  currentPrio = getUpperValueBoundAtCurrentState();
1232  break;
1233  case ExplorationHeuristic::GapPrio:
1234  currentPrio = getUpperValueBoundAtCurrentState() - getLowerValueBoundAtCurrentState();
1235  break;
1236  case ExplorationHeuristic::ProbabilityPrio:
1237  if (getCurrentMdpState() != noState()) {
1238  currentPrio = probabilityEstimation[getCurrentMdpState()] * transitionValue;
1239  } else {
1240  currentPrio = storm::utility::one<ValueType>();
