118 storm::modelchecker::RegionResult::Unknown, true);
119 EXPECT_EQ(expectedResult, regionCheckerMon->analyzeRegion(this->env(), allSatRegion, storm::modelchecker::RegionResultHypothesis::Unknown,
120 storm::modelchecker::RegionResult::Unknown, true, localMonRes));

126 storm::modelchecker::RegionResult::Unknown, true, localMonRes));
