503  template<typename SparseModelType, typename ConstantType>
504  std::pair<typename SparseModelType::ValueType, typename storm::storage::ParameterRegion<typename SparseModelType::ValueType>::Valuation> SparseParameterLiftingModelChecker<SparseModelType, ConstantType>::checkForPossibleMonotonicity(Environment const& env,
505  const storage::ParameterRegion<typename SparseModelType::ValueType> &region,
506  std::set<VariableType>& possibleMonotoneIncrParameters,
507  std::set<VariableType>& possibleMonotoneDecrParameters,
508  std::set<VariableType>& possibleNotMonotoneParameters,
509  std::set<VariableType>const& consideredVariables,
510  storm::solver::OptimizationDirection const& dir) {
511  typedef typename storm::storage::ParameterRegion<typename SparseModelType::ValueType>::Valuation Valuation;
512  typedef typename storm::storage::ParameterRegion<typename SparseModelType::ValueType>::CoefficientType CoefficientType;
513  ConstantType value = storm::solver::minimize(dir) ? 1 : 0;
514  Valuation valuation;
515  for (auto& var : consideredVariables) {
516  ConstantType previousCenter = -1;
517  bool monDecr = true;
518  bool monIncr = true;
519  auto valuationCenter = region.getCenterPoint();
520 
521  valuationCenter[var] = region.getLowerBoundary(var);
522  // TODO: make cmdline argument or 1/precision
523  int numberOfSamples = 50;
524  auto stepSize = (region.getUpperBoundary(var) - region.getLowerBoundary(var)) / (numberOfSamples - 1);
525 
526  while (valuationCenter[var] <= region.getUpperBoundary(var)) {
527  // Create valuation
528  ConstantType valueCenter = getInstantiationChecker().check(env, valuationCenter)->template asExplicitQuantitativeCheckResult<ConstantType>()[*this->parametricModel->getInitialStates().begin()];
529  if (storm::solver::minimize(dir) ? valueCenter <= value : valueCenter >= value) {
530  value = valueCenter;
531  valuation = valuationCenter;
532  }
533  // Calculate difference with result for previous valuation
534  ConstantType diffCenter = previousCenter - valueCenter;
535  assert (previousCenter == -1 || (diffCenter >= -1 && diffCenter <= 1));
536  if (previousCenter != -1) {
537  assert (previousCenter != -1 && previousCenter != -1);
538  monDecr &= diffCenter > 0 && diffCenter > 0 && diffCenter > 0; // then previous value is larger than the current value from the initial states
539  monIncr &= diffCenter < 0 && diffCenter < 0 && diffCenter < 0;
540  }
541  previousCenter = valueCenter;
542  if (!monDecr && ! monIncr) {
543  break;
544  }
545  valuationCenter[var] += stepSize;
546  }
547  if (monIncr) {
548  possibleMonotoneParameters.insert(var);
549  possibleMonotoneIncrParameters.insert(var);
550  } else if (monDecr) {
551  possibleMonotoneParameters.insert(var);
552  possibleMonotoneDecrParameters.insert(var);
553  } else {
554  possibleNotMonotoneParameters.insert(var);
555  }
556  }
557  return std::make_pair(storm::utility::convertNumber<typename SparseModelType::ValueType>(value), std::move(valuation));
558 
559  }
