458  STORM_LOG_INFO("Under-approx result for refinement improved after " << statistics.totalTime << " in refinement step #"
459  << (statistics.refinementSteps.value() + 1) << ". New value is '"
460  << newValue << "'.");
461  }
462  } else {
463  break;
464  }
465  }
466 
467  if (storm::utility::resources::isTerminate()) {
468  break;
469  } else {
470  ++statistics.refinementSteps.value();
471  // Don't make too many outputs (to avoid logfile clutter)
472  if (statistics.refinementSteps.value() <= 1000) {
473  STORM_LOG_INFO("Completed iteration #" << statistics.refinementSteps.value() << ". Current checktime is " << statistics.totalTime << ".");
474  computingLowerBound = false;
475  computingUpperBound = false;
476  if (options.discretize) {
477  STORM_LOG_INFO("\tOver-approx MDP has size " << overApproximation->getExploredMdp()->getNumberOfStates() << ".");
478  (min ? computingLowerBound : computingUpperBound) = true;
479  }
480  if (options.unfold) {
481  STORM_LOG_INFO("\tUnder-approx MDP has size " << underApproximation->getExploredMdp()->getNumberOfStates() << ".");
482  (min ? computingUpperBound : computingLowerBound) = true;
483  }
484  if (computingLowerBound && computingUpperBound) {
485  STORM_LOG_INFO("\tCurrent result is [" << result.lowerBound << ", " << result.upperBound << "].");
486  } else if (computingLowerBound) {
487  STORM_LOG_INFO("\tCurrent result is ≥" << result.lowerBound << ".");
488  } else if (computingUpperBound) {
489  STORM_LOG_INFO("\tCurrent result is ≤" << result.upperBound << ".");
490  }
491  STORM_LOG_WARN_COND(statistics.refinementSteps.value() < 1000, "Refinement requires more than 1000 iterations.");
492  }
493  }
494  if (overApproxFixPoint && underApproxFixPoint) {
495  STORM_LOG_INFO("Refinement fixpoint reached after " << statistics.refinementSteps.value() << " iterations.\n");
496  statistics.refinementFixpointDetected = true;
497  break;
498  }
499  }
500  }
501  // Print model information of final over- / under-approximation MDP
502  if (options.discretize && overApproximation->hasComputedValues()) {
503  auto printOverInfo = [&overApproximation]() {
504  std::stringstream str;
505  str << "Explored and checked Over-Approximation MDP:\n";
506  overApproximation->getExploredMdp()->printModelInformationToStream(str);
507  return str.str();
508  };
509  STORM_LOG_INFO(printOverInfo());
510  }
511  if (options.unfold && underApproximation->hasComputedValues()) {
512  auto printUnderInfo = [&underApproximation]() {
513  std::stringstream str;
514  str << "Explored and checked Under-Approximation MDP:\n";
515  underApproximation->getExploredMdp()->printModelInformationToStream(str);
516  return str.str();
517  };
518  STORM_LOG_INFO(printUnderInfo());
519  std::shared_ptr<storm::models::sparse::Model<ValueType>> scheduledModel = underApproximation->getExploredMdp();
520  if (!options.useStateEliminationCutoff) {
521  storm::models::sparse::StateLabeling newLabeling(scheduledModel->getStateLabeling());
522  auto nrPreprocessingScheds = min ? underApproximation->getNrSchedulersForUpperBounds() : underApproximation->getNrSchedulersForLowerBounds();
523  for (uint64_t i = 0; i < nrPreprocessingScheds; ++i) {
524  newLabeling.addLabel("sched_" + std::to_string(i));
525  }
526  newLabeling.addLabel("cutoff");
527  newLabeling.addLabel("clipping");
528  newLabeling.addLabel("finite_mem");
529 
530  auto transMatrix = scheduledModel->getTransitionMatrix();
531  for (uint64_t i = 0; i < scheduledModel->getNumberOfStates(); ++i) {
532  if (newLabeling.getStateHasLabel("truncated", i)) {
533  uint64_t localChosenActionIndex = underApproximation->getSchedulerForExploredMdp()->getChoice(i).getDeterministicChoice();
534  auto rowIndex = scheduledModel->getTransitionMatrix().getRowGroupIndices()[i];
535  if (scheduledModel->getChoiceLabeling().getLabelsOfChoice(rowIndex + localChosenActionIndex).size() > 0) {
536  auto label = *(scheduledModel->getChoiceLabeling().getLabelsOfChoice(rowIndex + localChosenActionIndex).begin());
537  if (label.rfind("clip", 0) == 0) {
538  newLabeling.addLabelToState("clipping", i);
539  auto chosenRow = transMatrix.getRow(i, 0);
540  auto candidateIndex = (chosenRow.end() - 1)->getColumn();
541  transMatrix.makeRowDirac(transMatrix.getRowGroupIndices()[i], candidateIndex);
542  } else if (label.rfind("mem_node", 0) == 0) {
543  newLabeling.addLabelToState("finite_mem", i);
544  newLabeling.addLabelToState("cutoff", i);
545  } else {
546  newLabeling.addLabelToState(label, i);
547  newLabeling.addLabelToState("cutoff", i);
548  }
549  }
550  }
551  }
552  newLabeling.removeLabel("truncated");
553 
554  transMatrix.dropZeroEntries();
555  storm::storage::sparse::ModelComponents<ValueType> modelComponents(transMatrix, newLabeling);
556  if (scheduledModel->hasChoiceLabeling()) {
557  modelComponents.choiceLabeling = scheduledModel->getChoiceLabeling();
558  }
559  storm::models::sparse::Mdp<ValueType> newMDP(modelComponents);
560  auto inducedMC = newMDP.applyScheduler(*(underApproximation->getSchedulerForExploredMdp()), true);
561  scheduledModel = std::static_pointer_cast<storm::models::sparse::Model<ValueType>>(inducedMC);
562  } else {
563  auto inducedMC = underApproximation->getExploredMdp()->applyScheduler(*(underApproximation->getSchedulerForExploredMdp()), true);
564  scheduledModel = std::static_pointer_cast<storm::models::sparse::Model<ValueType>>(inducedMC);
565  }
566  result.schedulerAsMarkovChain = scheduledModel;
567  if (min) {
568  result.cutoffSchedulers = underApproximation->getUpperValueBoundSchedulers();
569  } else {
570  result.cutoffSchedulers = underApproximation->getLowerValueBoundSchedulers();
571  }
572  }

761 template<typename ValueType>
762 ValueType getGap(ValueType const& l, ValueType const& u) {
763  STORM_LOG_ASSERT(l >= storm::utility::zero<ValueType>() && u >= storm::utility::zero<ValueType>(),
764  "Gap computation currently does not handle negative values.");
765  if (storm::utility::isInfinity(u)) {
766  if (storm::utility::isInfinity(l)) {
767  return storm::utility::zero<ValueType>();
768  } else {
769  return u;
770  }
771  } else if (storm::utility::isZero(u)) {
772  STORM_LOG_ASSERT(storm::utility::isZero(l), "Upper bound is zero but lower bound is " << l << ".");
773  return u;
774  } else {
775  STORM_LOG_ASSERT(!storm::utility::isInfinity(l), "Lower bound is infinity, but upper bound is " << u << ".");
776  // get the relative gap
777  return storm::utility::abs<ValueType>(u - l) * storm::utility::convertNumber<ValueType, uint64_t>(2) / (l + u);
778  }

781 template<typename PomdpModelType, typename BeliefValueType, typename BeliefMDPType>
782 bool BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType, BeliefMDPType>::buildOverApproximation(
783  storm::Environment const& env, std::set<uint32_t> const& targetObservations, bool min, bool computeRewards, bool refine,
784  HeuristicParameters const& heuristicParameters, std::vector<BeliefValueType>& observationResolutionVector,
785  std::shared_ptr<BeliefManagerType>& beliefManager, std::shared_ptr<ExplorerType>& overApproximation) {
786  // Detect whether the refinement reached a fixpoint.
787  bool fixPoint = true;
788 
789  statistics.overApproximationBuildTime.start();
790  storm::storage::BitVector refinedObservations;
791  if (!refine) {
792  // If we build the model from scratch, we first have to set up the explorer for the overApproximation.
793  if (computeRewards) {
794  overApproximation->startNewExploration(storm::utility::zero<ValueType>());
795  } else {
796  overApproximation->startNewExploration(storm::utility::one<ValueType>(), storm::utility::zero<ValueType>());
797  }
798  } else {
799  // If we refine the existing overApproximation, our heuristic also wants to know which states are reachable under an optimal policy
800  overApproximation->computeOptimalChoicesAndReachableMdpStates(heuristicParameters.optimalChoiceValueEpsilon, true);
801  // We also need to find out which observation resolutions needs refinement.
802  // current maximal resolution (needed for refinement heuristic)
803  auto obsRatings = getObservationRatings(overApproximation, observationResolutionVector);
804  // If there is a score < 1, we have not reached a fixpoint, yet
805  auto numericPrecision = storm::utility::convertNumber<BeliefValueType>(options.numericPrecision);
806  if (std::any_of(obsRatings.begin(), obsRatings.end(),
807  [&numericPrecision](BeliefValueType const& value) { return value + numericPrecision < storm::utility::one<BeliefValueType>(); })) {
808  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because there are still observations to refine.");
809  fixPoint = false;
810  }
811  refinedObservations = storm::utility::vector::filter<BeliefValueType>(obsRatings, [&heuristicParameters](BeliefValueType const& r) {
812  return r <= storm::utility::convertNumber<BeliefValueType>(heuristicParameters.observationThreshold);
813  });
814  STORM_LOG_DEBUG("Refining the resolution of " << refinedObservations.getNumberOfSetBits() << "/" << refinedObservations.size() << " observations.");
815  for (auto const& obs : refinedObservations) {
816  // Increment the resolution at the refined observations.
817  // Use storm's rational number to detect overflows properly.
818  storm::RationalNumber newObsResolutionAsRational = storm::utility::convertNumber<storm::RationalNumber>(observationResolutionVector[obs]) *
819  storm::utility::convertNumber<storm::RationalNumber>(options.resolutionFactor);
820  static_assert(storm::NumberTraits<BeliefValueType>::IsExact || std::is_same<BeliefValueType, double>::value, "Unhandled belief value type");
821  if (!storm::NumberTraits<BeliefValueType>::IsExact &&
822  newObsResolutionAsRational > storm::utility::convertNumber<storm::RationalNumber>(std::numeric_limits<double>::max())) {
823  observationResolutionVector[obs] = storm::utility::convertNumber<BeliefValueType>(std::numeric_limits<double>::max());
824  } else {
825  observationResolutionVector[obs] = storm::utility::convertNumber<BeliefValueType>(newObsResolutionAsRational);
826  }
827  }
828  overApproximation->restartExploration();
829  }
830  statistics.overApproximationMaxResolution = storm::utility::ceil(*std::max_element(observationResolutionVector.begin(), observationResolutionVector.end()));
831 
832  // Start exploration
833  storm::utility::Stopwatch explorationTime;
834  if (options.explorationTimeLimit != 0) {
835  explorationTime.start();
836  }
837  bool timeLimitExceeded = false;
838  std::map<uint32_t, typename ExplorerType::SuccessorObservationInformation> gatheredSuccessorObservations; // Declare here to avoid reallocations
839  uint64_t numRewiredOrExploredStates = 0;
840  while (overApproximation->hasUnexploredState()) {
841  if (!timeLimitExceeded && options.explorationTimeLimit != 0 &&
842  static_cast<uint64_t>(explorationTime.getTimeInSeconds()) > options.explorationTimeLimit) {
843  STORM_LOG_INFO("Exploration time limit exceeded.");
844  timeLimitExceeded = true;
845  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because the exploration time limit is exceeded.");
846  fixPoint = false;
847  }
848 
849  uint64_t currId = overApproximation->exploreNextState();
850  bool hasOldBehavior = refine && overApproximation->currentStateHasOldBehavior();
851  if (!hasOldBehavior) {
852  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because a new state is explored");
853  fixPoint = false; // Exploring a new state!
854  }
855  uint32_t currObservation = beliefManager->getBeliefObservation(currId);
856  if (targetObservations.count(currObservation) != 0) {
857  overApproximation->setCurrentStateIsTarget();
858  overApproximation->addSelfloopTransition();
859  } else {
860  // We need to decide how to treat this state (and each individual enabled action). There are the following cases:
861  // 1 The state has no old behavior and
862  // 1.1 we explore all actions or
863  // 1.2 we truncate all actions
864  // 2 The state has old behavior and was truncated in the last iteration and
865  // 2.1 we explore all actions or
866  // 2.2 we truncate all actions (essentially restoring old behavior, but we do the truncation step again to benefit from updated bounds)
867  // 3 The state has old behavior and was not truncated in the last iteration and the current action
868  // 3.1 should be rewired or
869  // 3.2 should get the old behavior but either
870  // 3.2.1 none of the successor observation has been refined since the last rewiring or exploration of this action
871  // 3.2.2 rewiring is only delayed as it could still have an effect in a later refinement step
872 
873  // Find out in which case we are
874  bool exploreAllActions = false;
875  bool truncateAllActions = false;
876  bool restoreAllActions = false;
877  bool checkRewireForAllActions = false;
878  // Get the relative gap
879  ValueType gap = getGap(overApproximation->getLowerValueBoundAtCurrentState(), overApproximation->getUpperValueBoundAtCurrentState());
880  if (!hasOldBehavior) {
881  // Case 1
882  // If we explore this state and if it has no old behavior, it is clear that an "old" optimal scheduler can be extended to a scheduler that
883  // reaches this state
884  if (!timeLimitExceeded && gap >= heuristicParameters.gapThreshold && numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
885  exploreAllActions = true; // Case 1.1
886  } else {
887  truncateAllActions = true; // Case 1.2
888  overApproximation->setCurrentStateIsTruncated();
889  }
890  } else if (overApproximation->getCurrentStateWasTruncated()) {
891  // Case 2
892  if (!timeLimitExceeded && overApproximation->currentStateIsOptimalSchedulerReachable() && gap > heuristicParameters.gapThreshold &&
893  numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
894  exploreAllActions = true; // Case 2.1
895  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because a previously truncated state is now explored.");
896  fixPoint = false;
897  } else {
898  truncateAllActions = true; // Case 2.2
899  overApproximation->setCurrentStateIsTruncated();
900  if (fixPoint) {
901  // Properly check whether this can still be a fixpoint
902  if (overApproximation->currentStateIsOptimalSchedulerReachable() && !storm::utility::isZero(gap)) {
903  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because we truncate a state with non-zero gap "
904  << gap << " that is reachable via an optimal sched.");
905  fixPoint = false;
906  }
907  // else {}
908  // In this case we truncated a state that is not reachable under optimal schedulers.
909  // If no other state is explored (i.e. fixPoint remains true), these states should still not be reachable in subsequent iterations
910  }
911  }
912  } else {
913  // Case 3
914  // The decision for rewiring also depends on the corresponding action, but we have some criteria that lead to case 3.2 (independent of the
915  // action)
916  if (!timeLimitExceeded && overApproximation->currentStateIsOptimalSchedulerReachable() && gap > heuristicParameters.gapThreshold &&
917  numRewiredOrExploredStates < heuristicParameters.sizeThreshold) {
918  checkRewireForAllActions = true; // Case 3.1 or Case 3.2
919  } else {
920  restoreAllActions = true; // Definitely Case 3.2

933  // First, check whether this action has been rewired since the last refinement of one of the successor observations (i.e. whether rewiring
934  // would actually change the successor states)
935  assert(overApproximation->currentStateHasOldBehavior());

937  overApproximation->currentStateHasSuccessorObservationInObservationSet(action, refinedObservations)) {

942  STORM_LOG_INFO_COND(!fixPoint, "Not reaching a refinement fixpoint because we rewire a state.");

944  } else {

958  }
959  } // else { Case 3.2.1 }

985  }
986  }

989  overApproximation->addTransitionsToExtraStates(action, truncationProbability);
990  overApproximation->computeRewardAtCurrentState(action, truncationValueBound);
991  } else {
992  overApproximation->addTransitionsToExtraStates(action, truncationValueBound, truncationProbability - truncationValueBound);
993  }
994  }
995  } else {
996  // Case 3.2
997  overApproximation->restoreOldBehaviorAtCurrentState(action);
998  }
999  }
1000  if (expandedAtLeastOneAction) {
1001  ++numRewiredOrExploredStates;
1002  }
1003  }
1004

1006  if (pomdp().hasChoiceLabeling()) {
1007  auto rowIndex = pomdp().getTransitionMatrix().getRowGroupIndices()[beliefManager->getRepresentativeState(currId)];
1008  if (pomdp().getChoiceLabeling().getLabelsOfChoice(rowIndex + action).size() > 0) {
1009  overApproximation->addChoiceLabelToCurrentState(action, *(pomdp().getChoiceLabeling().getLabelsOfChoice(rowIndex + action).begin()));
1010  }
1011  }
1012  }
1013 
1014  if (storm::utility::resources::isTerminate()) {
1015  break;
1016  }
1017  }
1018 
1019  if (storm::utility::resources::isTerminate()) {
1020  // don't overwrite statistics of a previous, successful computation
1021  if (!statistics.overApproximationStates) {
1022  statistics.overApproximationBuildAborted = true;
1023  statistics.overApproximationStates = overApproximation->getCurrentNumberOfMdpStates();
1024  }
1025  statistics.overApproximationBuildTime.stop();
1026  return false;
1027  }

1298 void BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType, BeliefMDPType>::clipToGrid(uint64_t clippingStateId, bool computeRewards, bool min,
1299  std::shared_ptr<BeliefManagerType>& beliefManager,
1300  std::shared_ptr<ExplorerType>& beliefExplorer) {
1301  // Add all transitions to states which are already in the MDP, clip all others to a grid
1302  // To make the resulting MDP smaller, we eliminate intermediate successor states when clipping is applied
1303  for (uint64_t action = 0, numActions = beliefManager->getBeliefNumberOfChoices(clippingStateId); action < numActions; ++action) {
1304  auto rewardBound = utility::zero<BeliefValueType>();
1305  auto successors = beliefManager->expand(clippingStateId, action);
1306  auto absDelta = utility::zero<BeliefValueType>();
1307  for (auto const& successor : successors) {
1308  // Add transition if successor is in explored space.
1309  // We can directly add the transitions as there is at most one successor for each observation
1310  // Therefore no belief can be clipped to an already added successor
1311  bool added = beliefExplorer->addTransitionToBelief(action, successor.first, successor.second, true);
1312  if (!added) {
1313  // The successor is not in the explored space. Clip it
1314  statistics.nrClippingAttempts = statistics.nrClippingAttempts.value() + 1;
1315  auto clipping = beliefManager->clipBeliefToGrid(
1316  successor.first, options.clippingGridRes, computeRewards ? beliefExplorer->getStateExtremeBoundIsInfinite() : storm::storage::BitVector());
1317  if (clipping.isClippable) {
1318  // The belief is not on the grid and there is a candidate with finite reward
1319  statistics.nrClippedStates = statistics.nrClippedStates.value() + 1;
1320  // Transition probability to candidate is (probability to successor) * (clipping transition probability)
1321  BeliefValueType transitionProb =
1322  (utility::one<BeliefValueType>() - clipping.delta) * utility::convertNumber<BeliefValueType>(successor.second);
1323  beliefExplorer->addTransitionToBelief(action, clipping.targetBelief, utility::convertNumber<BeliefMDPType>(transitionProb), false);
1324  // Collect weighted clipping values
1325  absDelta += clipping.delta * utility::convertNumber<BeliefValueType>(successor.second);
1326  if (computeRewards) {
1327  // collect cumulative reward bounds
1328  auto localRew = utility::zero<BeliefValueType>();
1329  for (auto const& deltaValue : clipping.deltaValues) {
1330  localRew += deltaValue.second *
1331  utility::convertNumber<BeliefValueType>((beliefExplorer->getExtremeValueBoundAtPOMDPState(deltaValue.first)));
1332  }
1333  if (localRew == utility::infinity<BeliefValueType>()) {
1334  STORM_LOG_WARN("Infinite reward in clipping!");
1335  }
1336  rewardBound += localRew * utility::convertNumber<BeliefValueType>(successor.second);
1337  }
1338  } else if (clipping.onGrid) {
1339  // If the belief is not clippable, but on the grid, it may need to be explored, too
1340  bool inserted = beliefExplorer->addTransitionToBelief(action, successor.first, successor.second, false);
1341  } else {
1342  // Otherwise, the reward for all candidates is infinite, clipping does not make sense. Cut it off instead
1343  absDelta += utility::convertNumber<BeliefValueType>(successor.second);
1344  rewardBound += utility::convertNumber<BeliefValueType>(successor.second) *
1345  utility::convertNumber<BeliefValueType>(min ? beliefExplorer->computeUpperValueBoundAtBelief(successor.first)
1346  : beliefExplorer->computeLowerValueBoundAtBelief(successor.first));
1347  }
1348  }
1349  }
1350  // Add the collected clipping transition if necessary
1351  if (absDelta != utility::zero<BeliefValueType>()) {
1352  if (computeRewards) {
1353  if (rewardBound == utility::infinity<BeliefValueType>()) {
1354  // If the reward is infinite, add a transition to the sink state to collect infinite reward
1355  beliefExplorer->addTransitionsToExtraStates(action, utility::zero<BeliefMDPType>(), utility::convertNumber<BeliefMDPType>(absDelta));
1356  } else {
1357  beliefExplorer->addTransitionsToExtraStates(action, utility::convertNumber<BeliefMDPType>(absDelta));
1358  BeliefValueType totalRewardVal = rewardBound / absDelta;
1359  beliefExplorer->addClippingRewardToCurrentState(action, utility::convertNumber<BeliefMDPType>(totalRewardVal));
1360  }
1361  } else {
1362  beliefExplorer->addTransitionsToExtraStates(action, utility::zero<BeliefMDPType>(), utility::convertNumber<BeliefMDPType>(absDelta));
1363  }
1364  }
1365  if (computeRewards) {
1366  beliefExplorer->computeRewardAtCurrentState(action);
1367  }
1368  }
1369 }

1406  utility::convertNumber<BeliefMDPType>(clipping.delta));
1407  }
1408  beliefExplorer->addChoiceLabelToCurrentState(localActionIndex, "clip");

1410  } else {

1412  // If the belief is not clippable, but on the grid, it may need to be explored, too
1413  beliefExplorer->markAsGridBelief(clippingStateId);

1415  }

1464 template<typename PomdpModelType, typename BeliefValueType, typename BeliefMDPType>
1465 BeliefValueType BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType, BeliefMDPType>::rateObservation(
1466  typename ExplorerType::SuccessorObservationInformation const& info, BeliefValueType const& observationResolution, BeliefValueType const& maxResolution) {
1467  auto n = storm::utility::convertNumber<BeliefValueType, uint64_t>(info.support.size());
1468  auto one = storm::utility::one<BeliefValueType>();
1469  if (storm::utility::isOne(n)) {
1470  // If the belief is Dirac, it has to be approximated precisely.
1471  // In this case, we return the best possible rating
1472  return one;
1473  } else {
1474  // Create the rating for this observation at this choice from the given info
1475  auto obsChoiceRating = storm::utility::convertNumber<BeliefValueType, ValueType>(info.maxProbabilityToSuccessorWithObs / info.observationProbability);
1476  // At this point, obsRating is the largest triangulation weight (which ranges from 1/n to 1)
1477  // Normalize the rating so that it ranges from 0 to 1, where
1478  // 0 means that the actual belief lies in the middle of the triangulating simplex (i.e. a "bad" approximation) and 1 means that the belief is precisely
1479  // approximated.
1480  obsChoiceRating = (obsChoiceRating * n - one) / (n - one);
1481  // Scale the ratings with the resolutions, so that low resolutions get a lower rating (and are thus more likely to be refined)
1482  obsChoiceRating *= observationResolution / maxResolution;
1483  return obsChoiceRating;
1484  }
1485 }

1487 template<typename PomdpModelType, typename BeliefValueType, typename BeliefMDPType>
1488 std::vector<BeliefValueType> BeliefExplorationPomdpModelChecker<PomdpModelType, BeliefValueType, BeliefMDPType>::getObservationRatings(
1489  std::shared_ptr<ExplorerType> const& overApproximation, std::vector<BeliefValueType> const& observationResolutionVector) {
1490  uint64_t numMdpStates = overApproximation->getExploredMdp()->getNumberOfStates();
1491  auto const& choiceIndices = overApproximation->getExploredMdp()->getNondeterministicChoiceIndices();
1492  BeliefValueType maxResolution = *std::max_element(observationResolutionVector.begin(), observationResolutionVector.end());
1493 
1494  std::vector<BeliefValueType> resultingRatings(pomdp().getNrObservations(), storm::utility::one<BeliefValueType>());
1495 
1496  std::map<uint32_t, typename ExplorerType::SuccessorObservationInformation> gatheredSuccessorObservations; // Declare here to avoid reallocations
1497  for (uint64_t mdpState = 0; mdpState < numMdpStates; ++mdpState) {
1498  // Check whether this state is reached under an optimal scheduler.
1499  // The heuristic assumes that the remaining states are not relevant for the observation score.
1500  if (overApproximation->stateIsOptimalSchedulerReachable(mdpState)) {
1501  for (uint64_t mdpChoice = choiceIndices[mdpState]; mdpChoice < choiceIndices[mdpState + 1]; ++mdpChoice) {
1502  // Similarly, only optimal actions are relevant
1503  if (overApproximation->actionIsOptimal(mdpChoice)) {
1504  // score the observations for this choice
1505  gatheredSuccessorObservations.clear();
1506  overApproximation->gatherSuccessorObservationInformationAtMdpChoice(mdpChoice, gatheredSuccessorObservations);
1507  for (auto const& obsInfo : gatheredSuccessorObservations) {
1508  auto const& obs = obsInfo.first;
1509  BeliefValueType obsChoiceRating = rateObservation(obsInfo.second, observationResolutionVector[obs], maxResolution);
1510 
1511  // The rating of the observation will be the minimum over all choice-based observation ratings
1512  resultingRatings[obs] = std::min(resultingRatings[obs], obsChoiceRating);
1513  }
1514  }
1515  }
1516  }
1517  }
1518  return resultingRatings;
1519 }
