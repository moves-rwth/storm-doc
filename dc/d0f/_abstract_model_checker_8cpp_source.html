64 template<typename ModelType>
65 std::unique_ptr<CheckResult> AbstractModelChecker<ModelType>::computeProbabilities(Environment const& env,
66  CheckTask<storm::logic::Formula, SolutionType> const& checkTask) {
67  storm::logic::Formula const& formula = checkTask.getFormula();
68 
69  if (formula.isStateFormula() || formula.hasQualitativeResult()) {
70  return this->computeStateFormulaProbabilities(env, checkTask.substituteFormula(formula));
71  }
72 
73  if (formula.isHOAPathFormula()) {
74  return this->computeHOAPathProbabilities(env, checkTask.substituteFormula(formula.asHOAPathFormula()));
75  } else if (formula.isConditionalProbabilityFormula()) {
76  return this->computeConditionalProbabilities(env, checkTask.substituteFormula(formula.asConditionalFormula()));
77  } else if (formula.info(false).containsComplexPathFormula()) {
78  // we need to do LTL model checking
79  return this->computeLTLProbabilities(env, checkTask.substituteFormula(formula.asPathFormula()));
80  } else if (formula.isBoundedUntilFormula()) {
81  return this->computeBoundedUntilProbabilities(env, checkTask.substituteFormula(formula.asBoundedUntilFormula()));
82  } else if (formula.isReachabilityProbabilityFormula()) {
83  return this->computeReachabilityProbabilities(env, checkTask.substituteFormula(formula.asReachabilityProbabilityFormula()));
84  } else if (formula.isGloballyFormula()) {
85  return this->computeGloballyProbabilities(env, checkTask.substituteFormula(formula.asGloballyFormula()));
86  } else if (formula.isUntilFormula()) {
87  return this->computeUntilProbabilities(env, checkTask.substituteFormula(formula.asUntilFormula()));
88  } else if (formula.isNextFormula()) {
89  return this->computeNextProbabilities(env, checkTask.substituteFormula(formula.asNextFormula()));
90  }
91  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The given formula '" << formula << "' is invalid.");
92 }

252 template<typename ModelType>
253 std::unique_ptr<CheckResult> AbstractModelChecker<ModelType>::checkStateFormula(Environment const& env,
254  CheckTask<storm::logic::StateFormula, SolutionType> const& checkTask) {
255  storm::logic::StateFormula const& stateFormula = checkTask.getFormula();
256  if (stateFormula.isBinaryBooleanStateFormula()) {
257  return this->checkBinaryBooleanStateFormula(env, checkTask.substituteFormula(stateFormula.asBinaryBooleanStateFormula()));
258  } else if (stateFormula.isUnaryBooleanStateFormula()) {
259  return this->checkUnaryBooleanStateFormula(env, checkTask.substituteFormula(stateFormula.asUnaryBooleanStateFormula()));
260  } else if (stateFormula.isBooleanLiteralFormula()) {
261  return this->checkBooleanLiteralFormula(env, checkTask.substituteFormula(stateFormula.asBooleanLiteralFormula()));
262  } else if (stateFormula.isProbabilityOperatorFormula()) {
263  return this->checkProbabilityOperatorFormula(env, checkTask.substituteFormula(stateFormula.asProbabilityOperatorFormula()));
264  } else if (stateFormula.isRewardOperatorFormula()) {
265  return this->checkRewardOperatorFormula(env, checkTask.substituteFormula(stateFormula.asRewardOperatorFormula()));
266  } else if (stateFormula.isTimeOperatorFormula()) {
267  return this->checkTimeOperatorFormula(env, checkTask.substituteFormula(stateFormula.asTimeOperatorFormula()));
268  } else if (stateFormula.isLongRunAverageOperatorFormula()) {
269  return this->checkLongRunAverageOperatorFormula(env, checkTask.substituteFormula(stateFormula.asLongRunAverageOperatorFormula()));
270  } else if (stateFormula.isAtomicExpressionFormula()) {
271  return this->checkAtomicExpressionFormula(env, checkTask.substituteFormula(stateFormula.asAtomicExpressionFormula()));
272  } else if (stateFormula.isAtomicLabelFormula()) {
273  return this->checkAtomicLabelFormula(env, checkTask.substituteFormula(stateFormula.asAtomicLabelFormula()));
274  } else if (stateFormula.isBooleanLiteralFormula()) {
275  return this->checkBooleanLiteralFormula(env, checkTask.substituteFormula(stateFormula.asBooleanLiteralFormula()));
276  } else if (stateFormula.isGameFormula()) {
277  return this->checkGameFormula(env, checkTask.substituteFormula(stateFormula.asGameFormula()));
278  } else if (stateFormula.isMultiObjectiveFormula()) {
279  if (env.modelchecker().multi().isLexicographicModelCheckingSet()) {
280  return this->checkLexObjectiveFormula(env, checkTask.substituteFormula(stateFormula.asMultiObjectiveFormula()));
281  } else {
282  return this->checkMultiObjectiveFormula(env, checkTask.substituteFormula(stateFormula.asMultiObjectiveFormula()));
283  }
284  } else if (stateFormula.isQuantileFormula()) {
285  return this->checkQuantileFormula(env, checkTask.substituteFormula(stateFormula.asQuantileFormula()));
286  }
287  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The given formula '" << stateFormula << "' is invalid.");
288 }

306 template<typename ModelType>
307 std::unique_ptr<CheckResult> AbstractModelChecker<ModelType>::checkBinaryBooleanStateFormula(
308  Environment const& env, CheckTask<storm::logic::BinaryBooleanStateFormula, SolutionType> const& checkTask) {
309  storm::logic::BinaryBooleanStateFormula const& stateFormula = checkTask.getFormula();
310  STORM_LOG_THROW(stateFormula.getLeftSubformula().isStateFormula() && stateFormula.getRightSubformula().isStateFormula(),
311  storm::exceptions::InvalidArgumentException, "The given formula is invalid.");
312 
313  std::unique_ptr<CheckResult> leftResult =
314  this->check(env, checkTask.template substituteFormula<storm::logic::Formula>(stateFormula.getLeftSubformula().asStateFormula()));
315  std::unique_ptr<CheckResult> rightResult =
316  this->check(env, checkTask.template substituteFormula<storm::logic::Formula>(stateFormula.getRightSubformula().asStateFormula()));
317 
318  STORM_LOG_THROW(leftResult->isQualitative() && rightResult->isQualitative(), storm::exceptions::InternalTypeErrorException,
319  "Expected qualitative results.");
320 
321  if (stateFormula.isAnd()) {
322  leftResult->asQualitativeCheckResult() &= rightResult->asQualitativeCheckResult();
323  } else if (stateFormula.isOr()) {
324  leftResult->asQualitativeCheckResult() |= rightResult->asQualitativeCheckResult();
325  } else {
326  STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The given formula '" << stateFormula << "' is invalid.");
327  }
328 
329  return leftResult;
330 }
