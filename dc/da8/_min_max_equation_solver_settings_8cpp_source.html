74storm::solver::MinMaxMethod MinMaxEquationSolverSettings::getMinMaxEquationSolvingMethod() const {
75 std::string minMaxEquationSolvingTechnique = this->getOption(solvingMethodOptionName).getArgumentByName("name").getValueAsString();
76 if (minMaxEquationSolvingTechnique == "value-iteration" || minMaxEquationSolvingTechnique == "vi") {
77 return storm::solver::MinMaxMethod::ValueIteration;
78 } else if (minMaxEquationSolvingTechnique == "policy-iteration" || minMaxEquationSolvingTechnique == "pi") {

80 } else if (minMaxEquationSolvingTechnique == "linear-programming" || minMaxEquationSolvingTechnique == "lp") {
81 return storm::solver::MinMaxMethod::LinearProgramming;
82 } else if (minMaxEquationSolvingTechnique == "ratsearch" || minMaxEquationSolvingTechnique == "rs") {
83 return storm::solver::MinMaxMethod::RationalSearch;
84 } else if (minMaxEquationSolvingTechnique == "interval-iteration" || minMaxEquationSolvingTechnique == "ii") {
85 return storm::solver::MinMaxMethod::IntervalIteration;
86 } else if (minMaxEquationSolvingTechnique == "sound-value-iteration" || minMaxEquationSolvingTechnique == "svi") {
87 return storm::solver::MinMaxMethod::SoundValueIteration;
88 } else if (minMaxEquationSolvingTechnique == "optimistic-value-iteration" || minMaxEquationSolvingTechnique == "ovi") {
89 return storm::solver::MinMaxMethod::OptimisticValueIteration;
90 } else if (minMaxEquationSolvingTechnique == "topological") {
91 return storm::solver::MinMaxMethod::Topological;
92 } else if (minMaxEquationSolvingTechnique == "vi-to-pi") {
93 return storm::solver::MinMaxMethod::ViToPi;
94 } else if (minMaxEquationSolvingTechnique == "acyclic") {
95 return storm::solver::MinMaxMethod::Acyclic;
96 }

98 STORM_LOG_THROW(false, storm::exceptions::IllegalArgumentValueException,
99 "Unknown min/max equation solving technique '" << minMaxEquationSolvingTechnique << "'.");
