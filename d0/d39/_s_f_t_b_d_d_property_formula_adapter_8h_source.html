73 std::vector<ValueType> check(size_t const chunksize = 0) {
74 auto const bdds{formulasToBdd()};
75
76 std::map<uint64_t, Bdd> BDDToBdd{};
77 for (auto const &bdd : bdds) {
78 BDDToBdd[bdd.GetBDD()] = bdd;
79 }
80
81 std::map<uint64_t, std::vector<double>> bddToReversedTimepoints{};
82 // Note that a vector of timepoints is necessary as formula-BDDs can
83 // occur multiple times. The vector is reversed as it later allows to
84 // pop the results from the back which is more efficient.
85 for (size_t i{0}; i < bdds.size(); ++i) {
86 auto const reversedIndex{bdds.size() - i - 1};
87 auto const &bdd{bdds[reversedIndex]};
88 auto const &formula{formulas[reversedIndex]};
89 auto const timebound{getTimebound(formula)};
90
91 bddToReversedTimepoints[bdd.GetBDD()].push_back(timebound);
92 }
93
94 std::map<uint64_t, std::vector<double>> bddToReversedProbabilities{};
95 for (auto const &pair : bddToReversedTimepoints) {
96 auto const bdd{BDDToBdd.at(pair.first)};
97 bddToReversedProbabilities[pair.first] = checker->getProbabilitiesAtTimepoints(bdd, pair.second, chunksize);
98 }
99
100 std::vector<ValueType> rval{};
101 rval.reserve(bdds.size());
102 for (size_t i{0}; i < bdds.size(); ++i) {
103 auto const &bdd{bdds[i]};
104 auto &tmpVec{bddToReversedProbabilities.at(bdd.GetBDD())};
105 rval.push_back(tmpVec.back());
106 tmpVec.pop_back();
107 }
108
109 return rval;
110 }

133 static void checkForm(FormulaVector const &formulas) {
134 for (auto const &formula : formulas) {
135 if (formula->isProbabilityOperatorFormula()) {
136 auto const probabilityOperator{std::static_pointer_cast<storm::logic::ProbabilityOperatorFormula const>(formula)};
137 auto const subFormula{probabilityOperator->getSubformula().asSharedPointer()};
138 if (subFormula->isBoundedUntilFormula()) {
139 auto const boundedUntil{std::static_pointer_cast<storm::logic::BoundedUntilFormula const>(subFormula)};
140
141 auto const leftSide{boundedUntil->getLeftSubformula().asSharedPointer()};
142 if (!leftSide->isTrueFormula()) {
143 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Left side is not a TrueFormula.");
144 }
145
146 auto const rightSide{boundedUntil->getRightSubformula().asSharedPointer()};
147 if (!rightSide->isStateFormula()) {
148 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Right side is not a StateFormula.");
149 }
150
151 if (!boundedUntil->hasUpperBound()) {
152 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "UpperBound must be set.");
153 } else if (boundedUntil->hasUpperBound() && boundedUntil->hasLowerBound()) {
154 // Check if '[F = x phi]' was used.
155 if (boundedUntil->getUpperBound().evaluateAsDouble() != boundedUntil->getLowerBound().evaluateAsDouble()) {
156 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException,
157 "upperBound is set wrongly. "
158 "Only lowerBound == upperBound is Supported.");
159 }
160 }
161 } else {
162 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "SubFormula is not a BoundedUntilFormula.");
163 }
164 } else {
165 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Only ProbabilityOperatorFormulas are supported.");
166 }
167 }

192 static bool checkBoundsSame(FormulaCPointer const &formula) {
193 auto const probabilityOperator{std::static_pointer_cast<storm::logic::ProbabilityOperatorFormula const>(formula)};
194 auto const boundedUntil{std::static_pointer_cast<storm::logic::BoundedUntilFormula const>(probabilityOperator->getSubformula().asSharedPointer())};
195
196 if (!boundedUntil->hasLowerBound() || !boundedUntil->hasUpperBound()) {
197 return false;
198 } else if (boundedUntil->getUpperBound().evaluateAsDouble() == boundedUntil->getLowerBound().evaluateAsDouble()) {
199 return true;
200 }
201
202 return false;
203 }

274 Bdd binaryStateFormulaToBdd(BinaryStateFormulaCPointer const &formula) const {
275 auto const leftBdd{FormulaToBdd(formula->getLeftSubformula().asSharedPointer())};
276 auto const rightBdd{FormulaToBdd(formula->getRightSubformula().asSharedPointer())};
277
278 if (formula->isAnd()) {
279 return leftBdd & rightBdd;
280 } else if (formula->isOr()) {
281 return leftBdd | rightBdd;
282 }
283
284 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Illegal BinaryStateFormula: " << formula->toString());
285 return getSylvanBddManager()->getZero();
