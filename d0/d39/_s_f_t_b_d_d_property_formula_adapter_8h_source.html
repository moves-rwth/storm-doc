83  std::map<uint64_t, std::vector<double>> bddToReversedTimepoints{};
84  // Note that a vector of timepoints is necessary as formula-BDDs can
85  // occur multiple times. The vector is reversed as it later allows to
86  // pop the results from the back which is more efficient.
87  for (size_t i{0}; i < bdds.size(); ++i) {
88  auto const reversedIndex{bdds.size() - i - 1};
89  auto const &bdd{bdds[reversedIndex]};
90  auto const &formula{formulas[reversedIndex]};
91  auto const timebound{getTimebound(formula)};
92 
93  bddToReversedTimepoints[bdd.GetBDD()].push_back(timebound);
94  }
95 
96  std::map<uint64_t, std::vector<double>> bddToReversedProbabilities{};
97  for (auto const &pair : bddToReversedTimepoints) {
98  auto const bdd{BDDToBdd.at(pair.first)};
99  bddToReversedProbabilities[pair.first] = checker->getProbabilitiesAtTimepoints(bdd, pair.second, chunksize);
100  }
101 
102  std::vector<ValueType> rval{};
103  rval.reserve(bdds.size());
104  for (size_t i{0}; i < bdds.size(); ++i) {
105  auto const &bdd{bdds[i]};
106  auto &tmpVec{bddToReversedProbabilities.at(bdd.GetBDD())};
107  rval.push_back(tmpVec.back());

135  static void checkForm(FormulaVector const &formulas) {
136  for (auto const &formula : formulas) {
137  if (formula->isProbabilityOperatorFormula()) {
138  auto const probabilityOperator{std::static_pointer_cast<storm::logic::ProbabilityOperatorFormula const>(formula)};
139  auto const subFormula{probabilityOperator->getSubformula().asSharedPointer()};
140  if (subFormula->isBoundedUntilFormula()) {
141  auto const boundedUntil{std::static_pointer_cast<storm::logic::BoundedUntilFormula const>(subFormula)};
142 
143  auto const leftSide{boundedUntil->getLeftSubformula().asSharedPointer()};
144  if (!leftSide->isTrueFormula()) {
145  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Left side is not a TrueFormula.");
146  }
147 
148  auto const rightSide{boundedUntil->getRightSubformula().asSharedPointer()};
149  if (!rightSide->isStateFormula()) {
150  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Right side is not a StateFormula.");
151  }
152 
153  if (!boundedUntil->hasUpperBound()) {
154  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "UpperBound must be set.");
155  } else if (boundedUntil->hasUpperBound() && boundedUntil->hasLowerBound()) {
156  // Check if '[F = x phi]' was used.
157  if (boundedUntil->getUpperBound().evaluateAsDouble() != boundedUntil->getLowerBound().evaluateAsDouble()) {
158  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException,
159  "upperBound is set wrongly. "
160  "Only lowerBound == upperBound is Supported.");
161  }
162  }
163  } else {
164  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "SubFormula is not a BoundedUntilFormula.");
165  }
166  } else {
167  STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Only ProbabilityOperatorFormulas are supported.");
168  }
169  }
