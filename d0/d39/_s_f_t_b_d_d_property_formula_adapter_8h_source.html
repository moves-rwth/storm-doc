109 std::vector<ValueType> check(size_t const chunksize = 0) {
110#ifdef STORM_HAVE_SYLVAN
111 auto const bdds{formulasToBdd()};
112
113 std::map<uint64_t, Bdd> BDDToBdd{};
114 for (auto const &bdd : bdds) {
115 BDDToBdd[bdd.GetBDD()] = bdd;
116 }
117
118 std::map<uint64_t, std::vector<double>> bddToReversedTimepoints{};
119 // Note that a vector of timepoints is necessary as formula-BDDs can
120 // occur multiple times. The vector is reversed as it later allows to
121 // pop the results from the back which is more efficient.
122 for (size_t i{0}; i < bdds.size(); ++i) {
123 auto const reversedIndex{bdds.size() - i - 1};
124 auto const &bdd{bdds[reversedIndex]};
125 auto const &formula{formulas[reversedIndex]};
126 auto const timebound{getTimebound(formula)};
127
128 bddToReversedTimepoints[bdd.GetBDD()].push_back(timebound);
129 }
130
131 std::map<uint64_t, std::vector<double>> bddToReversedProbabilities{};
132 for (auto const &pair : bddToReversedTimepoints) {
133 auto const bdd{BDDToBdd.at(pair.first)};
134 bddToReversedProbabilities[pair.first] = checker->getProbabilitiesAtTimepoints(bdd, pair.second, chunksize);
135 }
136
137 std::vector<ValueType> rval{};
138 rval.reserve(bdds.size());
139 for (size_t i{0}; i < bdds.size(); ++i) {
140 auto const &bdd{bdds[i]};
141 auto &tmpVec{bddToReversedProbabilities.at(bdd.GetBDD())};
142 rval.push_back(tmpVec.back());
143 tmpVec.pop_back();
144 }
145
146 return rval;
147#else
148 STORM_LOG_THROW(false, storm::exceptions::MissingLibraryException,
149 "This version of Storm was compiled without support for Sylvan. Yet, a method was called that requires this support. Please choose a "
150 "version of Storm with Sylvan support.");
151#endif
152 }

177 static void checkForm(FormulaVector const &formulas) {
178#ifdef STORM_HAVE_SYLVAN
179 for (auto const &formula : formulas) {
180 if (formula->isProbabilityOperatorFormula()) {
181 auto const probabilityOperator{std::static_pointer_cast<storm::logic::ProbabilityOperatorFormula const>(formula)};
182 auto const subFormula{probabilityOperator->getSubformula().asSharedPointer()};
183 if (subFormula->isBoundedUntilFormula()) {
184 auto const boundedUntil{std::static_pointer_cast<storm::logic::BoundedUntilFormula const>(subFormula)};
185
186 auto const leftSide{boundedUntil->getLeftSubformula().asSharedPointer()};
187 if (!leftSide->isTrueFormula()) {
188 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Left side is not a TrueFormula.");
189 }
190
191 auto const rightSide{boundedUntil->getRightSubformula().asSharedPointer()};
192 if (!rightSide->isStateFormula()) {
193 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Right side is not a StateFormula.");
194 }
195
196 if (!boundedUntil->hasUpperBound()) {
197 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "UpperBound must be set.");
198 } else if (boundedUntil->hasUpperBound() && boundedUntil->hasLowerBound()) {
199 // Check if '[F = x phi]' was used.
200 if (boundedUntil->getUpperBound().evaluateAsDouble() != boundedUntil->getLowerBound().evaluateAsDouble()) {
201 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException,
202 "upperBound is set wrongly. "
203 "Only lowerBound == upperBound is Supported.");
204 }
205 }
206 } else {
207 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "SubFormula is not a BoundedUntilFormula.");
208 }
209 } else {
210 STORM_LOG_THROW(false, storm::exceptions::NotSupportedException, "Only ProbabilityOperatorFormulas are supported.");
211 }
212 }
213#else
214 STORM_LOG_THROW(false, storm::exceptions::MissingLibraryException,
215 "This version of Storm was compiled without support for Sylvan. Yet, a method was called that requires this support. Please choose a "
216 "version of Storm with Sylvan support.");
217#endif
218 }

249 static bool checkBoundsSame(FormulaCPointer const &formula) {
250 auto const probabilityOperator{std::static_pointer_cast<storm::logic::ProbabilityOperatorFormula const>(formula)};
251 auto const boundedUntil{std::static_pointer_cast<storm::logic::BoundedUntilFormula const>(probabilityOperator->getSubformula().asSharedPointer())};
252
253 if (!boundedUntil->hasLowerBound() || !boundedUntil->hasUpperBound()) {
254 return false;
255 } else if (boundedUntil->getUpperBound().evaluateAsDouble() == boundedUntil->getLowerBound().evaluateAsDouble()) {
256 return true;
257 }
258
259 return false;
260 }

331 Bdd binaryStateFormulaToBdd(BinaryStateFormulaCPointer const &formula) const {
332 auto const leftBdd{FormulaToBdd(formula->getLeftSubformula().asSharedPointer())};
333 auto const rightBdd{FormulaToBdd(formula->getRightSubformula().asSharedPointer())};
334
335 if (formula->isAnd()) {

337 } else if (formula->isOr()) {
