263template<typename SparseCtmcModelType>
264std::unique_ptr<CheckResult> SparseCtmcCslModelChecker<SparseCtmcModelType>::computeSteadyStateDistribution(Environment const& env) {
265 // Initialize helper
266 auto probabilisticTransitions = this->getModel().computeProbabilityMatrix();
267 storm::modelchecker::helper::SparseDeterministicInfiniteHorizonHelper<ValueType> helper(probabilisticTransitions, this->getModel().getExitRateVector());
268
269 // Compute result
270 std::vector<ValueType> result;
271 auto const& initialStates = this->getModel().getInitialStates();
272 uint64_t numInitStates = initialStates.getNumberOfSetBits();
273 if (numInitStates == 1) {
274 result = helper.computeLongRunAverageStateDistribution(env, *initialStates.begin());
275 } else {
276 STORM_LOG_WARN("Multiple initial states found. A uniform distribution over initial states is assumed.");
277 ValueType initProb = storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(numInitStates);
278 result = helper.computeLongRunAverageStateDistribution(env, [&initialStates, &initProb](uint64_t const& stateIndex) {
279 return initialStates.get(stateIndex) ? initProb : storm::utility::zero<ValueType>();
280 });
281 }
282
283 // Return CheckResult
284 return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(result)));
285}
