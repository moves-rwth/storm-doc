273 template<typename SparseCtmcModelType>
274 std::unique_ptr<CheckResult> SparseCtmcCslModelChecker<SparseCtmcModelType>::computeSteadyStateDistribution(Environment const& env) {
275  // Initialize helper
276  auto probabilisticTransitions = this->getModel().computeProbabilityMatrix();
277  storm::modelchecker::helper::SparseDeterministicInfiniteHorizonHelper<ValueType> helper(probabilisticTransitions, this->getModel().getExitRateVector());
278 
279  // Compute result
280  std::vector<ValueType> result;
281  auto const& initialStates = this->getModel().getInitialStates();
282  uint64_t numInitStates = initialStates.getNumberOfSetBits();
283  if (numInitStates == 1) {
284  result = helper.computeLongRunAverageStateDistribution(env, *initialStates.begin());
285  } else {
286  STORM_LOG_WARN("Multiple initial states found. A uniform distribution over initial states is assumed.");
287  ValueType initProb = storm::utility::one<ValueType>() / storm::utility::convertNumber<ValueType, uint64_t>(numInitStates);
288  result = helper.computeLongRunAverageStateDistribution(env, [&initialStates, &initProb](uint64_t const& stateIndex) {
289  return initialStates.get(stateIndex) ? initProb : storm::utility::zero<ValueType>();
290  });
291  }
292 
293  // Return CheckResult
294  return std::unique_ptr<CheckResult>(new ExplicitQuantitativeCheckResult<ValueType>(std::move(result)));
295 }
