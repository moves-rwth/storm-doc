10template<typename ParametricType, typename ConstantType>
11ParameterLifter<ParametricType, ConstantType>::ParameterLifter(storm::storage::SparseMatrix<ParametricType> const& pMatrix,
12 std::vector<ParametricType> const& pVector, storm::storage::BitVector const& selectedRows,
13 storm::storage::BitVector const& selectedColumns, bool generateRowLabels,
14 bool useMonotonicityInFuture) {
15 // get a mapping from old column indices to new ones
16 oldToNewColumnIndexMapping = std::vector<uint_fast64_t>(selectedColumns.size(), selectedColumns.size());
17 uint_fast64_t newIndex = 0;
18 for (auto const& oldColumn : selectedColumns) {
19 oldToNewColumnIndexMapping[oldColumn] = newIndex++;
20 }
21
22 // create vector, such that the occuringVariables for all states can be stored
23 occurringVariablesAtState = std::vector<std::set<VariableType>>(pMatrix.getColumnCount());
24
25 // Stores which entries of the original matrix/vector are non-constant. Entries for non-selected rows/columns are omitted
26 auto nonConstMatrixEntries = storm::storage::BitVector(pMatrix.getEntryCount(), false); // this vector has to be resized later
27 auto nonConstVectorEntries = storm::storage::BitVector(selectedRows.getNumberOfSetBits(), false);
28 // Counters for selected entries in the pMatrix and the pVector
29 uint_fast64_t pMatrixEntryCount = 0;
30 uint_fast64_t pVectorEntryCount = 0;
31
32 // The matrix builder for the new matrix. The correct number of rows and entries is not known yet.
33 storm::storage::SparseMatrixBuilder<ConstantType> builder(0, selectedColumns.getNumberOfSetBits(), 0, true, true, selectedRows.getNumberOfSetBits());
34 rowGroupToStateNumber = std::vector<uint_fast64_t>();
35 uint_fast64_t newRowIndex = 0;
36 for (auto const& rowIndex : selectedRows) {
37 builder.newRowGroup(newRowIndex);
38 rowGroupToStateNumber.push_back(rowIndex);
39
40 // Gather the occurring variables within this row and set which entries are non-constant
41 std::set<VariableType> occurringVariables;
42 for (auto const& entry : pMatrix.getRow(rowIndex)) {
43 if (selectedColumns.get(entry.getColumn())) {
44 if (!storm::utility::isConstant(entry.getValue())) {
45 storm::utility::parametric::gatherOccurringVariables(entry.getValue(), occurringVariables);
46 nonConstMatrixEntries.set(pMatrixEntryCount, true);
47 }
48 ++pMatrixEntryCount;
49 } else {
50 if (!storm::utility::isConstant(entry.getValue())) {
51 storm::utility::parametric::gatherOccurringVariables(entry.getValue(), occurringVariables);
52 }
53 }
54 }
55
56 ParametricType const& pVectorEntry = pVector[rowIndex];
57 std::set<VariableType> vectorEntryVariables;
58 if (!storm::utility::isConstant(pVectorEntry)) {
59 storm::utility::parametric::gatherOccurringVariables(pVectorEntry, vectorEntryVariables);
60 if (generateRowLabels) {
61 // If row labels are to be generated, we do not allow unspecified valuations.
62 // Therefore, we also 'lift' parameters that only occurr on a vector.
63 occurringVariables.insert(vectorEntryVariables.begin(), vectorEntryVariables.end());
64 }
65 nonConstVectorEntries.set(pVectorEntryCount, true);
66 }
67 ++pVectorEntryCount;
68 // Compute the (abstract) valuation for each row
69 auto rowValuations = getVerticesOfAbstractRegion(occurringVariables);
70 for (auto const& val : rowValuations) {
71 if (generateRowLabels) {
72 rowLabels.push_back(val);
73 }
74
75 // Insert matrix entries for each valuation. For non-constant entries, a dummy value is inserted and the function and the valuation are collected.
76 // The placeholder for the collected function/valuation are stored in the matrixAssignment. The matrixAssignment is completed after the matrix is
77 // finished
78 for (auto const& entry : pMatrix.getRow(rowIndex)) {
79 if (selectedColumns.get(entry.getColumn())) {
80 if (storm::utility::isConstant(entry.getValue())) {
81 builder.addNextValue(newRowIndex, oldToNewColumnIndexMapping[entry.getColumn()],
82 storm::utility::convertNumber<ConstantType>(entry.getValue()));
83 } else {
84 builder.addNextValue(newRowIndex, oldToNewColumnIndexMapping[entry.getColumn()], storm::utility::one<ConstantType>());
85 ConstantType& placeholder = functionValuationCollector.add(entry.getValue(), val);
86 matrixAssignment.push_back(std::pair<typename storm::storage::SparseMatrix<ConstantType>::iterator, ConstantType&>(
87 typename storm::storage::SparseMatrix<ConstantType>::iterator(), placeholder));
88 }
89 }
90 }
91
92 // Insert the vector entry for this row
93 if (storm::utility::isConstant(pVectorEntry)) {
94 vector.push_back(storm::utility::convertNumber<ConstantType>(pVectorEntry));
95 } else {
96 vector.push_back(storm::utility::one<ConstantType>());
97 AbstractValuation vectorVal(val);
98 for (auto const& vectorVar : vectorEntryVariables) {
99 if (occurringVariables.find(vectorVar) == occurringVariables.end()) {
100 assert(!generateRowLabels);
101 vectorVal.addParameterUnspecified(vectorVar);
102 }
103 }
104 ConstantType& placeholder = functionValuationCollector.add(pVectorEntry, vectorVal);
105 vectorAssignment.push_back(
106 std::pair<typename std::vector<ConstantType>::iterator, ConstantType&>(typename std::vector<ConstantType>::iterator(), placeholder));
107 }
108
109 ++newRowIndex;
110 }
111 if (useMonotonicityInFuture) {
112 // Save the occuringVariables of a state, needed if we want to use monotonicity
113 for (auto& var : occurringVariables) {
114 occuringStatesAtVariable[var].insert(rowIndex);
115 }
116 occurringVariablesAtState[rowIndex] = std::move(occurringVariables);
117 }
118 }
119
120 // Matrix and vector are now filled with constant results from constant functions and place holders for non-constant functions.
121 matrix = builder.build(newRowIndex);
122 vector.shrink_to_fit();
123 matrixAssignment.shrink_to_fit();
124 vectorAssignment.shrink_to_fit();
125 nonConstMatrixEntries.resize(pMatrixEntryCount);
126
127 // Now insert the correct iterators for the matrix and vector assignment
128 auto matrixAssignmentIt = matrixAssignment.begin();
129 uint_fast64_t startEntryOfRow = 0;
130 for (uint_fast64_t group = 0; group < matrix.getRowGroupCount(); ++group) {
131 uint_fast64_t startEntryOfNextRow = startEntryOfRow + matrix.getRow(group, 0).getNumberOfEntries();
132 for (uint_fast64_t matrixRow = matrix.getRowGroupIndices()[group]; matrixRow < matrix.getRowGroupIndices()[group + 1]; ++matrixRow) {
133 auto matrixEntryIt = matrix.getRow(matrixRow).begin();
134 for (uint_fast64_t nonConstEntryIndex = nonConstMatrixEntries.getNextSetIndex(startEntryOfRow); nonConstEntryIndex < startEntryOfNextRow;
135 nonConstEntryIndex = nonConstMatrixEntries.getNextSetIndex(nonConstEntryIndex + 1)) {
136 matrixAssignmentIt->first = matrixEntryIt + (nonConstEntryIndex - startEntryOfRow);
137 ++matrixAssignmentIt;
138 }
139 }
140 startEntryOfRow = startEntryOfNextRow;
141 }
142 STORM_LOG_ASSERT(matrixAssignmentIt == matrixAssignment.end(), "Unexpected number of entries in the matrix assignment.");
143
144 auto vectorAssignmentIt = vectorAssignment.begin();
145 for (auto const& nonConstVectorEntry : nonConstVectorEntries) {
146 for (uint_fast64_t vectorIndex = matrix.getRowGroupIndices()[nonConstVectorEntry]; vectorIndex != matrix.getRowGroupIndices()[nonConstVectorEntry + 1];
147 ++vectorIndex) {
148 vectorAssignmentIt->first = vector.begin() + vectorIndex;
149 ++vectorAssignmentIt;
150 }
151 }
152 STORM_LOG_ASSERT(vectorAssignmentIt == vectorAssignment.end(), "Unexpected number of entries in the vector assignment.");
153}

156void ParameterLifter<ParametricType, ConstantType>::specifyRegion(storm::storage::ParameterRegion<ParametricType> const& region,
157 storm::solver::OptimizationDirection const& dirForParameters) {
158 // write the evaluation result of each function,evaluation pair into the placeholders
159 functionValuationCollector.evaluateCollectedFunctions(region, dirForParameters);
160
161 // apply the matrix and vector assignments to write the contents of the placeholder into the matrix/vector
162 for (auto& assignment : matrixAssignment) {
163 STORM_LOG_WARN_COND(
164 !storm::utility::isZero(assignment.second),
165 "Parameter lifting on region "
166 << region.toString()
167 << " affects the underlying graph structure (the region is not strictly well defined). The result for this region might be incorrect.");
168 assignment.first->setValue(assignment.second);
169 }
170
171 for (auto& assignment : vectorAssignment) {
172 *assignment.first = assignment.second;
173 }

211template<typename ParametricType, typename ConstantType>
212std::vector<typename ParameterLifter<ParametricType, ConstantType>::AbstractValuation>
213ParameterLifter<ParametricType, ConstantType>::getVerticesOfAbstractRegion(std::set<VariableType> const& variables) const {
214 std::size_t const numOfVertices = std::pow(2, variables.size());
215 std::vector<AbstractValuation> result(numOfVertices);
216
217 for (uint_fast64_t vertexId = 0; vertexId < numOfVertices; ++vertexId) {
218 // interprete vertexId as a bit sequence
219 // the consideredVariables.size() least significant bits of vertex will always represent the next vertex
220 //(00...0 = lower boundaries for all variables, 11...1 = upper boundaries for all variables)
221 uint_fast64_t variableIndex = 0;
222 for (auto const& variable : variables) {
223 if ((vertexId >> variableIndex) % 2 == 0) {
224 result[vertexId].addParameterLower(variable);
225 } else {
226 result[vertexId].addParameterUpper(variable);
227 }
228 ++variableIndex;
229 }
230 }
231 return result;
232}

281template<typename ParametricType, typename ConstantType>
282typename ParameterLifter<ParametricType, ConstantType>::AbstractValuation ParameterLifter<ParametricType, ConstantType>::AbstractValuation::getSubValuation(
283 std::set<VariableType> const& pars) const {
284 AbstractValuation result;
285 for (auto const& p : pars) {
286 if (std::find(lowerPars.begin(), lowerPars.end(), p) != lowerPars.end()) {
287 result.addParameterLower(p);
288 } else if (std::find(upperPars.begin(), upperPars.end(), p) != upperPars.end()) {
289 result.addParameterUpper(p);
290 } else if (std::find(unspecifiedPars.begin(), unspecifiedPars.end(), p) != unspecifiedPars.end()) {
291 result.addParameterUnspecified(p);
292 } else {
293 STORM_LOG_THROW(false, storm::exceptions::UnexpectedException,
294 "Tried to obtain a subvaluation for parameters that are not specified by this valuation");
295 }
296 }
297 return result;
298}

333template<typename ParametricType, typename ConstantType>
334ConstantType& ParameterLifter<ParametricType, ConstantType>::FunctionValuationCollector::add(ParametricType const& function,
335 AbstractValuation const& valuation) {
336 ParametricType simplifiedFunction = function;
337 storm::utility::simplify(simplifiedFunction);
338 std::set<VariableType> variablesInFunction;
339 storm::utility::parametric::gatherOccurringVariables(simplifiedFunction, variablesInFunction);
340 AbstractValuation simplifiedValuation = valuation.getSubValuation(variablesInFunction);
341 // insert the function and the valuation
342 // Note that references to elements of an unordered map remain valid after calling unordered_map::insert.
343 auto insertionRes = collectedFunctions.insert(std::pair<FunctionValuation, ConstantType>(
344 FunctionValuation(std::move(simplifiedFunction), std::move(simplifiedValuation)), storm::utility::one<ConstantType>()));
345 return insertionRes.first->second;
346}

348template<typename ParametricType, typename ConstantType>
349void ParameterLifter<ParametricType, ConstantType>::FunctionValuationCollector::evaluateCollectedFunctions(
350 storm::storage::ParameterRegion<ParametricType> const& region, storm::solver::OptimizationDirection const& dirForUnspecifiedParameters) {
351 for (auto& collectedFunctionValuationPlaceholder : collectedFunctions) {
352 ParametricType const& function = collectedFunctionValuationPlaceholder.first.first;
353 AbstractValuation const& abstrValuation = collectedFunctionValuationPlaceholder.first.second;
354 ConstantType& placeholder = collectedFunctionValuationPlaceholder.second;
355 auto concreteValuations = abstrValuation.getConcreteValuations(region);
356 auto concreteValuationIt = concreteValuations.begin();
357 placeholder = storm::utility::parametric::evaluate<ConstantType>(function, *concreteValuationIt);
358 for (++concreteValuationIt; concreteValuationIt != concreteValuations.end(); ++concreteValuationIt) {
359 ConstantType currentResult = storm::utility::parametric::evaluate<ConstantType>(function, *concreteValuationIt);
360 if (storm::solver::minimize(dirForUnspecifiedParameters)) {
361 placeholder = std::min(placeholder, currentResult);
362 } else {
363 placeholder = std::max(placeholder, currentResult);
364 }
365 }
366 }
