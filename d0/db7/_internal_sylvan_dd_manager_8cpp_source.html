141BDD InternalDdManager<DdType::Sylvan>::getBddEncodingLessOrEqualThanRec(uint64_t minimalValue, uint64_t maximalValue, uint64_t bound, BDD cube,
142 uint64_t remainingDdVariables) const {
143 if (maximalValue <= bound) {
144 return sylvan_true;
145 } else if (minimalValue > bound) {
146 return sylvan_false;
147 }
148
149 STORM_LOG_ASSERT(remainingDdVariables > 0, "Expected more remaining DD variables.");
150 uint64_t newRemainingDdVariables = remainingDdVariables - 1;
151 BDD elseResult =
152 getBddEncodingLessOrEqualThanRec(minimalValue, maximalValue & ~(1ull << newRemainingDdVariables), bound, sylvan_high(cube), newRemainingDdVariables);
153 bdd_refs_push(elseResult);
154 BDD thenResult =
155 getBddEncodingLessOrEqualThanRec(minimalValue | (1ull << newRemainingDdVariables), maximalValue, bound, sylvan_high(cube), newRemainingDdVariables);
156 bdd_refs_push(elseResult);
157 BDD result = sylvan_makenode(sylvan_var(cube), elseResult, thenResult);
158 bdd_refs_pop(2);
159 return result;
160}

246void InternalDdManager<DdType::Sylvan>::execute(std::function<void()> const& f) const {
247 // Only wake up the sylvan (i.e. lace) threads when they are suspended.
248 std::exception_ptr e = nullptr; // propagate exception
249 if (suspended) {
250 lace_resume();
251 suspended = false;
252 RUN(execute_sylvan, &f, &e);
253 lace_suspend();
254 suspended = true;
255 } else {
256 // The sylvan threads are already running, don't suspend afterwards.
257 RUN(execute_sylvan, &f, &e);
258 }
259 if (e) {
260 std::rethrow_exception(e);
261 }
