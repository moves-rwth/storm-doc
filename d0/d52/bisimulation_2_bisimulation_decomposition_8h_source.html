21inline BisimulationType resolveBisimulationTypeChoice(BisimulationTypeChoice c) {
22 switch (c) {
23 case BisimulationTypeChoice::Strong:
24 return BisimulationType::Strong;
25 case BisimulationTypeChoice::Weak:
26 return BisimulationType::Weak;
27 case BisimulationTypeChoice::FromSettings:
28 if (storm::settings::getModule<storm::settings::modules::BisimulationSettings>().isWeakBisimulationSet()) {
29 return BisimulationType::Weak;
30 } else {
31 return BisimulationType::Strong;
32 }
33 }
34 return BisimulationType::Strong;
35}
