119storm::storage::PomdpMemoryPattern POMDPSettings::getMemoryPattern() const {
120 auto pattern = this->getOption(memoryPatternOption).getArgumentByName("name").getValueAsString();
121 if (pattern == "trivial") {
122 return storm::storage::PomdpMemoryPattern::Trivial;
123 } else if (pattern == "fixedcounter") {
124 return storm::storage::PomdpMemoryPattern::FixedCounter;
125 } else if (pattern == "selectivecounter") {
126 return storm::storage::PomdpMemoryPattern::SelectiveCounter;
127 } else if (pattern == "ring") {
128 return storm::storage::PomdpMemoryPattern::SelectiveRing;
129 } else if (pattern == "fixedring") {
130 return storm::storage::PomdpMemoryPattern::FixedRing;
131 } else if (pattern == "settablebits") {
132 return storm::storage::PomdpMemoryPattern::SettableBits;
133 } else if (pattern == "full") {
134 return storm::storage::PomdpMemoryPattern::Full;
135 }
136 STORM_LOG_THROW(false, storm::exceptions::InvalidArgumentException, "The name of the memory pattern is unknown.");
137}
