258void BitVector::resize(uint_fast64_t newLength, bool init) {
259 if (newLength > bitCount) {
260 uint_fast64_t newBucketCount = newLength >> 6;
261 if ((newLength & mod64mask) != 0) {
262 ++newBucketCount;
263 }
264
265 if (newBucketCount > this->bucketCount()) {
266 uint64_t* newBuckets = new uint64_t[newBucketCount];
267 std::copy_n(buckets, this->bucketCount(), newBuckets);
268 if (init) {
269 if (this->bucketCount() > 0) {
270 newBuckets[this->bucketCount() - 1] |= ((1ull << (64 - (bitCount & mod64mask))) - 1ull);
271 }
272 std::fill_n(newBuckets + this->bucketCount(), newBucketCount - this->bucketCount(), -1ull);
273 } else {
274 std::fill_n(newBuckets + this->bucketCount(), newBucketCount - this->bucketCount(), 0);
275 }
276 delete[] buckets;
277 buckets = newBuckets;
278 bitCount = newLength;
279 } else {
280 // If the underlying storage does not need to grow, we have to insert the missing bits.
281 if (init) {
282 buckets[this->bucketCount() - 1] |= ((1ull << (64 - (bitCount & mod64mask))) - 1ull);
283 }
284 bitCount = newLength;
285 }
286 truncateLastBucket();
287 } else {
288 uint_fast64_t newBucketCount = newLength >> 6;
289 if ((newLength & mod64mask) != 0) {
290 ++newBucketCount;
291 }
292
293 // If the number of buckets needs to be reduced, we resize it now. Otherwise, we can just truncate the
294 // last bucket.
295 if (newBucketCount < this->bucketCount()) {
296 uint64_t* newBuckets = new uint64_t[newBucketCount];
297 std::copy_n(buckets, newBucketCount, newBuckets);
298 delete[] buckets;
299 buckets = newBuckets;
300 bitCount = newLength;
301 }
302 bitCount = newLength;
303 truncateLastBucket();
304 }

415void BitVector::increment() {
416 uint64_t firstUnsetIndex = getNextUnsetIndex(0);
417
418 // If there is no unset index, we clear the whole vector
419 if (firstUnsetIndex == this->bitCount) {
420 this->clear();
421 } else {
422 // All previous buckets have to be set to zero
423 uint64_t bucketIndex = firstUnsetIndex >> 6;
424 std::fill_n(buckets, bucketIndex, 0);
425
426 // modify the bucket in which the unset entry lies in
427 uint64_t& bucket = this->buckets[bucketIndex];
428 uint64_t indexInBucket = firstUnsetIndex & mod64mask;
429 if (indexInBucket > 0) {
430 // Clear all bits before the index
431 uint64_t mask = ~(-1ull << (64 - indexInBucket));
432 bucket &= mask;
433 }
434
435 // Set the bit at the index
436 uint64_t mask = 1ull << (63 - indexInBucket);
437 bucket |= mask;
438 }

560uint_fast64_t BitVector::getAsInt(uint_fast64_t bitIndex, uint_fast64_t numberOfBits) const {
561 STORM_LOG_ASSERT(numberOfBits <= 64, "Number of bits must be <= 64.");
562 uint64_t bucket = bitIndex >> 6;
563 uint64_t bitIndexInBucket = bitIndex & mod64mask;
564
565 uint64_t mask;
566 if (bitIndexInBucket == 0) {
567 mask = -1ull;
568 } else {
569 mask = (1ull << (64 - bitIndexInBucket)) - 1ull;
570 }
571
572 if (bitIndexInBucket + numberOfBits < 64) {
573 // If the value stops before the end of the bucket, we need to erase some lower bits.
574 mask &= ~((1ull << (64 - (bitIndexInBucket + numberOfBits))) - 1ull);
575 return (buckets[bucket] & mask) >> (64 - (bitIndexInBucket + numberOfBits));
576 } else if (bitIndexInBucket + numberOfBits > 64) {
577 // In this case, the integer "crosses" the bucket line.
578 uint64_t result = (buckets[bucket] & mask);
579 ++bucket;
580
581 // Compute the remaining number of bits.
582 numberOfBits -= (64 - bitIndexInBucket);
583
584 // Shift the intermediate result to the right location.
585 result <<= numberOfBits;
586
587 // Strip away everything from the second bucket that is beyond the final index and add it to the
588 // intermediate result.
589 mask = ~((1ull << (64 - numberOfBits)) - 1ull);
590 uint64_t lowerBits = buckets[bucket] & mask;
591 result |= (lowerBits >> (64 - numberOfBits));
592
593 return result;
594 } else {
595 // In this case, it suffices to take the current mask.
596 return buckets[bucket] & mask;
597 }
598}

623void BitVector::setFromInt(uint_fast64_t bitIndex, uint_fast64_t numberOfBits, uint64_t value) {
624 STORM_LOG_ASSERT(numberOfBits <= 64, "Number of bits must be <= 64.");
625 STORM_LOG_ASSERT(numberOfBits == 64 || (value >> numberOfBits) == 0,
626 "Integer value (" << value << ") too large to fit in the given number of bits (" << numberOfBits << ").");
627
628 uint64_t bucket = bitIndex >> 6;
629 uint64_t bitIndexInBucket = bitIndex & mod64mask;
630
631 uint64_t mask;
632 if (bitIndexInBucket == 0) {
633 mask = -1ull;
634 } else {
635 mask = (1ull << (64 - bitIndexInBucket)) - 1ull;
636 }
637
638 if (bitIndexInBucket + numberOfBits < 64) {
639 // If the value stops before the end of the bucket, we need to erase some lower bits.
640 mask &= ~((1ull << (64 - (bitIndexInBucket + numberOfBits))) - 1ull);
641 buckets[bucket] = (buckets[bucket] & ~mask) | (value << (64 - (bitIndexInBucket + numberOfBits)));
642 } else if (bitIndexInBucket + numberOfBits > 64) {
643 // Write the part of the value that falls into the first bucket.
644 buckets[bucket] = (buckets[bucket] & ~mask) | (value >> (numberOfBits + (bitIndexInBucket - 64)));
645 ++bucket;
646
647 // Compute the remaining number of bits.
648 numberOfBits -= (64 - bitIndexInBucket);
649
650 // Shift the bits of the value such that the already set bits disappear.
651 value <<= (64 - numberOfBits);
652
653 // Put the remaining bits in their place.
654 mask = ((1ull << (64 - numberOfBits)) - 1ull);
655 buckets[bucket] = (buckets[bucket] & mask) | value;
656 } else {
657 buckets[bucket] = (buckets[bucket] & ~mask) | value;
658 }
659}

661bool BitVector::empty() const {
662 uint64_t* last = buckets + bucketCount();
663 uint64_t* it = std::find_if(buckets, last, [](uint64_t const& a) { return a != 0; });
664 return it == last;
665}

667bool BitVector::full() const {
668 if (bitCount == 0) {
669 return true;
670 }
671 // Check that all buckets except the last one have all bits set.
672 uint64_t* last = buckets + bucketCount() - 1;
673 for (uint64_t const* it = buckets; it < last; ++it) {
674 if (*it != -1ull) {
675 return false;
676 }
677 }
678
679 // Now check whether the relevant bits are set in the last bucket.
680 uint64_t mask = ~((1ull << (64 - (bitCount & mod64mask))) - 1ull);
681 if ((*last & mask) != mask) {
682 return false;
683 }
684 return true;
685}

700uint_fast64_t BitVector::getNumberOfSetBitsBeforeIndex(uint_fast64_t index) const {
701 uint_fast64_t result = 0;
702
703 // First, count all full buckets.
704 uint_fast64_t bucket = index >> 6;
705 for (uint_fast64_t i = 0; i < bucket; ++i) {
706 // Check if we are using g++ or clang++ and, if so, use the built-in function
707#if (defined(__GNUG__) || defined(__clang__))
708 result += __builtin_popcountll(buckets[i]);
709#elif defined WINDOWS
710#include <nmmintrin.h>
711 // If the target machine does not support SSE4, this will fail.
712 result += _mm_popcnt_u64(bucketVector[i]);
713#else
714 uint_fast32_t cnt;
715 uint_fast64_t bitset = buckets[i];
716 for (cnt = 0; bitset; cnt++) {
717 bitset &= bitset - 1;
718 }
719 result += cnt;
720#endif
721 }
722
723 // Now check if we have to count part of a bucket.
724 uint64_t tmp = index & mod64mask;
725 if (tmp != 0) {
726 tmp = ~((1ll << (64 - (tmp & mod64mask))) - 1ll);
727 tmp &= buckets[bucket];
728 // Check if we are using g++ or clang++ and, if so, use the built-in function
729#if (defined(__GNUG__) || defined(__clang__))
730 result += __builtin_popcountll(tmp);
731#else
732 uint_fast32_t cnt;
733 uint64_t bitset = tmp;
734 for (cnt = 0; bitset; cnt++) {
735 bitset &= bitset - 1;
736 }
737 result += cnt;
738#endif
739 }
740
741 return result;
742}

814template<bool Value, bool Backward>
815uint_fast64_t BitVector::getNextIndexWithValue(uint64_t const* dataPtr, uint64_t startingIndex, uint64_t endIndex) {
816 if (startingIndex >= endIndex) {
817 return Backward ? startingIndex : endIndex;
818 }
819
820 uint64_t currentBucketIndexOffset = Backward ? endIndex - 1 : startingIndex;
821 uint_fast8_t currentBitInBucket = currentBucketIndexOffset & mod64mask;
822 uint64_t const* bucketIt = dataPtr + (currentBucketIndexOffset >> 6);
823 currentBucketIndexOffset = (currentBucketIndexOffset >> 6 << 6);
824
825 // Get relevant contents of the first bucket (the one that contains the bit with index currentBucketIndexOffset + currentBitInBucket)
826 uint64_t relevantBitsInBucket;
827 if constexpr (Backward) {
828 relevantBitsInBucket = -1ull << (63 - currentBitInBucket); // 111..111'1'000...000 where the last '1' is at the currentBitInBucket
829 } else {
830 relevantBitsInBucket = -1ull >> currentBitInBucket; // 000..000'1'111..111 where the first '1' is at the currentBitInBucket
831 }
832 uint64_t currentBucket = Value ? (*bucketIt & relevantBitsInBucket) : (*bucketIt | ~relevantBitsInBucket);
833
834 // Find the right bucket
835 if (currentBucket == (Value ? 0ull : -1ull)) {
836 // The first bucket does not contain a bit with the desired value...
837 do {
838 // Move to next bucket (if there is some)
839 if constexpr (Backward) {
840 if (currentBucketIndexOffset <= startingIndex) {
841 // No bucket found!
842 return startingIndex;
843 }
844 --bucketIt;
845 currentBucketIndexOffset -= 64; // does not underflow: currentBucketIndexOffset is greater than startIndex and always a multiple of 64
846 } else {
847 ++bucketIt;
848 currentBucketIndexOffset += 64;
849 if (currentBucketIndexOffset >= endIndex) {
850 // No bucket found!
851 return endIndex;
852 }
853 }
854 // Check if the bucket contains our bit
855 } while ((*bucketIt) == (Value ? 0ull : -1ull));
856 // At this point we have found our bucket, but it is not the first one
857 currentBucket = *bucketIt;
858 currentBitInBucket = Backward ? 63u : 0u; // search within the bucket starting at the last or at the first bit
859 }
860
861 if constexpr (!Value) {
862 currentBucket = ~currentBucket; // invert so that we always search for a '1' from this point
863 }
864 // At this point, currentBucket definitely contains a 1-bit and all bits (Backward ? after : before) the currentBitInBucket are zero
865 STORM_LOG_ASSERT(currentBucket != 0ull, "Bitvector's getNextIndexWithValue method in invalid state.");
866
867#if (defined(__GNUG__) || defined(__clang__))
868 // Use fast and easy builtin functions to find the correct bit index
869 if constexpr (Backward) {
870 // take max since the startIndex might point somewhere into the current bucket so the found bit might come before the startIndex
871 return std::max<uint64_t>(startingIndex,
872 currentBucketIndexOffset + 64ull - __builtin_ctzll(currentBucket)); // make sure to return +1 index after the found 1
873 } else {
874 // take min since the endIndex might point somewhere into the current bucket so the found bit might come after the endIndex
875 return std::min<uint64_t>(endIndex, currentBucketIndexOffset + __builtin_clzll(currentBucket));
876 }
877#else
878 // Find the correct bit index manually
879 uint64_t compareMask = 1ull << (63 - currentBitInBucket); // 000..000'1'000..000 with '1' at currentBitInBucket position
880 while (!static_cast<bool>(currentBucket & compareMask)) {
881 if constexpr (Backward) {
882 compareMask <<= 1ull;
883 --currentBitInBucket;
884 } else {
885 compareMask >>= 1ull;
886 ++currentBitInBucket;
887 }
888 }
889 if constexpr (Backward) {
890 return std::max(startingIndex, currentBucketIndexOffset + currentBitInBucket + 1ull); // make sure to return +1 index after the found 1
891 } else {
892 return std::min(endIndex, currentBucketIndexOffset + currentBitInBucket);
893 }
894#endif
895}

897storm::storage::BitVector BitVector::getAsBitVector(uint_fast64_t start, uint_fast64_t length) const {
898 STORM_LOG_ASSERT(start + length <= bitCount, "Invalid range.");
899#ifdef ASSERT_BITVECTOR
900 BitVector original(*this);
901#endif
902 storm::storage::BitVector result(length, false);
903
904 uint_fast64_t offset = start % 64;
905 uint64_t* getBucket = buckets + (start / 64);
906 uint64_t* insertBucket = result.buckets;
907 uint_fast64_t getValue;
908 uint_fast64_t writeValue = 0;
909 uint_fast64_t noBits = 0;
910 if (offset == 0) {
911 // Copy complete buckets
912 for (; noBits + 64 <= length; ++getBucket, ++insertBucket, noBits += 64) {
913 *insertBucket = *getBucket;
914 }
915 } else {
916 // Get first bits up until next bucket
917 getValue = *getBucket;
918 writeValue = (getValue << offset);
919 noBits += (64 - offset);
920 ++getBucket;
921
922 // Get complete buckets
923 for (; noBits + 64 <= length; ++getBucket, ++insertBucket, noBits += 64) {
924 getValue = *getBucket;
925 // Get bits till write bucket is full
926 writeValue |= (getValue >> (64 - offset));
927 *insertBucket = writeValue;
928 // Get bits up until next bucket
929 writeValue = (getValue << offset);
930 }
931 }
932
933 // Write last bits
934 uint_fast64_t remainingBits = length - noBits;
935 STORM_LOG_ASSERT(getBucket != buckets + bucketCount(), "Bucket index incorrect.");
936 // Get remaining bits
937 getValue = (*getBucket >> (64 - remainingBits)) << (64 - remainingBits);
938 STORM_LOG_ASSERT(remainingBits < 64, "Too many remaining bits.");
939 // Write bucket
940 STORM_LOG_ASSERT(insertBucket != result.buckets + result.bucketCount(), "Bucket index incorrect.");
941 if (offset == 0) {
942 *insertBucket = getValue;
943 } else {
944 writeValue |= getValue >> (64 - offset);
945 *insertBucket = writeValue;
946 if (remainingBits > offset) {
947 // Write last bits in new value
948 writeValue = (getValue << offset);
949 ++insertBucket;
950 STORM_LOG_ASSERT(insertBucket != result.buckets + result.bucketCount(), "Bucket index incorrect.");
951 *insertBucket = writeValue;
952 }
953 }
954
955#ifdef ASSERT_BITVECTOR
956 // Check correctness of getter
957 for (uint_fast64_t i = 0; i < length; ++i) {
958 if (result.get(i) != get(start + i)) {
959 STORM_LOG_ERROR("Getting of bits not correct for index " << i);
960 STORM_LOG_ERROR("Getting from " << start << " with length " << length);
961 std::stringstream stream;
962 printBits(stream);
963 stream << '\n';
964 result.printBits(stream);
965 STORM_LOG_ERROR(stream.str());
966 STORM_LOG_ASSERT(false, "Getting of bits not correct.");
967 }
968 }
969 for (uint_fast64_t i = 0; i < bitCount; ++i) {
970 if (i < start || i >= start + length) {
971 if (original.get(i) != get(i)) {
972 STORM_LOG_ERROR("Getting did change bitvector at index " << i);
973 STORM_LOG_ERROR("Getting from " << start << " with length " << length);
974 std::stringstream stream;
975 printBits(stream);
976 stream << '\n';
977 original.printBits(stream);
978 STORM_LOG_ERROR(stream.str());
979 STORM_LOG_ASSERT(false, "Getting of bits not correct.");
980 }
981 }
982 }
983
984#endif
985 return result;
986}

988void BitVector::setFromBitVector(uint_fast64_t start, BitVector const& other) {
989#ifdef ASSERT_BITVECTOR
990 BitVector original(*this);
991#endif
992 STORM_LOG_ASSERT(start + other.bitCount <= bitCount, "Range invalid.");
993
994 uint_fast64_t offset = start % 64;
995 uint64_t* insertBucket = buckets + (start / 64);
996 uint64_t* getBucket = other.buckets;
997 uint_fast64_t getValue;
998 uint_fast64_t writeValue = 0;
999 uint_fast64_t noBits = 0;
1000 if (offset == 0) {
1001 // Copy complete buckets
1002 for (; noBits + 64 <= other.bitCount; ++insertBucket, ++getBucket, noBits += 64) {
1003 *insertBucket = *getBucket;
1004 }
1005 } else {
1006 // Get first bits up until next bucket
1007 getValue = *getBucket;
1008 writeValue = (*insertBucket >> (64 - offset)) << (64 - offset);
1009 writeValue |= (getValue >> offset);
1010 *insertBucket = writeValue;
1011 noBits += (64 - offset);
1012 ++insertBucket;
1013
1014 // Get complete buckets
1015 for (; noBits + 64 <= other.bitCount; ++insertBucket, noBits += 64) {
1016 // Get all remaining bits from other bucket
1017 writeValue = getValue << (64 - offset);
1018 // Get bits from next bucket
1019 ++getBucket;
1020 getValue = *getBucket;
1021 writeValue |= getValue >> offset;
1022 *insertBucket = writeValue;
1023 }
1024 }
1025
1026 // Write last bits
1027 uint_fast64_t remainingBits = other.bitCount - noBits;
1028 STORM_LOG_ASSERT(remainingBits < 64, "Too many remaining bits.");
1029 STORM_LOG_ASSERT(insertBucket != buckets + bucketCount(), "Bucket index incorrect.");
1030 STORM_LOG_ASSERT(getBucket != other.buckets + other.bucketCount(), "Bucket index incorrect.");
1031 // Get remaining bits of bucket
1032 getValue = *getBucket;
1033 if (offset > 0) {
1034 getValue = getValue << (64 - offset);
1035 }
1036 // Get unchanged part of bucket
1037 writeValue = (*insertBucket << remainingBits) >> remainingBits;
1038 if (remainingBits > offset && offset > 0) {
1039 // Remaining bits do not come from one bucket -> consider next bucket
1040 ++getBucket;
1041 STORM_LOG_ASSERT(getBucket != other.buckets + other.bucketCount(), "Bucket index incorrect.");
1042 getValue |= *getBucket >> offset;
1043 }
1044 // Write completely
1045 writeValue |= getValue;
1046 *insertBucket = writeValue;
1047
1048#ifdef ASSERT_BITVECTOR
1049 // Check correctness of setter
1050 for (uint_fast64_t i = 0; i < other.bitCount; ++i) {
1051 if (other.get(i) != get(start + i)) {
1052 STORM_LOG_ERROR("Setting of bits not correct for index " << i);
1053 STORM_LOG_ERROR("Setting from " << start << " with length " << other.bitCount);
1054 std::stringstream stream;
1055 printBits(stream);
1056 stream << '\n';
1057 other.printBits(stream);
1058 STORM_LOG_ERROR(stream.str());
1059 STORM_LOG_ASSERT(false, "Setting of bits not correct.");
1060 }
1061 }
1062 for (uint_fast64_t i = 0; i < bitCount; ++i) {
1063 if (i < start || i >= start + other.bitCount) {
1064 if (original.get(i) != get(i)) {
1065 STORM_LOG_ERROR("Setting did change bitvector at index " << i);
1066 STORM_LOG_ERROR("Setting from " << start << " with length " << other.bitCount);
1067 std::stringstream stream;
1068 printBits(stream);
1069 stream << '\n';
1070 original.printBits(stream);
1071 STORM_LOG_ERROR(stream.str());
1072 STORM_LOG_ASSERT(false, "Setting of bits not correct.");
1073 }
1074 }
1075 }
1076#endif
1077}

1079bool BitVector::compareAndSwap(uint_fast64_t start1, uint_fast64_t start2, uint_fast64_t length) {
1080 if (length < 64) {
1081 // Just use one number
1082 uint_fast64_t elem1 = getAsInt(start1, length);
1083 uint_fast64_t elem2 = getAsInt(start2, length);
1084 if (elem1 < elem2) {
1085 // Swap elements
1086 setFromInt(start1, length, elem2);
1087 setFromInt(start2, length, elem1);
1088 return true;
1089 }
1090 return false;
1091 } else {
1092 // Use bit vectors
1093 BitVector elem1 = getAsBitVector(start1, length);
1094 BitVector elem2 = getAsBitVector(start2, length);
1095
1096 if (!(elem1 < elem2)) {
1097 // Elements already sorted
1098#ifdef ASSERT_BITVECTOR
1099 // Check that sorted
1100 for (uint_fast64_t i = 0; i < length; ++i) {
1101 if (get(start1 + i) > get(start2 + i)) {
1102 break;
1103 }
1104 STORM_LOG_ASSERT(get(start1 + i) >= get(start2 + i), "Bit vector not sorted for indices " << start1 + i << " and " << start2 + i);
1105 }
1106#endif
1107 return false;
1108 }
1109
1110#ifdef ASSERT_BITVECTOR
1111 BitVector check(*this);
1112#endif
1113
1114 // Swap elements
1115 setFromBitVector(start1, elem2);
1116 setFromBitVector(start2, elem1);
1117
1118#ifdef ASSERT_BITVECTOR
1119 // Check correctness of swapping
1120 bool tmp;
1121 for (uint_fast64_t i = 0; i < length; ++i) {
1122 tmp = check.get(i + start1);
1123 check.set(i + start1, check.get(i + start2));
1124 check.set(i + start2, tmp);
1125 }
1126 STORM_LOG_ASSERT(*this == check, "Swapping not correct");

1128 // Check that sorted

1131 break;

1134 }
1135#endif
1136
1137 return true;
1138 }
