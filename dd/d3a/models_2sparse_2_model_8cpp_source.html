34  template <typename ValueType, typename RewardModelType>
35  void Model<ValueType, RewardModelType>::assertValidityOfComponents(storm::storage::sparse::ModelComponents<ValueType, RewardModelType> const& components) const {
36 
37  // More costly checks are only asserted to avoid doing them in release mode.
38 
39  uint_fast64_t stateCount = this->getNumberOfStates();
40  uint_fast64_t choiceCount = this->getTransitionMatrix().getRowCount();
41 
42  // general components for all model types.
43  STORM_LOG_THROW(this->getTransitionMatrix().getColumnCount() == stateCount, storm::exceptions::IllegalArgumentException, "Invalid column count of transition matrix.");
44  STORM_LOG_ASSERT(components.rateTransitions || this->hasParameters() || this->getTransitionMatrix().isProbabilistic(), "The matrix is not probabilistic.");
45  STORM_LOG_THROW(this->getStateLabeling().getNumberOfItems() == stateCount, storm::exceptions::IllegalArgumentException, "Invalid item count (" << this->getStateLabeling().getNumberOfItems() << ") of state labeling (states: " << stateCount << ").");
46  for (auto const& rewardModel : this->getRewardModels()) {
47  STORM_LOG_THROW(!rewardModel.second.hasStateRewards() || rewardModel.second.getStateRewardVector().size() == stateCount, storm::exceptions::IllegalArgumentException, "Invalid size (" << rewardModel.second.getStateRewardVector().size() << ") of state reward vector (states:" << stateCount << ").");
48  STORM_LOG_THROW(!rewardModel.second.hasStateActionRewards() || rewardModel.second.getStateActionRewardVector().size() == choiceCount, storm::exceptions::IllegalArgumentException, "Invalid size of state reward vector.");
49  STORM_LOG_ASSERT(!rewardModel.second.hasTransitionRewards() || rewardModel.second.getTransitionRewardMatrix().isSubmatrixOf(this->getTransitionMatrix()), "The transition reward matrix is not a submatrix of the transition matrix, i.e. there are rewards for transitions that do not exist.");
50  }
51  STORM_LOG_THROW(!this->hasChoiceLabeling() || this->getChoiceLabeling().getNumberOfItems() == choiceCount, storm::exceptions::IllegalArgumentException, "Invalid item count of choice labeling.");
52  STORM_LOG_THROW(!this->hasStateValuations() || this->getStateValuations().getNumberOfStates() == stateCount, storm::exceptions::IllegalArgumentException, "Invalid choice count for choice origins.");
53  STORM_LOG_THROW(!this->hasChoiceOrigins() || this->getChoiceOrigins()->getNumberOfChoices() == choiceCount, storm::exceptions::IllegalArgumentException, "Invalid choice count for choice origins.");
54 
55  // Branch on type of nondeterminism
56  if (this->isOfType(ModelType::Dtmc) || this->isOfType(ModelType::Ctmc)) {
57  STORM_LOG_THROW(this->getTransitionMatrix().hasTrivialRowGrouping(), storm::exceptions::IllegalArgumentException, "Can not create deterministic model: Transition matrix has non-trivial row grouping.");
58  STORM_LOG_THROW(stateCount == this->getTransitionMatrix().getRowCount(), storm::exceptions::IllegalArgumentException, "Can not create deterministic model: Number of rows of transition matrix does not match state count.");
59  STORM_LOG_THROW(stateCount == this->getTransitionMatrix().getColumnCount(), storm::exceptions::IllegalArgumentException, "Can not create deterministic model: Number of columns of transition matrix does not match state count.");
60  STORM_LOG_ERROR_COND(!components.player1Matrix.is_initialized(), "Player 1 matrix given for a model that is no stochastic game (will be ignored).");
61  } else if (this->isOfType(ModelType::Mdp) || this->isOfType(ModelType::MarkovAutomaton) || this->isOfType(ModelType::Pomdp) || this->isOfType(ModelType::Smg)) {
62  STORM_LOG_THROW(stateCount == this->getTransitionMatrix().getRowGroupCount(), storm::exceptions::IllegalArgumentException, "Can not create nondeterministic model: Number of row groups (" << this->getTransitionMatrix().getRowGroupCount() << ") of transition matrix does not match state count (" << stateCount << ").");
63  STORM_LOG_THROW(stateCount == this->getTransitionMatrix().getColumnCount(), storm::exceptions::IllegalArgumentException, "Can not create nondeterministic model: Number of columns of transition matrix does not match state count.");
64  STORM_LOG_ERROR_COND(!components.player1Matrix.is_initialized(), "Player 1 matrix given for a model that is no stochastic game (will be ignored).");
65  } else if (this->isOfType(ModelType::S2pg)) {
66  STORM_LOG_THROW(components.player1Matrix.is_initialized(), storm::exceptions::IllegalArgumentException, "No player 1 matrix given for stochastic game.");
67  STORM_LOG_ASSERT(components.player1Matrix->isProbabilistic(), "Can not create stochastic game: There is a row in the p1 matrix with not exactly one entry.");
68  STORM_LOG_THROW(stateCount == components.player1Matrix->getRowGroupCount(), storm::exceptions::IllegalArgumentException, "Can not create stochastic game: Number of row groups of p1 matrix does not match state count.");
69  STORM_LOG_THROW(this->getTransitionMatrix().getRowGroupCount() == components.player1Matrix->getColumnCount(), storm::exceptions::IllegalArgumentException, "Can not create stochastic game: Number of row groups of p2 matrix does not match column count of p1 matrix.");
70  } else {
71  STORM_LOG_THROW(false, storm::exceptions::IllegalArgumentException, "Invalid model type.");
72  }
73 
74  // Branch on continuous/discrete timing
75  if (this->isOfType(ModelType::Ctmc) || this->isOfType(ModelType::MarkovAutomaton)) {
76  STORM_LOG_THROW(components.rateTransitions || components.exitRates.is_initialized(), storm::exceptions::IllegalArgumentException, "Can not create continuous time model: no rates are given.");
77  STORM_LOG_THROW(!components.exitRates.is_initialized() || components.exitRates->size() == stateCount, storm::exceptions::IllegalArgumentException, "Size of exit rate vector does not match state count.");
78  STORM_LOG_THROW(this->isOfType(ModelType::Ctmc) || components.markovianStates.is_initialized(), storm::exceptions::IllegalArgumentException, "Can not create Markov Automaton: no Markovian states given.");
79  } else {
80  STORM_LOG_WARN_COND(!components.rateTransitions && !components.exitRates.is_initialized(), "Rates specified for discrete-time model. The rates will be ignored.");
81  }
82  STORM_LOG_WARN_COND(this->isOfType(ModelType::MarkovAutomaton) || !components.markovianStates.is_initialized(), "Markovian states given for a model that is not a Markov automaton (will be ignored).");
83 
84  // Treat stochastic multiplayer games
85  if (this->isOfType(ModelType::Smg)) {
86  STORM_LOG_THROW(components.statePlayerIndications.is_initialized(), storm::exceptions::IllegalArgumentException, "Can not create stochastic multiplayer game: Missing player indications.");
87  // playerNameToIndexMap is optional.
88  STORM_LOG_THROW(stateCount == components.statePlayerIndications->size(), storm::exceptions::IllegalArgumentException, "Size of state player indications (" << components.statePlayerIndications->size() << ") of SMG does not match state count (" << stateCount << ").");
89  } else {
90  STORM_LOG_WARN_COND(!components.statePlayerIndications.is_initialized(), "statePlayerIndications given for a model that is not a stochastic multiplayer game (will be ignored).");
91  STORM_LOG_WARN_COND(!components.playerNameToIndexMap.is_initialized(), "playerNameToIndexMap given for a model that is not a stochastic multiplayer game (will be ignored).");
92  }

370  template<typename ValueType, typename RewardModelType>
371  void Model<ValueType, RewardModelType>::writeDotToStream(std::ostream& outStream, size_t maxWidthLabel, bool includeLabeling, storm::storage::BitVector const* subsystem, std::vector<ValueType> const* firstValue, std::vector<ValueType> const* secondValue, std::vector<uint_fast64_t> const* stateColoring, std::vector<std::string> const* colors, std::vector<uint_fast64_t>*, bool finalizeOutput) const {
372  outStream << "digraph model {" << std::endl;
373 
374  // Write all states to the stream.
375  for (uint_fast64_t state = 0, highestStateIndex = this->getNumberOfStates() - 1; state <= highestStateIndex; ++state) {
376  if (subsystem == nullptr || subsystem->get(state)) {
377  outStream << "\t" << state;
378  if (includeLabeling || firstValue != nullptr || secondValue != nullptr || stateColoring != nullptr || hasStateValuations()) {
379  outStream << " [ ";
380 
381  // If we need to print some extra information, do so now.
382  if (includeLabeling || firstValue != nullptr || secondValue != nullptr || hasStateValuations()) {
383  outStream << "label = \"" << state;
384  if (hasStateValuations()) {
385  std::string stateInfo = getStateValuations().getStateInfo(state);
386  std::vector<std::string> results;
387  boost::split(results, stateInfo, [](char c) { return c == ',';});
388  storm::utility::outputFixedWidth(outStream, results, maxWidthLabel);
389  }
390  outStream << ": ";
391 
392  // Now print the state labeling to the stream if requested.
393  if (includeLabeling) {
394  outStream << "{";
395  storm::utility::outputFixedWidth(outStream, this->getLabelsOfState(state), maxWidthLabel);
396  outStream << "}";
397  }
398 
399  outStream << this->additionalDotStateInfo(state);
400 
401  // If we are to include some values for the state as well, we do so now.
402  if (firstValue != nullptr || secondValue != nullptr) {
403  outStream << " [";
404  if (firstValue != nullptr) {
405  outStream << (*firstValue)[state];
406  if (secondValue != nullptr) {
407  outStream << ", ";
408  }
409  }
410  if (secondValue != nullptr) {
411  outStream << (*secondValue)[state];
412  }
413  outStream << "]";
414  }
415  outStream << "\"";
416 
417  // Now, we color the states if there were colors given.
418  if (stateColoring != nullptr && colors != nullptr) {
419  outStream << ", ";
420  outStream << " style = filled, fillcolor = " << (*colors)[(*stateColoring)[state]];
421  }
422  }
423  outStream << " ]";
424  }
425  outStream << ";" << std::endl;
426  }
427  }
428 
429  // If this methods has not been called from a derived class, we want to close the digraph here.
430  if (finalizeOutput) {
431  outStream << "}" << std::endl;
432  }
433  }
